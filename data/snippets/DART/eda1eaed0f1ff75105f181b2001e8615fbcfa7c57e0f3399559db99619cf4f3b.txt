/// This function calculates the sum of two numbers
/// Input: two numbers `a` and `b`
/// Output: the sum of `a` and `b`
int sum(int a, int b) {
  return a + b;
}

/// Main function
void main() {
  // Input
  int x = 5;
  int y = 7;

  // Output
  print(sum(x, y)); // Prints 12 as the sum of x and y
}
