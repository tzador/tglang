/// Returns the sum of two numbers 
int add(int a, int b) {
  return a + b;
}

/// The main function 
void main() {
  /// Assigning variables 
  var num1 = 10;
  var num2 = 20;

  /// Calling the add function and storing the result in a variable 
  var result = add(num1, num2);

  /// Printing the result 
  print('The sum of $num1 and $num2 is $result.');
}
