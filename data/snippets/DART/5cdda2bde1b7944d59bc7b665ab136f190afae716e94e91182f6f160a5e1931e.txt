/// This is a complex code snippet in DART
void main() {
  // This variable will store a list of numbers
  List<num> numbers = [1, 2, 3, 4, 5];
  // This variable will store the total sum of the numbers in the list
  num sum = 0;
  // This for loop will iterate through the numbers list
  // and add each number to the sum variable
  for (num number in numbers) {
    sum += number;
  }
  // This will print the final sum
  print("The sum of the numbers is: $sum");
}
