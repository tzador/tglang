/// This is an example function in Dart
void main() {
  print('Hello, World!'); // This will print 'Hello, World!'
  String name = 'John Doe'; // A variable to store a name
  for (var i = 0; i < 10; i++) { // A for loop to iterate 10 times
    print('My name is $name'); // Printing the name using string interpolation
  }
}
