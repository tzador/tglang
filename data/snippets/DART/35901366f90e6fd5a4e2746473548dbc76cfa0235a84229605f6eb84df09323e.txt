//This creates a function that takes in two integer parameters and returns their sum.
//Parameters:
//  x: the first integer
//  y: the second integer
//Output:
//  sum: the sum of the two integers
int sum(int x, int y){ 
  int sum = x + y; //declares and assigns the sum of the two integers to a variable
  return sum; //returns the value of the sum variable
}
