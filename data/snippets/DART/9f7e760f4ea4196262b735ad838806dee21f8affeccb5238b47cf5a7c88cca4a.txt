void main() {
  // This is a complex code snippet in DART.
  var numbers = [1, 2, 3, 4, 5]; // List of numbers.
  var evenNumbers = []; // Empty list to store even numbers.
  numbers.forEach((number) {
    if (number % 2 == 0) { // Checks if number is even.
      evenNumbers.add(number); // Adds even number to evenNumbers list.
    }
  });
  print(evenNumbers); // Prints out list of even numbers.
}
