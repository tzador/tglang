/* This is a multi-line comment in CSS */

h1 {
  /* Selects all h1 elements */
  color: blue; /* Sets text color to blue */
  font-size: 24px; /* Sets font size to 24 pixels */
}

p, span, a {
  /* Selects all p, span, and a elements */
  font-family: sans-serif; /* Sets font family to sans-serif */
  text-decoration: none; /* Removes underline from text */
}

ul li {
  /* Selects all li elements inside a ul element */
  list-style-type: none; /* Removes bullet points */
  color: #FFA500; /* Sets text color to orange */
}

div > button {
  /* Selects all button elements inside a div element */
  background-color: #000000; /* Sets background color to black */
  color: #FFFFFF; /* Sets text color to white */
  border: 2px solid #008000; /* Sets border to 2px solid green */
  padding: 10px; /* Adds 10px of padding to all sides */
  font-weight: bold; /* Sets font weight to bold */
}
