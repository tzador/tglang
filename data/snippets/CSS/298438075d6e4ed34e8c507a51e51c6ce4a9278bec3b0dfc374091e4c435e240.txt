/* Selects all anchor tags within a div with the class "container" */
div.container a {
  /* Applies a font size of 16 pixels */
  font-size: 16px;
  /* Specifies a sans-serif font */
  font-family: sans-serif;
  /* Adds a text decoration of none */
  text-decoration: none;
  /* Adds a padding of 10 pixels to the top and bottom, 20 pixels to the left and right */
  padding: 10px 20px;
  /* Sets the background color to a light grey */
  background-color: #cccccc;
  /* Adds a border of 1 pixel solid black */
  border: 1px solid #000000;
  /* Adds a border radius of 5 pixels */
  border-radius: 5px;
  /* Makes the text uppercase */
  text-transform: uppercase;
  /* Adds a transition effect of 0.3 seconds */
  transition: 0.3s;
}

/* Changes the text color to white when the anchor tag is hovered over */
div.container a:hover {
  color: #ffffff;
}
