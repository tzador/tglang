/* CSS Selectors for HTML elements */
p {
    font-size: 16px; /* sets the font size to 16px for all paragraph elements */
    color: #000000; /* sets the font color to black */
}

h1, h2 {
    text-align: center; /* centers the text for h1 and h2 elements */
    font-weight: bold; /* makes the text bold */
}

ul li {
    padding: 5px; /* adds padding of 5px to all list items within an unordered list */
    font-style: italic; /* makes the text italic for all list items within an unordered list */
}

a {
    text-decoration: none; /* removes the default underlining for links */
    color: #FF0000; /* sets the font color to red */
}

/* CSS Selectors based on class and ID */
#header {
    background-color: #333333; /* sets the background color to a dark grey for the element with the ID 'header' */
    width: 100%; /* sets the width to 100% */
    height: 50px; /* sets the height to 50px */
}

.btn {
    border: 1px solid #000000; /* sets a border of 1px solid black for all elements with the class 'btn' */
    padding: 10px; /* adds padding of 10px */
    text-transform: uppercase; /* transforms the text to uppercase */
}
