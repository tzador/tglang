/* This is a CSS code snippet that creates a custom button with a hover effect */

/* Select the button element and add box sizing */
button {
  box-sizing: border-box; 
}

/* Set a default background color and border for the button */
button {
  background-color: #4CAF50;
  border: none;
}

/* Add styling for the button's text */
button {
  color: white;
  padding: 14px 25px;
  text-align: center;
  text-decoration: none;
  display: inline-block;
}

/* Add a hover effect to the button */
button:hover {
  background-color: #3E8E41;
}

/* Add padding to the hover effect to create a "pressed" look */
button:active {
  background-color: #367f38;
  padding: 15px 25px 13px 27px;
}
