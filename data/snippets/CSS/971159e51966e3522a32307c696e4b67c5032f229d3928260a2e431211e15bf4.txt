/* This is a comment in CSS */
 
/* Selects all paragraphs and changes font to Arial */
p{
font-family: Arial;
}
 
/* Selects all headings and adds a 2 pixel solid red border */
h1, h2, h3{
border: 2px solid red;
}
 
/* Changes the background color of all elements with class "container" to light blue */
.container{
background-color: lightblue;
}
 
/* Selects all links and changes text color to dark blue */
a{
color: darkblue;
}
 
/* Selects the first letter of every element with class "text" and changes font size to 24px */
.text::first-letter{
font-size: 24px;
}
 
/* Changes the color of all elements with class "button" when hovered over */
.button:hover{
color: green;
}
 
/* Adds a 3D box shadow effect to all elements with class "box" */
.box{
box-shadow: 5px 5px 5px 5px rgba(0, 0, 0, 0.5);
}
 
/* Centers all elements with class "image" */
.image{
margin: 0 auto;
display: block;
}
 
/* Floats all elements with class "sidebar" to the right */
.sidebar{
float: right;
}
 
/* Changes the font size of the element with ID "title" to 36px */
#title{
font-size: 36px;
}
