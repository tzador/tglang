/* This is a CSS code snippet that includes multiple
selectors and properties for styling a website */

/* Selects all paragraph elements within a div
and changes the font size, color, and alignment */
div p {
  font-size: 20px;
  color: #333;
  text-align: justify;
}

/* Selects all anchor tags with a class of "link"
and changes the font size, color, and adds a border */
a.link {
  font-size: 16px;
  color: #666;
  border-bottom: 2px solid #ccc;
}

/* Selects the first child element of a list item
and adds a background color and padding */
li:first-child {
  background-color: #eee;
  padding: 5px;
}

/* Selects all image elements with a class of "logo"
and changes the width and adds a border radius */
img.logo {
  width: 150px;
  border-radius: 5px;
}

/* Selects all h2 elements and adds a margin bottom */
h2 {
  margin-bottom: 15px;
}

/* Selects all navigation list items and displays them
in a horizontal row and adds padding and a border radius */
nav li {
  display: inline-block;
  padding: 10px;
  border-radius: 10px;
}

/* Selects all elements with a class of "container"
and adds a max width, margin, and padding */
.container {
  max-width: 1200px;
  margin: 0 auto;
  padding: 20px;
}

/* Selects all elements with an id of "footer"
and adds a background color, text color, and padding */
#footer {
  background-color: #222;
  color: #fff;
  padding: 20px;
}
