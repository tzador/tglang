/* Selects all paragraphs within a specific class and applies a font size and margin */
.my-class p {
  font-size: 16px;
  margin: 20px;
}

/* Selects all unordered lists and applies a red background color */
ul {
  background-color: red;
}

/* Selects all hyperlinks with a specific class and applies a border and text color */
a.my-link {
  border: 1px solid black;
  color: blue;
}

/* Selects the first child of a target element and applies a font weight  */
.target-element:first-child {
  font-weight: bold;
}

/* Selects all input elements with type attribute equal to "text" */
input[type="text"] {
  padding: 10px;
  border: 1px solid black;
}

/* Selects all elements with a data-attribute and applies a font style and color */
[data-attribute] {
  font-style: italic;
  color: #ff9900;
}
