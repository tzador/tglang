/* This is a multi-line comment in CSS */
body {
  background-color: #e6f9ff; /* Sets the background color to a light blue */
  font-family: Helvetica, Arial, sans-serif; /* Defines the font family for text */
  margin: 0; /* Removes default margin from the body element */
}

h1 {
  color: #0066cc; /* Sets the color of h1 elements to a dark blue */
  font-size: 36px; /* Sets the font size to 36 pixels */
  text-align: center; /* Centers the text in h1 elements */
}

ul {
  list-style-type: none; /* Removes bullet points from unordered lists */
  margin: 20px; /* Adds a 20px margin around the unordered list */
  padding: 0; /* Removes default padding from unordered lists */
}

li {
  margin-bottom: 10px; /* Adds a 10px margin at the bottom of each list item */
}

a {
  color: #003366; /* Sets the color for anchor elements to a darker blue */
  text-decoration: none; /* Removes default underline from anchor elements */
}

a:hover {
  /* This is a pseudo-class that changes the style of an element when the user hovers over it */
  text-decoration: underline; /* Underlines the anchor element on hover */
}
