/* Complex code example using CSS */

/* Sets the text color */
color: #333333;

/* Sets the background color */
background-color: #ffffff;

/* Applies a box shadow with specified values */
box-shadow: 2px 2px 5px #cccccc;

/* Applies a border with specified values */
border: 1px solid #999999;

/* Sets the size of the element */
width: 500px;
height: 300px;

/* Applies a border radius */
border-radius: 10px;

/* Hides any overflowing content */
overflow: hidden;

/* Applies a linear gradient background */
background-image: linear-gradient(#ffffff, #cccccc);

/* Sets the font family */
font-family: sans-serif;

/* Applies a hover effect */
a:hover {
  color: #cccccc;
  text-decoration: underline;
}

/* Defines a media query for mobile devices */
@media (max-width: 600px) {
  /* Changes the font size */
  font-size: 12px;
  
  /* Changes the width and height of the element */
  width: 300px;
  height: 200px;
  
  /* Sets the background image */
  background-image: url("mobile-background.jpg");
}
