/*Selects all div elements within the body tag and applies a flex display with a column direction*/
body div {
    display: flex;
    flex-direction: column;
}

/*The following are specific styles for the div elements within the body*/

/*Sets a background color of light blue*/
div {
    background-color: #b3cde0;
}

/*Aligns content to the center*/
div {
    align-items: center;
}

/*Applies a margin of 10 pixels to all sides*/
div {
    margin: 10px;
}

/*Sets a padding of 5 pixels on the top and bottom, and 10 pixels on the left and right*/
div {
    padding: 5px 10px;
}

/*Adds a border with a solid line style, thickness of 2 pixels, and a dark blue color*/
div {
    border: 2px solid #22568d;
}

/*Sets the width to 500 pixels*/
div {
    width: 500px;
}

/*Sets the height to 400 pixels*/
div {
    height: 400px;
}

/*Applies a drop shadow with a horizontal offset of 0 pixels, vertical offset of 3 pixels, blur radius of 5 pixels, and a color of gray*/
div {
    box-shadow: 0 3px 5px gray;
}
