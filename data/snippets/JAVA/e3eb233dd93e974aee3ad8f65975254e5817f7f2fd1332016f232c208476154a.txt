public class Main {
  /*
  This is a multi-line comment with description of the Main class.
  */  
  public static void main(String[] args) {
    // This is a single-line comment.
    
    // Declaring two variables of type int.
    int num1, num2;
    
    // Initializing the variables with values.
    num1 = 10;
    num2 = 5;
    
    // Performing addition and storing the result in a variable.
    int sum = num1 + num2;
    
    // Printing the result to the console.
    System.out.println("Sum of " + num1 + " and " + num2 + " is " + sum);
    
    // Creating a for loop to print numbers from 1 to 10.
    for (int i = 1; i <= 10; i++) {
      System.out.println(i);
    }
    
    // Declaring and initializing a string variable.
    String message = "Hello world!";
    
    // Printing out the string to the console.
    System.out.println(message);
  }
}
