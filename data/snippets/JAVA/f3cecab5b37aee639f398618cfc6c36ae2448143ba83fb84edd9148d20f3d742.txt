// This is a multi-line comment in Java
public class Main { // This is the beginning of a class
  public static void main(String[] args) { // This is the main method
    int x = 5; // This is a variable declaration and initialization
    System.out.println("Value of x: " + x); // This prints the value of x
    for(int i = 0; i < 3; i++) { // This is a for loop
      System.out.println("Loop iteration: " + i); // This prints the current iteration of the loop
      if(i == 2) { // This is an if statement
        System.out.println("Reached the final iteration"); // If the condition is true, this statement will be printed
      }
    }
  }
}
