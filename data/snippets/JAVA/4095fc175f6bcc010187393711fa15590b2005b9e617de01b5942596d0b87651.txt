//This code snippet creates a method that returns the sum of two numbers
//Inputs: int num1 - the first number to be added, int num2 - the second number to be added
//Output: the sum of num1 and num2
public int addNumbers(int num1, int num2) {
	int result = num1 + num2; //calculates the sum and stores it in a variable
	return result; //returns the sum
}
