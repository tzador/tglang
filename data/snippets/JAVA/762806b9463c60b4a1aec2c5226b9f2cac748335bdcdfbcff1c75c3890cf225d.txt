public static void main(String args[]) {
int x = 10;
int y = 20;
if(x > 5 && y < 25) { //if statement
System.out.println("x is greater than 5 and y is less than 25."); //prints the statement if condition is met
} else { //else statement
System.out.println("x is not greater than 5 or y is not less than 25."); //prints this statement if condition is not met
}
}
