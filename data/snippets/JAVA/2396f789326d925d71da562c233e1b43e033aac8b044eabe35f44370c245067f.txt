// This code snippet is a class named "User"
// It contains two private variables, a constructor, and two public methods

public class User {

    // Private variables for storing user information
    private String name;
    private int age;

    // Constructor for creating new instances of User class
    public User(String name, int age) {
        this.name = name;
        this.age = age;
    }

    // Public method for setting the user's name
    public void setName(String name) {
        this.name = name;
    }

    // Public method for setting the user's age
    public void setAge(int age) {
        this.age = age;
    }
}
