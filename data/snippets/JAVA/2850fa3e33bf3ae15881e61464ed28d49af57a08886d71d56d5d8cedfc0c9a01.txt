/**
 * This is a complex program that performs a series of calculations
 * and outputs the result.
 */

// Define variables
double num1 = 5.5;
double num2 = 2.2;
int num3 = 10;

// Perform calculations
double result1 = (num1 + num2) / num3;

// Print results
System.out.println("The result of (num1 + num2) / num3 is: " + result1);
System.out.println("The result of num1 * num3 is: " + result2);
System.out.println("The result of num3 - num1 is: " + result3);
