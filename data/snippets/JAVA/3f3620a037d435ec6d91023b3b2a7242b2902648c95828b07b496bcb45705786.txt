// This program calculates the sum of two numbers and checks if it is greater than 10

//Declaring and initializing variables
int num1 = 5;
int num2 = 8;
int sum = num1 + num2; //Calculating sum

//Checking if sum is greater than 10
if(sum > 10){
    System.out.println("Sum is greater than 10"); //Printing output
} else {
    System.out.println("Sum is not greater than 10"); //Printing output
}
