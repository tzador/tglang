; This is a sample INI file

# Comments in INI files usually start with a semicolon or a hash character

[SectionOne] ; This is a section in the INI file
key1=value1 ; This is a key-value pair in the section

[AnotherSection] ; This is another section in the INI file
key2=value2
key3=value3
key4=value4
key5=value5

[SectionThree]
key6=value6

; Sections and keys do not have to be in a specific order
