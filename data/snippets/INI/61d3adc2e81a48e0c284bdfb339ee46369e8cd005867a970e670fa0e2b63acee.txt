; Sample INI configuration file

; Comments start with a semicolon and are ignored by parsers

[database]
; Database section
server = localhost ; setting the server to localhost
port = 3306 ; setting the port to 3306
username = admin ; setting the username to admin
password = password123 ; setting the password to password123

[server]
; Server section
name = My Server ; setting the server name to My Server
ip = 192.168.1.1 ; setting the IP address to 192.168.1.1
port = 8080 ; setting the port to 8080
