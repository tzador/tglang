; This is a comment in INI format

; First section
[section 1] ; This is the section header

; Key-value pairs
key1 = "value1" ; This is a string value
key2 = 123 ; This is an integer value
key3 = True ; This is a boolean value

; Nested section
[section 2]
sub_key1 = "sub_value1"
sub_key2 = 456
sub_section1
    sub_sub_key1 = "sub_sub_value1"
    sub_sub_key2 = False

; Another nested section
[section 3]
sub_key3 = "sub_value3"
sub_section2 ; This is an empty section
