;This is a sample configuration file in INI format.

[General] ;This section contains general settings for the program
name = My Program ;The name of the program
version = 1.0 ;The version number of the program

[Database] ;This section contains settings for the database
server = localhost ;The server IP address
username = admin ;The username for the database
password = 12345 ;The password for the database

[Themes] ;This section contains settings for the program's themes
default = light ;The default theme for the program
dark = #1C1C1C ;The dark theme color

[Files] ;This section contains settings for file paths
data_dir = C:\ProgramData\MyProgram ;The directory for program data
log_dir = C:\ProgramData\MyProgram\Logs ;The directory for program logs
