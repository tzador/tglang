; INI configuration file
; This is a sample configuration file for a web application
; Comments are preceded by a semicolon (;)

[application] ; Application settings section
name = "My Web App" ; Name of the application
version = 1.0.0 ; Version number

[database] ; Database settings section
host = "localhost" ; Database host address
port = 3306 ; Database port number
username = "admin" ; Database username
password = "secretpassword" ; Database password

[server] ; Server settings section
host = "127.0.0.1" ; Server host address
port = 3000 ; Server port number
mode = "production" ; Server mode (production or development)
