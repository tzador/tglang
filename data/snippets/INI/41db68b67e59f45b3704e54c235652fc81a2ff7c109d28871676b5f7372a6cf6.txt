## Example INI configuration file

; This is a comment line
[Section 1] ; This is a section

key1=value1 ; This is a key-value pair

key2="value2" ; This is a key-value pair with quotation marks

key3 = "multiple words value with spaces" ; This is a key-value pair with spaces in value

[Section 2] ; Adding another section

; This is a boolean key-value pair
key4 = true

[Section 3]
; This is an integer key-value pair
key5 = 123

[Section 4]
; This is a list key-value pair
key6 = item1, item2, item3
