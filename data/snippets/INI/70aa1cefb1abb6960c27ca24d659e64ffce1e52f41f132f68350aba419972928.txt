; This is a comment in INI format
# Another type of comment in INI format

[Section A] ; This is a section name
Key1=Hello ; This is a key-value pair
Key2=123
Key3="This is a long string!"

[Section B] ; Another section
Key1=Bye
Key2=456
Key3="Another long string!"

[Section C] ; Yet another section
Key1=123 ; Integer value
Key2=3.14159 ; Float value
Key3=True ; Boolean value
Key4=False

[Section D] ; Final section
Key1=1, 2, 3 ; List value
Key2=red, green, blue ; List value
Key3={"name": "John", "age": 30} ; Nested dictionary value
