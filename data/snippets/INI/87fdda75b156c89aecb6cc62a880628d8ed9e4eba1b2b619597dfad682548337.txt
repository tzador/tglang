; This is a comment in INI format
    
[Section1] ; This is a section header in INI format

key1 = value1 ; This is a key-value pair in INI format

key2 = true ; This is a boolean key-value pair in INI format

[Section2] ; This is another section header in INI format

key3 = "This is a string" ; This is a key-value pair with a string value
key4 = 10 ; This is a key-value pair with an integer value
key5 = 3.14 ; This is a key-value pair with a float value

[Section3] ; This is a third section header in INI format 

key6 = 1,2,3 ; This is a key-value pair with a list of integer values
key7 = "alpha","beta","gamma" ; This is a key-value pair with a list of string values
key8 = true,false,true ; This is a key-value pair with a list of boolean values
