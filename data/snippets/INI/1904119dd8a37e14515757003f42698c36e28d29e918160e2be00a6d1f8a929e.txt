; This is a sample INI file
; It contains multiple sections and keys with values

[Section1] ; First section
key1 = value1 ; Key1 and its value
key2 = "This is a string value"

[Section2] ; Second section
key1 = value with spaces ; Key1 and its value with spaces
key2 = 3.14 ; Key2 and its floating point value
key3 = true ; Key3 and its boolean value

[Section3] ; Third section
key1 = 1, 2, 3, 4, 5 ; Key1 and its list of values
key2 = [1, 2, 3] ; Key2 and its list of values inside brackets
key3 = This is a long
multi-line value
; Comments can be added after a semicolon (;)
