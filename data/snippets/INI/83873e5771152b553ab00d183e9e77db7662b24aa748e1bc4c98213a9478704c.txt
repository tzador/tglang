; This is an example INI file
; Comments start with a semicolon

[Section1] ; Section name must be enclosed in square brackets
key1 = value1 ; Key-value pair must be separated by an equal sign
key2 = value2 ; Multiple key-value pairs can be in a section

[Section2]
key3 = value3
key4 = value4
key5 = value5

[Section2.Subsection1] ; Subsections can be created using a dot notation
key6 = value6

[Section3]
boolean = true ; Booleans can be represented as true or false
float = 3.14 ; Float values can be represented with a decimal point
string = "This is a string" ; Strings must be enclosed in double quotes
