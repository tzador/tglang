; This is a sample INI file
; Contains settings for a game

[section1]              ; defines a new section
setting1 = value1       ; first setting with corresponding value
setting2 = 5            ; second setting with numerical value
setting3 = true         ; third setting with boolean value

[section2]              ; another section
setting1 = Hello        ; first setting with string value
setting2 = world        ; second setting with string value
setting3 = 3.14159      ; third setting with float value
setting4 = option1, option2, option3     ; fourth setting with multiple options separated by comma
; Comments can be added with a semicolon at the beginning of the line
