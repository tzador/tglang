# This is a sample configuration file
; Lines that start with semicolons are comments
[section1] ; Section for general settings
key=value ; This is a key value pair, where 'key' is the key and 'value' is the value
array[] = 1 ; This shows that array keys don't need to be in order
array[] = 2 ; The values will be added to the array in the order they appear
[section2] ; Section for advanced settings
key2 = "This value has quotes" ; Values can be enclosed in quotes
semicolon=value; ; You can escape special characters such as semicolons by adding semicolons themselves
[empty_section] ; Sections don't need to have any keys or values in them
