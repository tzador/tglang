; This is a sample INI file

; Global section
[Global]
; Set global options
globalOption = true
globalOption2 = false

; Server section
[Server]
; Specify server options
serverOption = "localhost"
serverOption2 = 8080
serverOption3 = "user:password"

; Paths section
[Paths]
; Set data and log paths
dataPath = C:\Users\Example\Documents\data
logPath = C:\Users\Example\Documents\logs

; Database section
[Database]
; Specify database options
dbOption = "SQL"
dbOption2 = "mysql://user:password@localhost:3306/db_name"

; Users section
[Users]
; Add user information
user1 = "John"
user2 = "Jane"
user3 = "Bob"
