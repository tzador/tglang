; Sample INI file 
[section1] ; Comments can be added after a semi-colon
key1="value1" ; Keys and values are separated by an equals sign
key2=123 ; Values can be numbers
key3=true ; Values can be Boolean
key4=example@example.com ; Values can also be strings
[section2] ; Sections can be used to group related keys and values
key1=subkey1,value1 ; Keys can have multiple values separated by a comma
key2=subkey2,value2
key3=subkey3,value3
key4=subkey4,value4
[section3] ; Empty sections can also be created
; Keys and values can be added to the sections later on
key1="new value"
key2="another value"
