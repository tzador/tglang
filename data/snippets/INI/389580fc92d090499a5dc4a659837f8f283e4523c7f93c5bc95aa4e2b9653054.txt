; This is a comment in INI file format
[section]
key1 = value1 ; This is a key-value pair in a section
key2 = "value2 with a space" ; Quotes can be used for values with spaces
key3 = 12345 ; You can also have numeric values
key4 = true ; Booleans are represented as 'true' or 'false'
key5 = 
    line 1
    line 2
    line 3 ; Multiline values are created using multiple lines
key6 = [sub-section] ; Sub-sections can be created using square brackets
subkey1 = true ; This subkey belongs to the sub-section
[sub-section2]
subkey2 = "complex value with special characters - %$#@" ; Sub-sections can have their own key-value pairs
