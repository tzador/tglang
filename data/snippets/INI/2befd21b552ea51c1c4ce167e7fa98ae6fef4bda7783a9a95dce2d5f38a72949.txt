#This is a sample INI file
;Comments start with a semicolon and are ignored
[Section 1]
;Each section should be enclosed in square brackets
key1 = value1 #Keys are on the left side, values on the right
key2 = value2
key3 = true
key4 = "This is a string" #Strings should be enclosed in double quotes

[Section 2]
#Another section can be created by adding another set of square brackets
key5 = 3.14 #Values can also be numbers
key6 = false
key7 = [1, 2, 3] #Lists can be defined using square brackets
key8 = {"a": 1, "b":2} #Dictionaries can be defined using curly brackets
