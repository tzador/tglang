; Lines starting with a hash or semicolon are comments
[section1] ; Sections are enclosed in square brackets
key1=value1 ; Keys and values are separated by an equal sign

[section2]

list_values = item1, item2, item3

[section4]
# Values can also be nested
list_values = item1, sub_list1, sub_item1, sub_item2, sub_item3, item2
