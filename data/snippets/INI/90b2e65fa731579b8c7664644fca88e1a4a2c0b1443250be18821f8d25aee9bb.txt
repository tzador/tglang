# Comments start with a semicolon

; Define a new section called 'Network'
[Network]

; Set the network interface
Interface = en0

; Set the IP address to be used
IP = 192.168.1.5

; Set the subnet mask for the network
Subnet = 255.255.255.0

; Define a new section called 'Database'
[Database]

; Set the database type
Type = MySQL

; Set the username for database access
Username = admin

; Set the password for database access
Password = password123

; Set the database name
DatabaseName = my_database
