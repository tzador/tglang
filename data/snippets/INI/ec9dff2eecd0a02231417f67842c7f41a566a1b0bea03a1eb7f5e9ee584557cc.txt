; Global settings section
[GLOBAL]
setting1 = value1
setting2 = value2

; Section for specific application settings
[APP_SETTINGS]

setting2 = setting2_value

[NETWORK_SETTINGS]
; Comment for IP address

; Comment for subnet mask
SubnetMask = 255.255.255.0
; Comment for gateway
Gateway = 192.168.1.254
; Comment for DNS server
