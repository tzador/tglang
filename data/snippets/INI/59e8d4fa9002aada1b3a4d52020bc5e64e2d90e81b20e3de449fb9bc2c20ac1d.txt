; INI file example
[section] ; This is a comment on a section
key=value ; This is a key-value pair
int_key=123 ; This is an integer value
float_key=3.14 ; This is a float value
array_key[]=item1 ; This is an array with one value
array_key[]=item2 ; This is an array with two values
[another_section] ; This is another section
string_array[]=hello ; This is a string array with one value
string_array[]=world ; This is a string array with two values
multi_line_string=""" ; This is a multi-line string
This is line 1
This is line 2
This is line 3
""" ; This is the end of the multi-line string
