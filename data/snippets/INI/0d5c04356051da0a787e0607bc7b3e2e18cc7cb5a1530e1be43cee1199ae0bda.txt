; This is a configuration file example using the INI format

[settings] ; Section for general settings
name = "John Doe" ; Name of the user
age = 25 ; Age of the user
gender = male ; Gender of the user
email = johndoe@example.com ; Email address of the user

[account] ; Section for account information
username = johndoe ; Username for logging in
password = Pa$$word! ; Password for logging in
account_type = premium ; Type of account, can be free or premium

[notifications] ; Section for notification settings
enable_email = true ; Enable email notifications
enable_push = false ; Enable push notifications
min_level = 3 ; Minimum notification level (1 = lowest, 5 = highest)

[database] ; Section for database configuration
host = localhost ; Database host address
port = 3306 ; Database port number
username = root ; Database username
password = Pa$$w0rd ; Database password
database_name = my_database ; Name of the database
