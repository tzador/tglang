#include <stdio.h>

int main()
{
    // initialize variables
    int num1 = 5;
    int num2 = 7;
    float num3 = 10.5;

    /* print a message */
    printf("Welcome to my code snippet!\n");

    /* calculate the sum of num1 and num2 */
    int sum = num1 + num2;

    /* print the sum */
    printf("The sum of %d and %d is %d\n", num1, num2, sum);

    /* calculate the product of sum and num3 */
    float product = sum * num3;

    /* print the product */
    printf("The product of %d and %f is %f\n", sum, num3, product);

}
