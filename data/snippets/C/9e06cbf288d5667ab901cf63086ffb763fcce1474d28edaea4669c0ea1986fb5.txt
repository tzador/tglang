// This program checks whether a given number is prime or not

#include <stdio.h>

int main() {
   int num, i, flag = 0;

   printf("Enter a positive integer: ");
   scanf("%d", &num);

   for (i = 2; i <= num / 2; ++i) {

      // condition for non-prime number
      if (num % i == 0) {
         flag = 1;
         break;
      }
   }

   if (flag == 0)
      printf("%d is a prime number.", num);
   else
      printf("%d is not a prime number.", num);

   return 0;
}
