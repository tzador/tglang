#include<stdio.h>

int main(void) {
  // Declare and initialize variables
  int first_num = 5;
  int second_num = 10;
  int result;

  // Calculate the sum of the two numbers
  result = first_num + second_num;

  // Print the result to the console
  printf("The sum of %d and %d is %d", first_num, second_num, result);

  return 0;
}
