#include <stdio.h> // include header file for input/output operations

int main() { // main function
    int num1 = 5; // declare and initialize variable 'num1' with value 5
    int num2 = 10; // declare and initialize variable 'num2' with value 10
    int result = num1 + num2; // declare and initialize variable 'result' with sum of num1 and num2

    printf("The result of num1 + num2 is: %d", result); // print the result

    return 0; // return 0 to indicate successful execution
}
