#include <stdio.h> // includes standard input/output header file

int main() { // main function

    /* Initializing variables */
    int num1 = 5;
    int num2 = 10;
    int result;

    /* Addition operation */
    result = num1 + num2;

    /* Printing the result */
    printf("The sum of %d and %d is %d", num1, num2, result);

    return 0; // return statement
}
