#include <stdio.h> //Including standard input/output library

int main() { //Main function

  int num1 = 5; //Declaring and initializing variable num1 with value 5
  int num2 = 10; //Declaring and initializing variable num2 with value 10
  int sum = num1 + num2; //Calculating the sum of num1 and num2 and storing it in the variable sum

  printf("The sum of %d and %d is %d\n", num1, num2, sum); //Printing the sum using printf function
  
  return 0; //Returning 0 to end the program
}
