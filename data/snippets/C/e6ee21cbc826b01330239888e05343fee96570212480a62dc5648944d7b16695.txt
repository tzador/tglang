// This program calculates the sum of two numbers

#include <stdio.h>      // Header file for standard input/output operations

int main() {

    // Declaration and initialization of variables
    int num1 = 10;
    int num2 = 5;
    int sum = 0;

    // Calculating the sum
    sum = num1 + num2;

    // Printing the result
    printf("The sum of %d and %d is %d", num1, num2, sum);

    return 0;
}
