#include <stdio.h>

int main()
{
  int n, reverse = 0, original, remainder;
  
  printf("Enter an integer: ");
  scanf("%d", &n);
  
  original = n;
  
  //reversing integer
  while (n != 0)
  {
    remainder = n % 10;
    reverse = reverse * 10 + remainder;
    n /= 10;
  }
  
  printf("Reverse of %d is %d", original, reverse);
  return 0;
}
