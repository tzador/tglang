// This is a code snippet in C

#include<stdio.h>                        // This header file imports the standard input/output library

int main() {                             // This is the main function that all C programs must have

    int num1 = 10;                       // Declaring and assigning the value 10 to variable num1
    float num2 = 23.45;                  // Declaring and assigning the value 23.45 to variable num2
    char letter = 'A';                   // Declaring and assigning character 'A' to variable letter

    printf("num1 = %d\n", num1);         // Printing the value of num1 using the printf function and %d placeholder
    printf("num2 = %f\n", num2);         // Printing the value of num2 using the printf function and %f placeholder
    printf("letter = %c\n", letter);     // Printing the value of letter using the printf function and %c placeholder

    return 0;                            // Returning 0 to indicate successful execution of the program
}

// Output:
// num1 = 10
// num2 = 23.45
// letter = A
