#include <stdio.h>

// Function to calculate the factorial of a number
int factorial(int n){
    if(n == 0){
        return 1;
    }
    else{
        return n * factorial(n-1);
    }
}

int main(){
    int num;
    printf("Enter a positive integer: ");
    scanf("%d", &num);

    printf("Factorial of %d is %d", num, factorial(num));
    
    return 0;
}
