/* This is a program that calculates the factorial of a number */
#include <stdio.h> 

int main() 
{ 
   int num, i, fact = 1; 

   printf("Enter a number: "); 
   scanf("%d", &num); 

   /* Loop to calculate factorial */ 
   for (i = 1; i <= num; i++) 
        fact = fact * i; 

   printf("Factorial of %d is %d", num, fact); 

   return 0; 
}
