#include <stdio.h>

/* This program takes in two numbers and prints the sum */
int main(void) {

    // Declare and initialize the variables
    int num1 = 5;
    int num2 = 10;
    int sum = num1 + num2;

    // Print the sum
    printf("The sum of %d and %d is %d.", num1, num2, sum);

    return 0;
}
