#include <stdio.h>

int main() {
    // This program calculates the sum of two numbers
    int num1 = 5; // first number
    int num2 = 10; // second number
    int sum = num1 + num2; // calculates sum
    
    // Print the result
    printf("The sum of %d and %d is %d", num1, num2, sum);
    
    return 0;
}
