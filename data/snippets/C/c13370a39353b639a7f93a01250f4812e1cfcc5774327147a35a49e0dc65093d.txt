#include <stdio.h>

int main() {
    int num1 = 2; //initialize first number
    int num2 = 6; //initialize second number
    int sum; //initialize sum variable

    sum = num1 + num2; //calculate sum and assign to sum variable

    printf("%d + %d = %d", num1, num2, sum); //print out sum in the format "num1 + num2 = sum"

    return 0;
}
