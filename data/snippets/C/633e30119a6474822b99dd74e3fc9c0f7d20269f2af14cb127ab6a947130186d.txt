/* This is a code snippet written in the programming language C */
#include <stdio.h> // This is a preprocessor directive that includes the standard input-output header file

int main() { // This is the main function, the entry point of the program
    int num1 = 10; // Initializing an integer variable with the value 10
    int num2 = 5; // Initializing another integer variable with the value 5
    int sum = num1 + num2; // Performing addition and storing the result in a variable
    printf("%d", sum); // Printing out the sum using the printf function
    return 0; // Returning 0 to indicate that the program executed successfully
}
