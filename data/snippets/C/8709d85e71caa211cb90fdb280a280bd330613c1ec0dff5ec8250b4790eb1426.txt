/* This C program demonstrates an example of bubble sort algorithm */

#include <stdio.h> 

/* Function to bubble sort an array */
void bubbleSort(int array[], int n) 
{ 
   int i, j; 
   for (i = 0; i < n-1; i++)
   { 
       for (j = 0; j < n-i-1; j++)
       { 
           if (array[j] > array[j+1])
           { 
              int temp = array[j]; 
              array[j] = array[j+1];
              array[j+1] = temp;
           } 
       } 
   } 
} 

// Function to print the sorted array
void printArray(int array[], int n) 
{ 
   int i; 
   for (i=0; i < n; i++) 
       printf("%d ", array[i]); 
   printf("\n"); 
} 

// Main function
int main()
{ 
    int array[] = {64, 34, 25, 12, 22, 11, 90}; 
    int n = sizeof(array)/sizeof(array[0]); 
    bubbleSort(array, n); 
    printArray(array, n); 
    return 0; 
} 

/* Output:
11 12 22 25 34 64 90
*/
