#include<stdio.h> //this is a preprocessor directive for input/output functions

int main() {

    //declare and initialize variables
    int num1 = 5;
    int num2 = 6;
    int result;

    //perform addition
    result = num1 + num2;

    //print result to console
    printf("The result of %d + %d = %d", num1, num2, result);

    return 0;
}
