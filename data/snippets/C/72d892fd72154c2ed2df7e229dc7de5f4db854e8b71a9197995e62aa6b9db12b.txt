#include <stdio.h> // This is a preprocessor directive to import the standard input/output library

int main() { // This is the main function, the starting point of the program
    int num1 = 10; // Define a variable named num1 with a value of 10
    int num2 = 20; // Define a variable named num2 with a value of 20 
    int sum = num1 + num2; // Define a variable named sum and set its value to the sum of num1 and num2 
    printf("The sum of %d and %d is %d", num1, num2, sum); // Print the sum in the specified format
    return 0; // Return 0 to indicate successful execution of the program
}
