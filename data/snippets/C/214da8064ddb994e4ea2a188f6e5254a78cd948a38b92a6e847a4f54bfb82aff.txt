#include<stdio.h>

int main()
{
    /* this code calculates the sum of two integers and stores the result in a variable */
    
    int num1 = 10, num2 = 20; // declaring and initializing two integers
    
    int sum = num1 + num2; // storing the sum of the two integers in a variable
    
    printf("The sum of %d and %d is %d.", num1, num2, sum); // printing the result
    
    return 0;
}
