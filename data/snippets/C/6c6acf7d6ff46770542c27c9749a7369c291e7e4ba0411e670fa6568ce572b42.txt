#include <stdio.h>
int main()
{
    // This is a comment
    int num = 5; // This is an integer variable
    float pi = 3.14; // This is a float variable
    char letter = 'A'; // This is a character variable
    double bigNum = 1234567890; // This is a double variable
    printf("The value of num is %d\n", num); 
    printf("The value of pi is %f\n", pi);
    printf("The value of letter is %c\n", letter);
    printf("The value of bigNum is %lf\n", bigNum);
    return 0;
}
