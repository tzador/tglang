function quadratic_solver(a, b, c)
    """
    This function solves the quadratic equation axÂ² + bx + c = 0
    using the quadratic formula
    """
    # Calculate discriminant
    delta = b^2 - 4*a*c
    # Check for complex roots
    if delta < 0
        println("The roots of the equation are complex numbers.")
        return
    end
    # Calculate square root of discriminant
    sqrt_delta = sqrt(delta)
    # Calculate solutions
    x1 = (-b + sqrt_delta)/(2*a)
    x2 = (-b - sqrt_delta)/(2*a)
    # Print solutions
    println("The solutions of the equation are:")
    println("x1 = $x1 and x2 = $x2")
    # Return solutions
    return x1, x2
end

# Example usage
a = 1
b = -5
c = 6
println(quadratic_solver(a, b, c))

# Output
The solutions of the equation are:
x1 = 3.0 and x2 = 2.0
(3.0, 2.0)
