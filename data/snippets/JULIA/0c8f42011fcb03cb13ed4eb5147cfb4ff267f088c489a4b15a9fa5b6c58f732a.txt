# Define a function called "add"
function add(x,y)
    return x + y # Return the sum of two inputs
end

# Define a function called "multiply"
function multiply(x,y)
    return x * y # Return the product of two inputs
end

# Store the result of adding 2 and 3 in a variable "sum"
sum = add(2,3)

# Store the result of multiplying 4 and 5 in a variable "product"
product = multiply(4,5)

# Print the values of "sum" and "product"
println("Sum: ", sum)
println("Product: ", product)
