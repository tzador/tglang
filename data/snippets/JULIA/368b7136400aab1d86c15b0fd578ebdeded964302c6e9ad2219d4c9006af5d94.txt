# Define a function named "calculate_product"
function calculate_product(x, y)
    # Calculate the product between x and y
    product = x * y
    
    # Assign the value of the product to a new variable "final_product"
    final_product = product
    
    # Print the final product
    println("The final product is: ", final_product)
    
    # Return the final product
    return final_product
end

# Call the function with input values of 5 and 3
calculate_product(5, 3)
