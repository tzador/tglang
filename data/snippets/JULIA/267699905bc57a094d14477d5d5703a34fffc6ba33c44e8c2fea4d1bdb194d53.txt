function power(n)
    """
    This function calculates the power of a number n.
    Parameters:
        n: the number to be calculated
    Returns:
        the power of the number
    """
    x = 1
    y = 2.5
    # initialize x and y 
    
    while y > 0
        x = x * n
        y = y - 1
        # iterate through the loop y times
    
    return x
    # return the final result
end

result = power(5)
# assign the function output to result variable 

println("The power of 5 is $result.")
