# Define a function that calculates the sum of two numbers

function calculate_sum(x, y)
    z = x + y # Assign z to the sum of x and y
    return z # Return z as the output of the function
end

# Call the function and assign the output to a variable

result = calculate_sum(5, 10) # Assign the output of calculate_sum function with x = 5 and y = 10 to the variable result

# Print the result

println("The sum of 5 and 10 is ", result) # Print the result in a formatted string

# Expected output: The sum of 5 and 10 is 15
