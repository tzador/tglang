# Defining a function that calculates the average of two numbers
function average(x, y)
    return (x + y)/2  # return the result of adding x and y, then dividing by 2
end

# Declaring two variables and assigning values
a = 5
b = 3

# Printing the average of a and b using the defined function
println("The average of $a and $b is $(average(a, b))")
