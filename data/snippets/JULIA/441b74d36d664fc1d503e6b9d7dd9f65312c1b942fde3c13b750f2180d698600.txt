#Define a function to perform quicksort algorithm
function quicksort(arr, first, last)

    #Check base case
    if first >= last
        return
    end

    i = first

    while i <= j
        while arr[i] < pivot

        end
        while arr[j] > pivot
            j -= 1
        end
        if i <= j
            temp = arr[i]
            arr[i] = arr[j]
            arr[j] = temp
            i += 1
            j -= 1
        end
    end
    
    #Recursively call quicksort on subarrays
    quicksort(arr, first, j)

end
