(* Example F# code for calculating the area of a circle *)
let pi = 3.14159 // define pi as a constant
let radius = 5.0 // set the radius of the circle
let area = pi * radius ** 2.0 // calculate the area using the formula pi * r^2
// print the result to the console
printfn "The area of the circle with a radius of %f is %f" radius area
(* Output: The area of the circle with a radius of 5.0 is 78.53975 *)
