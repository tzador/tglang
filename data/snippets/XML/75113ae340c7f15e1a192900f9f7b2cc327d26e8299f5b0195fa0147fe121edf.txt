<!DOCTYPE html> //Declares document type as HTML
<html> //Opening tag for the HTML document
	<head> //Opening tag for the head section
		<title>My Website</title> //Title of the website
		<link rel="stylesheet" type="text/css" href="styles.css"/> //Linking external CSS file
	</head> //Closing tag for the head section
	<body> //Opening tag for the body section
		<h1>Hello, World!</h1> //Heading tag with text
		<p>This is my website.</p> //Paragraph tag with text
		<ul> //Opening tag for unordered list
			<li>Item 1</li> //List item 1
			<li>Item 2</li> //List item 2
			<li>Item 3</li> //List item 3
		</ul> //Closing tag for unordered list
		<form> //Opening tag for form
			<label for="name">Name:</label> //Label for input field
			<input type="text" id="name" name="name" required/> //Input field for name, with required attribute
			<button type="submit">Submit</button> //Submit button
		</form> //Closing tag for form
	</body> //Closing tag for the body section
</html> //Closing tag for the HTML document
