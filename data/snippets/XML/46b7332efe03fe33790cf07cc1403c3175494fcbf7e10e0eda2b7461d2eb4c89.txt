<?xml version="1.0" encoding="UTF-8"?> <!-- This line specifies the version and encoding of the XML file -->
<!DOCTYPE html> <!-- This is the document type declaration for an HTML document -->
<html> <!-- This is the root element of the HTML document -->
  <head> <!-- The head element contains information about the document -->
    <title>My Website</title> <!-- This is the title of the webpage -->
    <meta charset="utf-8"> <!-- This specifies the character encoding of the webpage -->
    <meta name="description" content="A website for showcasing my work"> <!-- This is a meta tag for providing a brief description of the webpage -->
    <meta name="keywords" content="website, portfolio, work"> <!-- This is a meta tag for providing keywords for the webpage -->
  </head>
  <body> <!-- The body element contains the visible content of the webpage -->
    <h1>Hello World!</h1> <!-- This is a heading element -->
    <p>This is my website where I showcase my work.</p> <!-- This is a paragraph element -->
    <ul> <!-- This is an unordered list element -->
      <li>Project 1</li> <!-- This is a list item for project 1 -->
      <li>Project 2</li> <!-- This is a list item for project 2 -->
      <li>Project 3</li> <!-- This is a list item for project 3 -->
    </ul>
    <hr> <!-- This is a horizontal rule element -->
    <ol> <!-- This is an ordered list element -->
      <li>Blog post 1</li> <!-- This is a list item for blog post 1 -->
      <li>Blog post 2</li> <!-- This is a list item for blog post 2 -->
      <li>Blog post 3</li> <!-- This is a list item for blog post 3 -->
    </ol>
  </body>
</html>
