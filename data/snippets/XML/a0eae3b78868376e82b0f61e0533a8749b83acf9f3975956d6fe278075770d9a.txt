<!-- This is a comment in XML -->
<!DOCTYPE html> <!-- This is the DOCTYPE declaration -->
<html> <!-- This is the opening root element -->
  <head> <!-- This is the head element, where metadata and scripts are placed -->
    <title>This is the title of the webpage</title> <!-- This is the title element -->
    <meta charset="UTF-8"> <!-- This is the metadata element for specifying the character encoding of the document -->
    <script type="text/javascript" src="script.js"></script> <!-- This is the script element for linking an external javascript file -->
  </head>
  <body> <!-- This is the body element, where all the content is placed -->
    <h1>Welcome to my webpage!</h1> <!-- This is a heading element -->
    <p>I hope you enjoy your stay.</p> <!-- This is a paragraph element -->
    <ul> <!-- This is an unordered list element -->
      <li>This is the first item in the list.</li> <!-- This is a list item element -->
      <li>This is the second item in the list.</li>
      <li>This is the third item in the list.</li>
    </ul>
  </body>
</html>
