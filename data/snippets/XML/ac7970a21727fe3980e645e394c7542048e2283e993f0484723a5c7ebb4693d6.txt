<!-- This is an example of an XML document with a complex structure -->
<catalog>
    <!-- This element is for catalog metadata -->
    <metadata>
        <!-- The creator of the catalog -->
        <creator>John Smith</creator>
        <!-- The date the catalog was created -->
        <creation_date>2021-09-20</creation_date>
    </metadata>
    <!-- This element is for the items in the catalog -->
    <items>
        <!-- The first item in the catalog -->
        <item>
            <!-- The name of the item -->
            <name>Product A</name>
            <!-- The description of the item -->
            <description>This is a description of Product A</description>
            <!-- The price of the item -->
            <price>$20.00</price>
            <!-- Optional attribute for the color of the product -->
            <color>red</color>
        </item>
        <!-- The second item in the catalog -->
        <item>
            <!-- The name of the item -->
            <name>Product B</name>
            <!-- The description of the item -->
            <description>This is a description of Product B</description>
            <!-- The price of the item -->
            <price>$15.00</price>
        </item>
        <!-- The third item in the catalog -->
        <item>
            <!-- The name of the item -->
            <name>Product C</name>
            <!-- The description of the item -->
            <description>This is a description of Product C</description>
            <!-- The price of the item -->
            <price>$12.50</price>
            <!-- Optional attribute for the size of the product -->
            <size>small</size>
            <!-- Optional attribute for the color of the product -->
            <color>blue</color>
        </item>
    </items>
</catalog>
