<!-- This is a sample XML document -->

<!-- Defines the root element of the document -->
<root>

    <!-- Comments can be added within tags -->
    <element1>Value 1</element1>

    <!-- Attributes can be included in tags -->
    <element2 attribute="value">Value 2</element2>

    <!-- Elements can be nested within each other -->
    <parent>
        <child>Child element</child>
        <child>Another child element</child>
    </parent>

    <!-- Elements can have both attributes and nested elements -->
    <complex attribute="value">
        <subelement>Subelement</subelement>
    </complex>

</root>
