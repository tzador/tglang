<!-- This is an example of a complex XML code snippet -->
<html>
  <head>
    <!-- This is the head tag with child title and meta tags -->
    <title>This is an example title</title>
    <meta charset="UTF-8" />

    <!-- This is a comment within the head tag -->
    <!-- It is important to always include a charset meta tag for proper encoding -->

    <script lang="javascript" type="text/javascript">
      // This is a JavaScript code block within the head tag
      // It will be executed when the page is loaded
      window.onload = function() {
        alert("Welcome to this example website!");
      }
    </script>
  </head>

  <body>
    <!-- This is the body tag with child image, div, and p tags -->
    <img src="example.jpg" alt="Example Image">
    <div id="example-div">
      <!-- This is a div tag with child h1 and p tags -->
      <h1>This is an example heading</h1>
      <p>
        This is an example paragraph with <strong>bold</strong> and <em>italic</em> text.
      </p>
    </div>

    <!-- This is a self-closing tag with attributes -->
    <p class="example-paragraph">This is another example paragraph.</p>
  </body>
</html>
