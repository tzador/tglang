<!-- This is an XML code snippet -->

<!DOCTYPE html> <!-- set the document type to HTML -->

<html lang="en"> <!-- start the HTML document with 'en' as the language -->

<head> <!-- start the head section -->
  <title>My Website</title> <!-- set the title of the page -->
  <meta charset="utf-8"> <!-- set the character encoding to UTF-8 for international characters -->
  <meta name="description" content="This is a website about coding."> <!-- set a brief description of the website -->
  <meta name="author" content="John Doe"> <!-- set the author of the website -->
  <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- set the viewport for responsive design -->
</head> <!-- end the head section -->

<body> <!-- start the body section -->
  <h1>Welcome to My Website!</h1> <!-- add a heading to the page -->
  <p>This is a website dedicated to all things coding. From HTML to Python, we've got you covered!</p> <!-- add a paragraph of text -->
  <ul> <!-- start an unordered list -->
    <li>HTML</li> <!-- add a list item for HTML -->
    <li>CSS</li> <!-- add a list item for CSS -->
    <li>JavaScript</li> <!-- add a list item for JavaScript -->
    <li>Python</li> <!-- add a list item for Python -->
  </ul> <!-- end the unordered list -->
</body> <!-- end the body section -->

</html> <!-- end the HTML document -->
