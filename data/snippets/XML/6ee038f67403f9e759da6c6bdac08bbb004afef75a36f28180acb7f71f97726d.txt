<!-- This is a comment in XML -->
<!DOCTYPE html> <!-- This is the document type declaration of an XML document -->
<html> <!-- This tag marks the start of the HTML document -->

<head> <!-- This tag contains meta data about the HTML document -->
    <title>This is the title of the HTML document</title> <!-- This tag sets the title of the document -->
    <style> <!-- This tag contains CSS code to style the document -->
        body {
            background-color: red;
        }
        h1 {
            font-size: 24px;
            color: white;
        }
    </style>
</head>

<body> <!-- This tag contains the content of the HTML document -->
    <h1>Hello World! This is my first HTML document in XML</h1> <!-- This tag sets the heading of the document -->
    <p>This is a paragraph in the document</p> <!-- This tag contains a paragraph -->
    <div> <!-- This tag creates a division in the document -->
        <ul> <!-- This tag creates an unordered list -->
            <li>Item 1</li> <!-- This tag adds an item to the list -->
            <li>Item 2</li>
            <li>Item 3</li>
        </ul>
    </div>
</body>

</html> <!-- This tag marks the end of the HTML document -->
