<!-- This is an XML declaration that specifies the version and encoding of the document -->
<?xml version="1.0" encoding="UTF-8"?>

<!-- This is a DTD (Document Type Definition) that defines the structure of the document -->
<!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Transitional//EN" "http://www.w3.org/                      
TR/xhtml1/DTD/xhtml1-transitional.dtd">

<!-- This is the root element of our document -->
<html>

<!-- This is the head section that contains information about the document -->
<head>
  <title>Example XML Document</title>
  
  <!-- This is a meta tag that specifies the character set of the document -->
  <meta http-equiv="Content-Type" content="text/html; charset=utf-8" />

  <!-- This is a style sheet link that specifies the CSS file for styling the document -->
  <link rel="stylesheet" type="text/css" href="style.css" />
</head>

<!-- This is the body section that contains the actual content of the document -->
<body>
  <!-- This is a paragraph element that contains some text -->
  <p>This is an example XML document.</p>

  <!-- This is an unordered list element that contains three list items -->
  <ul>
    <li>List Item 1</li>
    <li>List Item 2</li>
    <li>List Item 3</li>
  </ul>

  <!-- This is an anchor element that creates a link to an external website -->
  <a href="https://www.example.com">Click here to go to Example website</a>
</body>

<!-- This is the closing tag for the root element -->
</html>
