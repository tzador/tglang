<!-- This is an example of a complex XML code snippet -->
<!-- It contains multiple elements, attributes, and namespaces -->

<!-- The 'xml' element is the root element in every XML document -->
<xml version="1.0" encoding="UTF-8">

<!-- The 'person' element represents a person and has three attributes -->
<person gender="male" age="25" nationality="American">

<!-- The 'name' element represents the person's name -->
<name>
    <!-- The text content of the element is the person's full name -->
    John Doe
</name>

<!-- The 'address' element represents the person's mailing address -->
<address>
    <!-- The 'street' element represents the street name -->
    <street>
        <!-- The text content of the element is the street name -->
        Main Street
    </street>
    <!-- The 'city' element represents the city name -->
    <city>
        <!-- The text content of the element is the city name -->
        New York
    </city>
    <!-- The 'state' element represents the state name -->
    <state>
        <!-- The text content of the element is the state name -->
        New York
    </state>
    <!-- The 'country' element represents the country name -->
    <country>
        <!-- The text content of the element is the country name -->
        United States
    </country>
    <!-- The 'zip' element represents the zip code -->
    <zip>
        <!-- The text content of the element is the zip code -->
        10001
    </zip>
</address>

<!-- The 'hobbies' element represents the person's hobbies -->
<hobbies>
    <!-- The 'hobby' element represents a single hobby and has an 'id' attribute -->
    <hobby id="1">
        <!-- The text content of the element is the name of the hobby -->
        Hiking
    </hobby>
    <hobby id="2">
        Yoga
    <hobby>
    <hobby id="3">
        Reading
    </hobby>
</hobbies>
</person>
</xml>
