    <item type="food"> <!-- The first item in the list is of type 'food' -->
        <name>Bananas</name> <!-- The name element specifies the name of the item -->
        <price>$2.99</price> <!-- The price element specifies the price of the item -->
        <quantity>12</quantity> <!-- The quantity element specifies the quantity of the item -->
        <expirationDate>12/31/2021</expirationDate> <!-- The expirationDate element specifies when the item expires -->
    </item>

    <item type="household"> <!-- The second item in the list is of type 'household' -->
        <name>Dish soap</name>
        <price>$4.99</price>
        <quantity>1</quantity>
        <brand>Mr. Clean</brand> <!-- The brand element specifies the brand of the item -->
        <size>16 oz.</size> <!-- The size element specifies the size of the item -->
    </item>

    <item type="clothing">
        <name>T-shirt</name>
        <price>$12.99</price>
        <quantity>3</quantity>
        <size>Medium</size>

    </item>
