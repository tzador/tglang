<!-- This is an XML document that represents a web page -->
<webPage>
  <!-- This is the opening tag for the <head> element -->
  <head>

    <!-- This is the opening tag for the <title> element -->
    <title>This is a web page</title>

    <!-- This is the opening tag for the <meta> element with attribute values -->
    <meta charset="UTF-8" content="This is a web page" name="description"/>

    <!-- This is the opening tag for the <link> element with attribute values -->
    <link rel="stylesheet" type="text/css" href="styles.css"/>
  </head>

  <!-- This is the opening tag for the <body> element -->
  <body>

    <!-- This is the opening tag for the <h1> element -->
    <h1>Welcome to my web page!</h1>

    <!-- This is the opening tag for the <p> element -->
    <p>This is a paragraph of text.</p>

    <!-- This is the opening tag for the <ul> element -->
    <ul>

      <!-- This is the opening tag for the <li> element -->
      <li>List item 1</li>

      <!-- This is the opening tag for the <li> element -->
      <li>List item 2</li>

      <!-- This is the opening tag for the <li> element -->
      <li>List item 3</li>
    </ul>

    <!-- This is the opening tag for the <a> element with attribute values -->
    <a href="https://www.example.com" target="_blank">Visit my website</a>
  </body>
</webPage>
