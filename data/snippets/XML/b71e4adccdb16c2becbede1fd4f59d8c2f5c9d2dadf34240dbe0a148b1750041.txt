<!-- This is an example of an XML document with various elements -->

<!-- Root element -->
<root>

    <!-- First child element -->
    <child1 attribute="value">
        Text content of the first child element
    </child1>

    <!-- Second child element -->
    <child2>
        <!-- Grandchild element -->
        <grandchild1>
            Text content of the grandchild element
        </grandchild1>
    </child2>

    <!-- Third child element -->
    <child3>
        <!-- First sub-child element -->
        <sub-child1>
            <!-- First great-grandchild element -->
            <great-grandchild1>
                Text content of the first great-grandchild element
            </great-grandchild1>
        </sub-child1>

        <!-- Second sub-child element -->
        <sub-child2>
            <!-- Second great-grandchild element -->
            <great-grandchild2>
                Text content of the second great-grandchild element
            </great-grandchild2>
        </sub-child2>
    </child3>
    
    <!-- Fourth child element -->
    <child4>
        <!-- Child element with inline text content -->
        <inline-child>This is an inline child element</inline-child>
    </child4>

</root>
