<!--This is an example of an XML document-->
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE note SYSTEM "note.dtd">
<!--The root element of our document-->
<note>
    <!--The child element of our document-->
    <to>John Smith</to>
    <!--One more child element-->
    <from>Jane Doe</from>
    <!--Yet another child element-->
    <heading>Reminder</heading>
    <!--Final child element with a nested element and attribute-->
    <body>
        <para>This is a reminder to bring your laptop <b type="bold">tomorrow</b> for our meeting.</para>
    </body>
</note>
