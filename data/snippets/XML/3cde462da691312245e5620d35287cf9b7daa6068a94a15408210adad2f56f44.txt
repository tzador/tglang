<?xml version="1.0" encoding="UTF-8"?> <!-- This is a standard header for XML documents -->
<!DOCTYPE html> <!-- This specifies the document type as HTML-->
<html> <!-- This is the root element of the document -->
    <head> <!-- This is the element for the HTML document's header -->
        <title>My Website</title> <!-- This is the title of the document -->
        <meta charset="UTF-8"/> <!-- This specifies the character encoding for the document -->
        <meta name="author" content="John Doe"/> <!-- This specifies the author of the document -->
    </head>
    <body> <!-- This is the element for the HTML document's body -->
        <h1>Welcome to My Website</h1> <!-- This is a heading element -->
        <p>This is a paragraph element in XML format.</p> <!-- This is a paragraph element -->
        <img src="image.jpg" alt="My Image"/> <!-- This is an image element with a source and alternate text -->
        <ul> <!-- This is an unordered list element -->
            <li>Item 1</li> <!-- This is a list item element -->
            <li>Item 2</li>
            <li>Item 3</li>
        </ul>
        <table> <!-- This is a table element -->
            <tr> <!-- This is a table row element -->
                <th>Column Heading 1</th> <!-- This is a table header element -->
                <th>Column Heading 2</th>
                <th>Column Heading 3</th>
            </tr>
            <tr>
                <td>Row 1, Column 1</td> <!-- This is a table data element -->
                <td>Row 1, Column 2</td>
                <td>Row 1, Column 3</td>
            </tr>
            <tr>
                <td>Row 2, Column 1</td>
                <td>Row 2, Column 2</td>
                <td>Row 2, Column 3</td>
            </tr>
        </table>
    </body>
</html>
