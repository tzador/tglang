--[[
    This is a comment block in LUA
    This code snippet demonstrates nested for loops to print a multiplication table
--]]

-- Loop to iterate through numbers 1 to 10
for i = 1, 10 do

    -- Loop to iterate through numbers 1 to 10
    for j = 1, 10 do

        -- Calculate and print the product of i * j
        print(i * j)

    end
end
