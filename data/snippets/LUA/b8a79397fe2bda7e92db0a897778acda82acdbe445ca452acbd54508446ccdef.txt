-- This is an example function that takes in two integers and returns their sum
function add_numbers(num1, num2)
    -- Inputs:
    --      num1: First integer
    --      num2: Second integer
    -- Output:
    --      Sum of both integers
    return (num1 + num2)
end

-- This is an example for loop that iterates through a given range and prints the numbers
for i = 1, 10 do
    print(i)
end

-- This is an example if statement that checks if a variable is equal to a certain value
if num1 == 5 then
    print("Number 1 is equal to 5")

else

end
