/*--------- Sample Program in BISON ---------*/

%{
#include <stdio.h>

/*-------------------------------------------*/

void printMessage(char* msg) {
    printf("The message is: %s\n", msg);
}

%}

/*-------------------------------------------*/
/* Lexical Analysis Rules */

alpha       [A-Za-z]
digit       [0-9]

%%

"hello"     { printMessage("Hello, world!"); }
"goodbye"   { printMessage("Goodbye, world!"); }

{digit}+    { printf("Number: %s\n", yytext); }
{alpha}{digit}+ { printf("ID: %s\n", yytext); }

.       	{ printf("Unknown token: %c\n", yytext[0]); }

%%

/*-------------------------------------------*/
/* Code generated by Bison */

int main() {
    yylex();
    return 0;
}

int yywrap() {
    return 1;
}

int yyerror(char* msg) {
    printf("Error encountered: %s\n", msg);
    return -1;
}
