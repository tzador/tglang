%token NUMBER IDENTIFIER

%%

program: 
    { /* code for program logic */ }
    statement_list
    ;

statement_list:
    /* empty */
    | statement
    | statement_list statement
    ;

statement:
    expression
    | if_statement
    | for_statement
    ;

expression:
    IDENTIFIER '=' NUMBER
    | /* other types of expressions */
    ;

if_statement:
    IF '(' expression ')' '{' statement_list '}' 
    | IF '(' expression ')' '{' statement_list '}' ELSE '{' statement_list '}'
    ;

for_statement:
    FOR '(' IDENTIFIER '=' NUMBER ';' IDENTIFIER '<' NUMBER ';' IDENTIFIER '++' ')' '{' statement_list '}'
    ;

%%
