program pascal_example;  //starts the program
var  //declares variables
  num1,num2,sum : integer; //declares three integers
begin  //begins the code
  num1 := 5; //assigns 5 to variable num1
  num2 := 10; //assigns 10 to variable num2
  sum := num1+num2; //adds num1 and num2 and stores result in sum
  write('The sum of ', num1, ' and ', num2, ' is: ');
  write(sum); //prints out the sum
end. //ends the code
