var
  x: integer; //declaring a variable named 'x' of type integer
  y: real; //declaring a variable named 'y' of type real

begin //beginning of main program
  readln(x); //reading input from user and storing in 'x'
  y := x * 3.14159; //calculating y as 3.14159 times x
  writeln(y); //writing the value of y to console
end. //end of main program
