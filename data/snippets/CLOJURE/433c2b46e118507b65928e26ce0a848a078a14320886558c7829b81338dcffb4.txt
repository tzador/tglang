(defn print-product
  "Returns the product of two numbers"
  [x y]
  (* x y))

(defn print-sum
  "Returns the sum of two numbers"
  [x y]
  (* x y))

(defn print-difference
  "Returns the difference of two numbers"
  [x y]
  (- x y))

(defn print-quotient
  "Returns the quotient of two numbers"
  [x y]
  (/ x y))
