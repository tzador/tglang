(defn fib [n]                           ; "Defines a function named fib which takes in a single argument n."
  (if (= n 0)                           ; "Checks if n is equal to 0."
    0                                   ; "If n is 0, returns 0."
    (if (= n 1)                         ; "Checks if n is equal to 1."
      1                                 ; "If n is 1, returns 1."
      (+ (fib (- n 1)) (fib (- n 2))))  ; "If n is not 0 or 1, calls fib recursively on n-1 and n-2, then adds the results together."
(defn -main []                          ; "Defines a function named -main, which takes in no arguments."
  (println (fib 10)))                   ; "Calls the fib function with an argument of 10 and prints the result."
