// This code snippet creates a class called Calculator
// and defines several methods and properties

class Calculator {
    // This method takes in two numbers and returns their sum
    // It also includes documentation on the parameters and return value
    public function add(x:Number, y:Number):Number {
        return x + y;
    }

    // This method takes in two numbers and returns their difference
    // It also includes documentation on the parameters and return value
    public function subtract(x:Number, y:Number):Number {
        return x - y;
    }

    // This property holds the value of Pi
    // It is a constant and cannot be changed
    public static const PI:Number = 3.141592653589793;
}
