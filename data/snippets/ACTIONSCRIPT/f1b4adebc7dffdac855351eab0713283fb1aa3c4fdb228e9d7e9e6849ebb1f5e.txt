/*This code snippet is a function that takes in two parameters, num1 and num2, and returns the sum of the two numbers*/

function sum(num1:Number, num2:Number):Number {
    var result:Number = num1 + num2; //create a variable to hold the sum of the two numbers
    return result; //return the result
}

//call the function and pass in two numbers as arguments
sum(5, 10);
