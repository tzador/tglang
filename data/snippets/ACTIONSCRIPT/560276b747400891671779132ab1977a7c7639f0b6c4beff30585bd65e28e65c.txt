// This code snippet calculates the sum of two numbers using a for loop
// Define the two numbers to be added
var num1:int = 5;
var num2:int = 10;

// Define a variable to store the sum
var sum:int = 0;

// Use a for loop to iterate through the numbers and add them to the sum
for (var i:int = 0; i < num2; i++) {
    sum += num1;
}

// Output the result
trace(sum); // Output: 50
