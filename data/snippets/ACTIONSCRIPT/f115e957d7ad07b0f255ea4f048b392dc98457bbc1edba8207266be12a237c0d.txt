/**
 * This function adds two numbers
 * @param {Number} num1 - The first number to be added
 * @param {Number} num2 - The second number to be added
 * @return {Number} - The sum of the two given numbers
 */
function addNumbers(num1, num2) {
    return num1 + num2;
}

// This variable holds the result of adding 5 and 10
var sum = addNumbers(5, 10);

// Print the result to the console
trace(sum);
