// calculate the sum of two numbers

// declare variables to hold the two numbers
var num1:Number = 10;
var num2:Number = 15;

// declare and initialize a variable to hold the sum
var sum:Number = num1 + num2;

// print the sum to the console
trace(sum); // output: 25

// add 10 to the sum and store the result in a new variable
var newSum:Number = sum + 10;

// print the new sum to the console
trace(newSum); // output: 35
