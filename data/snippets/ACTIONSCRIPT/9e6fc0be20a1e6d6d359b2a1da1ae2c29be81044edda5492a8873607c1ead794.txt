/* This code snippet shows how to use conditional statements in ActionScript */
//
var userInput = "yes";
// check if user input is equal to "yes"
if (userInput === "yes") {
  // if condition is met, output "User confirmed"
  trace("User confirmed");
} else {
  // if condition is not met, output "User did not confirm"
  trace("User did not confirm");
}

// now let's check for another condition
var age = 25;
// check if age is greater than or equal to 18
if (age >= 18) {
  // if condition is met, output "User is an adult"
  trace("User is an adult");
} else {
  // if condition is not met, output "User is a minor"
  trace("User is a minor");
}
