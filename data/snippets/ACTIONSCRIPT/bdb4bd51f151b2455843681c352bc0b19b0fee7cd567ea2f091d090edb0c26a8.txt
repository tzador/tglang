// This is a simple ActionScript program that calculates the area of a rectangle

// Define two variables to store the length and width of the rectangle
var length:int = 10;
var width:int = 5;

// Calculate the area of the rectangle and store it in a variable
var area:int = length * width;

// Print the result using trace() function
trace("The area of the rectangle is " + area);

// Output: `The area of the rectangle is 50`
