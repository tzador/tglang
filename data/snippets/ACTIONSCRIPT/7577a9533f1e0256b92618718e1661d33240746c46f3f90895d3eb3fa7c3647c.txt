var num1:Number = 5; // declare and initialize a variable of type Number and assign it a value of 5
var num2:Number = 10; // declare and initialize another variable of type Number and assign it a value of 10
var result:Number; // declare a variable to store the result of the arithmetic operation
result = num1 + num2; //perform addition and store the result in the result variable
trace("The sum of " + num1 + " and " + num2 + " is " + result); // print out the result using the trace() function
// output: The sum of 5 and 10 is 15
