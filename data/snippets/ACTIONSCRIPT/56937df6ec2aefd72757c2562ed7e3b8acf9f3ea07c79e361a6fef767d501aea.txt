// This code snippet demonstrates a function that calculates the area of a rectangle

function calcArea(length:Number, width:Number):Number { // function declaration with parameters

    var area:Number = length * width; // variable declaration and calculation statement

    return area; //return statement

}

var rectangleArea:Number = calcArea(5, 10); // variable declaration and function call

trace("The area of the rectangle is: " + rectangleArea); // output statement
