// This is a code snippet in ACTIONSCRIPT

// Declare a variable
var num:int;

// Initialize the variable
num = 5;

// Define a function that takes in a number and multiplies it by 2
function multiplyBy2(number:int):int{
    return number * 2;
}

// Call the function with the variable as an argument and store the result in a new variable
var result:int = multiplyBy2(num);

// Print out the result
trace("The result of multiplying " + num + " by 2 is " + result);

// Output:
// The result of multiplying 5 by 2 is 10
