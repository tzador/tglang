// This code snippet shows how to create and use a function in ActionScript

// Function to calculate the sum of two numbers
function sum(x:int, y:int):int {
    return x + y;
}

// Using the function to calculate and store the result in a variable
var result:int = sum(10, 5);

// Printing the result to the console
trace("The sum of 10 and 5 is:", result);

// Output: The sum of 10 and 5 is: 15
