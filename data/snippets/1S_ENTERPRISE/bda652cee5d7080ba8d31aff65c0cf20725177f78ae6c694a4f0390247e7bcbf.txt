//This code snippet creates a payment document in 1C Enterprise
Document = CreateObject("Document.Платежное поручение");
Document.Организация = Справочники.Организации["Рога и копыта"];
Document.Дата = "01.09.2020";

//Add lines to the document
НомерСтроки = 0;
Строка = Document.Строки.Добавить();
НомерСтроки=НомерСтроки+1;

//Set parameters for the first payment
Строка.Номенклатура = Справочники.Номенклатура["Банковские услуги"];
Строка.Количество = 1;
Строка.Цена = 10000;
Строка.Сумма = Строка.Количество*Строка.Цена;

//Add additional parameters for the first payment
Строка.ДатаОплаты = "10.09.2020";
Строка.ВидОплаты = Перечисления.ВидыОплаты.БанковскийПеревод;
Строка.ОснованиеПлатежа = "Оплата банковских услуг";
Строка.НомерСчетаБанка = "12345";
Строка.ОснованиеДокумента = "Договор на оказание банковских услуг";

//Set parameters for the second payment
Строка = Document.Строки.Добавить();
НомерСтроки=НомерСтроки+1;
Строка.Номенклатура = Справочники.Номенклатура["Консультационные услуги"];
Строка.Количество = 2;
Строка.Цена = 5000;
Строка.Сумма = Строка.Количество*Строка.Цена;

//Add additional parameters for the second payment
Строка.ДатаОплаты = "15.09.2020";
Строка.ВидОплаты = Перечисления.ВидыОплаты.НаличныеДеньги;
Строка.ОснованиеПлатежа = "Оплата консультационных услуг";
Строка.ОснованиеДокумента = "Договор на оказание консультационных услуг";

//Save the document
РезультатСохранения = Document.Записать();

//Check if the document was successfully saved
Если РезультатСохранения Тогда
    //Print message if successful
    Сообщить("Документ успешно создан и сохранен");
Иначе
    //Print error message if unsuccessful
    Сообщить("Ошибка при создании и сохранении документа");
КонецЕсли;
