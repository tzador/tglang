; Создаем функцию для вычисления факториала
Функция ВычислитьФакториал(Число) Экспорт
	Перем Индекс, Результат;
	
	; Проверяем, является ли число целым 
	Если ЦЕЛ(Число) Тогда 
	
		; Инициализируем переменную для хранения результата
		Результат = 1;
		
		; Запускаем цикл счета факториала 
		Для Индекс = 1 По Число Цикл
		
			; Умножаем текущее значение результата на индекс цикла 
			Результат = Результат * Индекс;
			
		КонецЦикла;
		
		; Возвращаем результат
		Возврат Результат;
		
	ИначеНачало
		; Если число не целое, то выводим сообщение об ошибке
		Сообщить("Ошибка! Число должно быть целым.");
	КонецЕсли;
КонецФункции;

; Вызываем функцию и передаем ей значение для расчета факториала
ВычислитьФакториал(5);

; Результатом должно быть число 120 (т.к. 5! = 120)
