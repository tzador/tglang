# Описанение направления
НаправлениеСтажировки = СправочникСсылка.НаправленияСтажировки.Новый("000001")
НаправлениеСтажировки.Наименование = "Программирование"
НаправлениеСтажировки.Описание = "Направление для стажировки в области программирования"

# Создание объекта стажера
Стажер = СправочникСсылка.Стажеры.Новый()
Стажер.ФИО = "Петров Иван"
Стажер.Направление = НаправлениеСтажировки

# Создание обучающих курсов для стажера
ОбучающиеКурсы = СправочникСсылка.ОбучающиеКурсы.Объект()
ОбучающиеКурсы.Стажер = СправочникСсылка.Стажеры.Новый("000001")
ОбучающиеКурсы.Название = "Основы программирования"
ОбучающиеКурсы.Длительность = 24

# Начало стажировки для стажера
НачалоСтажировки = ФункцияНачалоСтажировки(Стажер, ОбучающиеКурсы)

# Реализация метода начала стажировки
Функция ФункцияНачалоСтажировки(Стажер, ОбучающиеКурсы)
    Если СправочникСсылка.Стажеры.СтрокаСоответствия(Стажер) Тогда
        НаправлениеСтажировки = СправочникСсылка.НаправленияСтажировки.СтрокаСоответствия(Стажер.Направление)
        Если НаправлениеСтажировки.Описание = "Направление для стажировки в области программирования" Тогда
            Если СправочникСсылка.ОбучающиеКурсы.СтрокаСоответствия(ОбучающиеКурсы) Тогда
                ОбучающиеКурсы.ПланДействий = "1. Ознакомиться с основными понятиями программирования
                2. Написать первую программу на языке 1S_ENTERPRISE
                3. Изучить основы работы с базами данных на примере 1C:Предприятие
                4. Разработать собственный проект на языке 1S_ENTERPRISE"
                # Запуск обучающих курсов
                ОбучающиеКурсы.Запустить()
                # Начало стажировки и присвоение стажеру соответствующего статуса
                Стажер.Статус = "На стажировке"
                Возврат Истина
            КонецЕсли
        КонецЕсли
    КонецЕсли
    Возврат Ложь
КонецФункции
