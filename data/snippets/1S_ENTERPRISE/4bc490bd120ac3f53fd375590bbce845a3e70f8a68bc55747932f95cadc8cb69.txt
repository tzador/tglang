/*Библиотека объектов учета*/
Колчан("БиблиотекаОбъектовУчета", ОткрытьОбложку(){

    //Декларация библиотеки объектов
    Модуль БиблиотекаОбъектовУчета
    РасширениеБиблиотеки

    //Создание интерфейса объекта
    Интерфейс СоздатьОбъектУчета = Неопределено;

    //Определение метода создания объекта учета
    СоздатьОбъектУчета.Метод("СоздатьОбъектУчета", Процедура (ТипОбъекта){

        //Объявление объекта
        ОбъектУчета = Неопределено;

        //Проверка типа объекта
        Если ТипОбъекта = "СчетФактура" Тогда

            //Создание объекта счета-фактуры
            ОбъектУчета = СоздатьОбъект("СчетФактура");
            //Добавление объекта в коллекцию объектов учета
            ОбъектыУчета.Добавить(ОбъектУчета);

        КонецЕсли;

        //Возврат созданного объекта учета
        Возврат ОбъектУчета;
    КонецПроцедуры);

    //Определение метода получения объекта учета по номеру
    СоздатьОбъектУчета.Метод("ПолучитьОбъектУчета", Процедура (НомерОбъекта){

        //Объявление объекта
        ОбъектУчета = Неопределено;

        //Получение объекта из коллекции по номеру
        ОбъектУчета = ОбъектыУчета.Получить(НомерОбъекта);

        //Возврат объекта учета
        Возврат ОбъектУчета;
    КонецПроцедуры);

    //Создание коллекции объектов учета
    ОбъектыУчета = Новый Массив();

    //Добавление в библиотеку метода создания объекта учета
    БиблиотекаОбъектовУчета.ДобавитьМетод(СоздатьОбъектУчета);

    //Возврат библиотеки объектов учета
    Возврат БиблиотекаОбъектовУчета;
});
