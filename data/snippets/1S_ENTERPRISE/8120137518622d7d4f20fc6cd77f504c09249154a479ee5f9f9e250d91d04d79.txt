Алгоритм ПрисвоитьИсточникДанныхСвойств(ТекущийЭлемент)
	
// Описание:
// Данный алгоритм производит присвоение свойств источника данных текущему элементу.

// Входные параметры:
// ТекущийЭлемент – переменная, содержащая ссылку на текущий элемент

// Выходные параметры:
// отсутствуют

// Объявляем переменные
Перем Свойства;
Перем ТекущееСвойство;
Перем МодельИсточникаДанных;
Перем СписокСвойствИсточника;
Перем ТекущееСвойствоИсточника;

// Получаем модель источника данных для текущего элемента
МодельИсточникаДанных = ТекущийЭлемент.ИсточникДанных.ПолучитьМодель();

// Получаем список свойств источника данных
СписокСвойствИсточника = МодельИсточникаДанных.Свойства.Список();

// Перебираем свойства источника данных
Для Каждого ТекущееСвойствоИсточника Из СписокСвойствИсточника Цикл

	// Получаем свойства источника для текущего свойства
	Свойства = ТекущийЭлемент.ИсточникДанных.Свойство(ТекущееСвойствоИсточника);

	// Присваиваем свойству текущего элемента значение свойства источника
	ТекущийЭлемент[ТекущееСвойство.Имя] = Свойства.Значение;

КонецЦикла;	

КонецАлгоритма
