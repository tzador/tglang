Процедура НайтиКлиента(ИмяКлиента, ФамилияКлиента)
// В этой процедуре мы будем искать клиента в базе данных по его имени и фамилии
// Переменная ИмяКлиента будет содержать имя клиента, а переменная ФамилияКлиента - его фамилию
// Объявляем переменную для хранения найденного клиента
Переменная НайденныйКлиент;
// Запрос к базе данных для поиска клиента
Запрос = Новый Запрос;
Запрос.Текст = "ВЫБРАТЬ * ИЗ Клиенты ГДЕ Имя = &ИмяКлиента И Фамилия = &ФамилияКлиента";
Запрос.УстановитьПараметр("ИмяКлиента", ИмяКлиента);
Запрос.УстановитьПараметр("ФамилияКлиента", ФамилияКлиента);
// Выполняем запрос и получаем результирующий набор данных с найденным клиентом
Результат = Запрос.Выполнить();
// Проверяем, найден ли клиент
Если Результат.Количество() > 0 Тогда
   // Получаем первую запись из результирующего набора данных
   Строка = Результат.Получить(0);
   // Создаем объект клиента на основе полученных данных из базы
   НайденныйКлиент = Новый Клиент;
   НайденныйКлиент.Имя = Строка.Имя;
   НайденныйКлиент.Фамилия = Строка.Фамилия;
   НайденныйКлиент.Возраст = Строка.Возраст;
   НайденныйКлиент.Адрес = Строка.Адрес;
   // Возвращаем найденного клиента из процедуры
   Возврат НайденныйКлиент;
ИначеЕсли Результат.Количество() = 0 Тогда
   // Если клиент не найден, возвращаем пустое значение
   Возврат Неопределено;
