//This code snippet creates a complex function that calculates the average of three given numbers and returns the result
Function CalculateAverage(number1, number2, number3)
   //Declare local variables
   Var sum, average;
   
   //Calculate the sum of the three numbers
   sum = number1 + number2 + number3;
   
   //Calculate the average
   average = sum / 3;
   
   //Return the average
   Return average;
EndFunction
