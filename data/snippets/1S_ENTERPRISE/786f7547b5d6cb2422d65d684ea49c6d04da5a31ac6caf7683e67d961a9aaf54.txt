Процедура ФормированиеОтчета(год, месяц, счетчик)
    Перем отчет, файл, строка;
    // Объявляем переменные

    отчет = СоздатьОбъект("Отчеты.ФинансовыйОтчет"); // Создаем объект класса "ФинансовыйОтчет"
    отчет.Год = год; // Устанавливаем значение года
    отчет.Месяц = месяц; // Устанавливаем значение месяца
    отчет.Счетчик = счетчик; // Устанавливаем значение счетчика

    файл = ОткрытьФайл("Отчет_Год" + Строка(год) + "_Месяц" + Строка(месяц) + ".txt"); // Открываем файл для записи

    строка = "Отчет за " + Строка(месяц) + " месяц " + Строка(год) + " года:"; // Формируем строку заголовка отчета

    Записать(файл, строка); // Записываем строку в файл
    Записать(файл, отчет.ПолучитьОтчет()); // Записываем содержимое отчета в файл

    ЗакрытьФайл(файл); // Закрываем файл

КонецПроцедуры
