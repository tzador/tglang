// Define variables
var a, b, c;

// Initialize variables with values
a = 5;
b = 10;
c = 15;

// Calculate the sum of variables
var sum = a + b + c;

// Print the result
Print(sum); // Output: 30

// Calculate the average of variables
var avg = sum / 3;

// Round the average to the nearest whole number
var roundedAvg = Round(avg);

// Print the rounded average
Print(roundedAvg); // Output: 10
