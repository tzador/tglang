// This code snippet creates a new object of type "Customer" and sets its properties

// Variable declaration
Customer MyCustomer;

// Object initialization
MyCustomer = New Customer;

// Setting properties
MyCustomer.Name = "John Smith";
MyCustomer.Age = 35;
MyCustomer.Address = "123 Main Street";
MyCustomer.PhoneNumber = "555-123-4567";
MyCustomer.Email = "johnsmith@email.com";

// Output the customer's information
Message("Name: " + MyCustomer.Name);
Message("Age: " + MyCustomer.Age);
Message("Address: " + MyCustomer.Address);
Message("Phone Number: " + MyCustomer.PhoneNumber);
Message("Email: " + MyCustomer.Email);

// Expected Output:
// Name: John Smith
// Age: 35
// Address: 123 Main Street
// Phone Number: 555-123-4567
// Email: johnsmith@email.com
