Процедура СоздатьНовыйДокумент()
	
	// Создаем отчет по продажам
	НовыйОтчет = Отчеты.Продажи.Создать();
	
	// Устанавливаем период для отчета
	НовыйОтчет.УстановитьПериодНачало(ДатаНачала);
	НовыйОтчет.УстановитьПериодКонец(ДатаКонца);
	
	// Получаем данные о продажах из базы данных
	Данные = Запрос.Выполнить("Выбрать * Из Продажи");
	
	// Устанавливаем источник данных для отчета
	НовыйОтчет.УстановитьИсточникДанных(Данные);
	
	// Устанавливаем параметры отчета (например, валюта)
	НовыйОтчет.УстановитьПараметр("Валюта", "Рубли");
	
	// Генерируем отчет и получаем ссылку на него
	Ссылка = НовыйОтчет.Сгенерировать();
	
	// Печатаем отчет
	НовыйОтчет.Напечатать(Ссылка);
