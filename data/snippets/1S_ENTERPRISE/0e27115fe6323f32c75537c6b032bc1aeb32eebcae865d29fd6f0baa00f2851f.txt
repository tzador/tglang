// Create object for holding employee information
ОбъектСотрудник = Новый Структура("Имя", "Должность", "Зарплата");

// Add employees to the structure
ОбъектСотрудник.Вставить("Петров", "менеджер", 50000);
ОбъектСотрудник.Вставить("Иванов", "бухгалтер", 60000);
ОбъектСотрудник.Вставить("Сидоров", "разработчик", 70000);

// Create document
Документ = Новый ДокументНастроек();
Документ.УстановитьЗаголовок("Список сотрудников");

// Add table to document
ТаблицаСотрудников = Документ.ДобавитьТаблицу();
ТаблицаСотрудников.УстановитьЗаголовокСтроки("Сотрудник", "Должность", "Зарплата");

// Loop through the employee structure and add data to table
Для Каждого Сотрудник Из ОбъектСотрудник Цикл
    ТаблицаСотрудников.ДобавитьСтроку(Сотрудник.Имя, Сотрудник.Должность, Сотрудник.Зарплата);
КонецЦикла;

// Generate XML document from settings document and save it to file
СтроковыйДокумент = Документ.Текст;
СтроковыйДокумент.Записать("СписокСотрудников.xml");

// Print success message
Сообщить("Список сотрудников успешно сгенерирован и сохранен в файле СписокСотрудников.xml.");
