//Creating a function that calculates the sum of two numbers

Function CalculateSum(firstNumber, secondNumber)
	//Declaring local variable to store the sum
	Var sum = firstNumber + secondNumber
	
	//Printing the sum to the console
	Console.Write("The sum of " + firstNumber + " and " + secondNumber + " is: " + sum)
EndFunction

//Calling the function with two numbers
CalculateSum(5, 10) //Output: The sum of 5 and 10 is: 15
