# Структура "Заказ"
# Содержит поля для хранения информации о заказе
Структура Заказ;
	Идентификатор заказа, строка; // Поле для хранения уникального идентификатора заказа
	УникальныйНомерЗаказа, число; // Поле для хранения уникального номера заказа
	Клиент, строка; // Поле для хранения информации о клиенте, сделавшем заказ
	Дата, дата; // Поле для хранения даты оформления заказа
	ИтоговаяСумма, число; // Поле для хранения итоговой суммы заказа
	
	# Метод "ДобавитьТовар"
	# Добавляет товар в заказ и изменяет итоговую сумму
	Процедура ДобавитьТовар(Товар, Количество, Цена)
		Если ТипЗначения(Ключ(Товар)) = Строка Тогда // Проверяем, что в качестве товара передано название товара, а не ссылка на объект
			НовыйОбъект = Новый Структура("Товар, Количество, Цена"); // Создаем новый объект, содержащий информацию о товаре, количестве и цене
			Заказ.Товары.Вставить(Товар, НовыйОбъект); // Вставляем новый объект в структуру "Товары" внутри структуры "Заказ"
			Заказ.ИтоговаяСумма = Заказ.ИтоговаяСумма + Количество * Цена; // Изменяем итоговую сумму заказа путем добавления к текущему значению произведения количества и цены
		Иначе
			Возврат "Ошибка: неверный формат данных о товаре";
		КонецЕсли;
	КонецПроцедуры;
	
	# Метод "УдалитьТовар"
	# Удаляет товар из заказа и изменяет итоговую сумму
	Процедура УдалитьТовар(Товар, Количество)
		Если Заказ.Товары.Содержит(Товар) Тогда // Проверяем, содержит ли структура "Товары" указанный товар
			СтарыйОбъект = Заказ.Товары.Найти(Товар); // Находим объект с информацией о товаре
			НовыйОбъект = Новый Структура("Товар, Количество"); // Создаем новый объект без информации о цене
			Заказ.Товары.Вставить(Товар, НовыйОбъект); // Заменяем старый объект на новый в структуре "Товары"
			Заказ.ИтоговаяСумма = Заказ.ИтоговаяСумма - Количество * СтарыйОбъект.Цена; // Изменяем итоговую сумму заказа путем вычитания из текущего значения произведения количества и цены
			Заказ.Товары.Удалить(Товар); // Удаляем старый объект из структуры "Товары"
		Иначе
			Возврат "Ошибка: указанный товар отсутствует в заказе";
		КонецЕсли;
	КонецПроцедуры;
КонецСтруктуры; // Конец описания структуры "Заказ"
