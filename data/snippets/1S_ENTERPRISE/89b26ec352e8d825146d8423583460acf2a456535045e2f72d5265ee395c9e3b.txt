&НаСервере
Процедура РасчетНДС(СтавкаНДС, СуммаБезНДС) Экспорт
	
	// Необходимо указать основную ставку НДС для расчета
	// Сумма без НДС должна быть в денежном формате
	СуммаСНДС = СуммаБезНДС * СтавкаНДС / (100 + СтавкаНДС); 
	
	// Если результат расчета меньше 1, то ставка НДС равна 0%
	// Иначе, ставка НДС будет округлена до ближайшего целого числа
	Если СуммаСНДС < 1 Тогда
		СтавкаНДС = 0;
	Иначе
		СтавкаНДС = Окр(СтавкаНДС, -2);
	КонецЕсли;
	
	// Возвращаем результат расчета - сумму НДС
	Возврат СуммаСНДС;
	
КонецПроцедуры

&НаКлиенте
Процедура РасчетНДС(СтавкаНДС, СуммаБезНДС) Экспорт

	// Получаем доступ к объекту "СтавкиНДС" из справочника "СтавкиНДС"
	// Для расчета налога в документах
	СтавкиНДС = Справочники.СтавкиНДС.НайтиПоКоду(СтавкаНДС);
	
	// Если объект найден, то получаем значение "Ставка"
	Если СтавкиНДС <> Неопределено Тогда
		СтавкаНДС = СтавкиНДС.Ставка;
	Иначе
		СтавкаНДС = 0;
	КонецЕсли;
	
	// Необходимо указать основную ставку НДС для расчета
	// Сумма без НДС должна быть в денежном формате
	СуммаСНДС = СуммаБезНДС * СтавкаНДС / (100 + СтавкаНДС); 
	
	// Если результат расчета меньше 1, то ставка НДС равна 0%
	// Иначе, ставка НДС будет округлена до ближайшего целого числа
	Если СуммаСНДС < 1 Тогда
		СтавкаНДС = 0;
	Иначе
		СтавкаНДС = Окр(СтавкаНДС, -2);
	КонецЕсли;
	
	// Возвращаем результат расчета - сумму НДС
	Возврат СуммаСНДС;
	
КонецПроцедуры
