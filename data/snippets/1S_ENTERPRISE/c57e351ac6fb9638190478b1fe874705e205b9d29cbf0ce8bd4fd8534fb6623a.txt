Функция ВычислитьСреднее(Массив)
	// Функция для вычисления среднего значения массива чисел
	//
	// Параметры:
	// Массив - массив чисел, для которого нужно вычислить среднее
	//
	// Возвращаемое значение:
	// Среднее значение массива чисел
	//
	// Использование:
	// Среднее = ВычислитьСреднее(Массив);

	Если ТипЗначения(Массив) <> Тип("Массив") Тогда
		// Если переданный параметр не является массивом, генерируем ошибку
		ВызватьИсключение "Переданный параметр должен быть массивом";
	КонецЕсли;

	Сумма = 0;
	КоличествоЭлементов = ПодсчитатьКоличествоЭлементов(Массив);

	Для Каждого Элемент Из Массив Цикл
		Сумма = Сумма + Элемент;
	КонецЦикла;

	Среднее = Сумма / КоличествоЭлементов;
	Возврат(Среднее);
КонецФункции

// Вызов функции для вычисления среднего значения массива из 5 чисел
Среднее = ВычислитьСреднее([1, 2, 3, 4, 5]);
