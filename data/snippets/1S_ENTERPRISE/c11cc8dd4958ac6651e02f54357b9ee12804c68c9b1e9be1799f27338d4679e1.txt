Функция РедактироватьДанные(Таблица, Параметры, Знач ТекущийРежим = "Редактирование")
    ПодключитьСинтезаторРечи(Таблица) # подключаем синтезатор речи
    Если ТекущийРежим = "Редактирование" Тогда # проверяем текущий режим работы
        ВызватьФорму(Таблица) # вызываем форму редактирования данных
    Иначе Если ТекущийРежим = "НовыйРежим" Тогда # проверяем новый режим работы
        БлокироватьДанные(Таблица) # блокируем данные для редактирования
    КонецЕсли
    Фильтр = Параметры.Фильтр # задаем фильтр для выборки данных
    Данные = Таблица.Выбрать(Фильтр) # выбираем данные из таблицы по фильтру
    Таблица.НоваяСтрока(Данные[0]) # создаем новую строку в таблице с данными из первой выбранной строки
    Таблица.Записать(Файл) # записываем изменения в файл
    Возврат Spy.Placeholder # выводим результат выполнения функции
