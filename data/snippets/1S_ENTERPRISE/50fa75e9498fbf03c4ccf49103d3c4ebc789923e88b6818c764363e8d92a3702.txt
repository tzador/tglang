#Использовать Модуль "Константы"
#Использовать Модуль "Справочники"

//Определяем переменные
Перем КонстантаНомер; //переменная для хранения номера 
Перем ИтогПоосновнымСредствам; //переменная для хранения итоговой суммы по основным средствам

//Получаем номер из константы
КонстантаНомер = КонстантаНоменклатуры.Номенклатура123; //значение константы "Номер123" - 12345

//Получаем справочник по номеру
СправочникНоменклатура = ПолучитьОбъект("Справочник.Номенклатура").НайтиПоНомеру(КонстантаНомер);

Если СправочникНоменклатура = Неопределено Тогда //проверяем, найден ли объект в справочнике
    Сообщить("Номенклатура не найдена в справочнике."); 
Иначе //если найден, то выполняем следующие действия
    //получаем список основных средств из справочника в переменную-список
    СписокОсновныхСредств = СправочникНоменклатура.ОсновныеСредства; 
    ИтогПоОсновнымСредствам = 0; //обнуляем переменную итога
    
    //перебираем каждое основное средство из списка
    Для Каждого ОсновноеСредство Из СписокОсновныхСредств Цикл 
        ИтогПоОсновнымСредствам = ИтогПоОсновнымСредствам + ОсновноеСредство.Стоимость; //добавляем стоимость каждого основного средства к итогу
    КонецЦикла;
    
    ИтогПоОсновнымСредствам = ИтогПоОсновнымСредствам * 1.2; //добавляем НДС к итогу
    
    Сообщить("Итоговая сумма по основным средствам с НДС: " + ИтогПоОсновнымСредствам); //выводим сообщение с итоговой суммой
КонецЕсли;

//Закрываем доступ к переменным и объектам
ОсвободитьСсылку(СправочникНоменклатура);
ОсвободитьСсылку(СписокОсновныхСредств);
ОсвободитьСсылку(ОсновноеСредство); //освобождаем ссылку на последнее использованное основное средство
