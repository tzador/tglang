#заполнение даннх
Запрос = Новый Запрос;
Запрос.Текст = "ВЫБРАТЬ
	        | СуммаДокумента,
	        | ДатаДокумента,
                | Контрагент
	        | ИЗ Документ.ОтгрузкаТоваровКлиентам КАК Докумет
	        | ВНУТРЕНЕЕ СОЕДИНЕНИЕ РегистрСведений.ЦеныНоменклатуры КАК Цены
	        | НА Документ.ОтгрузкаТоваровКлиентам.Ссылка = Цены.Ссылка
	        | ГДЕ Документ.ОтгрузкаТоваровКлиентам.Дата >= &День
	        | И Документ.ОтгрузкаТоваровКлиентам.Дата <= &День";
ДатаНачала = Дата(2021, 8, 1); #задаем начальную дату
ДатаКонца = Дата(2021, 8, 31); #задаем конечную дату
Запрос.УстановитьПараметр("День", ДатаНачала); #присваиваем параметр
Запрос.УстановитьПараметр("День", ДатаКонца); #присваиваем параметр
Результат = Запрос.Выполнить(); #выполняем запрос
#вывод результатов
Пока Результат.Следующий() Цикл
{
    СуммаДокумента = Результат.Получить("СуммаДокумента");
    ДатаДокумента = Результат.Получить("ДатаДокумента");
    Контрагент = Результат.Получить("Контрагент");
    
    Сообщить(СуммаДокумента + ", " + ДатаДокумента + ", " + Контрагент); #выводим результат в консоль
}
