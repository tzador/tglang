// Declare a procedure that calculates the total cost of items purchased
Procedure calculateTotalCost()
    // Initialize variables
    Var totalCost = 0;
    Var itemPrices = [12, 34, 56, 78, 90]; // An array of item prices

    // Loop through the array and add each price to the total cost
    For Each itemPrice In itemPrices Do
        totalCost = ItemPrice + TotalCost;
    EndDo;

    // Print the total cost to the console
    OutPut "Total cost: " + TotalCost;
EndProcedure;
