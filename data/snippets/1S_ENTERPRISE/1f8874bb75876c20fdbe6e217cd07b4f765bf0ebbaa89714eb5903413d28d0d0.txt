// This program calculates the area of a rectangle

#Область ТурбоДиагностикаПродуктов
Ссылка НаОбъект = Неопределено;
Перем ПерваяСторона, ВтораяСторона, Площадь;
 
// Выводим на экран приветствие
Сообщить("Добро пожаловать в программу для расчета площади прямоугольника!");

// Считываем значения первой и второй сторон прямоугольника с клавиатуры
ПерваяСторона = Ввод("Введите значение первой стороны прямоугольника:");
ВтораяСторона = Ввод("Введите значение второй стороны прямоугольника:");

// Проверяем введенные значения на числовой тип и наличие только положительных значений
Если ТипЗнч(ПерваяСторона) = Число И ТипЗнч(ВтораяСторона) = Число Тогда
	Если ПерваяСторона > 0 И ВтораяСторона > 0 Тогда
		// Вычисляем площадь прямоугольника
		Площадь = ПерваяСторона * ВтораяСторона;
		
		// Выводим результат расчета на экран
		Сообщить("Площадь прямоугольника равна " + Площадь);
		
		// Создаем новый объект для хранения расчета
		Ссылка НаОбъект = СоздатьОбъект("Объект.РасчетПлощадиПрямоугольника");
		
		// Записываем значения сторон и площади в объект
		Назначить(ПерваяСторона, "ПерваяСторона", НаОбъект);
		Назначить(ВтораяСторона, "ВтораяСторона", НаОбъект);
		Назначить(Площадь, "Площадь", НаОбъект);
		
		// Сохраняем объект
		СохранитьОбъект(НаОбъект);
		
		// Выводим сообщение об успешном завершении программы
		Сообщить("Расчет площади прямоугольника успешно выполнен и сохранен!");
	Иначе
		// Выводим сообщение об ошибке ввода данных
		Сообщить("Ошибка ввода данных. Значения сторон прямоугольника должны быть положительными числами.");
	КонецЕсли;
Иначе
	// Выводим сообщение об ошибке ввода данных
	Сообщить("Ошибка ввода данных. Значения сторон прямоугольника должны быть числами.");
КонецЕсли;

#Область ТурбоДиагностикаПродуктовКонца

// обработка исключений при возникновении ошибок
Перехват Исключение Исключение;
Если Исключение = Неопределено Тогда
	Сообщить("Произошла неизвестная ошибка. Проверьте корректность введенных данных.");
Иначе
	Сообщить("Произошла ошибка: " + Исключение.ТекстОшибки);
КонецЕсли;

// объявление класса для хранения результатов расчета площади прямоугольника
#Область Классов
Класс РасчетПлощадиПрямоугольника Дата, Владелец, ПерваяСторона, ВтораяСторона, Площадь
КонецКласса;
#КонецОбласти

// End of program
