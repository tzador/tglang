Перем СписокНоменклатуры;
//Описание переменной "СписокНоменклатуры" 
//предназначенной для хранения списка номенклатуры товаров

СписокНоменклатуры = Новый СписокЗначений;
//Создание нового объекта списка значений и присвоение его переменной "СписокНоменклатуры"

СписокНоменклатуры.Добавить(Номенклатура.НайтиПоНаименованию("Карандаш"));
//Добавление нового элемента в список, при этом происходит поиск номенклатуры "Карандаш" по наименованию 
//и если она найдена, то добавляется в список

СписокНоменклатуры.Добавить(Номенклатура.НайтиПоКоду("123"));
//Добавление нового элемента в список, при этом происходит поиск номенклатуры с кодом "123" и если она найдена, 
//то добавляется в список

СписокНоменклатуры.Сортировать("Код");
//Сортировка списка номенклатуры по полю "Код" (код товара)

Для Каждого Номенклатура Из СписокНоменклатуры Цикл
	Номенклатура.ПолучитьСвойство("Наименование");
	//Получение значения свойства "Наименование" для каждого элемента списка номенклатуры
КонецЦикла;

СписокНоменклатуры.УдалитьПоИндексу(0);
//Удаление первого элемента списка

СписокНоменклатуры.ВставитьПоИндексу(0, Номенклатура.НайтиПоНаименованию("Ручка"));
//Вставка нового элемента в список по указанному индексу, в данном случае - первый элемент списка

СписокНоменклатуры.Объединить(СписокЗначений);
//Объединение списка номенклатуры с другим списком значений

СписокНоменклатуры.НайтиПоЗначению("Карандаш");
//Поиск элемента списка номенклатуры по значению "Карандаш"

СписокНоменклатуры.Очистить();
//Очистка списка номенклатуры
