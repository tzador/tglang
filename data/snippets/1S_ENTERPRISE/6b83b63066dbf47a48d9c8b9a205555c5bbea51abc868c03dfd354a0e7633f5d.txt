Документ = Новый Документ("ЗаказНаДоставку");

// Выберем несколько полей для заполнения
Документ.Номенклатура = НоменклатураНайти("Книга");
Документ.Количество = 5;

// Заполним адрес доставки
Адрес = Справочники.Адреса.НайтиПоНаименованию("ул. Комсомольская, 12");
Документ.АдресДоставки = Адрес;

// Заполним данные о покупателе
Контрагент = Справочники.Контрагенты.НайтиПоНаименованию("ООО Любимая Книга");
Документ.Покупатель = Контрагент;

// Добавим товары в заказ
Документ.Товары.Добавить("Вечерний сборник", 2);
Документ.Товары.Добавить("Утренний сборник", 3);

// Создадим новую платежную операцию
ПлатежнаяОперация = Новая ПлатежнаяОперация();
ПлатежнаяОперация.ТипОперации = ТипыПлатежей.Приход;
ПлатежнаяОперация.Сумма = 1000;
ПлатежнаяОперация.Назначение = "Оплата за книги";

// Добавим платежную операцию в документ
Документ.ПлатежныеОперации.Добавить(ПлатежнаяОперация);

// Сохраним документ
Документ.Записать();
