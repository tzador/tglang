Функция РасчетАвансовыхДанных(СуммаЗаказа, СтавкаНДС, КурсВалюты)
    // Расчет авансовых данных для заказа
    СуммаАванса = СуммаЗаказа / 2; // Рассчитываем сумму аванса как половину от общей суммы заказа
    НДС = СуммаАванса * СтавкаНДС / 100; // Рассчитываем сумму НДС для аванса
    СуммаЗаказаВВалюте = СуммаЗаказа / КурсВалюты; // Переводим сумму заказа в валюту заказа
    КоличествоУчастников = АвансовыйГрафик.Количество(); // Получаем количество участников в авансовом графике
    СуммаАвансаНаУчастника = СуммаАванса / КоличествоУчастников; // Рассчитываем сумму аванса на одного участника
    Возврат Новый Структура("СуммаАванса", СуммаАванса, "НДС", НДС, "СуммаЗаказаВВалюте", СуммаЗаказаВВалюте, "КоличествоУчастников", КоличествоУчастников, "СуммаАвансаНаУчастника", СуммаАвансаНаУчастника); // Возвращаем структуру с расчитаными значениями
