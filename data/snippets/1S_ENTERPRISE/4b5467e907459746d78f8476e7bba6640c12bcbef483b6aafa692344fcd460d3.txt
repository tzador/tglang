Вичислить СредняяЗарплата(Зарплаты, Размер) Экспорт

Если ТипЗнч(Зарплаты) = Тип("Массив[Число]") И ТипЗнч(Размер) = Тип("Число") Тогда
    Начало
 
        Если Размер = 0 Тогда
            Возврат 0;
        КонецЕсли;

        Начало
            Пока Размер > 0 Цикл
                Зарплаты[Размер] = СокрЛП(Зарплаты[Размер], 2);
                Размер = Размер - 1;
            КонецЦикла;
        Конец;

        Начало
            Если Размер > 0 Тогда
                СредняяЗарплата = Сумма(Зарплаты)/Размер;
                СредняяЗарплата = СокрЛП(СредняяЗарплата, 2);
                Возврат СредняяЗарплата;
            КонецЕсли;
        Конец;
 
    Конец;
