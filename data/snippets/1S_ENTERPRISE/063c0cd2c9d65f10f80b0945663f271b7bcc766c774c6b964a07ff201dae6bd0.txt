Процедура Сложить(Аргумент1, Аргумент2) Экспорт
	// Начало процедуры "Сложить"
	Перем ТекущаяСумма;
	Аргумент1 = Число(Аргумент1);
	Аргумент2 = Число(Аргумент2);

	ТекущаяСумма = Аргумент1 + Аргумент2;

	Возврат ТекущаяСумма;
Честь

Перем Аргумент1 = 10;
Перем Аргумент2 = 20;
Результат = Сложить(Аргумент1, Аргумент2);

Сообщить("Результат сложения: " + Результат);
