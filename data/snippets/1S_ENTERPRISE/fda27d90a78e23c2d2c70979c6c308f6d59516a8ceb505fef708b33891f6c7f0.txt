# Товары
// Создаем реквизит "Код" 
КодТовара - Тип("Строка 10", Обязательный);
// Создаем реквизит "Название" 
Название - Тип("Строка 50", Обязательный);
// Создаем реквизит "Цена" 
Цена - Тип("Число", Обязательный);
// Создаем регистр сведений "Товары"
ТоварыРег - РегистрСведений.Создать("Товары");

// Создаем форму "СписокТоваров"
ФормаСписокТоваров - Форма.Создать("СписокТоваров");
// Добавляем колонку "Код" в форму
ФормаСписокТоваров.Item(0).Item(0).Item(0).Колонки.Add(КодТовара);
// Добавляем колонку "Название" в форму
ФормаСписокТоваров.Item(0).Item(0).Item(0).Колонки.Add(Название);
// Добавляем колонку "Цена" в форму 
ФормаСписокТоваров.Item(0).Item(0).Item(0).Колонки.Add(Цена);
// Отображаем форму
ФормаСписокТоваров.Show();

// Заполняем регистр сведений значениями
ТоварыРег.Добавить();
ТоварыРег.КодТовара = "001";
ТоварыРег.Название = "Книга";
ТоварыРег.Цена = 500;
ТоварыРег.Добавить();
ТоварыРег.КодТовара = "002";
ТоварыРег.Название = "Флешка";
ТоварыРег.Цена = 1000;

// Создаем форму "КарточкаТовара"
ФормаКарточкаТовара - Форма.Создать("КарточкаТовара");
// Добавляем поле "Код" на форму и привязываем к реквизиту
ФормаКарточкаТовара.Item(0).Item(0).Add(КодТовара);
// Добавляем поле "Название" на форму и привязываем к реквизиту
ФормаКарточкаТовара.Item(0).Item(1).Add(Название);
// Добавляем поле "Цена" на форму и привязываем к реквизиту
ФормаКарточкаТовара.Item(0).Item(2).Add(Цена);
// Отображаем форму
ФормаКарточкаТовара.Show();
