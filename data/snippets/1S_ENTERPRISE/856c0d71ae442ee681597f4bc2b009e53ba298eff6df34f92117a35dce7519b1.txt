Функция ТочностьЧисла(Знач Число, Знач Точность)
{
    // Для сохранения исходного знака числа
    При Знак(Число) = -1 Тогда
        Число = -Число;
    КонецЕсли;

    // Для получения целой части числа
    ЦелаяЧасть = Цел(Число);

    // Для получения дробной части числа
    ДробнаяЧасть = Число - ЦелаяЧасть;

    // Для округления числа до заданной точности
    ДробнаяЧасть = ЗначениеФункции("Окр", ДробнаяЧасть, Точность);

    Значение = ЦелаяЧасть + ДробнаяЧасть * Знак(Число);

    // Для возврата результата функции
    Возврат Значение;
}
