НайтиМаксимум = Процедура(Список) Экспорт
    // Обнуляем переменную Максимум для хранения максимального значения
    Максимум = 0;
    // Обходим все элементы списка
    Для Каждого Элемент Из Список Цикл
        // Сравниваем значение текущего элемента с текущим максимальным
        Если Элемент > Максимум Тогда
            Максимум = Элемент;
        КонецЕсли;
    КонецЦикла;

    // Возвращаем найденный максимум
    Возврат Максимум;
КонецПроцедуры;

// Пример использования
// Создаем список с числами
СписокЧисел = Новый СписокЗначений(1, 5, 7, 10, 2);
// Вызываем процедуру НайтиМаксимум и передаем ей список
Максимум = НайтиМаксимум(СписокЧисел);
// Выводим результат на экран
Сообщить("Максимальное значение: " + Максимум);
