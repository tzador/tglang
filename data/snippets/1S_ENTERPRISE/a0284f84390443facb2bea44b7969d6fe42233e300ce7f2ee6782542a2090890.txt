// Объявление класса "Справочник Номенклатура"

EndClass

// Свойство "Наименование"
Property Наименование
	Get
		Return Наименование;
	EndGet
	Set(Value)
		// Проверяем, является ли значение строкой
		If (TypeOf(Value) = "String") Then
			// Присваиваем значение свойству
			Наименование = Value;
		EndIf
	EndSet
EndProperty

// Метод "ДобавитьТовар"
Method ДобавитьТовар(Знач Параметры)
	// Создаем новый объект справочника "Номенклатура"
	НовыйНоменклатура = СоздатьОбъект("Справочник.Номенклатура");
	// Присваиваем параметры
	НовыйНоменклатура.Наименование = Параметры.Наименование;
	НовыйНоменклатура.Артикул = Параметры.Артикул;
	НовыйНоменклатура.Цена = Параметры.Цена;
	НовыйНоменклатура.ЕдиницаИзмерения = Параметры.ЕдиницаИзмерения;
	// Добавляем объект в справочник "Номенклатура"
	ЭтотОбъект.Добавить(НовыйНоменклатура);
EndMethod

// Метод "НайтиТоварПоНаименованию"
Method НайтиТоварПоНаименованию(Знач Наименование)
	// Переменная для хранения найденного товара
	НайденныйТовар;
	// Проходим по всем элементам справочника
	For Each Элемент Из ЭтотОбъект Do
		// Если наименование совпадает с искомым
		If (Элемент.Наименование = Наименование) Then
			// Присваиваем найденный товар в переменную
			НайденныйТовар = Элемент;
			// Прерываем цикл
			Break;
		EndIf
	EndDo
	// Возвращаем найденный товар
	Return НайденныйТовар;
EndMethod
