Процедура СоздатьОбъект(Знач ИмяТаблицы)
    
    // Создаем ссылку на метаданные таблицы
    Метаданные = Метаданные.Объекты[ИмяТаблицы].Метаданные;
    
    // Создаем ссылку на объект данных
    ОбъектДанных = СоздатьОбъект(Метаданные);
    
    // Добавляем нужные поля в объект данных
    ОбъектДанных.Поля.Добавить("Номер", Новый ПолеДанных(Тип("Число")));
    ОбъектДанных.Поля.Добавить("Название", Новый ПолеДанных(Тип("Строка")));
    ОбъектДанных.Поля.Добавить("Дата", Новый ПолеДанных(Тип("Дата")));
    
    // Добавляем ссылку на регистр сведений
    ОбъектДанных.Реквизиты.Добавить("Ссылка", Новый РеквизитСсылка("РегистрСведений.Производство.Дефекты"));
    
    // Добавляем реквизит типа "перечисление"
    ОбъектДанных.Реквизиты.Добавить("Статус", Новый РеквизитПеречисление(Тип("Перечисление.Статусы")));
    
    // Добавляем подчиненный объект данных
    ПодчиненныйОбъект = СоздатьОбъект(Метаданные.ГруппыОбъектов[ИмяТаблицы].Метаданные);
    ПолеДанных = Новый ПолеДанных(Тип("Число"));
    ПолеДанных.Свойство("Использование").Установить(Истина);
    ПодчиненныйОбъект.Поля.Добавить("Количество", ПолеДанных);
    ОбъектДанных.ПодчиненныеДанные.Добавить("ПодчиненныйОбъект", ПодчиненныйОбъект);
    
    // Устанавливаем действующий период для объекта данных
    ОбъектДанных.ПериодыДействия.УстановитьЗначениеЗаполняемого(ЗаполняемыеПериодыДействия.ДефолтнаяЗаполнитель);
    
    // Возвращаем созданный объект данных
    Возврат ОбъектДанных;
КонецПроцедуры
