#Структура СтруктураТовара - структура, которая описывает товар в подсистеме "Склад"
СтруктураТовара = Новый Структура;
СтруктураТовара.Вставить("Артикул", "ТГ-34521");
СтруктураТовара.Вставить("Наименование", "Карандаш");
СтруктураТовара.Вставить("Цена", 50);

#Структура СтруктураЗаказа - структура, которая описывает заказ в подсистеме "CRM"
СтруктураЗаказа = Новый Структура;
СтруктураЗаказа.Вставить("Номер", "З-21");
СтруктураЗаказа.Вставить("Дата", Новый Дата(2021, 6, 15));

#Процедура для обработки заказа
Процедура ОбработкаЗаказа(Знач Заказ, Знач Товар, Знач Количество)
    Если Заказ.Вставить("Товары", Новый Соответствие) Тогда
        Заказ.Товары.Вставить(Товар, Количество);
    Иначе
        Заказ.Товары.Вставить(Товар, 1);
    КонецЕсли;
КонецПроцедуры

#Вызов процедуры
ОбработкаЗаказа(СтруктураЗаказа, СтруктураТовара, 10);

#Вывод результатов
Сообщить("Заказ " + СтруктураЗаказа.Номер + " от " + СтруктураЗаказа.Дата + " содержит товар " + СтруктураТовара.Наименование + " в количестве " + СтруктураЗаказа.Товары[СтруктураТовара] + "шт.");
