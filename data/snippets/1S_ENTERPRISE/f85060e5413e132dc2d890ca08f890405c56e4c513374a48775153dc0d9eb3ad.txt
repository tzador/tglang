#Модуль Склад

// Создание объекта "Склад"
Склад = Новый Склад;

// Заполнение объекта данными
Склад.Наименование = "Основной склад";
Склад.Адрес = "г. Москва, ул. Ленина, д. 1";
Склад.Телефон = "8(800)555-35-35";

// Добавление товаров на склад
Товар1 = Новый Товар;
Товар1.Артикул = "001";
Товар1.Наименование = "Карандаши";
Товар1.Цена = 50;
Товар1.Количество = 100;

Товар2 = Новый Товар;
Товар2.Артикул = "002";
Товар2.Наименование = "Ручки";
Товар2.Цена = 80;
Товар2.Количество = 50;

Склад.ДобавитьТовар(Товар1);
Склад.ДобавитьТовар(Товар2);

// Вывод информации о складе
Сообщить("Наименование склада: " + Склад.Наименование);
Сообщить("Адрес: " + Склад.Адрес);
Сообщить("Телефон: " + Склад.Телефон);
Сообщить("Количество товаров на складе: " + Склад.КоличествоТоваров());

// Изменение цены товара с артикулом "001"
Товар1.Цена = 60;

// Удаление товара с артикулом "002"
Склад.УдалитьТовар("002");

// Вывод информации о товаре с артикулом "001"
Сообщить("Товар с артикулом 001: " + Склад.НайтиТовар("001").Наименование + ", " + Склад.НайтиТовар("001").Количество + " шт., " + Склад.НайтиТовар("001").Цена + " руб.");

// Расчет стоимости оставшегося товара с артикулом "001"
Сообщить("Стоимость оставшегося товара с артикулом 001: " + Склад.НайтиТовар("001").Цена * Склад.НайтиТовар("001").Количество + " руб.");
