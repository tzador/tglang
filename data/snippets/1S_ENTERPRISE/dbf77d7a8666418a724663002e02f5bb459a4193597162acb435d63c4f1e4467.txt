Процедура ПроверитьСистему(Аутентификация, ПроверкаУчетнойЗаписи) Экспорт
// Данная процедура проверяет систему на наличие нужной аутентификации и учетной записи
// Входные параметры:
// - Аутентификация - флаг, указывающий на необходимость проверки аутентификации
// - ПроверкаУчетнойЗаписи - флаг, указывающий на необходимость проверки учетной записи
    Если Аутентификация Тогда
        // Если установлен флаг аутентификации, проверяем доступность аутентификационного сервера
        АутентификационныйСервер = Новый АутентификационныйСервер();
        Если Не АутентификационныйСервер.Доступен() Тогда
            Сообщить("Ошибка! Аутентификационный сервер недоступен!");
            Возврат Ложь;
        КонецЕсли;
    КонецЕсли;
    Если ПроверкаУчетнойЗаписи Тогда
        // Если установлен флаг проверки учетной записи, проверяем ее корректность
        УчетнаяЗапись = Новый УчетнаяЗапись();
        Если Не УчетнаяЗапись.Активна() Тогда
            Сообщить("Ошибка! Эта учетная запись не активна!");
            Возврат Ложь;
        КонецЕсли;
    КонецЕсли;
    Сообщить("Система готова к работе!");
    Возврат Истина;
КонецПроцедуры
