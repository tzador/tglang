Процедура ВызватьЗаполнениеОбъекта(Объект, Шаблон, Параметры)
	Перем ОбъектПоиска, ВыборкаОбъектов, ПолеДанных, ТекстЗапроса;
	
	// Сначала получаем список объектов, соответствующих шаблону
	ТекстЗапроса = "ВЫБРАТЬ РАЗЛИЧНЫЕ
	|	Объект
	|ИЗ
	|	&ОбъектПоиска ЗДЕСЬ &Параметры";
	ВыборкаОбъектов = Запрос.Выполнить(ТекстЗапроса);
	
	// Затем заполняем каждый объект данными из шаблона
	ОбъектПоиска = Новый ЗначенияРеквизитов(Шаблон);
	Пока ВыборкаОбъектов.Следующий() Цикл
		Для Каждого ПолеДанных Из ОбъектПоиска Цикл
			ПолеДанных.УстановитьЗначение(Шаблон[ПолеДанных.Имя], ВыборкаОбъектов[ПолеДанных.Имя]);
		КонецЦикла;
		Объект.Добавить(ОбъектПоиска);
	КонецЦикла;
КонецПроцедуры
