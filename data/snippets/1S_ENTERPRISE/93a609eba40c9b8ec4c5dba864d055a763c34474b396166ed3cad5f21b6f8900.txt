# Данный модуль содержит функцию, реализующую сортировку списка чисел методом пузырька

Параметры:
  Список (Массив) - список чисел, который необходимо отсортировать по возрастанию

Функция СортировкаПузырьком(Список) Экспорт

  # Переменная для хранения длины списка
  Длина = Список.Количество();
  
  # Цикл для прохода по всем элементам списка
  Для Индекс = 0 По Длина-1 Цикл
  
    # Внутренний цикл для сравнения пар элементов и их обмена
    # На каждой итерации внешнего цикла, последний элемент списка будет уже отсортирован
    Для ВнутреннийИндекс = 0 По Длина - Индекс - 1 Цикл
    
      # Сравниваем текущий элемент списка с следующим
      Если Список[ВнутреннийИндекс] > Список[ВнутреннийИндекс+1] Тогда
        
        # Обмениваем элементы местами
        ТекущийЭлемент = Список[ВнутреннийИндекс];
        Список[ВнутреннийИндекс] = Список[ВнутреннийИндекс+1];
        Список[ВнутреннийИндекс+1] = ТекущийЭлемент;
      
      КонецЕсли;
      
    КонецЦикла;
    
  КонецЦикла;
  
  # Возвращаем отсортированный список
  Возврат Список;
  
КонецФункции;
