#DEFINITION
Функция ВычислитьСреднее(МассивЗначений)
    
    #LOCALS
    ИтоговоеЗначение = 0;
    КоличествоЗначений = 0;
    
    #CYCLE START
    Для Каждого Значение Из МассивЗначений Цикл
    
        #CALCULATION
        ИтоговоеЗначение = ИтоговоеЗначение + Значение;
        КоличествоЗначений = КоличествоЗначений + 1;
        
    КонецЦикла;
    #CYCLE END
    
    #CHECK IF EMPTY
    Если КоличествоЗначений = 0 Тогда
        Возврат 0;
    КонецЕсли;
    
    #CALCULATE AVERAGE
    Возврат ИтоговоеЗначение / КоличествоЗначений;
    
КонецФункции;
