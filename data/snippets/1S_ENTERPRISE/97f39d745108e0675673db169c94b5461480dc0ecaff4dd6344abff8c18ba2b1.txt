# Алгоритм обработки товара
Процедура ОбработатьТовар(Товар){
    Если Товар.Цена > 1000 Тогда
        ДобавитьТовар(Товар.Наименование, Товар.Цена, Товар.Количество, Товар.Производитель);
    КонецЕсли
}

# Процедура добавления товара
Процедура ДобавитьТовар(Наименование, Цена, Количество, Производитель){
    ПровайдерДанных = Новый ПровайдерДанных();
​
    Запрос = Новый Запрос;
    Запрос.Текст = "Вставить в Товары Значения (&Наименование, &Цена, &Количество, &Производитель)";
​
    Параметры = Запрос.ДобавитьПараметр("Наименование", Наименование);
    Параметры = Запрос.ДобавитьПараметр("Цена", Цена);
    Параметры = Запрос.ДобавитьПараметр("Количество", Количество);
    Параметры = Запрос.ДобавитьПараметр("Производитель", Производитель);
​
    ПровайдерДанных.ВыполнитьЗапрос(Запрос);
    Если ПровайдерДанных.Ошибок() > 0 Тогда
        Сообщить("Ошибка при добавлении товара.");
    КонецЕсли
}
