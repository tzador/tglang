Константа ПЕРВАЯКОНСТАНТА = 5; (* это первая константа, которая применяется в коде *)
Переменная ПЕРВАЯПЕРЕМЕННАЯ;
Переменная ВТОРАЯПЕРЕМЕННАЯ;
Область ДанныеДокумента(ПЕРЕМЕННАЯ, ВТОРАЯПЕРЕМЕННАЯ);
Функция УмножениеЧисел(Аргумент1, Аргумент2)
	Если Аргумент1 > 0 Тогда (* если первый аргумент больше 0 *)
		Возврат Аргумент1 * Аргумент2; (* умножаем первый аргумент на второй и возвращаем результат *)
	Иначе
		Возврат 0; (* если первый аргумент не больше 0, возвращаем 0 *)
	КонецЕсли;
КонецФункции;
Функция СреднееЧисло(МассивЧисел)
	Переменная Сумма = 0; (* переменная для хранения суммы значений *)
	Переменная Количество = 0; (* переменная для хранения количества чисел *)
	Для Каждого Число Из МассивЧисел Цикл (* для каждого числа в массиве *)
		Сумма = Сумма + Число; (* прибавляем число к сумме *)
		Количество = Количество + 1; (* увеличиваем количество чисел на 1 *)
	КонецЦикла;
	Возврат Сумма / Количество; (* вычисляем среднее значение и возвращаем результат *)
КонецФункции;
Процедура ВыводЧисел(МассивЧисел)
	Для Каждого Число Из МассивЧисел Цикл (* для каждого числа в массиве *)
		Сообщить(Число); (* выводим число в сообщении *)
	КонецЦикла;
КонецПроцедуры;
