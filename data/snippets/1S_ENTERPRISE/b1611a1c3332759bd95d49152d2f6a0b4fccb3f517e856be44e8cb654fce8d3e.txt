Документ = Новый Документ("ЗаказПоставщику");
// создаем новый документ типа 'ЗаказПоставщику'
Документ.Номер = "ЗП0001";
// присваиваем номер документа
Документ.Дата = ТекущаяДата();
// присваиваем текущую дату
Товары = Документ.Номенклатура.НайтиПоНаименованию("Ноутбук");
// находим в справочнике Номенклатура ноутбук
Товар = Товары.Получить();
// получаем объект товара
Товар.Количество = 10;
// задаем количество ноутбуков, которые нужно заказать
Товар.Цена = 50000.00;
// задаем цену за один ноутбук
Товар.Сумма = Товар.Количество * Товар.Цена;
// вычисляем сумму заказа
Документ.Товары.Добавить(Товар);
// добавляем товар в документ
Документ.ЗаполнитьПоНомеруСерии(Товар, "SN0001");
// заполняем номер серии товара
Документ.Сохранить();
// сохраняем документ
