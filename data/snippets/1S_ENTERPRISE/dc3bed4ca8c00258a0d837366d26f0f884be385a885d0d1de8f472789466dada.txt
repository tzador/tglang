// This code snippet calculates the sum of two numbers
// and outputs the result to the console

// Declaring and initializing variables
var num1 = 5; // first number
var num2 = 7; // second number
var result; // variable to store the result

// Using built-in function to calculate sum
result = num1 + num2; // sum of num1 and num2 is assigned to result

// Output the result to the console
Console.Write(result); // prints the result to the console
