Процедура ВыгрузкаДанных(Параметры)
	Перем ТаблицаДанных;
	ТаблицаДанных = Новый Структура;
	Соединение = Новый COMОбъект("ADODB.Connection"); // создаем объект для работы с базой данных
	СтрокаЗапроса = "SELECT * FROM Сотрудники WHERE Должность = 'Менеджер'"; // формируем запрос для выборки данных по сотрудникам
	Запрос = Новый COMОбъект("ADODB.Recordset"); // создаем объект для хранения результата запроса
	Запрос.Open(СтрокаЗапроса, Соединение, , , 1); // выполняем запрос и заполняем объект
	Если Запрос.RecordCount <> 0 Тогда
		Пока НЕ Запрос.Блокирование Цикл
			ТаблицаДанных.Вставить(Запрос.Fields("ФИО"), Запрос.Fields("Возраст"), Запрос.Fields("Отдел")); // добавляем полученные данные в структуру
			Запрос.MoveNext(); // переходим к следующей записи
		КонецЦикла;
	КонецЕсли;
	ТаблицаДанных.Сортировать("ФИО", Возраст: Убыв); // сортируем данные по ФИО в обратном порядке и по возрасту по убыванию
	Для Каждого Строка Из ТаблицаДанных Цикл
		Сообщить(Строка.ФИО + " работает в отделе " + Строка.Отдел + " и ему " + Строка.Возраст + " лет"); // выводим информацию о сотруднике
	КонецЦикла;
КонецПроцедуры

ВыгрузкаДанных(); // вызываем процедуру без параметров, так как нас интересуют все сотрудники-менеджеры
