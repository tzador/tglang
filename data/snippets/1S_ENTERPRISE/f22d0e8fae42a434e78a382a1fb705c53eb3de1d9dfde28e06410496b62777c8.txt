Процедура РассчитатьЗатраты(Количество, Цена)
	
	//Количество - входной параметр, определяющий количество товара
	//Цена - входной параметр, определяющий цену единицы товара
	//Затраты - выходной параметр, определяющий общую сумму затрат

	//Объявление переменных
	Перем Количество, Цена, Затраты, НДС;
	
	//Рассчет суммы без НДС
	Затраты = Количество * Цена;
	
	//Рассчет НДС
	НДС = Затраты * 0.2;
	
	//Расчет общей суммы затрат
	Затраты = Затраты + НДС;
	
	//Вывод результата
	Сообщение("Общие затраты: " + Затраты + "руб.");
	
КонецПроцедуры
