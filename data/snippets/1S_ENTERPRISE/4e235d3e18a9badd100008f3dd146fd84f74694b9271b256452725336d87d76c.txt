ВРЕЗУЛТАТ = Новый Массив;

ДЛЯ Каждого Элемент ИЗ Справочник.Валюты ЦИКЛ
    Если НЕ ЗначениеЗаполнено(Справочник.Валюты[Элемент].Курс) Тогда
        Продолжить;
    КонецЕсли;
    
    Если Не ЗначениеЗаполнено(Справочник.Валюты[Элемент].Кратность) Тогда
        Продолжить;
    КонецЕсли;
    
    Если Не ЗначениеЗаполнено(Справочник.Валюты[Элемент].Номинал) Тогда
        Продолжить;
    КонецЕсли;
    
    КурсВалюты = Справочник.Валюты[Элемент].Курс;
    Кратность = Справочник.Валюты[Элемент].Кратность;
    Номинал = Справочник.Валюты[Элемент].Номинал;
    
    Результат = КурсВалюты * Кратность / Номинал;
    Результат = Округлить(Результат, 4);
    ВРЕЗУЛТАТ.Добавить(Результат);
КонецЦикла;

Сообщить(ВРЕЗУЛТАТ);
