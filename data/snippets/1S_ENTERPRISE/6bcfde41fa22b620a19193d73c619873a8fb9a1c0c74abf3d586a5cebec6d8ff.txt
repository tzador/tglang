Функция НайтиСумму(число1, число2, число3, число4)
СтрокаТаблицы = Новый Массив;
Перем Условие1, Условие2, Условие3, ИтогДепозит;
Условие1 = Число1 > Число2;
Условие2 = Число3 < Число4;
Условие3 = Число1 = Число3 И Число2 = Число4;
Если Условие1 Тогда
   СтрокаТаблицы.Добавить(Символы.Строка(Число1, Число2));
Иначе Если Условие2 Тогда
   СтрокаТаблицы.Добавить(Символы.Строка(Число3, Число4));
Иначе Если Условие3 Тогда
   СтрокаТаблицы.Добавить(Символы.Строка(Число1, Число2, Число3, Число4));
Иначе
   СтрокаТаблицы.Добавить(Символы.Строка(Число1, Число2, Число4));
КонецЕсли;
Для каждого СтрокаТаблицы Из СтрокаТаблицы Цикл
   ИтогДепозит = ИтогДепозит + СтрокаТаблицы.Получить(0);
КонецЦикла;
Возврат ИтогДепозит;
КонецФункции;
