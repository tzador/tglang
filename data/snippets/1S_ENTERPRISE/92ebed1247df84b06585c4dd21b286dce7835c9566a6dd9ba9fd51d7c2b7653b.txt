Справочник.СоздатьОписаниеФормы("ФормаЦеныСтандартная",Истина, Истина); //Создание описания формы "ФормаЦеныСтандартная" c возможностью редактирования и копирования
Форма = ОписаниеФормы.СоздатьФорму(); //Создание объекта формы на основе описания
Форма.УстановитьЗаголовок("Цены для товаров"); //Установка заголовка формы
Форма.УстановитьПоложениеФормы(400, 200); //Установка позиции формы на экране
КолонкаТовар = Форма.Реквизиты.Добавить("Товар", Тип("СправочникСсылка.Номенклатура")); //Добавление колонки справочников "Товар" в форму
КолонкаЦена = Форма.Реквизиты.Добавить("Цена", Тип("Число")); //Добавление колонки типа "Число" в форму
КолонкаФирма = Форма.Реквизиты.Добавить("Фирма", Тип("СправочникСсылка.Контрагенты")); //Добавление колонки справочников "Фирма" в форму
КнопкаДобавить = Форма.Кнопки.Добавить("Добавить"); //Добавление кнопки "Добавить" в форму
КнопкаСохранить = Форма.Кнопки.Добавить("Сохранить"); //Добавление кнопки "Сохранить" в форму
КнопкаОтменить = Форма.Кнопки.Добавить("Отменить"); //Добавление кнопки "Отменить" в форму
Форма.Реквизиты.Разместить(КолонкаТовар, КолонкаЦена, КолонкаФирма); //Размещение реквизитов на форме в порядке добавления
Форма.Кнопки.Разместить(КнопкаДобавить, КнопкаСохранить, КнопкаОтменить); //Размещение кнопок на форме в порядке добавления
Форма.Открыть(); //Открытие формы для редактирования
Номенклатура = Номенклатура.НайтиПоКоду(12345); //Поиск существующей записи о номенклатуре по ее коду
Контрагент = Контрагенты.НайтиПоКоду(98765); //Поиск существующей записи о контрагенте по его коду
НоваяЗапись = Таблица.Добавить(); //Создание новой записи в таблице данных для редактирования
НоваяЗапись.Товар = Номенклатура; //Установка значения поля "Товар" для новой записи
НоваяЗапись.Цена = 1000; //Установка значения поля "Цена" для новой записи
НоваяЗапись.Фирма = Контрагент; //Установка значения поля "Фирма" для новой записи
