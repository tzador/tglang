Процедура РассчитатьНалог(Сумма) Экспорт
    ПредопределенныеПеременные = Новый Массив;
    ПредопределенныеПеременные.Добавить("НДС");
// Создаем объект-запрос для получения ставки НДС
    ЗапросНДС = Новый Запрос;
    ЗапросНДС.Текст = "ВЫБРАТЬ ПЕРВЫЕ 1
                      Измерения.НДССтавки
                      ГДЕ Номенклатура В (ВЫБРАТЬ РодительНоменклатуры ИЗ Товары)";
// Выполняем запрос
    Результат = ЗапросНДС.Выполнить();
// Получаем значение ставки НДС из результата запроса
    СтавкаНДС = Результат[0].НДССтавка;
    Налог = Сумма * СтавкаНДС / 100;
    Возврат Налог;
