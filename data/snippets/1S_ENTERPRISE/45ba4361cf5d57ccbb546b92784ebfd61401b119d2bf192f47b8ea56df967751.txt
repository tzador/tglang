// This code snippet illustrates a complex conditional statement in 1C Enterprise

Если Истина Тогда
    // Начало тела условия
    Если Истина Тогда
        // Выполняется если истина обоих условий
        Перем ИмяПеременной;
        Если Значение1 = Значение2 Тогда
            // Выполняется если оба значения равны
            ИмяПеременной = 10;
        КонецЕсли;
    Иначе
        // Выполняется если первое условие истинно, а второе - ложно
        Если Значение2 = 5 Тогда
            ИмяПеременной = 20;
        КонецЕсли;
    КонецЕсли;
Иначе ЕСЛИ НЕ Истина Тогда
    // Выполняется если первое условие ложно
    Перем НоваяПеременная;
    NewVar = "This is a new variable";
Иначе
    // Выполняется если все условия ложны
    Перем Ответ;
    Ответ = "None of the conditions were met";
КонецЕсли;
