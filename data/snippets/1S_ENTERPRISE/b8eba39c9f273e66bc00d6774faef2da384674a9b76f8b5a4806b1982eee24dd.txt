Функция ВыполнитьОперацию(Знач Операция, Знач СписокЗначений, Знач СтрокаСоединения) Экспорт
	
	// Описание:
	// Функция для выполнения операции с помощью соединения к базе данных
	
	// Параметры:
	// Операция (ПеречислениеТиповОпераций) - операция, которую необходимо выполнить
	// СписокЗначений (УникальныйТип) - список значений, необходимых для выполнения операции
	// СтрокаСоединения (Строка) - строка для соединения к базе данных
	
	// Возвращаемое Значение:
	// Булево - результат выполнения операции
	
	// Инициализация переменных
	УникальныйТипЗначение = СоздатьОбъект("УникальныйТип");
	УникальныйТипЗначение.Новый();
	Булево = Ложь;
	Операции = Новый Массив;

	// Добавляем доступные операции
	Операции.Add(ТипПеречисления.Создать("Операция1"));
	Операции.Add(ТипПеречисления.Создать("Операция2"));
	Операции.Add(ТипПеречисления.Создать("Операция3"));

	// Проверяем, что операция существует в списке доступных операций
	Если Не Операции.Содержит(Операция) Тогда

		// Выполняем соединение к базе данных
		ПодключениеКБД = ПодключитьКБазеДанных(СтрокаСоединения);
		Если ПодключениеКБД = Неопределено Тогда
			
			// В случае ошибки выводим сообщение
			Сообщение("Ошибка соединения к базе данных");
		Иначе
			
			// Выполняем операцию с помощью соединения к базе данных
			ВыборкаДанных = ПодключениеКБД.ВыполнитьЗапрос(СписокЗначений);
			Если ВыборкаДанных.Количество() > 0 Тогда 
				
				// Устанавливаем результат выполнения операции в Булево
				Булево = Истина;
			КонецЕсли;
		КонецЕсли;
	Иначе
		
		// Выводим сообщение об ошибке
		Сообщение("Невозможно выполнить операцию. Операция не существует в списке доступных операций.");
	КонецЕсли;

	// Возвращаем результат выполнения операции
	Возврат Булево;

КонецФункции
