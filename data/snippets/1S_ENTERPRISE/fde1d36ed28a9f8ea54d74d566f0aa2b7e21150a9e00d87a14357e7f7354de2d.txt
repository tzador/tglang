#Программный модуль
#ОписаниеМодуля "ИерархияСотрудников"

Процедура РассчитатьИерархиюСотрудников(Отдел)
	//Получим список всех сотрудников в отделе
	СписокСотрудников = Сотрудники.НайтиПоОтделу(Отдел);
	
	//Получим список подразделений в отделе
	СписокПодразделений = Отделы.НайтиПоРодительскому(Отдел);
	
	//Для каждого подразделения в списке
	Для Каждого Подразделение Из СписокПодразделений Цикл
		//Получим список сотрудников в подразделении
		СписокСотрудниковПодразделения = Сотрудники.НайтиПоОтделу(Подразделение);
		
		//Добавим каждого сотрудника в список общий список сотрудников в отделе
		СписокСотрудников.Добавить(СписокСотрудниковПодразделения);
		
		//Рекурсивно выполняем данную процедуру для текущего подразделения
		РассчитатьИерархиюСотрудников(Подразделение);
		
		//Установим сотрудникам их руководителя
		Для Каждого Сотрудник Из СписокСотрудниковПодразделения Цикл
			Сотрудник.Руководитель = Подразделение.НайтиРуководителя();
		
	КонецЦикла;
	
	//Возвращаем список сотрудников
	Возврат СписокСотрудников;
	
КонецПроцедуры
