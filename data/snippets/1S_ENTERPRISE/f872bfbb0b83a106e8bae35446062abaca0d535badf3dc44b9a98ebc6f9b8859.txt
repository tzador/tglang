#Директивы
#Область *Инструкция*
Введите "Hello, World!" как Строка; #Объявление переменной и ее инициализация

#Декларациянная обработка
Если Строка = "Hello, World!" ТОГДА #Условный оператор
    Вывести "Истинно" на Экран; #Вывод на экран
ИНАЧЕ
    Вывести "Ложно" на Экран; 
КонецЕсли;
    
#Циклические операции
Для i = 1 По 10 Цикл #Цикл с параметром
    Вывести i на Экран; #Вывод на экран текущего значения параметра
КонецЦикла;

Пока i < 10 Цикл #Цикл с предусловием
    Вывести i на Экран;
    Увеличить(i); #Увеличение значения параметра
КонецЦикла;

#Процедуры и функции
Процедура ВывестиСумму(Аргумент1, Аргумент2) #Декларация процедуры с двумя аргументами
    Сумма = Аргумент1 + Аргумент2; #Вычисление суммы
    Вывести Сумма на Экран; #Вывод на экран
КонецПроцедуры;

Функция ВычитаниеЧисел(Аргумент1, Аргумент2) #Декларация функции с двумя аргументами
    Разность = Аргумент1 - Аргумент2; #Вычисление разности
    Возврат Разность; #Возвращение результата
КонецФункции;

#Пример использования процедуры и функции
Введите А = 10 как Число;
Введите B = 5 как Число;

ВывестиСумму(А, B); #Вызов процедуры
Вывести ВычитаниеЧисел(А, B); #Вывод результата функции на экран

#Конструкция "Если...Иначе"
Если А < B Тогда
    Вывести "B больше A" на Экран;
Иначе
    Вывести "A больше или равно B" на Экран;
КонецЕсли;
