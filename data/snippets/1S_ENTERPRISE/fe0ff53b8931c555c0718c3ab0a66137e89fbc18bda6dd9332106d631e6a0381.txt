# Сначала создаем массив типа "Структура", внутри которой будут находиться различные значения для каждого компьютера

Имя_компьютеров.Добавить("Компьютер3");

Имя_компьютеров[0].Свойство1 = Значение1;
Имя_компьютеров[0].Свойство2 = Значение2;

Имя_компьютеров[1].Свойство2 = Значение4;

Сообщить(Имя_компьютеров.СтрокаОбъекта());
