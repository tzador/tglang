# Обработка Данных

// Эта функция считает среднее значение списка чисел
Функция Среднее(СписокЧисел)
{
    Перем Сумма = 0;
    Перем Колво = 0;
    Перем Среднее = 0;
    Для Каждого Число Из СписокЧисел Цикл
    {
        Сумма = Сумма + Число; // считаем сумму всех чисел
        Колво = Колво + 1; // считаем количество чисел
    }
    Среднее = Сумма / Колво; // находим среднее значение
    Возврат Среднее;
}

// Эта функция сортирует список чисел по возрастанию
Функция Сортировка(СписокЧисел)
{
    Перем Позиция = 1;
    Пока Позиция < Размер(СписокЧисел) Цикл
    {
        Если СписокЧисел[Позиция] > СписокЧисел[Позиция + 1] Тогда
        {
            Перем Временная = СписокЧисел[Позиция]; // сохраняем значение во временную переменную
            СписокЧисел[Позиция] = СписокЧисел[Позиция + 1]; // меняем местами значения
            СписокЧисел[Позиция + 1] = Временная;
        }
        Позиция = Позиция + 1;
    }
    Возврат СписокЧисел;
}

// Эта функция проверяет, является ли строка палиндромом
Функция Палиндром (Строка)
{
    Перем Строка1 = Верх(Строка); // переводим строку в верхний регистр
    Перем Строка2 = Нижн(Строка); // переводим строку в нижний регистр
    Перем Размер = Размер(Строка1);
    Для Каждого Символ Из Строка1 Цикл
    {
        Если Символ = Строка2[Размер] Тогда // проверяем симетричность символов
            Размер = Размер - 1;
        Иначе
            Возврат Ложь;
    }
    Возврат Истина;
}

// Пример использования функций
Перем Лист = Новый Список(3, 5, 1, 4, 2); // создаем список чисел
Вывести Среднее(Лист); // 3
Перем ОтсортированныйЛист = Сортировка(Лист); // сортируем список
Вывести ОтсортированныйЛист; // 1, 2, 3, 4, 5
Перем Строка = "арозаупаланалапуазора"; // строка-палиндром
Вывести Палиндром(Строка); // Истина
