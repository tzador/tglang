Отчет Структура Каталога
Параметры:
	глУровень - Число - Необязательный - Уровень структуры (по умолчанию 1)
Отчет = Новый Отчет;
Отчет.Параметры.глУровень = !глУровень;
Каталог = Отчет.Сформировать(ВидОтчета.Табличный,, "Идентификатор, Описание, ТипОбъекта, Уровень");
Набор = Новый НаборЗаписей;
Набор.Отбор.ТипОбъекта.Добавить("Каталог");
Набор.Отбор.Уровень.Добавить(Отчет.Параметры.глУровень);
Набор.Сортировка.Добавить("Уровень, Описание");
Набор.Загрузить(Каталог);
Для Каждого Запись Из Набор Цикл
	Идентификатор = Запись.Идентификатор;
	ТипОбъекта = Запись.ТипОбъекта;
	Уровень = Запись.Уровень;
	Описание = Запись.Описание;
	Если Запись.ЭтоГруппа Тогда
		Счетчик = Неопределено;
		Если СтрНайти(Идентификатор, ".") > 0 Тогда
			Счетчик = СокрЛП(Лев(Идентификатор, СтрНайти(Идентификатор, ".") - 1));
		КонецЕсли;
	Если Счетчик <> о Неопределено Тогда
		СчетГруппа = Неопределено;
		СчетГруппа = Новый СчетГруппа;
		СчетГруппа.Наименование = Описание;
		Затраты = СчетГруппа.Добавить(ВидСубконто.Затраты);
		Затраты.Наименование = Описание;
		КомплектМатериалов = Затраты.Добавить(ВидСубконто.КомплектМатериалов);
		КомплектМатериалов.Наименование = Описание;
	ФормированиеОбъекта = СчетГруппа.Добавить(ВидСубконто.ФормированиеОбъекта);
	ФормированиеОбъекта.Наименование = Описание;
	Продукция = ФормированиеОбъекта.Добавить(ВидСубконто.Продукция);
	Продукция.Наименование = Описание;
Продукция.Регистратор.Записать(Отчет);
Продукция.Записать(Истина);
КонецЦикла;
