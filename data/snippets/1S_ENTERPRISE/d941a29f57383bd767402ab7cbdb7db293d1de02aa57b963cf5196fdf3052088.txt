Процедура ПроверкаСостояния(Значение)
   Перем ТекстовоеСообщение, ФлагСостояния;
   
   Условие = Значение > 1000;
   Если Условие Тогда
      ФлагСостояния = Истина;
      Сообщить("Значение больше 1000");
   Иначе
      ФлагСостояния = Ложь;
      Сообщить("Значение меньше или равно 1000");
   КонецЕсли;
   
   Попытка
      В случае Когда ФлагСостояния Тогда
         Соединение = Новый Соединение;
         Соединение.Подключить();
      Когда Иначе
         Выражение = Новый Выражение;
         Выражение.Вычислить();
      КонецСлучая;
   Исключение
      Сообщить("Ошибка при выполнении проверки состояния");
   КонецПопытки;
КонецПроцедуры;
