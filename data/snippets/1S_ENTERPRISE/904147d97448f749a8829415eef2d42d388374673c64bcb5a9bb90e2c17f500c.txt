Функция СтроковаяВытаскиваетБуквы()
{

Перем Строка;
Перем Буква;
Строка = "Привет! Как дела?";

Для Индекс = 1 По СтрДлина(Строка) Цикл
{
    Буква = Сред(Строка, Индекс, 1); //Вытаскиваем букву из строки
    Если (Буква % 2 = 0) Тогда //Проверяем, является ли буква четной
    {
        Сообщить(Буква); //Выводим букву на экран
    }
}
Возврат;
}
 
Процедура ГенерацияЧисел(Мин, Макс)
{
    Пока Мин <= Макс Цикл //Генерируем числа от Мин до Макс
    {
        СлучайноеЧисло = СлучайноеЧисло(Мин, Макс);
        Если СтрДлина(СлучайноеЧисло) > 2 Тогда //Проверяем, является ли число больше двухзначного
        {
            Сообщить(СлучайноеЧисло); //Выводим число на экран
        }
        Иначе
        {
            Продолжить; //Идем на следующую итерацию цикла
        }
        Мин = Мин + 1; //Увеличиваем Мин на 1
    }
}
 
Пока Выбор = Да Цикл
{
    СтроковаяВытаскиваетБуквы(); //Вызываем функцию
    ГенерацияЧисел(10, 100); //Вызываем процедуру
    Сообщить("Хотите продолжить? (Да/Нет)");
    Считать(Выбор);
}
