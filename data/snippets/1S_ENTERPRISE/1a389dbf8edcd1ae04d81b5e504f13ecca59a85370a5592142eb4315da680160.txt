// This code snippet shows how to implement a sorting algorithm in 1C Enterprise programming language

// Define an array of unsorted numbers

Варианты = Новый Массив;

Варианты.Добавить(51);
Варианты.Добавить(32);
Варианты.Добавить(16);
Варианты.Добавить(78);
Варианты.Добавить(24);

// Создаем переменную для хранения длины массива
РазмерМассива = Варианты.Количество();

// Определяем переменные для хранения текущего и следующего элемента массива
ТекущийЭлемент = 0;
СледующийЭлемент = 0;

// Начинаем цикл сортировки
Для i = 0 По РазмерМассива Цикл
{
    // Определяем текущий элемент как первый элемент в неотсортированной части массива
    ТекущийЭлемент = Варианты[i];

    // Определяем следующий элемент как первый элемент в оставшейся части массива
    СледующийЭлемент = Варианты[i + 1];

    // Если текущий элемент больше следующего, меняем их местами
    Если (ТекущийЭлемент > СледующийЭлемент) Тогда
    {
        Варианты[i] = СледующийЭлемент;
        Варианты[i + 1] = ТекущийЭлемент;
    }
}

// Выводим отсортированный массив
Сообщить("Отсортированные числа: " + Варианты);
