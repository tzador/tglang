// This code snippet calculates the area of a circle

// Declaring variables
CircleR: Number; // Radius of the circle
ConstantPi: Real; // Constant value of Pi
Area: Real; // Variable to store the calculated area

// Setting values to variables
CircleR = 5; // Radius of the circle is 5
ConstantPi = 3.14;

// Calculating area of the circle
Area = ConstantPi * Square(CircleR); // Area = Pi * R^2

// Printing the result
Message("The area of the circle with radius " + CircleR + " is " + Area);

// Output:
// The area of the circle with radius 5 is 78.5
