var str = "Hello World!";
console.log(str);

// This code snippet creates a variable "str" and assigns it the value "Hello World!"
// Then, it prints the value of the variable to the console.
// The syntax used in 1C Enterprise is similar to JavaScript.
// Comments are created using double forward slashes.
// Variables are declared using the "var" keyword, followed by the variable name and its value.
// The console.log() function is used to print the value of the variable to the console.
// In this case, the output would be "Hello World!"
