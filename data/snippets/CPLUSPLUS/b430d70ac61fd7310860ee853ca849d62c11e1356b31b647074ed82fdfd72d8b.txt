// initialize variables
int num1 = 6;
int num2 = 9;
int result = 0;

// if-else statement to check if num1 is greater than num2
if(num1 > num2){
    // if true, perform multiplication
    result = num1 * num2;
} else {
    // if false, perform division
    result = num1 / num2;
}

// print result 
std::cout << "Result: " << result << std::endl;
