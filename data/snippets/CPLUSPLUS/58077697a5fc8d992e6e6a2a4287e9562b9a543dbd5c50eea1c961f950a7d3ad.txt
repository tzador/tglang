#include <iostream>  // include required library

using namespace std;  // use standard namespace

int main() {  // define main function
    int num1 = 5;  // declare and initialize variable
    int num2 = 10;  // declare and initialize variable
    int result;  // declare variable for result
    result = num1 + num2;  // perform addition and assign result to variable
    cout << "The sum of " << num1 << " and " << num2 << " is: " << result << endl; // print result to console
    return 0; // end program
}
