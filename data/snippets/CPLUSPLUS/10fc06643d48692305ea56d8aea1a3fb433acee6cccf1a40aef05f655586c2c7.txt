#include <iostream>                    //include the iostream library

using namespace std;                   //use the standard namespace

int main() {                           //declare the main function with an integer return type
    int a = 10;                        //declare and initialize an integer variable 'a' with value 10
    float b = 5.5;                     //declare and initialize a float variable 'b' with value 5.5
    cout << "The sum of a and b is: " << a + b << endl;  //output the sum of variables 'a' and 'b' using cout
    return 0;                          //return 0 to exit the main function
}                                      //end of main

/* Output:
The sum of a and b is: 15.5
*/
