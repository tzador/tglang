#include <iostream> // Include input/output library

using namespace std; // Use standard namespace for convenience

int main() // Main function
{
    int num1 = 5; // Initialize first variable
    int num2 = 10; // Initialize second variable
    double result; // Declare result variable

    // Perform complex calculation
    result = (num1 + num2) * (num1 - num2);

    // Print output to console
    cout << "Result of calculation: " << result << endl;

    return 0; // End of program 
}
