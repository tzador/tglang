#include <iostream> //include the standard library for input/output operations

int main() //main function
{
    int num1 = 5; //initialize variable num1 with value 5
    int num2 = 10; //initialize variable num2 with value 10
    int result = num1 + num2; //add num1 and num2 and store the result in variable result

    std::cout << "The sum of " << num1 << " and " << num2 << " is: " << result << std::endl; //print the result to the console

    return 0; //return 0 to indicate successful execution
}
