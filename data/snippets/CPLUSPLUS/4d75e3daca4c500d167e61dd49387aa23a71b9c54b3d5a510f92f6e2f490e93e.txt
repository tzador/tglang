#include <iostream>

// Main function
int main() {
    // Initialize variables
    int num1 = 5;
    int num2 = 7;
    int sum;

    // Add numbers and store result in sum
    sum = num1 + num2;

    // Print sum to console
    std::cout << "Sum of " << num1 << " and " << num2 << " is " << sum << std::endl;

    // End program
    return 0;
}

// Output:
// Sum of 5 and 7 is 12
