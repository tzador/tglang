// This is a function that takes in two integers and returns the sum of their squares
// x: first integer
// y: second integer
// returns: the sum of their squares
int sumOfSquares(int x, int y) {
    // Initializing variables
    int squareX, squareY;
    
    // Calculating squares
    squareX = x * x;
    squareY = y * y;
    
    // Returning the sum of squares
    return squareX + squareY;
}
