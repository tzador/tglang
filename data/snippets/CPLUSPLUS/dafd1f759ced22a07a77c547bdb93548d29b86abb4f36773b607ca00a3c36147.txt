int main() { //this is the main function
   int a = 5; //initialize variable a with value 5
   int b = 7; //initialize variable b with value 7
   int c = 0; //initialize variable c with value 0

   //perform arithmetic operations
   c = a + b; //add a and b, store result in c
   c = c / 2; //divide c by 2, store new result in c
   c = c - a; //subtract a from c, store new result in c

   //print output
   cout << "The value of c is: " << c << endl; //output the value of c

   return 0; //return 0 to indicate successful execution
}
