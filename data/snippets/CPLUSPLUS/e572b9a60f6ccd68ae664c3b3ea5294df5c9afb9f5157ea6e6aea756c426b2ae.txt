// Example code snippet in C++
#include <iostream>

int main() {
  std::cout << "Hello World!"; //Prints out "Hello World!" to the console
  int a = 5; //Declares an integer variable named 'a' and assigns the value of 5 to it
  int b = 10; //Declares an integer variable named 'b' and assigns the value of 10 to it
  int result = a + b; //Declares an integer variable named 'result' and assigns the value of a + b to it
  std::cout << result; //Prints out the value stored in 'result' to the console
  return 0; //Returns 0 to indicate successful execution of the program
}
