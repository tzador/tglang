(defun partition (list pivot)
  "This function splits a list into two sublists based on a given pivot element."
  (let ((smaller '())
        (greater '())
        (pivot-val (car pivot)))
    (dolist (elem list)
      (if (< elem pivot-val)
          (setf smaller (cons elem smaller))
          (setf greater (cons elem greater))))
    (list smaller greater)))
