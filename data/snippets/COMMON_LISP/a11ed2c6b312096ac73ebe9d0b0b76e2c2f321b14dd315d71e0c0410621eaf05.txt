(with-open-file (in-file "input.txt" :direction :input)
  (with-open-file (out-file "output.txt" :direction :output)
    (format out-file "Input string:~%~%" (read-line in-file))
    (format out-file "Input integer:~%~%" (read in-file)))
    (format out-file "Input float:~%~%" (read in-file))
    (format out-file "Input boolean:~%~%" (read in-file))))
