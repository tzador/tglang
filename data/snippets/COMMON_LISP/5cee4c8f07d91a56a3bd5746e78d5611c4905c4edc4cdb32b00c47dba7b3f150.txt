(with-open-file (file "data.txt" :direction :input :if-does-not-exist :error)
  (let ((input-strings (make-array 10 :element-type 'string :initial-element ""))
        (matching-strings (map 'list #'(lambda (x) (if (search "lion" x) x ""))
                              (read-line file nil)))
    (loop for i in 0 to 9
          do (setf (aref input-strings i) (nth i matching-strings)))
    (dolist (string matching-strings)
      (format t "Found matching string: ~a" string))
    (let ((count (count-if #'(lambda (x) (eql 5 (length x))) matching-strings)))
      (format t "Number of strings with length 5: ~a" count))))
