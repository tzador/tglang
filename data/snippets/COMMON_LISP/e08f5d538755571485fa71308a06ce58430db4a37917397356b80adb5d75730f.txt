(defun quicksort (list)
   (if (null list)
       nil
       (flet ((< (a b) (< a b)))
            (let* ((pivot (first list)) 
                   (smaller (remove-if-not #'< list :key #'identity)) 
                   (larger (remove-if #'< list :key #'identity)))
                (append (quicksort smaller) 
                        (list pivot) 
                        (quicksort larger))))))
