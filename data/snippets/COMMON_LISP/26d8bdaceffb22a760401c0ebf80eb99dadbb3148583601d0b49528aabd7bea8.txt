(defmethod binary-search ((array vector) (searched-item t)
                           &key (start 0) (end (length array)))
  "Uses binary search to find the index of the searched item in a sorted vector."
  (if (>= end start)
      (let* ((mid (floor (+ start end) 2))
             (mid-item (aref array mid)))
        (cond ((= searched-item mid-item) mid)
              ((< searched-item mid-item)
               (binary-search array searched-item :start start :end (1- mid)))
              ((> searched-item mid-item)
               (binary-search array searched-item :start (1+ mid) :end end))))
      (format nil "Item ~A was not found in the vector." searched-item)))
