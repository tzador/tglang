(defparameter *cases* 
  `(("apple" ,(+ 4 5) "red") 
    ("banana" ,(* 2 3) "yellow") 
    ("orange" ,(- 6 3) "orange")))
(format t "Fruit Cases:~%")
(dotimes (i (length *cases*))
  (let ((fruit (first (elt *cases* i)))
        (quantity (second (elt *cases* i)))
        (color (third (elt *cases* i))))
    (format t "- ~A: ~D ~A~%" fruit quantity color)))
