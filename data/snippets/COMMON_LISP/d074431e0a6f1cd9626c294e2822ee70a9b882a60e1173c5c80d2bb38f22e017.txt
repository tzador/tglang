;; Define a function that multiplies its argument by 2
(defun multiply-by-two (x) ; Define a named function with one argument
  (* x 2)) ; Multiply the argument by 2 and return the result
(multiply-by-two 10) ; Call the function with the argument 10 and return the result (20)
(multiply-by-two 5) ; Call the function with the argument 5 and return the result (10)
