(defun bubble-sort (lst)
    "This function implements the bubble sort algorithm on a list of numbers."
    (let ((n (length lst)))
        (do ((i 0 (+ i 1)))
            ((>= i (- n 1)))
            (do ((j 0 (+ j 1)))
                ((>= j (- n i 1)))
                (if (> (nth j lst) (nth (+ j 1) lst))
                    (rotatef (nth j lst) (nth (+ j 1) lst))))))
    lst)
