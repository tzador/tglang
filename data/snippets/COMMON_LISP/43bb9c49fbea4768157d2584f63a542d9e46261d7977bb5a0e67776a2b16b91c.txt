(let* ((numbers '(1 2 3 4 5))
       (doubled-numbers (mapcar #'(lambda (num) (* num 2)) numbers))
       (summed-numbers (reduce #'+ doubled-numbers)))
  (format t "The original numbers were: ~A~%" numbers)
  (format t "The doubled numbers are: ~A~%" doubled-numbers)
  (format t "The sum of the doubled numbers is: ~A" summed-numbers))
