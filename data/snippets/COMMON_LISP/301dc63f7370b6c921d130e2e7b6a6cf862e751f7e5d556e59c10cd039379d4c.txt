(defun bubble-sort (array)
  "Sorts the elements of an array in ascending order using the bubble sort algorithm."
  (let ((n (length array)))
    (do ((i 0 (+ i 1)))
        ((>= i (- n 1)))
      (do ((j 0 (+ j 1)))
          ((>= j (- n i 1)))
          (when (> (aref array j) (aref array (+ j 1)))
            (rotatef (aref array j) (aref array (+ j 1)))))))
  array)
