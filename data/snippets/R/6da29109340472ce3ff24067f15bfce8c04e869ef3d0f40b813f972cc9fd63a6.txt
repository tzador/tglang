# This code snippet calculates the mean and standard deviation of a vector
my_vector <- c(10, 12, 14, 16, 18, 20) # create a vector with 6 elements
my_mean <- mean(my_vector) # calculate the mean of the vector using the built-in mean() function
my_sd <- sd(my_vector) # calculate the standard deviation of the vector using the built-in sd() function
