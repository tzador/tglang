for (i in 1:10) {  # Using a for loop to iterate through numbers 1 to 10
  if (i %% 2 == 0) {  # Checking if the number is even using modulus operator
    print("The number is even.")  # If it is even, print this statement
  } else {  # Otherwise
    print("The number is odd.")  #Print this statement
  }
}
