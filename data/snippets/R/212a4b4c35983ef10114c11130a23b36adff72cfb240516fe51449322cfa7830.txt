# This is a function that calculates the mean of two numbers

``mean_calculation <- function(num1, num2) {
  # Assign zero to a variable
  total <- 0

  # Calculate the sum of the two numbers
  total <- num1 + num2

  # Divide the sum by 2 to get the mean
  mean <- total / 2

  # Print the result
  print(mean)

  # Return the mean as output
  return(mean)
}
