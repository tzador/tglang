# This code snippet calculates the mean and standard deviation of a numeric vector

# Create a sample vector
my_vector <- c(4, 5, 6, 1, 3, 2)

# Calculate the mean
mean <- mean(my_vector)

# Output the mean
cat("The mean is", mean)

# Calculate the standard deviation
sd <- sd(my_vector)

# Output the standard deviation
cat("The standard deviation is", sd)
