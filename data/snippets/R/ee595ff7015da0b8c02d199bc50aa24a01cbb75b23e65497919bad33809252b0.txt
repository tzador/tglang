# This code snippet uses a function to return the sum of two numbers
# Define a function with two arguments
sum_func <- function(num1, num2) {
  # Use the return keyword to return the sum of the two numbers
  return (num1 + num2)
}
# Call the function with two integers as arguments and store the result in a variable
result <- sum_func(10, 25)
# Print the result
print(result)
