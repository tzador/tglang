: FACTORIAL ( n -- n! )
  1 SWAP         ( n 1 )
  BEGIN
    DUP 1 =      ( n n 1= )
    IF           ( n )
      DROP      ( )
      EXIT      ( )
    ELSE         ( n )
      DUP 1       ( n n 1 )
      -          ( n n-1 )
      OVER       ( n n-1 n )
      *          ( n (n-1)*n )
    THEN         ( n )
  AGAIN          ( n )
  DROP           ( )
;
