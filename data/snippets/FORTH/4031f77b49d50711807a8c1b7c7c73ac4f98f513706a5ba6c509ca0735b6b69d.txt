: MEMCPY ( dest src size -- ) ( copied from Gforth)
  >r					(save size)
  >r					( save src)
 	>r					(save dest)
  begin					( start loop)
    dup					( duplicate size)
    0=					( check if size is zero)
    if					( if it is zero)
      2drop				( drop everything and exit)
    else
      begin				( start loop)
        r@				( load dest)
        c@				( get dest char)
        r@ 				( load src)
        c@				( get src char)
        swap			( swap char places)
        c!				( store char in dest)
        1+ 				( increment dest)
        r@ 				( load src)
        1+ 				( increment src)
        dup				( duplicate size)
        1- 				( decrement size)
      0=					( check if size is zero)
      until 				( if it is, exit loop)
    then 					(if it is not, continue loop)
  r>						( restore dest)
  drop 					( drop size)
  r> 						( restore src)
  drop 					( drop src)
;							( end of word)
