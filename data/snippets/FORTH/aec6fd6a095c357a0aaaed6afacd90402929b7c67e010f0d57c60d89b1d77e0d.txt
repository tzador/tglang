: fibonacci			( n -- nth Fibonacci number )
	1 over <=			( check if n is 1 or 0 )
	if				( if so, return n )
		drop			( remove extra 1 from stack )
	else				( otherwise )
		1				( start with first two numbers in sequence )
		1 swap			( swap top two numbers )
		begin			( start loop )
			dup			( duplicate top of stack )
			rot			( rotate top three numbers )
			+			( add top two numbers )
			rot			( rotate top three numbers )
			dup			( duplicate top of stack )
			1 swap -		( subtract 1 from top number )
		2 over >		( check if reached nth number )
		until			( repeat until nth number )
		nip				( remove extra number from stack )
	then				( end if )
;					( end definition )
