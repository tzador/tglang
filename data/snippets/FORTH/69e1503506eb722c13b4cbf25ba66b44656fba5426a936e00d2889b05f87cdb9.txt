: FACTORIAL    ( n -- n! )
    1           ( n -- n! 1 )
    ROT         ( n -- 1 n n! )
    0           ( n -- 1 n n! 0 )
    DO          ( Start a DO LOOP with the values on the stack)
        OVER    ( Get the current factorial value on top of the loop index)
        1+      ( Add 1 to the loop index)
        SWAP    ( Swap the loop index and factorial value)
        *       ( Multiply the loop index with the factorial value)
    LOOP        ( Go back to the beginning of the loop)
    DROP        ( Drop the loop index, leaving only the factorial value)
;               ( End of word definition)
