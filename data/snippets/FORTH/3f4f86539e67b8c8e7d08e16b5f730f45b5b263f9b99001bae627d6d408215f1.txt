: DOUBLE-ARRAY ( n -- arr )
  2 * array (create array with double the size)
  swap (put the value n on top of the array)
  0 (initialize the loop counter)
  do (start of loop)
    dup (duplicate the top value on the stack)
    2* (multiply it by 2)
    i (push the loop counter value onto the stack)
    2! (store the multiplied value at the index in the array)
    1 (increment the loop counter)
    +loop (end of loop, jumps back to do)
  loop (end of loop)
  nip (remove the loop counter from the stack)
;
