: factorial ( n -- n! ) 
  dup 0 = if          (check if input is equal to 0)
    drop 1            (if yes, drop input and push 1 onto stack)
  else                (if no, continue with calculation)
    dup 1 -           (push input - 1 onto stack)
    factorial        (recursively call factorial on input - 1)
    * swap           (multiply result of recursive call with input)
  then ;              (end if statement and end definition)
