(define (quick-sort list)
  (if (null? list)
  '()
  (let ((pivot (car list)))
    (let ((smaller-elements (filter (lambda (x)
      (<= x pivot))
      (cdr list))))
        (let ((larger-elements (filter (lambda (x)
          (> x pivot))
          (cdr list))))
          (append (quick-sort smaller-elements) (cons pivot (quick-sort larger-elements))))))))
