;; Define a function to check if a given number is prime
(define (is-prime? n)
  (define (is-divisible? div)
    (and (not (= 0 (modulo n div)))
         (not (= n div))))
  (define (try-dividers current)
    (cond ((> current (sqrt n)) #t)
          ((is-divisible? current) #f)
          (else (try-dividers (+ current 1)))))
  (try-dividers 2))
