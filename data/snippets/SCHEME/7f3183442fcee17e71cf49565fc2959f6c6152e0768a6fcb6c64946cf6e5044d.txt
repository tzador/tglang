(define (quicksort lst)
   (cond ((null? lst) '())
         ((null? (cdr lst)) lst)
         (else (let ((pivot (car lst))
                     (less-than (lambda (x) (< x pivot)))
                     (greater-than (lambda (x) (> x pivot))))
                 (append (quicksort (filter less-than lst))     (list pivot)
                         (quicksort (filter greater-than lst)))))))
