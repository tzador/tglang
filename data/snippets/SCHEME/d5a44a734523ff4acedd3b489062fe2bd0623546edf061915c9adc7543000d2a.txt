;; This code performs a binary search on a given list
(define (binary-search ls x)
    (let loop ((ls ls) 
        (low 0) 
        (high (- (length ls) 1)))
        (if (<= low high)
            (let ((mid (floor (/ (+ low high) 2))))
                (cond ((< (nth mid ls) x) (loop ls (+ mid 1) high))
                      ((> (nth mid ls) x) (loop ls low (- mid 1)))
                      (else mid)))
            #f)))
