class Animal
  # Variables
  @[Animal] var name : String
  @[Animal] var species : String
  
  # Methods
  # Constructor method
  def initialize(name : String, species : String)
    @name = name
    @species = species
  end
  
  # Getter method
  def get_name
    @name
  end
  
  # Setter method
  def set_name(name : String)
    @name = name
  end
end

# Inheritance

  def self.class_method
    puts "I am a dog, I'm from the 'Dog' class!"

    "I'm fetching #{toy}!"
  end
end
