# Calculate the average of a given set of numbers
BEGIN {
  # Initialize variables
  sum = 0; 
  count = 0; 
}
{
  # Iterate through each number
  for (i = 1; i <= NF; i++) { 
    # Add to the sum
    sum += $i; 
    # Increase the count
    count++; 
  } 
}
# Calculate and print the average
END {
  # Divide the sum by the count
  avg = sum / count; 
  # Print the result
  print "The average is " avg; 
}
