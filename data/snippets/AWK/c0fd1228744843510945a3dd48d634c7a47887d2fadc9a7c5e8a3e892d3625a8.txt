# The following code snippet uses the AWK programming language to perform calculations and print the results

# Define variables for input values
BEGIN {var1 = 5; var2 = 10;}

# Calculate the sum of the two variables
sum = var1 + var2;

# Calculate the product of the two variables
product = var1 * var2;

# Print the results with annotations
print "The sum of " var1 " and " var2 " is: " sum "\n" # Outputs: The sum of 5 and 10 is: 15
print "The product of " var1 " and " var2 " is: " product # Outputs: The product of 5 and 10 is: 50
