server { #[This line starts a new server block]
    listen 80; #[Specifies the port to listen on]
    server_name example.com; #[Specifies the domain name for the server]
    location / { #[This line starts a new location block]
        root /var/www/example.com; #[Specifies the root directory for the server]
    } #[Closing curly brace to end the location block]
    location /images/ { #[This location block handles requests for the /images/ location]
        try_files $uri.jpg $uri.jpeg $uri.png @image_proxy; #[Checks for existing files with these extensions and if not found, goes to the @image_proxy block]
        return 404; #[Returns a 404 error for any requests to this location]
    } #[Closing curly brace to end the location block]
    location @image_proxy { #[This location block handles requests for the @image_proxy named location]
        rewrite ^/images/(.*).(jpg|jpeg|png)$ /image-proxy/$1.$2; #[Rewrites the request to the image-proxy directory]
        proxy_pass http://127.0.0.1:8001; #[Forwards the request to the specified proxy server]
    } #[Closing curly brace to end the location block]
} #[Closing curly brace to end the server block]
