Set i /1*4/, j /i+1*4/;
Parameter a(i,j) 'matrix a';
a(i,j) = uniform(1,10);
Display a;
Equations e1(i), e2(j), e3;
Variables x(i,j), y;
e1(i).. sum(j, a(i,j)*x(i,j)) =l= 100;
e2(j).. sum(i, x(i,j)) =g= 10;
e3.. y =e= sum(i, sum(j, x(i,j)));
Model m /all/;
Solve m using lp maximizing y;
Display x.l, y.l;
