set i 'index' /i1*i5/;
parameter a(i) 'arbitrary parameter' /i1 100, i2 300, i3 500, i4 700, i5 900/;
variable x(i) 'decision variable';
equation obj 'objective function'.. sum(i, a(i)*x(i)) =e= 1000;
equation c1 'constraint1'.. sum(i, x(i)) =e= 1;
equation c2 'constraint2'.. sum(i, sqr(x(i)) - sqr(x(i-1))) =l= 0;
equation c3 'constraint3'.. x(i) =g= 0;
model test /all/;
option optcr=0.001;
solve test using LP maximizing obj;
display x.l;
