set I "set of all origin nodes"/1*4/;
set J "set of all destination nodes"/5*7/;
parameters c(i,j) "cost for each transportation arc";
c(i,j) = 10*i + j;
variables x(i,j) "flow of commodities on each arc";
equations supply(i) "flow balance at each origin node", demand(j) "demand balance at each destination node", obj "objective function";
supply(i).. sum(j, x(i,j)) =e= 10;
demand(j).. sum(i, x(i,j)) =e= 8;
obj.. sum(i,j, c(i,j)*x(i,j)) =e= z;
model transport /all/;
solve transport using lp minimizing z;
display x.l, x.marginal;
