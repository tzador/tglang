set products 'set of products' /A, B, C/;
alias (products, p);
parameter demand(p) 'demand for each product';
demand(p) = 100 * sin(p / 10);
parameter unit_cost(p) 'unit cost for each product';
unit_cost(p) = min(50 * p, 200);
variable x(p) 'quantity of each product to produce';
equation obj 'objective function';
obj.. sum(p, unit_cost(p) * x(p)) =e= sum(p, demand(p));
equation constraint 'production constraint';
constraint.. sum(p, x(p)) =e= 100;
model production /all/;
solve production maximizing obj using lp;
