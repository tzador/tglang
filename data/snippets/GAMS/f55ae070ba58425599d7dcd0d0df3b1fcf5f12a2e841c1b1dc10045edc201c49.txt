set i 'todo items'/1,2,3/;
set j 'categories'/low,medium,high/;
parameter c(i) 'completion time'/1 20, 2 30, 3 60/;
parameter q(j) 'effort required' /low 20, medium 30, high 60/;
variable x(i,j) 'completion status';
equation total_time;
total_time.. sum((i,j), x(i,j)*c(i)) =g= 100;
equation total_effort;
total_effort.. sum((i,j), x(i,j)*q(j)) =l= 50;
model todo 'to-do list' / all /;
solve todo using lp maximizing total_time;
display x.l, c.l, q.l;
