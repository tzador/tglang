Set i 'Index of Rows' /1, 2, 3, 4/;
Set j 'Index of Columns' /1, 2, 3/;
Alias (i,di), (j,dj);
Parameter a(i,j) 'Dummy Data';
a(i,j) = uniform(1,10);
Variable x(i,j) 'Optimization Variables';
Equation obj 'Objective Function';
obj.. sum((i,j),x(i,j)*a(i,j)) =e= 10;
Equation con1(i), con2(j) 'Constraints';
con1(i).. sum(j,x(i,j)) =e= di;
con2(j).. sum(i,x(i,j)) =e= dj;
Model test 'Sample Model' /all/;
Solve test using MIP minimizing obj;
Display x.l, obj.l, a;
