(* Code to generate 10 random integers and find the sum of their squares *)
numbers = RandomInteger[{0, 100}, 10]; (* Generates 10 random integers between 0 and 100 *)
squared = Map[#^2 &, numbers]; (* Maps the function 'square' to each of the numbers *)
sum = Total@squared; (* Finds the total sum of all squared numbers *)
Print["The sum of squares of the 10 random integers is ", sum]; (* Prints the output with appropriate message *)
