(* Function for finding eigenvalues and eigenvectors of a matrix *)
findEigen[matrixList_] :=                (* Define a function with parameter matrixList *)
Module[{                                (* Start a Module, a localized variable environment *)
eigen = Eigenvalues[matrixList],        (* Use Eigenvalues function to find eigenvalues *)
eigenv = Eigenvectors[matrixList]},      (* Use Eigenvectors function to find eigenvectors *)
{                                       (* Start a list of results *)
"Eigenvalues:" -> eigen,                (* Label and output eigenvalues *)
"Eigenvectors:" -> eigenv               (* Label and output eigenvectors *)
}]                                      (* End the list and the Module *)
