(* create a function that takes in a list of numbers and returns their sum *)
(* @param list a list of numbers *)
(* @return the sum of all the numbers in the list *)
TotalSum[list_] := Total[list] (* calculate the total of the list using the Total function *)
(* example usage *)
TotalSum[{1, 2, 3, 4, 5}] (* returns 15 as the sum of the numbers *)
