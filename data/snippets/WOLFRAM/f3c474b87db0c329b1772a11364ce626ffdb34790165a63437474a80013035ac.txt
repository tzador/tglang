(* This is a simple program in Wolfram Language that calculates the sum of two numbers *)

(* Define variables for input *)
number1 = 5;
number2 = 10;

(* Calculate the sum of the two numbers *)
sum = number1 + number2;

(* Output the result *)
Print["The sum of ", number1, " and ", number2, " is ", sum];

(* Output: The sum of 5 and 10 is 15 *)
