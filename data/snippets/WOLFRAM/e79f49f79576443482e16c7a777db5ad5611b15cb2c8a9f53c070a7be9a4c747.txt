(*This code snippet uses the Wolfram Language to calculate the cosine of an angle in degrees)

(*Declare a variable called angle and assign it a value of 45 degrees*)
angle = 45 Degree;

(*Use the Cosine function to find the cosine of the angle*)
cosine_angle = Cos[angle];

(*Print the result*)
Print["The cosine of", angle, "degrees is", cosine_angle];

(*Output: The cosine of 45 degrees is 0.707107*)
