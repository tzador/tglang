(* Define a function that calculates the factorial of a number *)
factorial[n_Integer] := Module[{result = 1}, (* initialize the result variable *)
    For[i = 1, i <= n, i++, (* loop through the numbers 1 to n *)
        result = result * i; (* multiply the current result by the current number *)
    ]; (* end of the for loop *)
    result (* return the final result *)
]; (* end of the function definition *)

(* Calculate the factorial of the number 5 *)
factorial[5] (* output: 120 *)
