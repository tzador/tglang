(* This is a code snippet that uses a built-in function in Wolfram called Expand *)
(* The Expand function takes in a polynomial expression and expands it into its constituent terms *)

(* Define a polynomial expression using Wolfram's polynomial syntax *)
polynomial = 3 x^2 - 5 x + 2

(* Use the Expand function to expand the polynomial *)
expanded_polynomial = Expand[polynomial]

(* The output will be equivalent to 3 x^2 - 5 x + 2 *)
