(* This code snippet solves the quadratic equation ax^2 + bx + c = 0 
and returns the two possible solutions*)
Clear["Global`*"] (* clears any previous variables *)
a = Input["Enter the coefficient of x^2: "]; (* prompts user to input a value for a*)
b = Input["Enter the coefficient of x: "]; (* prompts user to input a value for b*)
c = Input["Enter the constant term: "]; (* prompts user to input a value for c*)
discriminant = b^2 - 4*a*c; (* calculates the discriminant *)
If[discriminant > 0, (* checks if discriminant is positive *)
    sol1 = (-b + Sqrt[discriminant])/(2*a); (* calculates the first solution *)
    sol2 = (-b - Sqrt[discriminant])/(2*a); (* calculates the second solution *)
    print["The solutions to the quadratic equation are:", sol1, "and", sol2], (* prints the solutions *)
discriminant == 0, (* checks if discriminant is zero *)
    sol = -b/(2*a); (* calculates the only solution *)
    print["The only solution to the quadratic equation is:", sol], (* prints the solution *)
True, (* checks if discriminant is negative *)
    print["The quadratic equation has no real solutions."]]; (* prints error message *)
