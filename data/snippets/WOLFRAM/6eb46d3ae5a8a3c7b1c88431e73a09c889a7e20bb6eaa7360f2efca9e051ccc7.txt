(*This code defines a function that calculates the volume of a sphere*)
sphereVolume[r_]:= (4/3)*Pi*r^3; (*Calculates the volume of a sphere with radius r*)
Print["Enter the radius of the sphere:"]; (*Asks for user input*)
r = Input[]; (*Sets user input as value for r*)
Print["The volume of the sphere with radius ", r, " is:"]; (*Prints output message*)
sphereVolume[r] (*Calls the function and passes in the user input*)
