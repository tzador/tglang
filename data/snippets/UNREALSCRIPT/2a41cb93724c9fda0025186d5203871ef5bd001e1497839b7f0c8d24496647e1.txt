/** 
 * This is a complex function that calculates the sum of two numbers and 
 * returns their product.
 *
 * @param int A The first number.
 * @param int B The second number.
 * @returns int The product of the two numbers.
 */
function int GetProduct(int A, int B)
{
  local int Sum;
  Sum = A + B; // calculating the sum
  return A * B; // returning the product
}
