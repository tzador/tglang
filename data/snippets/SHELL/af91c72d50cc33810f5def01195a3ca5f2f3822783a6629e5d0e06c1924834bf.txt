#!/bin/sh
# This is a shell script
echo "Hello World!" # Print the greeting
# Declare variables
NAME="John"
AGE=20
# Print the user's name
echo "My name is $NAME."
# Print the user's age
echo "I am $AGE years old."
# Calculate the user's age in dog years
DOG_AGE=$(($AGE * 7))
# Print the user's age in dog years
echo "In dog years, I am $DOG_AGE years old."
