h1(#header)[This is a header]  //This creates a header with the id 'header'
p{color:green}. This is a paragraph with green text  //This creates a paragraph with green text
bc.. Your code goes here  //This creates a code block
ul  //This starts a list
  * This is the first item  //This creates a list item
  * This is the second item  //This creates another list item
h3(#subheader){color:red}. This is a subheader with red text  //This creates a subheader with the id 'subheader' and red text
table  //This starts a table
|_. Header 1 |_. Header 2 |  //This creates table headers
| This is a row | This is another row |  //This creates table rows
| Another row | Yet another row |  //This creates more table rows
