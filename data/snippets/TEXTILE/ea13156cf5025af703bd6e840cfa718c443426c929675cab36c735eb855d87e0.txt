# Header 1                                       // Hash indicates HTML header 1
{
  *Some bold text.*                               // Asterisk indicates bold text
  __Some underlined text.__                        // Double underscore indicates underlined text
  - Some bulleted list item 1                      // Dash indicates bulleted list item
  - Some bulleted list item 2 
  - Some bulleted list item 3
  !Some important note.                            // Exclamation point indicates a note
  h3. Header 3                                     // HTML header 3
  |Table Heading 1 |Table Heading 2 |              // Vertical bar indicates table row 
  |Table Cell 1    |Table Cell 2    |
  |Table Cell 3    |Table Cell 4    |
  bq. Block quote that spans multiple lines        // bq. Indicates a block quote
  *Some more bold text.* 
  h4. Header 4                                     // HTML header 4
  -# Ordered list item 1                           // Dash followed by hash for ordered list 
  -# Ordered list item 2                           // The hash indicates the number of the item
  -# Ordered list item 3
  h5. Header 5 
  "Link text":http://example.com                   // Double quotes to indicate link text 
}
