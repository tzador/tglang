Option Explicit

'This is a VBSCRIPT code snippet that calculates the circumference of a circle given its radius

'Initialize variables
Dim radius, circumference, PI

'Assign values to variables
radius = 5
PI = 3.14

'Calculate circumference using formula: circumference = 2 * PI * radius
circumference = 2 * PI * radius

'Print result
MsgBox "The circumference of a circle with radius " & radius & " is " & circumference

'End of code
End
