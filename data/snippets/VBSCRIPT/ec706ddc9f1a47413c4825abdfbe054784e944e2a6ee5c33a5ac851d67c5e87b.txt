' Declare variables 
Dim num1, num2, sum 

' Initialize values 
num1 = 5 
num2 = 10 

' Perform addition 
sum = num1 + num2 

' Print result 
MsgBox "The sum of " & num1 & " and " & num2 & " is " & sum ' This is a comment 

' Check if sum is greater than 15 
If sum > 15 Then 
    ' Multiply sum by 2 and print result 
    sum = sum * 2 
    MsgBox "The new sum is " & sum ' This is another comment 
End If
