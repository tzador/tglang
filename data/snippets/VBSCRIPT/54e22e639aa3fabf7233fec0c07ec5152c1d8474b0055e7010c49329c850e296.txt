If num > 0 Then ' Checks if the variable "num" is greater than 0
    MsgBox "Positive" ' If true, displays a message "Positive"
ElseIf num < 0 Then ' Checks if the variable "num" is less than 0
    MsgBox "Negative" ' If true, displays a message "Negative"
Else ' If both conditions are false
    MsgBox "Zero" ' Displays a message "Zero"
End If ' Ends the conditional statement
