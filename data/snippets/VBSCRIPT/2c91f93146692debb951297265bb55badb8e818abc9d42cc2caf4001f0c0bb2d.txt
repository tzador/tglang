' This code snippet performs a simple calculation of the sum of two numbers
' and displays the result in a message box

' Declare two variables to store the numbers for calculation
Dim num1
Dim num2

' Assign values to the variables
num1 = 5
num2 = 10

' Calculate the sum of the two numbers
Dim sum
sum = num1 + num2

' Display the result in a message box
MsgBox "The sum of " & num1 & " and " & num2 & " is " & sum
