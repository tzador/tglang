' This is a VBScript code snippet that demonstrates a simple math function
' It calculates the area of a circle given the radius
'
' Declaring variables
Dim radius, pi
' Assigning values to variables
radius = 5
pi = 3.14
' Calculating the area of a circle using the formula A = pi * r * r
Dim area
area = pi * radius * radius
' Printing the result
MsgBox "The area of the circle with radius " & radius & " is " & area & " square units."
