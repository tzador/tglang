' This code snippet calculates the sum of two numbers and displays the result

Dim num1, num2, sum   ' declaring variables

num1 = 5   ' assigning value to num1
num2 = 8   ' assigning value to num2

sum = num1 + num2   ' calculating the sum

MsgBox "The sum of " & num1 & " and " & num2 & " is " & sum   ' displaying the result in a message box

' Output: The sum of 5 and 8 is 13
