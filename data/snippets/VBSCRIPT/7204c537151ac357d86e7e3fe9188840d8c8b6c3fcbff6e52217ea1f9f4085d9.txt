' This is a comment in VBScript
Sub Calculator()

    ' Declare a variable to hold the first number
    Dim num1

    ' Declare a variable to hold the second number
    Dim num2

    ' Assign values to the variables
    num1 = 10
    num2 = 5

    ' Perform addition operation and assign result to a new variable
    Dim sum
    sum = num1 + num2

    ' Print the result to the console
    MsgBox "The result of addition is: " & sum

    ' Perform subtraction operation and assign result to a new variable
    Dim diff
    diff = num1 - num2

    ' Print the result to the console
    MsgBox "The result of subtraction is: " & diff

    ' Perform multiplication operation and assign result to a new variable
    Dim prod
    prod = num1 * num2

    ' Print the result to the console
    MsgBox "The result of multiplication is: " & prod

    ' Perform division operation and assign result to a new variable
    Dim quot
    quot = num1 / num2

    ' Print the result to the console
    MsgBox "The result of division is: " & quot

End Sub
