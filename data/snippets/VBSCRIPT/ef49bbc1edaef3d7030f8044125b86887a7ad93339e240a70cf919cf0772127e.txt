'''
' This code snippet shows how to calculate the area of a triangle using user input of the base and height
' 

' Prompt user for base length and assign to variable
base = InputBox("Enter base length of triangle:")

' Prompt user for height and assign to variable
height = InputBox("Enter height of triangle:")

' Calculate area using the formula area = (1/2) * base * height
area = (1/2) * base * height

' Display the calculated area to the user
MsgBox "The area of the triangle is: " & area
'''
