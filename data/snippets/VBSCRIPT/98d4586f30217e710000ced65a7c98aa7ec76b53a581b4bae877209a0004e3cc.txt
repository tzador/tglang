' This program calculates the area of a rectangle and displays the result in a message box

' Define the variables for the length and width of the rectangle
Dim length, width 

' Assign values to the length and width variables
length = 5 
width = 10 

' Calculate the area of the rectangle using the formula length * width
Dim area
area = length * width

' Display the result in a message box with appropriate label
MsgBox "The area of the rectangle is: " & area & " square units."
