'''' Description: This code snippet is a function that takes in two parameters, num1 and num2, and returns the sum of the two numbers.
'''' Usage: sum = MySum(num1, num2)
'''' Parameters:
'''' - num1: the first number to be added
'''' - num2: the second number to be added
Function MySum(num1, num2)
	' Declare and initialize a variable for the sum
	Dim sum
	' Add the two numbers and assign the sum to the variable
	sum = num1 + num2
	' Return the sum
	MySum = sum
End Function
