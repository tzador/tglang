' Declare a variable to store the first number
Dim num1
' Set the value of num1 to 10
num1 = 10

' Declare a variable to store the second number
Dim num2
' Set the value of num2 to 5
num2 = 5

' Declare a variable to store the result
Dim result
' Perform addition operation and store the result in 'result' variable
result = num1 + num2

' Print the result to the console
WScript.Echo "The result of addition is: " & result
