' Set up variables for calculation
Dim num1, num2, sum, product, difference

' Set initial values for num1 and num2
num1 = 5
num2 = 10

' Calculate sum, product, and difference
sum = num1 + num2 ' Addition
product = num1 * num2 ' Multiplication
difference = num1 - num2 ' Subtraction

' Print the results
MsgBox "The sum of " & num1 & " and " & num2 & " is " & sum
MsgBox "The product of " & num1 & " and " & num2 & " is " & product
MsgBox "The difference between " & num1 & " and " & num2 & " is " & difference
