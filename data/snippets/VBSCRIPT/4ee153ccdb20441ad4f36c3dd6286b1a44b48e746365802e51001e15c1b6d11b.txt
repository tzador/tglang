' This is a vbScript code snippet that calculates the sum of two numbers and prints the result

Dim num1, num2, sum

' defining variables to store input numbers and result of summation

num1 = InputBox("Enter first number:")
num2 = InputBox("Enter second number:")

' taking input numbers from the user using InputBox and storing them in variables

sum = num1 + num2

' calculating sum of the two numbers

WScript.Echo "The sum of " & num1 & " and " & num2 & " is " & sum

' printing the result using WScript.Echo and string concatenation
