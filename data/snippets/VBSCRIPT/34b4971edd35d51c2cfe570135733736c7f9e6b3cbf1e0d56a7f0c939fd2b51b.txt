' This code calculates and prints the average of three numbers
' Create variables for the numbers
Dim num1, num2, num3
' Assign values to the variables
num1 = 5
num2 = 10
num3 = 15
' Calculate the average
average = (num1 + num2 + num3) / 3
' Print the result
MsgBox "The average of " & num1 & ", " & num2 & ", and " & num3 & " is " & average
