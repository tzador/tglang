' This code snippet calculates the perimeter and area of a rectangle
' Define variables for length and width
Dim length, width
' Set length and width values
length = 10
width = 5
' Calculate perimeter
perimeter = 2 * (length + width)
' Calculate area
area = length * width
' Display results
MsgBox "Perimeter: " & perimeter & vbCrLf & "Area : " & area
