' This code snippet calculates the average of three numbers
Option Explicit

' Define variables
Dim num1, num2, num3
Dim sum, average

' Set values for variables
num1 = 10
num2 = 15
num3 = 20

' Calculate the sum of the three numbers
sum = num1 + num2 + num3

' Calculate the average by dividing the sum by the number of numbers (3)
average = sum / 3

' Print the result
MsgBox "The average of " & num1 & ", " & num2 & ", and " & num3 & " is " & average
