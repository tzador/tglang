' This code snippet calculates the average of two numbers 
'
' Declaring two variables for input numbers
Dim num1, num2 
num1 = 10 ' Assigning value to first number 
num2 = 15 ' Assigning value to second number 

' Calculating the sum of the two numbers
Dim sum 
sum = num1 + num2 

' Calculating the average of the two numbers
Dim average 
average = sum / 2 

' Displaying the result to the user
MsgBox "The average of " & num1 & " and " & num2 & " is " & average
