' This code will calculate the average of 5 values and print it to the console

' Declare and initialize variables
Dim num1, num2, num3, num4, num5 ' variables to hold the 5 values
Dim sum, average ' variables to hold the sum and average of the 5 values
num1 = 7
num2 = 12
num3 = 20
num4 = 5
num5 = 30

' Calculate the sum of the 5 values
sum = num1 + num2 + num3 + num4 + num5

' Calculate the average of the 5 values
average = sum / 5

' Print the average to the console
WScript.Echo "The average of the 5 values is " & average
