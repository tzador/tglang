<!-- This code snippet is a function that calculates the average of three numbers -->
Function average(num1, num2, num3)
    ' Declare variables to hold the calculated sum and average
    Dim sum, avg
    ' Calculate the sum of the three numbers
    sum = num1 + num2 + num3
    ' Calculate the average
    avg = sum / 3
    ' Return the average
    average = avg
End Function
