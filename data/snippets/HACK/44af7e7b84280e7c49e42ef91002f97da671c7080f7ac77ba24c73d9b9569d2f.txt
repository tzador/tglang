### Initializing variables
variable x = 5
variable y = 10
variable z = 15

### Creating a function
function add(x, y) {
  return x + y;
}

### Running the function
variable result = add(x, y);

### Conditional statement
if (result > z) { // If the result is greater than our variable z
  print("Result is greater than z"); // Print this message
} else { // If the result is not greater than z
  print("Result is not greater than z"); // Print this message
}
