# Initializing variables
INIT x INT
INIT y INT
INIT result INT
# Assigning values to variables
ASSIGN x 10
ASSIGN y 5
# Performing calculations and storing result in 'result' variable
ADD x y result # Add x and y
SUB result 3 result # Subtract 3 from result
MULT result 2 result # Multiply result by 2
DIV result 2 result # Divide result by 2
# Printing result
PRINT result # Output: 24
