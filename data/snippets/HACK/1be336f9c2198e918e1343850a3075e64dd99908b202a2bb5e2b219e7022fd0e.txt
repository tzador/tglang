# Set up variables
int a = 5, b = 3;          //Declare and initialize variables
int sum, difference;       //Declare variables for results

# Perform mathematical operations
sum = a + b;               //Addition operation
difference = a - b;        //Subtraction operation

# Print results
print("The sum of " + a + " and " + b + " is " + sum);             //Print addition result
print("The difference between " + a + " and " + b + " is " + difference);   //Print subtraction result
