// This code snippet is a basic function in the programming language HACK.

// Declaring the function with parameters
function helloWorld(name, age) {
  let greeting = `Hello, ${name}! You are ${age} years old.`; // Creating a string using template literals
  console.log(greeting); // Logging the greeting to the console
  console.log("Thank you for using HACK!"); // Logging a thank you message
}

// Calling the function and passing in arguments
helloWorld("John", 25); // Output: Hello, John! You are 25 years old.
// Thank you for using HACK!
