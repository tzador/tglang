# This code snippet calculates the average of 5 numbers
# Input: n1, n2, n3, n4, n5 (float numbers)
# Output: average (float number)

# Initialize variables with input numbers
n1 = 10
n2 = 15
n3 = 25
n4 = 7
n5 = 19

# Calculate sum of all numbers
sum = n1 + n2 + n3 + n4 + n5

# Calculate the average by dividing sum by the number of inputs (in this case, 5)
average = sum / 5

# Print the average
print("The average of the numbers is: ", average)
