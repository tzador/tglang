// Assign a function to a variable
`let func = function() {
    // Create local variables
    let count = 0;
    let message = "This is a long and complex code snippet in HACK";
    
    // Add a for loop
    for(let i = 0; i <= 10; i++) {
        // Update local variables
        count += i;
        message += " " + i;
    }
    
    // Print the final count and message
    console.log(count);
    console.log(message);
}`;

// Call the function
`func();`
