let i = 0; // This is a variable declaration
while (i < 10) { //This is a while loop
    if (i % 2 == 0) { // This is an if statement
        print("Even number"); // Prints "Even number" if condition is true
    }
    else {
        print("Odd number"); // Prints "Odd number" if condition is false
    }
    i++; // Increments i by 1
}

const double PI = 3.14; // This is a constant declaration
function circleArea(radius) { // This is a function declaration
    return PI * radius * radius; // This is a return statement that calculates the area of a circle
}
let radius = 5; // Declares the radius variable
print("The area of the circle is: " + circleArea(radius)); // Prints the area of the circle with radius 5
