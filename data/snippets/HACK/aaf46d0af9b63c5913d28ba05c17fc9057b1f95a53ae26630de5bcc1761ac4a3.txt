## INPUT
int x = 5;
int y = 3;

## PROCESS
// The following code calculates the sum of x and y
int sum = x + y;

// The following loop prints out all numbers from 1 to 10
for (int i = 1; i <= 10; i++) {
    print(i);
}

// This if statement checks if the sum is greater than or equal to 10
if (sum >= 10) {
    print("The sum is greater than or equal to 10");
}

// This while loop calculates the factorial of x
int factorial = 1;
while (x > 1) {
    factorial *= x;
    x--;
}

## OUTPUT
// The output of the code will be:
// The sum is greater than or equal to 10
// and the factorial of 5 is 120
