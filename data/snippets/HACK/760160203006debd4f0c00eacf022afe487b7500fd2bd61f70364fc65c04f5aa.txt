// This code snippet calculates the sum of two numbers and prints the result.

// create variables to store the two numbers
let num1 = 5;
let num2 = 10;

// create a function to add the two numbers
function addNumbers(num1, num2) {
  // add the numbers and store the result in a variable
  let sum = num1 + num2;

  // print the result
  console.log(`The sum of ${num1} and ${num2} is ${sum}`);
}

// call the function
addNumbers(num1, num2);

// Output: The sum of 5 and 10 is 15
