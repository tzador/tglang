*** This is a sample class declaration ***
CLASS ZCL_SAMPLE_CLASS DEFINITION.
  PUBLIC SECTION.
    *** Constants ***
    CONSTANTS:
      C_ONE TYPE I VALUE 1,
      C_TWO TYPE I VALUE 2,
      C_THREE TYPE I VALUE 3.

    *** Private Attributes ***
    PRIVATE SECTION.
      DATA:
        PRIVATE_VAR1 TYPE I,
        PRIVATE_VAR2 TYPE I,
        PRIVATE_VAR3 TYPE I.

  PROTECTED SECTION.
    *** Protected Attributes ***
      PROTECTED_VAR1 TYPE I,
      PROTECTED_VAR2 TYPE I,
      PROTECTED_VAR3 TYPE I.

  METHODS:
    *** Private Methods ***
    PRIVATE SECTION.
      *** This method multiplies two numbers ***
      METHOD MULTIPLY_NUMBERS
        IMPORTING
          I_NUMBER1 TYPE I
          I_NUMBER2 TYPE I
        RETURNING
          VALUE(R_RESULT) TYPE I.
      PRIVATE_SECTION:
        R_RESULT = I_NUMBER1 * I_NUMBER2.
    ENDMETHOD.

  METHODS:
    *** Public Methods ***
    PUBLIC SECTION.
      *** This method adds two numbers ***
      METHOD ADD_NUMBERS
        IMPORTING
          I_NUMBER1 TYPE I
          I_NUMBER2 TYPE I
        RETURNING
          VALUE(R_RESULT) TYPE I.

      PUBLIC SECTION:
        R_RESULT = I_NUMBER1 + I_NUMBER2.

      METHOD:
        *** This method divides two numbers ***
        DIVIDE_NUMBERS
          IMPORTING
            I_NUMBER1 TYPE I
            I_NUMBER2 TYPE I
          RETURNING
            VALUE(R_RESULT) TYPE I.
        R_RESULT = I_NUMBER1 / I_NUMBER2.
    ENDMETHOD.
ENDCLASS.
