* Create a new custom transparent table
TABLES: ZEMPLOYEE.

* Define fields for the table
DATA: BEGIN OF ZEMPLOYEE,
      EMP_ID TYPE NUMC, " Employee ID
      EMP_NAME TYPE STRING, " Employee's name
      DEPARTMENT TYPE CHAR10, " Employee's department
      SALARY TYPE DECIMALS, " Employee's salary
      END OF ZEMPLOYEE.

* Add a unique key to the table
CONSTANTS: C_KEY TYPE C VALUE '1'.
ZEMPLOYEE-KEYID = C_KEY.

* Initialize the data
ZEMPLOYEE-EMP_ID = '001'.
ZEMPLOYEE-EMP_NAME = 'John Smith'.
ZEMPLOYEE-DEPARTMENT = 'Sales'.
ZEMPLOYEE-SALARY = '5000'.

* Insert the data into the table
INSERT ZEMPLOYEE.

* Display a success message
WRITE: 'Employee data successfully inserted.'.

* Define a selection screen
SELECTION-SCREEN BEGIN OF BLOCK B1 WITH FRAME TITLE TEXT-001.

* Add input fields for selection
PARAMETERS: P_EMP_ID TYPE ZEMPLOYEE-EMP_ID,
            P_EMP_NAME TYPE ZEMPLOYEE-EMP_NAME,
            P_DEPARTMENT TYPE ZEMPLOYEE-DEPARTMENT,
            P_SALARY TYPE ZEMPLOYEE-SALARY.

SELECTION-SCREEN END OF BLOCK B1.

* Define variables for selection
DATA: L_EMP_ID TYPE ZEMPLOYEE-EMP_ID,
      L_EMP_NAME TYPE ZEMPLOYEE-EMP_NAME,
      L_DEPARTMENT TYPE ZEMPLOYEE-DEPARTMENT,
      L_SALARY TYPE ZEMPLOYEE-SALARY.

* Validate user input and perform selection
IF P_EMP_ID IS NOT INITIAL.
  L_EMP_ID = P_EMP_ID.
ELSE.
  L_EMP_ID = ZEMPLOYEE-EMP_ID.
ENDIF.

IF P_EMP_NAME IS NOT INITIAL.
  L_EMP_NAME = P_EMP_NAME.
ELSE.
  L_EMP_NAME = ZEMPLOYEE-EMP_NAME.
ENDIF.

IF P_DEPARTMENT IS NOT INITIAL.
  L_DEPARTMENT = P_DEPARTMENT.
ELSE.
  L_DEPARTMENT = ZEMPLOYEE-DEPARTMENT.
ENDIF.

IF P_SALARY IS NOT INITIAL.
  L_SALARY = P_SALARY.
ELSE.
  L_SALARY = ZEMPLOYEE-SALARY.
ENDIF.

SELECT * FROM ZEMPLOYEE INTO TABLE @DATA(IT_EMPLOYEE)
       WHERE EMP_ID = L_EMP_ID
       AND EMP_NAME = L_EMP_NAME
       AND DEPARTMENT = L_DEPARTMENT
       AND SALARY = L_SALARY.

* Display the selected data in a table
FORMAT COLOR COL_BACKGROUND INTENSIFIED ON.
DO 20 TIMES.
  WRITE: / SY-VLINE, ZEMPLOYEE-EMP_ID, SY-VLINE, ZEMPLOYEE-EMP_NAME, SY-VLINE,
            ZEMPLOYEE-DEPARTMENT, SY-VLINE, ZEMPLOYEE-SALARY, SY-VLINE.
ENDDO.
FORMAT COLOR COL_BACKGROUND OFF.
DO 20 TIMES.
  WRITE: / SY-VLINE, IT_EMPLOYEE-EMP_ID, SY-VLINE, IT_EMPLOYEE-EMP_NAME, SY-VLINE,
            IT_EMPLOYEE-DEPARTMENT, SY-VLINE, IT_EMPLOYEE-SALARY, SY-VLINE.
ENDDO.
