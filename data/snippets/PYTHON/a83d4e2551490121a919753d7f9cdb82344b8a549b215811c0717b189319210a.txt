# Function to calculate the sum of two numbers
def add(num1, num2):
    """
    # This is a docstring that describes the purpose of the function
    # Parameters:
        num1 (int): First number
        num2 (int): Second number
        
    # Returns:
        sum (int): Sum of the two numbers
        
    # This is a multiline comment inside the function
    """
    # This is a single line comment inside the function
    sum = num1 + num2 # Adding the two numbers
    return sum # Returning the sum

# Example inputs
num1 = 5
num2 = 10

# Calling the add() function and printing the result
print(add(num1, num2)) # Output: 15
