# Initializing a dictionary
dict = {'Name': 'John', 'Age': 23, 'Occupation': 'Engineer'}

# Adding new key-value pair
dict['Address'] = '123 Street, City'

# Looping through the dictionary
for key, value in dict.items():
    print(key + ': ' + str(value))
    
# Output:
# Name: John
# Age: 23
# Occupation: Engineer
# Address: 123 Street, City
