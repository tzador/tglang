# This program calculates the average score of a student in 3 subjects

# Define 3 variables for the scores
math_score = 90
science_score = 85
english_score = 95

# Calculate the total score by adding all 3 scores together
total_score = math_score + science_score + english_score

# Calculate the average score by dividing the total score by 3
average_score = total_score / 3

# Print the average score as an output
print("The average score is:", average_score)
