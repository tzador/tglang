# This is a complex code snippet in python

# Create a list of numbers
numbers = [1, 2, 3, 4, 5]

# Define a function to multiply each number in the list by 2
def multiply_by_two(list):
    multiplied_numbers = []
    for num in list:
        multiplied_numbers.append(num * 2)
    return multiplied_numbers
    
# Call the function and store the result in a variable
result = multiply_by_two(numbers)

# Print out the result
print(result)
