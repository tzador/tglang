# Define a function for calculating the sum of two numbers
def calculate_sum(num1, num2):
    # Add the two numbers together
    total = num1 + num2
    # Print a message with the result
    print("The sum of", num1, "and", num2, "is", total)
# Call the function with two given numbers and print the result
calculate_sum(5, 2) # Output: The sum of 5 and 2 is 7
