--This script will calculate the sum of two numbers and output the result

set num1 to 5 --set the first number
set num2 to 10 --set the second number
set result to num1 + num2 --calculate the sum
display dialog "The sum of " & num1 & " and " & num2 & " is " & result --display the result in a dialog box
return result --return the result to the user
