--This script will calculate the total cost of apples based on the price per pound and the weight of apples.
set applePrice to 1.50 --Price per pound
set appleWeight to 5 --Weight of apples in pounds
set totalPrice to applePrice * appleWeight --Calculate total price by multiplying the price per pound and weight
display dialog "The total cost of apples is $" & totalPrice --Display the total cost in a dialog box
