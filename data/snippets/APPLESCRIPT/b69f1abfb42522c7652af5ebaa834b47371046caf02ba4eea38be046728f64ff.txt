(*This program will take a list of numbers and return the average of all the numbers*)
set numbers to {5, 7, 2, 9, 10} (*the list of numbers*)
set sum to 0 (*variable to store the sum of all numbers*)
set count to 0 (*variable to store the count of numbers*)
repeat with num in numbers (*loop through the list*)
	set sum to sum + num (*add the current number to the sum*)
	set count to count + 1 (*increment count*)
end repeat
set average to sum / count (*calculate the average*)
return "The average of the numbers is " & average (*print the average*)
