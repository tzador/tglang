-- This code snippet is a simple calculator that performs basic arithmetic operations

set num1 to 10 -- set the first number
set num2 to 5 -- set the second number

-- addition
set sum to num1 + num2
log "The sum of " & num1 & " and " & num2 & " is " & sum

-- subtraction
set difference to num1 - num2
log "The difference between " & num1 & " and " & num2 & " is " & difference

-- multiplication
set product to num1 * num2
log "The product of " & num1 & " and " & num2 & " is " & product

-- division
set quotient to num1 / num2
log "The quotient of " & num1 & " divided by " & num2 & " is " & quotient

-- modulus (returns remainder after division)
set remainder to num1 mod num2
log "The remainder after dividing " & num1 & " by " & num2 & " is " & remainder

-- power
set power to num1 ^ num2
log "The result of " & num1 & " to the power of " & num2 & " is " & power
