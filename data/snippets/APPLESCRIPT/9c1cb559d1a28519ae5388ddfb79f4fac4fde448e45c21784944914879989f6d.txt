-- This is a code snippet in APPLESCRIPT that calculates the area of a circle
set radius to 5 -- set the radius of the circle
set pi to 3.14 -- set the value of pi
set area to pi * radius^2 -- calculate the area of the circle

-- output the result with an annotation
display dialog "The area of the circle with a radius of " & radius & " is " & area & " square units." with title "Circle Area Calculator"
