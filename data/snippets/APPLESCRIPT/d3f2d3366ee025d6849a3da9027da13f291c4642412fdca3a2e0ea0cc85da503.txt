--This script will calculate the average of three numbers.
set num1 to 10
set num2 to 15
set num3 to 20

--Calculating the sum of the three numbers.
set sum to num1 + num2 + num3

--Calculating the average by dividing the sum by 3.
set average to sum / 3

--Displaying the average.
display dialog "The average of three numbers is " & average
