-- This is a script to calculate the average of three numbers in Applescript

-- Declare variables to store the three numbers
set num1 to 10
set num2 to 15
set num3 to 20

-- Calculate the sum of the three numbers
set total to num1 + num2 + num3

-- Calculate the average of the three numbers
set average to total / 3

-- Print the average with a message
display dialog "The average of 10, 15, and 20 is " & average
