-- This code snippet uses AppleScript to perform a complex calculation
-- Define variables to hold values
set num1 to 10
set num2 to 5
set num3 to 2

-- Calculate the sum of num1 and num2
set sum to num1 + num2

-- Calculate the product of the sum and num3
set product to sum * num3

-- Print the result
display dialog "The result is " & product
