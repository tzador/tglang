(* This is a complex code snippet in AppleScript *)
property myName : "John" (* property declaration *)
set myAge to 25 (* variable declaration *)
if myName = "John" then (* if statement *)
    set myAge to myAge + 5 (* setting variable to new value *)
    log "My name is " & myName & " and I am " & myAge & " years old." (* logging statement *)
else if myName = "Jane" then (* else if statement *)
    set myAge to myAge - 5 (* setting variable to new value *)
    log "My name is " & myName & " and I am " & myAge & " years old." (* logging statement *)
else (* else statement *)
    log "Name not recognized." (* logging statement *)
end if (* end if statement *)
repeat with i from 1 to 10 (* repeat loop *)
    log "Loop iteration " & i (* logging statement *)
end repeat (* end repeat loop *)
try (* try statement *)
    set randomNum to random number from 1 to 100 (* setting variable to random number *)
    display dialog "Your random number is " & randomNum (* displaying dialog with random number *)
on error (* on error statement *)
    display dialog "An error occurred." (* displaying error dialog *)
end try (* end try statement *)
