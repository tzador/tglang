// Define a function that takes in two parameters of type number
function calculateSum(num1: number, num2: number): void {
	// Declare a variable of type number and assign it the sum of the parameters
	let sum: number = num1 + num2;
	// Print the sum to the console
	console.log(`The sum is: ${sum}`);
}

// Call the function with two numbers
calculateSum(5, 10);

// Output:
// The sum is: 15
