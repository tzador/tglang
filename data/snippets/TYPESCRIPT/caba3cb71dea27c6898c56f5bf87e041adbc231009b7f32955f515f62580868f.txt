/**
 * A function that takes in two numbers and multiplies them together.
 * @param {number} num1 - The first number to be multiplied.
 * @param {number} num2 - The second number to be multiplied.
 * @returns {number} - The result of the multiplication.
 */
function multiply(num1: number, num2: number): number {
    return num1 * num2;
}

// Example usage:
console.log(multiply(2, 3)); // Output: 6
console.log(multiply(4, 5)); // Output: 20
console.log(multiply(10, -2)); // Output: -20
