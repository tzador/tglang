// This is a simple TypeScript program that calculates the average of two numbers
// Author: John Doe

// Function to calculate the average of two numbers
function calculateAverage(x: number, y: number): number {
  return (x + y) / 2;
}

// Calling the function and storing the result in a variable
let result: number = calculateAverage(10, 20);

// Printing the result
console.log(`The average of 10 and 20 is ${result}`);
