syntax = "proto3";

package AddressBook;

message Person {
  string name = 1;
  int32 age = 2;
  repeated string emails = 3;
  repeated PhoneNumber phones = 4;

  enum PhoneType {
    MOBILE = 0;
    HOME = 1;
    WORK = 2;
  }

  message PhoneNumber {
    string number = 1;
    PhoneType type = 2;
  }
}

message AddressBook {
  repeated Person people = 1;
}
