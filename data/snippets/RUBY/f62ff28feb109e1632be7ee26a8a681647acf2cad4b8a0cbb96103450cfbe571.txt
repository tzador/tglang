# This is a method to calculate the sum of two numbers
def add(num1, num2)
    sum = num1 + num2
    return sum
end

# This is a method to calculate the difference between two numbers
def subtract(num1, num2)
    difference = num1 - num2
    return difference
end

# This is a method to calculate the product of two numbers
def multiply(num1, num2)
    product = num1 * num2
    return product
end

# This is a method to calculate the quotient of two numbers
def divide(num1, num2)
    quotient = num1 / num2
    return quotient
end

# This is a method to calculate the power of one number to another
def power(num1, num2)
    power = num1 ** num2
    return power
end

# This is a method to calculate the remainder when dividing two numbers
def modulus(num1, num2)
    remainder = num1 % num2
    return remainder
end

# Calculate the sum of 3 and 5
add(3, 5)

# Calculate the difference between 10 and 2
subtract(10, 2)

# Calculate the product of 4 and 7
multiply(4, 7)

# Calculate the quotient of 20 and 4
divide(20, 4)

# Calculate the power of 2 to the 5th power
power(2, 5)

# Calculate the remainder when dividing 15 by 4
modulus(15, 4)
