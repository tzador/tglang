# This code snippet calculates the sum of two given integers

def sum(a, b)
	# This function takes two parameters a and b
	# and returns the sum of the two integers
  return a + b

end

# Declaring and initializing two variables
number1 = 5
number2 = 10

# Calling the sum function with the two variables as arguments
result = sum(number1, number2)

# Printing the result
puts result #=> 15
