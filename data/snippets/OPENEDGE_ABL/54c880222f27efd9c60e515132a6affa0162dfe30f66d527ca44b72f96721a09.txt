{&FILE-NAME}:DEF VAR i AS INTEGER.           //defines an integer variable
{&FILE-NAME}:DEF VAR j AS INTEGER.           //defines another integer variable
{&FILE-NAME}:DEF VAR result AS INTEGER.      //defines a variable to hold the result
i = 10.                                      //assigns a value of 10 to variable i
j = 20.                                      //assigns a value of 20 to variable j
result = i + j.                              //performs addition and assigns result to variable
DISPLAY "The result of adding i and j is " result ".".   //displays the result
