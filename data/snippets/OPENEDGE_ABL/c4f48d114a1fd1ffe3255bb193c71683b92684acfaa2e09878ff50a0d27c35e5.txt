{^DEFINE TEMP-TABLE tt-info
  FIELD id AS INT
  FIELD name AS CHARACTER
  FIELD age AS INT
  FIELD salary AS DECIMAL}

USING System.IO.SystemWriter.

OUTPUT TO VALUE("info.txt") FORMAT "XLS"
  NO-CONSOLE-SNOOP
  NO-ERROR
  NO-MESSAGE
  NO-PREF
  NO-TITLE
  tt-info:
  START-ROW 2
  OPEN-INHERIT 1
  OUTPUT-REGION 11 BY FIRST ROWS
  FIELD-SEPARATOR X"2C"
  COLUMN-SEPARATOR X"09"
  COLHDG-SEPARATOR X"0D0A"
  NO-APPEND
  PROCESS-ROWS OF tt-info:

    WRITE tt-info.id
    WRITE tt-info.name
    WRITE tt-info.age
    WRITE DECIMAL tt-info.salary
END.
