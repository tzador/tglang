DEFINE VARIABLE i AS INTEGER NO-UNDO.
DEFINE VARIABLE num AS DECIMAL NO-UNDO.
DEFINE VARIABLE result AS LOGICAL NO-UNDO.
DEFINE VARIABLE val1 AS CHARACTER NO-UNDO.
DEFINE VARIABLE val2 AS CHARACTER NO-UNDO.
DO i=1 TO 10:
    ASSIGN num = 5.5 * i.
    ASSIGN val1 = STRING(num).
    ASSIGN val2 = STRING(num * 2).
    IF val1 = val2 THEN
        ASSIGN result = TRUE.
    ELSE
        ASSIGN result = FALSE.
    MESSAGE "Result is " val1 "equals" val2 ":" result VIEW-AS ALERT-BOX.
END.
