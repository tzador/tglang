TO SQUARE :SIZE
  REPEAT 4 [FD :SIZE RT 90]
END

TO SPIRAL :SIZE :ANGLE :INCR
  REPEAT 50 [
    FD :SIZE RT :ANGLE
    SETANGLE MAKE "ANGLE :ANGLE + :INCR
    SETSIZE :SIZE + 5
  ]
END

TO DRAWCIRCLE :SIZE :ANGLE :SIDES
  REPEAT :SIDES [
    FD :SIZE RT :ANGLE
    SETANGLE MAKE "ANGLE :ANGLE + 1
    SETSIZE :SIZE + 2
  ]
END

TO TREE :SIZE :ANGLE :LEVEL
  IF :LEVEL > 0 [
    FORWARD :SIZE
    RIGHT :ANGLE
    TREE :SIZE * 0.8 :ANGLE * 1.2 :LEVEL - 1
    LEFT :ANGLE * 2
    TREE :SIZE * 0.8 :ANGLE * 1.2 :LEVEL - 1
    RIGHT :ANGLE
    BACK :SIZE
  ]
END

TO FLOWER :SIZE :SIDES :COLOR :LEVEL
  REPEAT :SIDES [
    SETPC :COLOR
    CIRCLE :SIZE
    RIGHT 360 / :SIDES
  ]
  IF :LEVEL > 0 [
    FLOWER :SIZE * 0.8 :SIDES + 1 :COLOR + 10 :LEVEL - 1
  ]
END

TO POLYGON :SIZE :ANGLE :SIDES
  REPEAT :SIDES [
    FORWARD :SIZE
    RIGHT :ANGLE
  ]
END

TO FRACTAL :SIZE :ANGLE :LEVEL
  IF :LEVEL > 0 [
    SETPC :LEVEL * 10
    FD :SIZE
    RIGHT :ANGLE
    FRACTAL :SIZE / 2 :ANGLE * 2 :LEVEL - 1
    LEFT :ANGLE * 4
    FRACTAL :SIZE / 2 :ANGLE * 2 :LEVEL - 1
    RIGHT :ANGLE
    BK :SIZE 
  ]
END

TO HEXSPIRAL :SIZE :ANGLE :INCR
  IF :SIZE > 0 [
    REPEAT 6 [
      SPIRAL :SIZE :ANGLE :INCR
      HEXSPIRAL :SIZE * 0.95 :ANGLE :INCR
      RT 60
    ]
  ]
END

TO CIRCLEFRINGE
  REPEAT 360 [ 
    FLOWER 50 :COUNT * 10 :COLOR 5.5 
    RT 1 
  ] 
END
