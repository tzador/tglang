  jump 50 ; jumps forward 50 units
  forward 100 ; moves forward 100 units
  right 90 ; turns the turtle right 90 degrees
  back 50 ; moves backward 50 units
  left 45 ; turns the turtle left 45 degrees
  circle 30 ; creates a circle with a radius of 30 units
  penup ; lifts the turtle's pen off the canvas
  setpos 0 0 ; sets the turtle's position to (0,0)
  pendown ; puts the turtle's pen back on the canvas
