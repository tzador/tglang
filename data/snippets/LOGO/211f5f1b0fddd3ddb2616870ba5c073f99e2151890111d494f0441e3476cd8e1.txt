TO SPIRAL :SIZE :ANGLE :STEP
  REPEAT :SIZE [
    RIGHT :ANGLE
    FORWARD :STEP
    SIZE :SIZE - 1
    SPIRAL :SIZE :ANGLE :STEP + 1
  ]
END 

TO DRAW-CIRCLE :SIZE 
  REPEAT 360 [
    FORWARD :SIZE
    LEFT 1
    ] 
END 

TO DRAW-SPIRAL :SIZE 
  SPIRAL :SIZE 10 1 
  SPIRAL :SIZE 10 -1 
  REP_CIRCLE :SIZE 
END 

DRAW-SPIRAL 100
