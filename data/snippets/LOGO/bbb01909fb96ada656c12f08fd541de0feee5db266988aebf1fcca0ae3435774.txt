MAKE "COUNT 10
; SETS COUNT VARIABLE TO EQUAL 10

REPEAT :COUNT [
  SETPC random 255
  PENUP
  SETPOS random 500 random 500
  PENDOWN
  SETPC random 255

  REPEAT :COUNT [
    FORWARD 50
    LEFT 90
  ]
]
; REPEATS A SET OF RANDOMLY COLORED SQUARES WITHIN A RANDOMLY GENERATED POSITION FOR TEN TIMES

REPEAT 8 [
  MAKE "X POS
  MAKE "Y POS
  SETPOS :X :Y
  FD 50
  RT 45
]
; CREATES A CIRCULAR SPHERE PATTERN USING A SET OF DECLARED VARIABLES

REPEAT :COUNT [
  SETBG random 255
  PENDOWN
  MAKE "SIDE 15
  REPEAT 12 [
    FORWARD :SIDE
    LEFT 30
    FORWARD :SIDE
    LEFT 30
    FORWARD :SIDE + 10
    RIGHT 90
    FORWARD :SIDE + 10
    HOME
  ]
  PENUP
  SETPOS random 500 random 500
]
; CREATES A RANDOMLY COLORED AND POSITIONED FLOWER-LIKE PATTERN FOR TEN TIMES

PENUP
SETPOS 0 0
SETPENSIZE 10
SETPC 0 0 0
PENDOWN
REPEAT 6 [
  SETPOS 0 0
  FORWARD 150
  LEFT 60
]
; CREATES A HEXAGONAL SHAPE OUTLINE USING A SET OF DECLARED VARIABLES
