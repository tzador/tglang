to star //procedure to draw a star shape
  repeat 5 [ //repeat loop to draw 5 points of the star
    fd 100 //move forward 100 units
    rt 144 //turn right by 144 degrees to form an angle
    fd 100 //move forward 100 units again
    lt 72 //turn left by 72 degrees to form another angle
    rt 72 //return to original direction
  ]
end //end of procedure
