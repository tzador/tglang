; This code will draw a spiral shape using the LOGO programming language
 
 to spirals
   pendown
   repeat 36 [ forward 10 right 10]
   repeat 36 [ forward 20 right 10]
   repeat 36 [ forward 30 right 10]
   repeat 36 [ forward 40 right 10]
   repeat 36 [ forward 50 right 10]
   repeat 36 [ forward 60 right 10]
   repeat 36 [ forward 70 right 10]
   repeat 36 [ forward 80 right 10]
   repeat 36 [ forward 90 right 10]
 end

; Call the spirals function
spirals
