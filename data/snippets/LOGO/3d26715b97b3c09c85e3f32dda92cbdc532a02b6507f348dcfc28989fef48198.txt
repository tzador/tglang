REPEAT 4 [  ; a loop that will repeat 4 times
  FD 100  ; moves 100 pixels forward
  RT 90   ; turns right 90 degrees
  LT 45   ; turns left 45 degrees
  BK 50   ; moves 50 pixels backwards
  PU      ; lifts pen up
  FD 75   ; moves 75 pixels forward without drawing
  PD      ; puts pen down
  ARC 50  ; draws a 50 degree arc
  RIGHT 2 ; rotates right 2 degrees
]
