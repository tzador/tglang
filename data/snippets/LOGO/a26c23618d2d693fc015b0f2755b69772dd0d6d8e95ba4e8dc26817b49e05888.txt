repeat 10 [ 									; loop 10 times
  setheading 0                 ; set direction to north
  forward 50                   ; move forward 50 steps
  setheading 90                ; set direction to east
  forward 50                   ; move forward 50 steps
  setheading 180               ; set direction to south
  forward 50                   ; move forward 50 steps
  setheading 270               ; set direction to west
  forward 50                   ; move forward 50 steps
]
