; This code snippet draws a complex geometric pattern using the LOGO programming language

TO SPIRAL :SIZE :ANGLE
  IFELSE :SIZE > 20 [
    FORWARD :SIZE
    RIGHT :ANGLE
    SPIRAL (:SIZE - 5) :ANGLE
  ] [
    FORWARD :SIZE
  ]
END

TO SQUARE :LENGTH
  REPEAT 4 [
    FORWARD :LENGTH
    RIGHT 90
  ]
END

TO FLORAL :SIZE
  PENUP
  SETPOS (RANDOM 300 - 150) (RANDOM 300 - 150)
  PENDOWN
  SETHEADING RANDOM 360
  SPIRAL :SIZE 20
END

TO CIRCLE :RADIUS
  REPEAT 360 [
    FORWARD :RADIUS * SIN 1
    RIGHT 1
  ]
END

PENUP
SETPOS -100 -100

  RIGHT 90
]
