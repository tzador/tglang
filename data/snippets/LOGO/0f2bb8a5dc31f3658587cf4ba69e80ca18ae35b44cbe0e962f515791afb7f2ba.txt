TO SQUARE :SIZE 
  REPEAT 4 [ FD :SIZE RT 90 ] 
END 

TO TURTLE-POLYGON :SIDES :SIZE
  REPEAT :SIDES [FD :SIZE RT 360 / :SIDES] 
END 

TO SPIRAL :ITER :SIZE 
  PU 
  IF :ITER > 0 [ 
    PD 
    FD :ITER * :SIZE 
    RT 90 
    SPIRAL :ITER - 1 :SIZE + 1 
  ] 
END 

TO PYRAMID :SIZE 
  FOR [I 1 :SIZE] 
  [ 
    SQUARE I 
    PU FD 1 
    RT 90 FD 1 
    PD 
  ] 
END
