MAKE "WORD "HELLO ; Defines a variable 'WORD' and assigns the string "HELLO" to it
MAKE "NUM 5 ; Defines a variable 'NUM' and assigns the value 5 to it
REPEAT :NUM [ ; Starts a loop that will run 'NUM' number of times
  PRINT :WORD ; Prints the value of the variable 'WORD' to the screen
] ; Ends the loop
