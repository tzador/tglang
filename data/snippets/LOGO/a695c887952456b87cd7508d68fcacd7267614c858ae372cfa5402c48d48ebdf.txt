; This code draws a rainbow using LOGO graphics
; Set the initial position and heading of the turtle
TO RAINBOW
SETPOS (-160) 102, LEFT 135

; Begin drawing the rainbow
REPEAT 16 [
  PENWIDTH 6
  PENCOLOR "RED
  FORWARD 11
  RIGHT 5
  PENCOLOR "ORANGE
  FORWARD 11
  RIGHT 5
  PENCOLOR "YELLOW
  FORWARD 11
  RIGHT 5
  PENCOLOR "GREEN
  FORWARD 11
  RIGHT 5
  PENCOLOR "BLUE
  FORWARD 11
  RIGHT 5
  PENCOLOR "INDIGO
  FORWARD 11
  RIGHT 5
  PENCOLOR "VIOLET
  FORWARD 11
  RIGHT 5
]
END
