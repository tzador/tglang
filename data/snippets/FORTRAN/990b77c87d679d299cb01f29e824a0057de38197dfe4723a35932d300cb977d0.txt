PROGRAM MULTDIV 

  REAL :: A = 5.0
  REAL :: B = 3.0

 	! This program calculates the multiplication and division of two REAL numbers.

  A = A * B                ! Multiplication
  B = A / B                ! Division

  PRINT *, "Result: ", A   ! Prints the result of multiplication
  PRINT *, "Result: ", B   ! Prints the result of division

  END PROGRAM MULTDIV
