SUBROUTINE INTERPOLATE(X,Y,N,NINT,XINT,YINT)
  DIMENSION X(N),Y(N),XINT(NINT),YINT(NINT)
  DO I= 1,NINT
    J=2
    DO 1
      IF (XINT(I)>=X(J)) J=J+1
    NUM=J

    XINT1=X(NUM-2)
    XINT2=X(NUM-1)
    YINT1=Y(NUM-2)
    YINT2=Y(NUM-1)

    YINT(I)=YINT1+(YINT2-YINT1)*(XINT(I)-XINT1)/(XINT2-XINT1)

1    CONTINUE
  END
