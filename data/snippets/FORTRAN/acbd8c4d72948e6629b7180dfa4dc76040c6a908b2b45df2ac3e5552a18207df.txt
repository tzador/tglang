PROGRAM MAIN ! START OF MAIN PROGRAM
      IMPLICIT NONE ! STATEMENT TO ENSURE ALL VARIABLES ARE DECLARED
      INTEGER :: N, I, J ! DECLARING INTEGER VARIABLES N, I AND J
      REAL :: A(100,100), B(100,100), C(100,100) ! DECLARING 3-DIMENSIONAL REAL ARRAYS WITH DIMENSIONS 100 X 100 X 100
      
      DO I = 1, 100 ! START OF FIRST DO LOOP
            DO J = 1, 100 ! START OF SECOND DO LOOP
                  DO N = 1, 100 ! START OF THIRD DO LOOP
                        A(I,J,N) = I + J + N ! ASSIGNING VALUES TO ARRAY A
                        B(I,J,N) = 2.0 * I - J + N ! ASSIGNING VALUES TO ARRAY B
                  END DO ! END OF THIRD DO LOOP
            END DO ! END OF SECOND DO LOOP
      END DO ! END OF FIRST DO LOOP
      
      DO I = 1, 100 ! START OF FOURTH DO LOOP
            DO J = 1, 100 ! START OF FIFTH DO LOOP
                  DO N = 1, 100 ! START OF SIXTH DO LOOP
                        C(I,J,N) = A(I,J,N) * B(I,J,N) + 5.0 ! PERFORMING CALCULATIONS AND STORING THE RESULT IN ARRAY C
                  END DO ! END OF SIXTH DO LOOP
            END DO ! END OF FIFTH DO LOOP
      END DO ! END OF FOURTH DO LOOP
      
      PRINT *, 'Final result:' ! PRINTING A HEADER FOR THE FINAL OUTPUT
      
      DO I = 1, 100 ! START OF SEVENTH DO LOOP
            DO J = 1, 100 ! START OF EIGHTH DO LOOP
                  DO N = 1, 100 ! START OF NINTH DO LOOP
                        PRINT *, 'C(',I,',',J,',',N,') = ', C(I,J,N) ! PRINTING THE INDEX AND VALUE OF EACH ELEMENT IN ARRAY C
                  END DO ! END OF NINTH DO LOOP
            END DO ! END OF EIGHTH DO LOOP
      END DO ! END OF SEVENTH DO LOOP
      
      END PROGRAM MAIN ! END OF MAIN PROGRAM
