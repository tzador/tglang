PROGRAM CALCULATE_AVERAGE

	! This program calculates the average of a given set of numbers

	REAL :: num1, num2, num3, num4, num5, average

	! Declare variables for the numbers and the average

	num1 = 12.5
	num2 = 20.3
	num3 = 9.7
	num4 = 15.2
	num5 = 11.8

	! Assign values to the variables

	average = (num1 + num2 + num3 + num4 + num5) / 5

	! Calculate the average by adding all the numbers and dividing by the total number of numbers

	PRINT *, "The average is: ", average

	! Print the calculated average

	END PROGRAM CALCULATE_AVERAGE
