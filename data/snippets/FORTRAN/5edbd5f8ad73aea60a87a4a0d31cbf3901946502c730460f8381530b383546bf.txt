PROGRAM MAIN		! DECLARING THE PROGRAM
      IMPLICIT NONE		! TO AVOID UNDECLARED VARIABLES

      INTEGER :: A, B, C	! DECLARING INTEGER VARIABLES A, B, AND C

      ! PROMPTING THE USER FOR INPUT VALUES
      WRITE (*,*) 'ENTER VALUE FOR A:'
      READ (*,*) A
      WRITE (*,*) 'ENTER VALUE FOR B:'
      READ (*,*) B

      C = A + B		! ADDING THE VALUE OF A AND B, AND STORING IT IN C

      ! PRINTING THE RESULT
      WRITE (*,*), 'THE SUM OF A AND B IS:', C

      END PROGRAM MAIN	! END OF PROGRAM
