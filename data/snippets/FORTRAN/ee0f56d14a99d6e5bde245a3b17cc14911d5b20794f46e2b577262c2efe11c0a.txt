! EXAMPLE CODE FOR CALCULATING AND PRINTING THE SQUARE ROOT OF A NUMBER
      PROGRAM SQUARE_ROOT
! DECLARE VARIABLES
      REAL :: X, SQRT
! GET USER INPUT
      WRITE(*,*) 'ENTER A NUMBER TO FIND ITS SQUARE ROOT: '
      READ(*,*) X
! CALCULATE SQUARE ROOT
      SQRT = SQRT(X)
! PRINT RESULTS
      WRITE(*,*) 'THE SQUARE ROOT OF ', X, ' IS ', SQRT
      END PROGRAM
