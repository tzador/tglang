! This program calculates the sum of two numbers

program sum

    ! declare variables 
    real :: num1, num2, sum
    
    ! set initial values
    num1 = 5.5
    num2 = 3.2
    
    ! calculate sum
    sum = num1 + num2
    
    ! print output
    print *, 'The sum of',num1,'and',num2,'is',sum
    
end program sum
