subroutine matrix_multiplication(A,B,C,m,n,k)
! This subroutine performs matrix multiplication of two matrices, A and B, and stores the result in matrix C.! The number of rows in matrix A is m, the number of columns in matrix A and rows in matrix B is n,! and the number of columns in matrix B is k.implicit noneinteger :: m,n,k, i,j,l ! declaring the input variables m,n,k and loop indices i,j,l
real :: A(m,n), B(n,k), C(m,k)  ! declaring the input matrices A and B and the resulting matrix C! Initialize matrix C to zerodo i=1,m  do j=1,k     C(i,j) = 0.0  end doend do! Perform matrix multiplicationdo i=1,m  do j=1,k     do l=1,n        C(i,j) = C(i,j) + A(i,l) * B(l,j)     end do  end doend doend subroutine matrix_multiplication  ! ending the subroutine
