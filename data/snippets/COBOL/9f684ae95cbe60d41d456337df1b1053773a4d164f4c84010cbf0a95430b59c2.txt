IDENTIFICATION DIVISION.
      PROGRAM-ID. PROGRAM_NAME.
      AUTHOR. YOUR_NAME.

      DATA DIVISION.
      WORKING-STORAGE SECTION.
      77  STUDENT-COUNT PIC 999.
      77  TOTAL-MARKS PIC 99999.
      77  GRADE-CODE PIC A(1).

      01  STUDENT-RECORD.
          05  STUDENT-NAME PIC X(20).
          05  STUDENT-MARKS PIC 999.

      01  GRADE-TABLE.
          05  GRADE-CODE       PIC A(1) VALUE "A".
          05  GRADE-DESCRIPTION PIC X(20) VALUE "EXCELLENT".
          05  GRADE-POINTS     PIC 99    VALUE 10.
          05  GRADE-FLAGS      PIC A(1) VALUE " ".

          05  GRADE-CODE       PIC A(1) VALUE "B".
          05  GRADE-DESCRIPTION PIC X(20) VALUE "GOOD".
          05  GRADE-POINTS     PIC 99    VALUE 8.
          05  GRADE-FLAGS      PIC A(1) VALUE " ".

          05  GRADE-CODE       PIC A(1) VALUE "C".
          05  GRADE-DESCRIPTION PIC X(20) VALUE "SATISFACTORY".
          05  GRADE-POINTS     PIC 99    VALUE 6.
          05  GRADE-FLAGS      PIC A(1) VALUE " ".

          05  GRADE-CODE       PIC A(1) VALUE "D".
          05  GRADE-DESCRIPTION PIC X(20) VALUE "PASS".
          05  GRADE-POINTS     PIC 99    VALUE 4.
          05  GRADE-FLAGS      PIC A(1) VALUE " ".

          05  GRADE-CODE       PIC A(1) VALUE "F".
          05  GRADE-DESCRIPTION PIC X(20) VALUE "FAIL".
          05  GRADE-POINTS     PIC 99    VALUE 0.
          05  GRADE-FLAGS      PIC A(1) VALUE " ".

      PROCEDURE DIVISION.
      MAIN-SECTION.
         PERFORM 10 TIMES
         ACCEPT STUDENT-RECORD
         IF STUDENT-MARKS > 100
            DISPLAY "Invalid marks entered for student: " STUDENT-NAME
         END-IF
         ADD STUDENT-MARKS TO TOTAL-MARKS
         ADD 1 TO STUDENT-COUNT
         SEARCH GRADE-TABLE
         AT END
            MOVE "F" TO GRADE-CODE
         WHEN GRADE-CODE = GRADE-CODE
            MOVE GRADE-DESCRIPTION TO GRADE-FLAGS
         END-SEARCH.

       COMPILE-SECTION.
          DISPLAY "Total number of students: " STUDENT-COUNT
          DISPLAY "Average marks of class: " TOTAL-MARKS / STUDENT-COUNT
          DISPLAY "Grade code and points: " GRADE-CODE " - " GRADE-POINTS

       EXIT-SECTION.
          STOP RUN.
      END PROGRAM PROGRAM_NAME.
