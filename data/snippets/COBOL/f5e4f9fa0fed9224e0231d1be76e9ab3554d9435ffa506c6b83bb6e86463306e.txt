IDENTIFICATION DIVISION.
PROGRAM-ID. CALCULATOR.
DATA DIVISION.
WORKING-STORAGE SECTION.
01 NUM1 PIC 9(4).
01 NUM2 PIC 9(4).
01 RESULT PIC 9(5).
01 OPERATOR PIC X.
PROCEDURE DIVISION.
MAIN SECTION.
    DISPLAY "Enter first number: ".
    ACCEPT NUM1.
    DISPLAY "Enter second number: ".
    ACCEPT NUM2.
    DISPLAY "Enter operator (+, -, *, /): ".
    ACCEPT OPERATOR.
    PERFORM COMPUTE-RESULT.
    DISPLAY "The result is: ".
    DISPLAY RESULT.
    STOP RUN.

COMPUTE-RESULT.
28 IF OPERATOR = "+" THEN
29     ADD NUM1, NUM2 TO RESULT
30 ELSE IF OPERATOR = "-" THEN
31     SUBTRACT NUM2 FROM NUM1 GIVING RESULT
32 ELSE IF OPERATOR = "*" THEN
33     MULTIPLY NUM1 BY NUM2 GIVING RESULT
34 ELSE IF OPERATOR = "/" THEN
35     DIVIDE NUM1 BY NUM2 GIVING RESULT
36 ELSE
37     DISPLAY "Invalid operator entered."
38     DISPLAY "Please enter a valid operator (+, -, *, /): ".
39     ACCEPT OPERATOR.
40     PERFORM COMPUTE-RESULT.
41 END-IF.
