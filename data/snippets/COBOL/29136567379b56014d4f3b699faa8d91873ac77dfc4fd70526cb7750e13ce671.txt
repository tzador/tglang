IDENTIFICATION DIVISION.
       PROGRAM-ID. EXAMPLE-PROGRAM.
       DATA DIVISION.
       WORKING-STORAGE SECTION.
       	01 NUM-DATA PIC 9(4) VALUE 1234.
       	01 ALPHA-DATA PIC A(6) VALUE 'TEST'.
       PROCEDURE DIVISION.
       	PERFORM UNTIL ALPHADATA EQUALS 'DONE'
       	   IF NUM-DATA = 1234 THEN
       	     DISPLAY 'NUM-DATA IS EQUAL TO 1234'
       	   ELSE
       	     DISPLAY 'NUM-DATA IS NOT EQUAL TO 1234'
       	   END-IF
       	   ADD 1 TO NUM-DATA
       	END-PERFORM.
       	DISPLAY 'LOOP FINISHED'.
       	STOP RUN.
