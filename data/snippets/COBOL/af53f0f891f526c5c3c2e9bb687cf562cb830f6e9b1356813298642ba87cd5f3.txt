IDENTIFICATION DIVISION.
       PROGRAM-ID. PRINT-REPORT.
       DATA DIVISION.
       WORKING-STORAGE SECTION.
       01 RECORD-TOTAL.
          05 EMPLOYEE-COUNT PIC 9(3).
          05 TOTAL-SALARY PIC 9(7)V99.
          05 AVERAGE-SALARY PIC 9(6)V99.
          05 HIGHEST-SALARY PIC 9(6)V99.
          05 LOWEST-SALARY PIC 9(6)V99.
       01 EMPLOYEE-DETAIL.
          05 EMPLOYEE-NAME PIC X(20).
          05 EMPLOYEE-ID PIC 9(5).
          05 EMPLOYEE-SALARY PIC 9(6)V99.
       01 TEMPORARY STORAGE.
          05 COUNTER PIC 99.
          05 TOTAL-SALARY-TEMP PIC 9(8)V99.
          05 HIGHEST-SALARY-TEMP PIC 9(6)V99.
          05 LOWEST-SALARY-TEMP PIC 9(6)V99.
       FILE SECTION.
       01 EMPLOYEE-FILE.
          05 EMPLOYEE-RECORD.
             10 FILLER PIC X.
             10 EMPLOYEE-NAME PIC X(20).
             10 FILLER PIC X.
             10 EMPLOYEE-ID PIC 9(5).
             10 FILLER PIC X.
             10 EMPLOYEE-SALARY PIC 9(6)V99.
          05 FILLER PIC X.
       PROCEDURE DIVISION.
       OPEN INPUT EMPLOYEE-FILE
       READ EMPLOYEE-FILE
       AT END
          MOVE EMPLOYEE-COUNT TO TOTAL-SALARY-TEMP
          DIVIDE TOTAL-SALARY-TEMP BY EMPLOYEE-COUNT
          GIVING AVERAGE-SALARY
          DISPLAY 'Total number of employees: ' EMPLOYEE-COUNT
          DISPLAY 'Average salary: ' AVERAGE-SALARY
          DISPLAY 'Highest salary: ' HIGHEST-SALARY
          DISPLAY 'Lowest salary: ' LOWEST-SALARY
          STOP RUN
       END-READ
       ADD 1 TO COUNTER
       ADD EMPLOYEE-SALARY TO TOTAL-SALARY
       IF TEMP-COUNTER = ZERO
          MOVE EMPLOYEE-SALARY TO HIGHEST-SALARY-TEMP
          MOVE EMPLOYEE-SALARY TO LOWEST-SALARY-TEMP
       END-IF
       IF EMPLOYEE-SALARY > HIGHEST-SALARY-TEMP
          MOVE EMPLOYEE-SALARY TO HIGHEST-SALARY-TEMP
       END-IF
       IF EMPLOYEE-SALARY < LOWEST-SALARY-TEMP
          MOVE EMPLOYEE-SALARY TO LOWEST-SALARY-TEMP
       END-IF
       END-PERFORM
       CLOSE EMPLOYEE-FILE
       EXIT
