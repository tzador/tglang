WORKING-STORAGE SECTION.
01 COUNTER-VARIABLE PIC 9(3) VALUE ZERO.
01 SENTENCE-LENGTH PIC 9(3) VALUE ZERO.
01 STRING-LENGTH PIC 9(3) VALUE ZERO.
01 FINAL-STRING PIC X(50) VALUE SPACES.

PROCEDURE DIVISION.
MAIN-PROCEDURE.
    DISPLAY "Enter a sentence:".
    ACCEPT INPUT-STRING.

    PERFORM CALCULATE-LENGTH THRU CALC-END-PROCEDURE
        VARYING COUNTER-VARIABLE FROM 1 BY 1 
        UNTIL COUNTER-VARIABLE > SENTENCE-LENGTH

    DISPLAY "Final String: ", FINAL-STRING.
    STOP RUN.

CALCULATE-LENGTH.
        COMPUTE SENTENCE-LENGTH = FUNCTION LENGTH(INPUT-STRING).

CALC-END-PROCEDURE.
        CONTINUE.
        EVALUATE TRUE
            WHEN COUNTER-VARIABLE = 1
                MOVE INPUT-STRING(1:STRING-LENGTH)
                TO FINAL-STRING
            WHEN COUNTER-VARIABLE = SENTENCE-LENGTH
                MOVE INPUT-STRING(STRING-LENGTH+1:STRING-LENGTH)
                TO FINAL-STRING
            WHEN OTHER
                MOVE INPUT-STRING(STRING-LENGTH+1:STRING-LENGTH)
                TO FINAL-STRING
                MOVE FINAL-STRING(STRING-LENGTH+1:STRING-LENGTH)
                TO TEMP-VARIABLE
                STRING FINAL-STRING DELIMITED BY SPACE
                    " " TEMP-VARIABLE
                    INTO FINAL-STRING
        END-EVALUATE.
