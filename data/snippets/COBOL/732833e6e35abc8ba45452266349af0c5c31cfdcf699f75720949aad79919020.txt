DATA DIVISION.
       WORKING-STORAGE SECTION.
       COPY "WSDFLEAA"

       ATOMIC INTEGER NUM1, NUM2.
       ATOMIC INTEGER RESULT.

       PROCEDURE DIVISION.
       MAIN-PROCESS.
           SET NUM1 TO 10.
           SET NUM2 TO 5.
           COMPUTE RESULT = (NUM1 + NUM2) * NUM2 - NUM1.
           DISPLAY "The result is: " RESULT.
           END MAIN-PROCESS.
       END PROGRAM.
