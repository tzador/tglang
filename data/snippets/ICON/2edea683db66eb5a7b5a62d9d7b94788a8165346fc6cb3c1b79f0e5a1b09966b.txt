procedure main()
# This is a comment

# Initialize variables
sum := 0
i := 1
n := 10

# Loop to calculate sum of numbers from 1 to 10
while i <= n do {
    # Add current number to sum
    sum := sum + i 

    # Increment counter
    i := i + 1
}

# Print result
write "The sum of numbers from 1 to 10 is ", sum, "\n"

# End of program
end
