procedure main() # Procedure definition statement for main procedure
   write("Hello, world!") # Output statement to print "Hello, world!"
   n := 5 # Assignment statement setting variable n equal to 5
   loop # Loop statement
      write(n) # Output statement to print value of n
      n := n - 1 # Assignment statement to decrement n by 1
   again while n > 0 # While statement to loop while n is greater than 0
end # End of procedure definition statement for main procedure
