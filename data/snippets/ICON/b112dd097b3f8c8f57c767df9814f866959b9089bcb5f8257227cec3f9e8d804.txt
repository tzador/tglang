/* This code snippet is a program that calculates the volume of a box. */

procedure main() {
  ; This procedure calculates the volume of a box
  
  ; Declare variables for length, width, and height
  length := 4
  width := 6
  height := 3

  ; Calculate volume by multiplying length, width, and height
  volume := length * width * height

  ; Output the volume to the screen
  writeln("The volume of the box is ", volume)

  end
}
