; example of a complex code snippet in ICON

procedure main()

# define function that adds two numbers
define add(num1, num2)
    return num1 + num2
    
# initialize variables
let a = 5
let b = 10

# call add function and store result in variable c
let c = add(a, b)

# print the result
put "The sum of a and b is: ", c

end
