# Start with a procedure statement:
PROCEDURE main()

# Define variables:
INT num1, num2
STRING message

# Initialize variables:
num1 = 10
num2 = 5
message = "The result is: "

# Calculate sum:
ADD num1, num2

# Store result in variable:
num3 = RESULT

# Add result to message:
message := message || num3

TYPE message
