procedure main()
  ;; Define variables
  name = "John Doe"
  age = 25

  ;; Print values of variables
  print(name)
  print(age)

  ;; Loop through numbers from 1 to 10
  loop i = 1 to 10
    ;; Check if number is even
    if i % 2 = 0 then
      ;; Print number
      print("Even number: " || i)

  ;; Print "Finished" message
  print("Finished")

end
