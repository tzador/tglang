procedure double(input: integer)                 #This is a procedure named double that takes in an integer as an input
    double_val: integer                          #Declaring a variable named double_val of type integer
    double_val := input * 2                       #Multiplying the input by 2 and storing it in double_val
    print("The doubled value is ", double_val)    #Printing the doubled value with a message
end                                             #End of the procedure
