procedure mergeLists(x, y)
-- merges two lists and sorts them
while x not empty and y not empty do
    if x.first < y.first then
        z := z 路 x.first
        x := x[2:]
    else
        z := z 路 y.first
        y := y[2:]
    end
end
if x not empty then
    z := z 路 x
else
    z := z 路 y
end
return z
end
