// This code snippet creates a function to calculate the area of a rectangle

procedure calc_rect_area (width, height)
  // the procedure takes in two parameters, width and height
  area := width * height // calculates the area by multiplying the width by the height
  write("The area of the rectangle is ", area) // prints the calculated area
end

// Example usage
calc_rect_area(5, 10) // outputs 'The area of the rectangle is 50'
