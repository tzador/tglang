procedure printNumbers(n)
"""
Description: This procedure prints the numbers from 1 to n in ascending order.
Input: Integer n, the final number to be printed.
Output: The numbers from 1 to n printed in order.
Example input: 5
Example output: 1
                2
                3
                4
                5
"""

do i = 1 to n by 1 {
  write(i)
}
end
