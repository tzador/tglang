/* This is a sample code in ICON programming language */

procedure main()

    /* Variable declarations */
    local name := "John"
    local age := 27
    local height := 5.6

    /* Print output */
    out "Hello, my name is " || name || " and I am " || age || " years old."
    out "I am " || height || " feet tall."

end

/* Output:
Hello, my name is John and I am 27 years old.
I am 5.6 feet tall.
*/
