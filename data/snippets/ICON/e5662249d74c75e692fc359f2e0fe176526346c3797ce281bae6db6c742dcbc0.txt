procedure main() # This is the main procedure
let counter := 10 # This is a counter variable set to 10
if counter > 0 then # This is an if statement
  counter := counter - 1 # This decrements the counter by 1
  println("Counter is now: ", counter) # This prints the updated counter value
else # If counter is 0 or less then this else statement runs
  println("Counter is less than or equal to 0") # This prints an error message
end # End of if statement
end # End of main procedure
