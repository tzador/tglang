/* This code snippet is written in ICON programming language */
procedure main()   /* Main procedure that is executed when the program is run */
 
/* Initializing variables */
integer numa, numb      /* Two integer variables */
numa := 12           /* Assigning a value to variable numa */
numb := 8            /* Assigning a value to variable numb */
 
/* Comparing the two variables and outputting the result */
if (numa > numb) then    /* Checks if numa is greater than numb */
    write("numa is greater than numb.")      /* If condition is true, output the message */
else    /* If condition is false */
    write("numa is not greater than numb.")  /* Output a different message */
 
/* End of program */    /* This line denotes the end of the program */
end
