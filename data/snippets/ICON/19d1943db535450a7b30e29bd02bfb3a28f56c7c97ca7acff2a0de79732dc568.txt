procedure SEARCH (SUBJECT, PATTERN)
;SEARCH PROCEDURE - SEARCHES FOR A PATTERN IN A GIVEN SUBJECT

SUBJECT := REPLACE(SUBJECT, SPACE, NULL) ;REMOVE SPACES FROM SUBJECT
PATTERN := REPLACE(PATTERN, SPACE, NULL) ;REMOVE SPACES FROM PATTERN

n := SIZE(SUBJECT) ;GET LENGTH OF SUBJECT
m := SIZE(PATTERN) ;GET LENGTH OF PATTERN

IF m <= n
THEN
    FOR i := 1 TO n - m + 1 BY 1 DO
    ;LOOP THROUGH THE SUBJECT WITH A STEP SIZE OF 1

        j := 1 ;INTIALIZE COUNTER FOR PATTERN MATCHING

        WHILE j <= m AND SUBJECT[i + j - 1] = PATTERN[j]
        ;ITERATE THROUGH EACH LETTER OF THE PATTERN

            j := j + 1 ;INCREMENT PATTERN COUNTER

        END ;END WHILE LOOP

        IF j > m
        THEN
            PRINT("Pattern found at position " || i) ;PRINT POSITION OF MATCH

        END ;END IF

    END ;END FOR LOOP

ELSE
    PRINT("Pattern length cannot be greater than Subject length.") ;HANDLE ERROR

END ;END IF
