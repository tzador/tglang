;This is a code snippet in Assembly language
;It calculates the sum of two numbers and stores the result in a register
MOV AX, 4 ;Move the first number (4) into the AX register
MOV BX, 3 ;Move the second number (3) into the BX register
ADD AX, BX ;Add the values in AX and BX and store the result in AX
;The result is now stored in the AX register
