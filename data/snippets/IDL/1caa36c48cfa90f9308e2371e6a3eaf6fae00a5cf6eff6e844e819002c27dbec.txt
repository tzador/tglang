PRO FIBNUM, N, RESULT ;
; This program calculates the n-th value in the Fibonacci sequence ;
; Input: n, the value to calculate ;
; Output: result, the n-th value in the Fibonacci sequence ;
; Example: print, FIBNUM(6) ; ; ; This will print the 6th value, which is 8 ;
IF (N LT 0) THEN BEGIN ; ; If n is less than 0 ;
PRINT, 'n must be a positive integer!' ; ; Print error message ;
RETURN, 0 ; ; Return 0 and exit program ;
ENDIF ; ; End if statement ;
; If n is 0 or 1, return n ;
IF (N EQ 0) OR (N EQ 1) THEN BEGIN ; ; If n is 0 or 1 ;
RETURN, N ; ; Return n and exit program ;
ENDIF ; ; End if statement ;
; Initialize variables ;
A = 0 ; ; First value in sequence ;
B = 1 ; ; Second value in sequence ;
I = 2 ; ; Counter for loop ;
; Loop through sequence until reaching n-th value ;
WHILE (I LT N) DO BEGIN ; ; While loop condition ;
RESULT = A + B ; ; Calculate next value in sequence ;
A = B ; ; Update first value ;
B = RESULT ; ; Update second value ;
I = I + 1 ; ; Increment counter ;
ENDWHILE ; ; End while loop ;
; Print result ;
PRINT, 'The n-th value in the Fibonacci sequence is: ', RESULT ; ; Print result ;
; Return result and exit program ;
RETURN, RESULT ; ; Return result ;
END ; ; End program ;
