; This program calculates the average of three numbers
; Author: John Smith

; Inputs
a = 12; ; First number
b = 9; ; Second number
c = 17; ; Third number

; Calculations
sum = a + b + c ; Sum of the three numbers
avg = sum / 3 ; Average of the three numbers

; Output
print, "The average of", a, b, "and", c, "is", avg
