let rec fibonacci n =      (* defines a recursive function called fibonacci *)
  if n < 2 then             (* checks the base cases *)
    n                       (* returns n for n < 2 *)
  else                      (* if n is 2 or greater, we apply the recursive steps *)
    fibonacci (n-1) + fibonacci (n-2)   (* recursively calls the function with n-1 and n-2, then adds the results *)
