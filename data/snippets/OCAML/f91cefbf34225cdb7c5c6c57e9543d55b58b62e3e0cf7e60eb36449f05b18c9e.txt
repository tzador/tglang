let rec factorial n =           (* This function calculates the factorial of a number *)
  if n = 0 then 1               (* If n is 0, the factorial is 1 *)
  else n * factorial (n-1)      (* Otherwise, the factorial is n multiplied by factorial of n-1 *)
in                              (* End of function *)
print_string "Enter a number: "; (* Asks the user to enter a number *)
let num = read_int() in         (* Reads the input and stores it in variable num *)
print_string "The factorial of "; (* Prints the message *)
print_int num;                   (* Prints the number entered by the user *)
print_string " is ";             (* Prints an in between string *)
print_int (factorial num);       (* Calculates the factorial and prints the result *)
print_string "."                 (* Prints a period *)
