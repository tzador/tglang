let rec fibonacci n = (* Recursive function to calculate the nth Fibonacci number *)
  match n with (* Pattern matching based on the input *)
    | 0 -> 0 (* Base case: if n is 0, return 0 *)
    | 1 -> 1 (* Base case: if n is 1, return 1 *)
    | _ -> fibonacci (n-1) + fibonacci (n-2) (* Recursive call on previous two numbers *)
  
let () = (* Main function *)
  Printf.printf "Enter the value of n: "; (* Print message to enter input *)
  let n = read_int () in (* Read input from user *)
  Printf.printf "Fibonacci number at position %d is %d\n" n (fibonacci n) (* Print result *)
