<!--This is a sample HTML code snippet with various HTML elements and attributes-->

<!DOCTYPE html>  <!--This declares the document type as HTML-->

<html>  <!--This is the opening tag for the HTML document-->

<head>  <!--This is the head element of the HTML document-->

    <title>This is the title of the webpage</title>  <!--This is the title element, which displays the title of the webpage in the browser's tab-->
    
    <meta charset="UTF-8">  <!--This is a meta tag that specifies the character encoding for the webpage-->
    
    <meta name="description" content="This is a description of the webpage">  <!--This is a meta tag that provides a brief description of the webpage-->
    
    <meta name="keywords" content="HTML, code snippet, annotated, markup language">  <!--This is a meta tag that specifies keywords for the webpage, which helps with search engine optimization-->
    
    <style>  <!--This is the style element, which is used to add CSS styles to the webpage-->
    
    body {  <!--This is a CSS selector, which targets the body element of the HTML document-->
        font-family: Arial, sans-serif;  <!--This is a CSS property, which specifies the font family for the body-->
        background-color: #f2f2f2;  <!--This is a CSS property, which specifies the background color for the body-->
    }
    
    h1 {  <!--This is a CSS selector, which targets the h1 element of the HTML document-->
        color: #003366;  <!--This is a CSS property, which specifies the text color for the h1-->
        font-size: 24px;  <!--This is a CSS property, which specifies the font size for the h1-->
        font-weight: bold;  <!--This is a CSS property, which specifies the font weight for the h1-->
        text-align: center;  <!--This is a CSS property, which specifies the text alignment for the h1-->
    }
    
    </style>

</head>

<body>  <!--This is the body element of the HTML document-->

    <h1>This is a heading</h1>  <!--This is the h1 element, which displays a heading on the webpage-->
    
    <p>This is a sample paragraph.</p>  <!--This is the p element, which is used to display a paragraph on the webpage-->
    
    <a href="https://www.example.com">This is a link to example.com</a>  <!--This is the a element, which is used to create a link-->
    
    <img src="image.jpg" alt="This is an image">  <!--This is the img element, which is used to display an image on the webpage-->
    
    <ul>  <!--This is the ul element, which creates an unordered list-->
        <li>List item 1</li>  <!--This is a list item within the ul element-->
        <li>List item 2</li>
        <li>List item 3</li>
    </ul>
    
    <ol>  <!--This is the ol element, which creates an ordered list-->
        <li>List item 1</li>
        <li>List item 2</li>
        <li>List item 3</li>
    </ol>
    
    <table>  <!--This is the table element, which creates a table-->
        <tr>  <!--This is a table row-->
            <th>Name</th>  <!--This is a table header, which displays the first column-->
            <th>Age</th>
            <th>Country</th>
        </tr>
        <tr>
            <td>John</td>  <!--This is a table data, which displays the name of the person-->
            <td>25</td>
            <td>USA</td>
        </tr>
        <tr>
            <td>Jane</td>
            <td>30</td>
            <td>Canada</td>
        </tr>
    </table>

</body>

</html>  <!--This is the closing tag for the HTML document-->
