<!DOCTYPE html> <!-- Declares that this is an HTML document -->
<html> <!-- Opening tag for the HTML document -->
<head> <!-- Contains metadata about the document -->
  <title>Sample Page</title> <!-- Sets the title of the page -->
  <link rel="stylesheet" type="text/css" href="style.css"> <!-- Links to an external stylesheet -->
  <style> <!-- Defines internal styles for the page -->
    h1 {
      color: #ff0000; /* Sets the color of the header to red */
    }
  </style>
</head> 
<body> <!-- Contains the visible content of the document -->
  <h1>Hello World!</h1> <!-- Displays a header with the text "Hello World!" -->
  <p>This is a sample paragraph.</p> <!-- Displays a paragraph with sample content -->
  <img src="image.jpg" alt="Sample Image"> <!-- Inserts an image with alternate text -->
  <a href="https://www.example.com">Click here</a> <!-- Creates a link to a website -->
  <table> <!-- Creates a table with rows and columns -->
    <tr> <!-- Defines a row in the table -->
      <th>Header 1</th> <!-- Creates a header cell -->
      <th>Header 2</th> <!-- Creates another header cell -->
    </tr>
    <tr>
      <td>Cell 1</td> <!-- Creates a data cell in the first column -->
      <td>Cell 2</td> <!-- Creates a data cell in the second column -->
    </tr>
    <tr>
      <td>Cell 3</td> <!-- Creates a data cell in the first column of the second row -->
      <td>Cell 4</td> <!-- Creates a data cell in the second column of the second row -->
    </tr>
  </table>
  <script type="javascript"> <!-- Adds JavaScript code to the page -->
    alert("This is a sample alert."); // Displays an alert with the given message
  </script>
</body>
</html> <!-- Closing tag for the HTML document -->
