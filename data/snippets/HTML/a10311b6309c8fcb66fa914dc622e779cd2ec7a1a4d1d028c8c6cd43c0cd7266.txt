<!DOCTYPE html> 
<!-- This is the start of an HTML document -->
<html> 
<!-- The 'html' tag represents the root element of an HTML document -->
  <head> 
    <!-- The 'head' tag contains meta information about the document -->
    <title>Sample HTML Document</title> 
    <!-- The 'title' tag specifies the document title -->
    <meta charset="UTF-8"> 
    <!-- This meta tag specifies the character encoding for the document -->
    <meta name="description" content="This is a sample HTML document."> 
    <!-- This meta tag specifies a description of the document -->
    <meta name="keywords" content="HTML, code, snippet, complex"> 
    <!-- This meta tag specifies keywords for the document -->
    <meta name="author" content="John Doe"> 
    <!-- This meta tag specifies the author of the document -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> 
    <!-- This meta tag specifies the viewport for the document -->
  </head> 
  <body> 
    <!-- The 'body' tag represents the content of an HTML document -->
    <header> 
      <!-- The 'header' tag represents the header section of a document -->
      <nav> 
        <!-- The 'nav' tag represents a navigation section -->
        <ul> 
          <!-- The 'ul' tag represents an unordered list -->
          <li><a href="#section1">Section 1</a></li> 
          <!-- The 'li' tag represents a list item -->
          <li><a href="#section2">Section 2</a></li> 
        </ul> 
      </nav> 
    </header> 
    <main> 
      <!-- The 'main' tag represents the main content of a document -->
      <h1 id="section1">Section 1</h1> 
      <!-- The 'h1' tag represents a heading -->
      <p>This is some sample text in Section 1.</p> 
      <!-- The 'p' tag represents a paragraph -->
      <h2 id="section2">Section 2</h2> 
      <!-- The 'h2' tag represents a heading -->
      <p>This is some more sample text in Section 2.</p> 
    </main> 
    <footer> 
      <!-- The 'footer' tag represents the footer section of a document -->
      <p>Copyright Â© 2021 John Doe.</p> 
      <!-- The 'p' tag represents a paragraph -->
    </footer> 
  </body> 
</html>
