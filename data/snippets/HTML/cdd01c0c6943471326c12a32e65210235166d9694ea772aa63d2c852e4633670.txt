<!-- This is a comment in HTML -->

<!DOCTYPE html> <!-- This is a declaration of the document type -->

<html> <!-- This tag signifies start of the HTML document -->

<head> <!-- This tag contains meta information about the document -->

    <title>My Complex HTML Document</title> <!-- The title tag specifies the title of the document -->

    <meta name="description" content="A complex HTML document for demonstration purposes"> <!-- This meta tag contains a description of the document -->

    <link rel="stylesheet" type="text/css" href="style.css"> <!-- This link tag is used to include an external style sheet -->

</head>

<body> <!-- This tag contains the visible content of the document -->

    <h1>Heading 1</h1> <!-- This tag specifies a heading 1 element -->
    
    <div class="container"> <!-- This div tag creates a container with the class "container" -->

        <p>This is a paragraph inside a container.</p> <!-- This p tag contains a paragraph element inside the container div -->

        <ul> <!-- This tag specifies an unordered list -->

            <li>List item 1</li> <!-- This tag specifies a list item 1 -->
            <li>List item 2</li> <!-- This tag specifies a list item 2 -->
            <li>List item 3</li> <!-- This tag specifies a list item 3 -->

        </ul>

    </div>

    <a href="#section">Link to a section</a> <!-- This tag creates a link to the section with the id "section" -->

    <img src="image.jpg" alt="An image"> <!-- This tag specifies an image with the file name "image.jpg" and an alternative text "An image" -->

</body>

</html>
