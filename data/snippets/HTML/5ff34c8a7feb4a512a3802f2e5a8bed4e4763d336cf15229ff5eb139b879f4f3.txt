<!-- This is a comment in HTML -->
<!DOCTYPE html> <!-- This defines the document type -->
<html> <!-- This is the root element of an HTML document -->
<head> <!-- This element contains metadata about the document -->
  <title>My Website</title> <!-- This is the title of the webpage -->
  <link rel="stylesheet" href="style.css"> <!-- This links to an external stylesheet -->
</head>
<body> <!-- This is the body of the HTML document -->
  <h1>Welcome to my website!</h1> <!-- This is a heading element -->
  <p>This is a paragraph element.</p> <!-- This is a paragraph element -->
  <ul> <!-- This is an unordered list element -->
    <li>List item 1</li> <!-- This is a list item element -->
    <li>List item 2</li> <!-- This is a list item element -->
  </ul>
  <table> <!-- This is a table element -->
    <thead> <!-- This is the table head element -->
      <tr> <!-- This is a table row element -->
        <th>Column 1</th> <!-- This is a table header element -->
        <th>Column 2</th> <!-- This is a table header element -->
      </tr>
    </thead>
    <tbody> <!-- This is the table body element -->
      <tr> <!-- This is a table row element -->
        <td>Row 1, Column 1</td> <!-- This is a table data element -->
        <td>Row 1, Column 2</td> <!-- This is a table data element -->
      </tr>
      <tr> <!-- This is a table row element -->
        <td>Row 2, Column 1</td> <!-- This is a table data element -->
        <td>Row 2, Column 2</td> <!-- This is a table data element -->
      </tr>
    </tbody>
  </table>
  <form> <!-- This is a form element -->
    <label for="name">Name:</label> <!-- This is a label for a form input -->
    <input type="text" id="name" name="name"> <!-- This is a text input element -->
    <label for="email">Email:</label> <!-- This is a label for a form input -->
    <input type="email" id="email" name="email"> <!-- This is an email input element -->
    <input type="submit" value="Submit"> <!-- This is a submit button element -->
  </form>
  <script src="script.js"></script> <!-- This links to an external JavaScript file -->
</body>
