<!-- This is a comment in HTML -->
<!DOCTYPE html> <!-- Declaration of the HTML document type -->
<html lang="en"> <!-- Opening tag for the HTML document and declaring the language as English -->

<head> <!-- Opening tag for the head section -->
	<meta charset="UTF-8"> <!-- Character encoding for the document set to UTF-8 -->
	<title>My Website</title> <!-- Title of the webpage -->
	<link rel="stylesheet" href="styles.css"> <!-- Link to an external stylesheet -->
	<script src="script.js"></script> <!-- Link to an external JavaScript file -->
	<style> <!-- Opening tag for the style section -->
		body { /* CSS selector for the body element */
			background-color: #ffffff; /* Background color set to white */
		}
		h1 { /* CSS selector for the h1 element */
			font-size: 36px; /* Font size set to 36 pixels */
			color: #000000; /* Font color set to black */
		}
	</style> <!-- Closing tag for the style section -->
</head> <!-- Closing tag for the head section -->

<body> <!-- Opening tag for the body section -->
	<h1>Hello World!</h1> <!-- Heading with the text "Hello World!" -->
	<p>This is a paragraph.</p> <!-- Paragraph with text "This is a paragraph." -->
	
	<div> <!-- Opening tag for a div element -->
		<img src="image.jpg" alt="Image"> <!-- Image element with a source link and alternate text -->
		<br> <!-- Line break -->
		<a href="https://www.example.com">Link</a> <!-- Anchor element with a link to example.com -->
		<ul> <!-- Opening tag for an unordered list -->
			<li>List item 1</li> <!-- List item with text "List item 1" -->
			<li>List item 2</li> <!-- List item with text "List item 2" -->
		</ul> <!-- Closing tag for the unordered list -->
	</div> <!-- Closing tag for the div element -->
</body> <!-- Closing tag for the body section -->

</html> <!-- Closing tag for the HTML document -->
