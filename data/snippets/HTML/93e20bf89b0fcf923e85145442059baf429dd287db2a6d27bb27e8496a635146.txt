<!-- This is a comment in HTML -->

<!DOCTYPE html>

<html>
<!-- The opening and closing tags for the html document -->

<head>
<!-- The opening and closing tags for the head element -->

  <title>My Website</title> <!-- The title of the webpage -->

  <script type="text/javascript">
    // This is a JavaScript code snippet inside the head element
    // The opening and closing tags for the script element
    console.log("Hello, world!"); // Prints "Hello, world!" in the console
  </script>
  <style>
  /* This is a CSS code snippet inside the head element */
    h1 { /* Selects all h1 tags */
      color: blue; /* Changes the text color to blue */
      font-size: 30px; /* Changes the font size to 30px */
    }
  </style>
</head>

<body>
<!-- The opening and closing tags for the body element -->
  <h1>Hello, World!</h1> <!-- A h1 heading with the text "Hello, World!" -->
  <p>This is a paragraph</p> <!-- A paragraph with some text -->

  <img src="image.jpg" alt="An image"> <!-- An image element with a source and alt attribute -->
  <a href="https://www.example.com">Link</a> <!-- A link element with a URL destination and text -->
</body>

</html>
