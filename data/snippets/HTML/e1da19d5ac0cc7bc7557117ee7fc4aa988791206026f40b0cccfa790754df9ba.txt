<!DOCTYPE html>
<html>
  <head>
    <title>This is a test page</title>
    <!-- External CSS stylesheet -->
    <link rel="stylesheet" href="style.css">
    <!-- Internal JavaScript code -->
    <script>
      // Function to toggle visibility
      function toggleVisibility(elementId) {
        var e = document.getElementById(elementId);
        if (e.style.display === 'none') {
          e.style.display = 'block';
        } else {
          e.style.display = 'none';
        }
      }
    </script>
  </head>
  <body>
    <h1>Test Page</h1>

    <!-- Inline style attribute -->
    <div style="width: 500px; border: 1px solid black; padding: 10px;">
        <p>This is an example of inline style attribute.</p>
    </div>

    <!-- Form element -->
    <form>
      <!-- Label and input element -->
      <label for="username">Username:</label>
      <input type="text" id="username" name="username">

      <!-- Label and input element with placeholder attribute -->
      <label for="password">Password:</label>
      <input type="password" id="password" name="password" placeholder="Enter your password">

      <!-- Button element to trigger function on click -->
      <button onclick="toggleVisibility('output')">Submit</button>
    </form>

    <!-- Output element, hidden by default -->
    <div id="output" style="display: none;">
      <p>Output:</p>
      <!-- Paragraph element with dynamic text generated by JavaScript -->
      <p id="result"></p>
    </div>
  </body>
</html>
