<!DOCTYPE html>
<!-- This is the start of an HTML document -->

<html>
<!-- The <html> tag denotes the start of an HTML document -->

<head>
  <!-- The <head> tag contains meta information about the document, such as the document title -->
  <title>My Website</title>
  
  <!-- This is a comment, it is not displayed in the web browser -->
  
  <link rel="stylesheet" type="text/css" href="style.css">
  <!-- The <link> tag is used to link to an external CSS file -->
  <!-- The "rel" attribute specifies the relationship between the current document and the linked document -->
  <!-- The "type" attribute specifies the type of external resource being linked to -->
  <!-- The "href" attribute specifies the location of the linked resource -->
  
  <script type="text/javascript" src="script.js"></script>
  <!-- The <script> tag is used to link to an external JavaScript file -->
  <!-- The "type" attribute specifies the type of script -->
  <!-- The "src" attribute specifies the location of the script file -->
  
</head>

<body>
  <!-- The <body> tag contains all the visible content of the document -->
  
  <h1>Welcome to my website!</h1>
  <!-- The <h1> tag is used for headings, with <h1> being the largest size -->
  
  <p>This is a paragraph.</p>
  <!-- The <p> tag is used for paragraphs -->
  
  <ul>
    <!-- The <ul> tag is used for an unordered list -->
    
    <li>Item 1</li>
    <!-- The <li> tag is used for list items -->
    
    <li>Item 2</li>
    <li>Item 3</li>
    
  </ul>
  
  <ol>
    <!-- The <ol> tag is used for an ordered list -->
    
    <li>Item 1</li>
    <li>Item 2</li>
    <li>Item 3</li>
    
  </ol>
  
  <a href="https://www.example.com">Link</a>
  <!-- The <a> tag is used for links -->
  <!-- The "href" attribute specifies the location the link goes to -->
  
  <img src="image.png" alt="Image">
  <!-- The <img> tag is used for images -->
  <!-- The "src" attribute specifies the location of the image file -->
  <!-- The "alt" attribute specifies the alternate text for the image, in case the image cannot be displayed -->
  
  <table>
    <!-- The <table> tag is used for tables -->
    
    <tr>
      <!-- The <tr> tag is used for table rows -->
      
      <th>First Name</th>
      <!-- The <th> tag is used for table headers -->
      
      <th>Last Name</th>
      
    </tr>
    
    <tr>
      <td>John</td>
      <!-- The <td> tag is used for table cells -->
      
      <td>Doe</td>
      
    </tr>
    
    <tr>
      <td>Jane</td>
      <td>Smith</td>
    </tr>
    
  </table>
  
</body>

</html>
