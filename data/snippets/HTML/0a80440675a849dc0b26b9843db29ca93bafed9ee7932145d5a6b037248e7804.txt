<!DOCTYPE html> <!-- This is the doctype declaration, specifying the type of document -->

<html> <!-- This is the opening tag for the root element, the beginning of the HTML document -->

<head> <!-- This is the opening tag for the head element, where the document's metadata is contained -->

<title>My Website</title> <!-- This is the title of the HTML document -->

<!-- This is a comment, used to provide notes or explanations within the HTML code -->

<meta charset="utf-8"> <!-- This is a meta tag, specifying the character encoding of the document -->

<link rel="stylesheet" href="styles.css"> <!-- This is a link tag, linking an external stylesheet to the document -->

<style> <!-- This is the opening tag for the style element, where CSS code can be written -->

body { /* This is a CSS rule, targeting the body element */
    font-family: sans-serif; /* This is a CSS property, setting the font family of the body text */
    background-color: #F5F5F5; /* This is a CSS property, setting the background color of the body */
}

h1 { /* This is a CSS rule, targeting the heading element */
    color: #0047AB; /* This is a CSS property, setting the color of the heading text */
    font-size: 32px; /* This is a CSS property, setting the font size of the heading text */
    text-align: center; /* This is a CSS property, setting the text alignment of the heading */
}

p { /* This is a CSS rule, targeting the paragraph element */
    line-height: 1.5; /* This is a CSS property, setting the line height of the paragraph text */
    color: #333; /* This is a CSS property, setting the color of the paragraph text */
}

</style> <!-- This is the closing tag for the style element -->

</head> <!-- This is the closing tag for the head element -->

<body> <!-- This is the opening tag for the body element, where the visible content of the document is contained -->

<h1>Welcome to My Website</h1> <!-- This is a heading element, displaying the heading text -->

<p>This is a paragraph element, containing some text to display on the website.</p> <!-- This is a paragraph element, displaying the paragraph text -->

</body> <!-- This is the closing tag for the body element -->

</html> <!-- This is the closing tag for the root element, the end of the HTML document -->
