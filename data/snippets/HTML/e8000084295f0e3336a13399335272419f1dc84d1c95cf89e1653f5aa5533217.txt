<!DOCTYPE html> <!--Declare document type-->
<html> <!--Opening tag for HTML document-->
<head> <!--Opening tag for document header-->
    <title>This is a long and complex code snippet</title> <!--Title of the document-->
</head> <!--Closing tag for document header-->
<body> <!--Opening tag for document body-->
    <!-- A navigation bar using a list and anchor tags -->
    <nav> <!--Opening tag for navigation bar-->
        <ul> <!--Opening tag for unordered list-->
            <li><a href="#">Home</a></li> <!--List item for home page-->
            <li><a href="#">About</a></li> <!--List item for about page-->
            <li><a href="#">Contact</a></li> <!--List item for contact page-->
        </ul> <!--Closing tag for unordered list-->
    </nav> <!--Closing tag for navigation bar-->
    <div> <!--Opening tag for div element-->
        <h1>This is a heading</h1> <!--First level heading-->
        <h2>This is a subheading</h2> <!--Second level heading-->
        <p>This is a paragraph with <strong>bold</strong> and <em>italic</em> text.</p> <!--Paragraph with bold and italic text-->
        <!-- An image with a caption -->
        <figure> <!--Opening tag for figure element-->
            <img src="image.jpg" alt="Example Image"> <!--Image with source and alternative text-->
            <figcaption>This is an example image</figcaption> <!--Caption for image-->
        </figure> <!--Closing tag for figure element-->
    </div> <!--Closing tag for div element-->
    <!-- An unordered list with nested lists -->
    <ul> <!--Opening tag for unordered list-->
        <li>List item 1</li> <!--First list item-->
        <li>List item 2</li> <!--Second list item-->
        <li>List item 3
            <ul> <!--Nested unordered list-->
                <li>Nested list item 1</li> <!--First nested list item-->
                <li>Nested list item 2</li> <!--Second nested list item-->
                <li>Nested list item 3</li> <!--Third nested list item-->
            </ul> <!--Closing tag for nested unordered list-->
        </li> <!--Closing tag for third list item-->
        <li>List item 4</li> <!--Fourth list item-->
    </ul> <!--Closing tag for unordered list-->
    <!-- A table with header, body, and footer -->
    <table> <!--Opening tag for table element-->
        <thead> <!--Opening tag for table header-->
            <tr> <!--Table row for header-->
                <th>Column 1</th> <!--First column header-->
                <th>Column 2</th> <!--Second column header-->
            </tr> <!--Closing tag for table row-->
        </thead> <!--Closing tag for table header-->
        <tbody> <!--Opening tag for table body-->
            <tr> <!--Table row for body-->
                <td>Cell 1</td> <!--First cell of body-->
                <td>Cell 2</td> <!--Second cell of body-->
            </tr> <!--Closing tag for table row-->
            <tr> <!--Table row for body-->
                <td>Cell 3</td> <!--Third cell of body-->
                <td>Cell 4</td> <!--Fourth cell of body-->
            </tr> <!--Closing tag for table row -->
        </tbody> <!--Closing tag for table body-->
        <tfoot> <!--Opening tag for table footer-->
            <tr> <!--Table row for footer-->
                <th colspan="2">Footer</th> <!--Footer spanning two columns-->
            </tr> <!--Closing tag for table row-->
        </tfoot> <!--Closing tag for table footer-->
    </table> <!--Closing tag for table element-->
</body> <!--Closing tag for document body-->
</html> <!--Closing tag for HTML document-->
