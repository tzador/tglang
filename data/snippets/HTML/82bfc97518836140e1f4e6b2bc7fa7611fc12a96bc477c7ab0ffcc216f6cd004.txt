<!DOCTYPE html>
<html> <!-- Opening tag for the html element -->
  <head> <!-- Opening tag for the head element -->
    <title>My Website</title> <!-- Title element with content "My Website" -->
    <meta charset="UTF-8"> <!-- Meta tag specifying the character encoding -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- Meta tag for responsive design -->
    <style> <!-- Opening tag for style element -->
      body {
        background-color: #ebebeb; /* CSS property for background color with value in hex code */
        font-family: Arial, sans-serif; /* CSS property for font family with multiple values */
        margin: 50px; /* CSS property for margin with value in pixels */
        padding: 20px; /* CSS property for padding with value in pixels */
      }

      h1 {
        font-size: 36px; /* CSS property for font size with value in pixels */
        text-align: center; /* CSS property for text alignment */
        color: #333; /* CSS property for text color with value in hex code */
      }

      p {
        font-size: 18px; /* CSS property for font size with value in pixels */
        line-height: 1.5; /* CSS property for line height with value as a multiple */
      }
    </style> <!-- Closing tag for style element -->
  </head> <!-- Closing tag for head element -->
  <body> <!-- Opening tag for body element -->
    <h1>Welcome to my website!</h1> <!-- h1 element with content "Welcome to my website!" -->
    <p>This is a sample paragraph on my website.</p> <!-- p element with sample content -->
    <hr> <!-- Horizontal rule element -->
    <a href="https://www.example.com">Visit my website!</a> <!-- Anchor element with href attribute linking to example URL and anchor text as "Visit my website!" -->
  </body> <!-- Closing tag for body element -->
</html> <!-- Closing tag for html element -->
