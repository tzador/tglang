<!-- This is a comment in HTML code -->
<!DOCTYPE html> <!-- Specifies the document type to be HTML -->
<html> <!-- The root element of an HTML document -->
<head> <!-- Contains metadata about the document -->
    <title>Sample HTML Page</title> <!-- Specifies the document title -->
    <style type="text/css">  /* Defines a style sheet for the document */
        body {  /* Selects the <body> element */
            background-color: #eee; /* Sets the background color to light gray */
        }
        h1 { /* Selects all <h1> elements */
            color: blue; /* Sets the text color to blue */
            font-size: 32px; /* Sets the font size to 32 pixels */
        }
    </style>
</head>
<body> <!-- Contains the visible content of the HTML document -->
    <h1>This is a heading</h1> <!-- Defines a heading with level 1 -->
    <p>This is a paragraph.</p> <!-- Defines a paragraph -->
    <ul> <!-- Represents an unordered list -->
        <li>Item 1</li> <!-- Defines a list item -->
        <li>Item 2</li>
        <li>Item 3</li>
    </ul>
    <table> <!-- Represents an HTML table -->
        <tr> <!-- Represents a table row -->
            <th>Column 1</th> <!-- Defines a table header cell -->
            <th>Column 2</th>
        </tr>
        <tr>
            <td>Row 1, Column 1</td> <!-- Defines a table data cell -->
            <td>Row 1, Column 2</td>
        </tr>
        <tr>
            <td>Row 2, Column 1</td>
            <td>Row 2, Column 2</td>
        </tr>
    </table>
</body>
</html>
