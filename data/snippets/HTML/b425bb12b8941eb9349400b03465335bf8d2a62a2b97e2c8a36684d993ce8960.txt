  </head> <!-- This is a closing tag denoting the end of the head section -->

    <h1>Welcome to my website!</h1> <!-- This is an opening and closing tag denoting a heading in the HTML page -->

    <img src="images/example.png" alt="Example Image"> <!-- This is an image tag used for displaying an image in the HTML page -->
    <ul> <!-- This is an opening tag denoting the start of an unordered list -->
      <li>Item 1</li> <!-- This is a list item in an unordered list -->
      <li>Item 2</li> <!-- This is a list item in an unordered list -->
      <li>Item 3</li> <!-- This is a list item in an unordered list -->
    </ul> <!-- This is a closing tag denoting the end of the unordered list -->
    <ol> <!-- This is an opening tag denoting the start of an ordered list -->
      <li>Item A</li> <!-- This is a list item in an ordered list -->
      <li>Item B</li> <!-- This is a list item in an ordered list -->
      <li>Item C</li> <!-- This is a list item in an ordered list -->
    </ol> <!-- This is a closing tag denoting the end of the ordered list -->
  </body> <!-- This is a closing tag denoting the end of the body section -->
