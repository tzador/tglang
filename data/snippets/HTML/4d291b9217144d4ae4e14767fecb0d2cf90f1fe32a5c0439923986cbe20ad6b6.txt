<!DOCTYPE html> <!-- Defines the document type as HTML -->
<html> <!-- Opens the HTML document -->
<head> <!-- Defines the document's head section -->
    <title>My Website</title> <!-- Sets the document's title -->
    <meta charset="UTF-8"> <!-- Sets the character encoding for the document -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- Sets the viewport for responsive design -->
    <style> <!-- Defines the document's style(s) -->
        body {
            background-color: #F2F2F2; /* Sets the background color to light grey */
            font-family: Arial, sans-serif; /* Sets the font for the document */
            color: #333333; /* Sets the font color to dark grey */
        }
        h1 {
            font-size: 32px; /* Sets the font size for h1 headings */
            text-align: center; /* Centers the heading */
            font-weight: bold; /* Sets the font weight to bold */
            text-decoration: underline; /* Adds an underline to the heading */
        }
        p {
            font-size: 18px; /* Sets the font size for paragraphs */
            line-height: 1.5; /* Sets the spacing between lines */
            margin-bottom: 20px; /* Sets the bottom margin to create space between paragraphs */
        }
    </style>
</head>
<body> <!-- Defines the document's body section -->
    <h1>Welcome to My Website</h1> <!-- Creates a heading -->
    <p>This is a paragraph on my website. It contains some information about what the website is about and what visitors can expect to find on it.</p> <!-- Creates a paragraph -->
    <img src="image.jpg" alt="Example Image"> <!-- Inserts an image with alternative text -->
    <ul> <!-- Creates an unordered list -->
        <li>Item 1</li> <!-- Creates a list item -->
        <li>Item 2</li>
        <li>Item 3</li>
    </ul>
    <table> <!-- Creates a table -->
        <tr> <!-- Creates a table row -->
            <th>Column 1</th> <!-- Creates a table header -->
            <th>Column 2</th>
            <th>Column 3</th>
        </tr>
        <tr>
            <td>Row 1, Column 1</td> <!-- Creates a table cell -->
            <td>Row 1, Column 2</td>
            <td>Row 1, Column 3</td>
        </tr>
        <tr>
            <td>Row 2, Column 1</td>
            <td>Row 2, Column 2</td>
            <td>Row 2, Column 3</td>
        </tr>
    </table>
</body>
</html>
