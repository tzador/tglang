<!DOCTYPE html> <!-- Declaring the document type as HTML -->
<html lang="en"> <!-- Opening the root HTML element with a language attribute of "en" -->
<head> <!-- Opening the head element -->
    <meta charset="UTF-8"> <!-- Setting the document's character encoding to UTF-8 -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- Setting the viewport for responsive design -->
    <title>Code Snippet</title> <!-- Setting the document title -->
</head> <!-- Closing the head element -->
<body> <!-- Opening the body element -->
    <h1>This is a Code Snippet</h1> <!-- Creating a heading with the text "This is a Code Snippet" -->
    <p>This snippet shows how to generate a complex code snippet using HTML.</p> <!-- Creating a paragraph with instructional text -->
    <div> <!-- Opening a div element -->
        <ul> <!-- Opening an unordered list element -->
            <li>First Item</li> <!-- Creating a list item with the text "First Item" -->
            <li>Second Item</li> <!-- Creating a list item with the text "Second Item" -->
            <li>Third Item</li> <!-- Creating a list item with the text "Third Item" -->
        </ul> <!-- Closing the unordered list element -->
    </div> <!-- Closing the div element -->
    <style> <!-- Opening a style element -->
        /* Styling for the page */
        body {
            font-family: Arial, sans-serif; /* Using Arial as the font for the entire page */
        }
        h1 {
            color: blue; /* Setting the heading's text color to blue */
        }
        p {
            font-size: 14px; /* Setting the font size for paragraphs to 14 pixels */
        }
        li {
            list-style-type: circle; /* Using a circle bullet point for list items */
        }
    </style> <!-- Closing the style element -->
</body> <!-- Closing the body element -->
</html> <!-- Closing the root HTML element -->
