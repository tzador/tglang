<!-- This is a comment in HTML -->
<!DOCTYPE html>
<html> <!-- The opening tag for the html document -->
<head> <!-- The head tag contains meta information -->
  <title>HTML Code Snippet</title> <!-- This is the title of the webpage -->
  <style>
    /* This is a CSS code snippet that adds styling to our HTML document */
    h1 {
      color: blue; /* Specifies the color of the heading */
      font-size: 30px; /* Specifies the font size of the heading */
      font-family: Arial, sans-serif; /* Specifies the font family of the heading */
    }
  </style>
</head>
<body> <!-- The body tag contains the visible content of the webpage -->
  <h1>Hello World!</h1> <!-- This is a heading with text "Hello World!" -->
  <p>This is a paragraph.</p> <!-- This is a paragraph with text "This is a paragraph." -->
  <ul> <!-- The unordered list tag -->
    <li>Item 1</li> <!-- List item 1 -->
    <li>Item 2</li> <!-- List item 2 -->
    <li>Item 3</li> <!-- List item 3 -->
  </ul>
  <ol> <!-- The ordered list tag -->
    <li>Item 1</li> <!-- List item 1 -->
    <li>Item 2</li> <!-- List item 2 -->
    <li>Item 3</li> <!-- List item 3 -->
  </ol>
  <img src="image.jpg" alt="Image"> <!-- This is an image tag with the source and alt attributes -->
  <a href="https://www.example.com">Link</a> <!-- This is a link tag with the href attribute -->
</body>
</html> <!-- The closing tag for the html document -->
