<!DOCTYPE html> 
<!--This is the doctype declaration, it tells the browser that this is an HTML document-->
<html lang="en"> 
<!--This is the opening tag for the HTML document, and the "lang" attribute indicates the language of the document-->
<head> 
<!--This is the head section, where we can include metadata and link to external resources-->
    <meta charset="UTF-8"> 
    <!--This meta tag sets the character encoding for the document-->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> 
    <!--This meta tag sets the viewport, which is important for responsive design-->
    <title>Code Snippet</title> 
    <!--This is the title of the page, which is displayed in the browser tab-->
    <style> 
    /*This is the opening tag for the style element, where we can write CSS code*/
        body { 
        /*This is a CSS selector, it selects the body element and applies the following CSS rules*/
            font-family: Arial, sans-serif; 
            /*This is a CSS property and value, it sets the font family for the body element*/
            background-color: #f0f0f0; 
            /*This sets the background color for the body element*/
        } 
    </style> 
</head> 
<body> 
<!--This is the opening tag for the body element, where we can write the content of the page-->
    <h1>This is a Heading</h1> 
    <!--This is a heading element, there are six levels of headings from h1 to h6-->
    <p>This is a paragraph element. Lorem ipsum dolor sit amet, consectetur adipiscing elit. Sed tincidunt leo enim, vitae consequat velit venenatis at. Maecenas suscipit ultricies sapien eu interdum.</p> 
    <!--This is a paragraph element, used for text content-->
    <ul> 
    <!--This is an unordered list element-->
        <li>Item 1</li> 
        <!--This is a list item element-->
        <li>Item 2</li> 
    </ul> 
    <!--Closing tag for the unordered list element-->
    <ol> 
    <!--This is an ordered list element-->
        <li>Item 1</li> 
        <!--This is a list item element-->
        <li>Item 2</li> 
    </ol> 
    <!--Closing tag for the ordered list element-->
    <img src="image.jpg" alt="Image"> 
    <!--This is an image element, the "src" attribute specifies the path to the image, and the "alt" attribute provides alternative text for screen readers and in case the image cannot be loaded-->
    <a href="https://www.example.com">Link to Example</a> 
    <!--This is an anchor element, which creates a link to the specified URL-->
    <table> 
    <!--This is a table element-->
        <tr> 
        <!--This is a table row element-->
            <th>Header 1</th> 
            <!--This is a table header cell element-->
            <th>Header 2</th> 
        </tr> 
        <!--Closing tag for the table row element-->
        <tr> 
            <td>Data 1</td> 
            <!--This is a table data cell element-->
            <td>Data 2</td> 
        </tr> 
        <!--Closing tag for the table row element-->
    </table> 
    <!--Closing tag for the table element-->
</body> 
</html> 
<!--Closing tag for the html document-->
