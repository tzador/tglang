<!DOCTYPE html> <!-- This is used to specify that the document is an HTML file -->
<html> <!-- This is the root element of an HTML page -->
<head> <!-- This is used to contain metadata about the HTML page -->
	<meta charset="UTF-8"> <!-- This specifies the character encoding for the HTML document -->
	<title>My Website</title> <!-- This specifies the title of the HTML page -->
	<link rel="stylesheet" href="style.css"> <!-- This links to an external CSS stylesheet for styling the HTML page -->
	<script src="script.js"></script> <!-- This links to an external JavaScript file for adding functionality to the HTML page -->
</head>
<body> <!-- This is used to contain the main content of the HTML page -->
	<header> <!-- This is used to define the header section of the HTML page -->
		<h1>Welcome to My Website!</h1> <!-- This specifies the main heading of the header section -->
	</header>
	<main> <!-- This is used to define the main section of the HTML page -->
		<section> <!-- This is used to define a section within the main section -->
			<h2>About Me</h2> <!-- This specifies the heading for the section -->
			<p>Hello, my name is John and I am a web developer.</p> <!-- This specifies the paragraph content for the section -->
		</section>
		<section> <!-- This is used to define another section within the main section -->
			<h2>My Projects</h2>
			<ul> <!-- This is used to create an unordered list of items -->
				<li>Project 1</li> <!-- Each list item represents a project -->
				<li>Project 2</li>
				<li>Project 3</li>
			</ul>
		</section>
	</main>
	<footer> <!-- This is used to define the footer section of the HTML page -->
		<p>Â© 2021 My Website</p> <!-- This specifies the copyright information for the footer -->
	</footer>
</body>
</html>
