<!DOCTYPE html> <!-- declares document type -->
<html> <!-- declares start of html document -->
  <head> <!-- declares head section -->
    <title>This is a test</title> <!-- declares title of the page -->
    <style> <!-- declares style section -->
      body { <!-- declares body element -->
        background-color: red; /* sets background color to red */
      }

      p { <!-- declares p element -->
        font-size: 12px; /* sets font size to 12 pixels */
        font-weight: bold; /* sets font weight to bold */
      }
    </style>
  </head>
  <body> <!-- declares body section -->
    <h1>Heading 1</h1> <!-- declares a heading level 1 -->
    <p>This is a test paragraph.</p> <!-- declares a paragraph -->
    <ul> <!-- declares an unordered list -->
      <li>Item 1</li> <!-- declares list item 1 -->
      <li>Item 2</li> <!-- declares list item 2 -->
      <li>Item 3</li> <!-- declares list item 3 -->
    </ul>
    <a href="www.example.com">Link</a> <!-- declares a hyperlink -->
    <img src="image.jpg" alt="Example Image"> <!-- declares an image with a source and alternate text -->
  </body>
</html> <!-- declares end of html document -->
