<!DOCTYPE html> <!--Defines the document type as HTML-->
<html> <!--Opening tag for HTML document-->
<head> <!--Opening tag for the head element-->
  <title>My Website</title> <!--Defines the title of the website-->
  <style> <!--Opening tag for the style element-->
    body {  /*Defines the style of the body element*/
      font-family: Arial, sans-serif; /*Sets the font for the body*/
      background-color: #F5F5F5; /*Sets the background color for the body*/
    }
    h1 { /*Defines the style of the h1 element */
      color: #3F51B5; /*Sets the color of the h1 element*/
      font-size: 24px; /*Sets the font size of the h1 element*/
      text-align: center; /*Aligns the text to the center of the h1 element*/
      font-weight: bold; /*Sets the font weight to bold for the h1 element*/
    }
    p { /*Defines the style of the p element*/
      color: #616161; /*Sets the color of the p element*/
      font-size: 16px; /*Sets the font size of the p element*/
      text-align: justify; /*Aligns the text in the p element to justify*/
    }
  </style> <!--Closing tag for the style element-->
</head> <!--Closing tag for the head element-->
<body> <!--Opening tag for the body element-->
  <h1>Welcome to My Website</h1> <!--Creates a heading 1 element with the text "Welcome to My Website"-->
  <p>This is a <strong>long</strong> and <em>complex</em> HTML code snippet.</p> <!--Creates a paragraph element with styled text using the strong and em elements-->
</body> <!--Closing tag for the body element-->
</html> <!--Closing tag for the HTML document-->
