<!DOCTYPE html> <!--Defines document type as HTML-->
<html> <!--Start of HTML document-->
<head> <!--Start of the head section-->
	<title>My Website</title> <!--Defines the title of the page-->
	<meta charset="UTF-8"> <!--Defines the character encoding for the page-->
	<meta name="description" content="This is my website"> <!--Defines the page's description-->
	<meta name="author" content="John Doe"> <!--Defines the author of the page-->
	<link rel="stylesheet" href="style.css"> <!--Links to external stylesheet-->
	<style> <!--Start of internal CSS-->
		h1 {color: blue;} <!--Sets color of h1 element to blue-->
		p {font-size: 20px;} <!--Sets font size for all paragraphs to 20 pixels-->
	</style> <!--End of internal CSS-->
</head> <!--End of head section-->
<body> <!--Start of body section-->
	<h1>Welcome to my website!</h1> <!--Defines main heading-->
	<p>This is an example of a paragraph in HTML.</p> <!--Defines paragraph element-->
	<ul> <!--Start of unordered list-->
		<li>Item 1</li> <!--Defines first list item-->
		<li>Item 2</li> <!--Defines second list item-->
		<li>Item 3</li> <!--Defines third list item-->
	</ul> <!--End of unordered list-->
	<a href="https://www.example.com">Click here</a> <!--Defines link element-->
	<img src="image.jpg" alt="Example Image"> <!--Defines image element with alternate text-->
	<form> <!--Start of form-->
		<label for="name">Name:</label> <!--Defines label for input field-->
		<input type="text" id="name" name="name"> <!--Defines input field for user to enter their name-->
		<input type="submit" value="Submit"> <!--Defines submit button-->
	</form> <!--End of form-->
</body> <!--End of body section-->
</html> <!--End of HTML document-->
