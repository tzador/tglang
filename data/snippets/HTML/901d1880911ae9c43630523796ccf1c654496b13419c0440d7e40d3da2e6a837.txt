<!-- This is a comment in HTML -->
<!DOCTYPE html> <!-- Declares the document as an HTML5 document -->
<html> <!-- Indicates the start of an HTML document -->
<head> <!-- Contains info about the document -->
	<title>This is a title</title> <!-- Sets the title of the document -->
	<meta charset="UTF-8"> <!-- Declares the character encoding -->
	<style> <!-- Defines the style of the document -->
		body { background-color: #F5F5F5; } <!-- Sets the background color of the document body -->
		h1 { color: #008080; font-size: 24px; } <!-- Sets the color and font size of the heading -->
		ul { list-style-type: none; } <!-- Removes default bullet points from unordered lists -->
	</style>
</head>
<body> <!-- Contains the visible content of the document -->
	<h1>This is a Heading</h1> <!-- Displays the heading -->
	<p>This is a paragraph</p> <!-- Displays a paragraph of text -->
	<ul> <!-- Starts an unordered list -->
		<li>Item 1</li> <!-- Displays the first item in the list -->
		<li>Item 2</li> <!-- Displays the second item in the list -->
	</ul> <!-- Ends the unordered list -->
	<a href="https://www.example.com">This is a link</a> <!-- Creates a hyperlink to the specified URL -->
	<img src="https://www.example.com/image.jpg" alt="Image"> <!-- Displays an image with a specified source and alternate text -->
</body>
</html> <!-- Indicates the end of an HTML document -->
