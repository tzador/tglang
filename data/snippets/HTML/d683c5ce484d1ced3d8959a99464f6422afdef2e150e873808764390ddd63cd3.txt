<!-- This is a comment in HTML -->

<!DOCTYPE html> <!--Declares the document type as HTML-->

<html> <!--Start of HTML document-->

<head> <!--Start of the head section-->
  <meta charset="UTF-8"> <!--Specifies character encoding for the document-->
  <!-- We can also add in a title for the document using the <title> tag -->
  <title>My HTML Document</title>
</head> <!--End of head section-->

<body> <!--Start of body section-->
  <h1>My Header</h1> <!--Adds a level one heading to the document-->
  <p>This is a paragraph in my document.</p> <!--Adds a paragraph to the document-->
  <ul> <!--Start of unordered list-->
    <li>Item 1</li> <!--First item in the list-->
    <li>Item 2</li> <!--Second item in the list-->
    <li>Item 3</li> <!--Third item in the list-->
  </ul> <!--End of unordered list-->
  <a href="https://example.com">Link</a> <!--Creates a link with the URL and text-->
</body> <!--End of body section-->

</html> <!--End of HTML document-->
