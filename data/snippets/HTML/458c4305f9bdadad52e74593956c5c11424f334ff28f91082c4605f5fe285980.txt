<!DOCTYPE html> <!--Defines the document type as HTML-->
<html lang="en"> <!--Start of HTML document, with language set to English-->
<head> <!--Start of head section-->
    <meta charset="UTF-8"> <!--Specifies the character encoding for the document-->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!--Specifies the viewport settings for mobile devices-->
    <title>Code Snippet</title> <!--Sets the title of the document-->
    <style> <!--Start of style section-->
        /*CSS code can be added here to style the document*/
    </style> <!--End of style section-->
</head> <!--End of head section-->
<body> <!--Start of body section-->
    
    <header> <!--Start of header section-->
        <h1>This is a code snippet!</h1> <!--Adds a heading to the header section-->
        <nav> <!--Start of navigation bar-->
            <ul> <!--Start of unordered list-->
                <li><a href="#">Home</a></li> <!--Adds a list item with a link to the home page-->
                <li><a href="#">About</a></li> <!--Adds a list item with a link to the about page-->
                <li><a href="#">Contact</a></li> <!--Adds a list item with a link to the contact page-->
            </ul> <!--End of unordered list-->
        </nav> <!--End of navigation bar-->
    </header> <!--End of header section-->
    
    <main> <!--Start of main section-->
        <h2>Here's a complex code snippet in HTML:</h2> <!--Adds a subheading-->
        <pre><code> <!--Start of preformatted text-->
            &lt;div id="container"&gt; <!--Adds a division element with an id of "container"-->
                &lt;h3&gt;This is a complex code snippet.&lt;/h3&gt; <!--Adds a heading to the division element-->
                &lt;p&gt;It includes various HTML tags and attributes.&lt;/p&gt; <!--Adds a paragraph to the division element-->
                &lt;img src="example.png" alt="Example Image"&gt; <!--Adds an image to the division element with a source and alternate text-->
            &lt;/div&gt; <!--End of division element-->
        </code></pre> <!--End of preformatted text-->
    </main> <!--End of main section-->
    
    <footer> <!--Start of footer section-->
        <p>&copy; 2021 Code Snippet</p> <!--Adds a paragraph with the current year and document name-->
    </footer> <!--End of footer section-->
    
</body> <!--End of body section-->
</html> <!--End of HTML document-->
