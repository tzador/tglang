<!DOCTYPE html> // Specifies that this is an HTML document
<html> // Opening tag for the HTML document
<head> // Opening tag for the head section
  <title>My Website</title> // Specifies the title of the document
  <meta charset="UTF-8"> // Specifies the character encoding for the document
  <link rel="stylesheet" href="style.css"> // Links the external CSS file to the HTML document
</head> // Closing tag for the head section
<body> // Opening tag for the body section
  <header> // Opening tag for the header section
    <h1>Welcome to My Website</h1> // Heading 1 element for the website title
    <nav> // Opening tag for the navigation section
      <ul> // Unordered list for navigation links
        <li><a href="#home">Home</a></li> // List item for home link
        <li><a href="#about">About</a></li> // List item for about link
        <li><a href="#services">Services</a></li> // List item for services link
        <li><a href="#contact">Contact</a></li> // List item for contact link
      </ul> // Closing tag for unordered list
    </nav> // Closing tag for navigation section
  </header> // Closing tag for header section
  <main> // Opening tag for main section
    <section id="home"> // Opening tag for home section
      <h2>Home</h2> // Heading 2 element for home section title
      <p>Welcome to my website! This is the home page.</p> // Paragraph element for introduction
    </section> // Closing tag for home section
    <section id="about"> // Opening tag for about section
      <h2>About</h2> // Heading 2 element for about section title
      <p>This website is all about me and my skills.</p> // Paragraph element for website description
    </section> // Closing tag for about section
    <section id="services"> // Opening tag for services section
      <h2>Services</h2> // Heading 2 element for services section title
      <ul> // Unordered list for services offered
        <li>Web Design</li> // List item for web design service
        <li>Graphic Design</li> // List item for graphic design service
        <li>Content Creation</li> // List item for content creation service
      </ul> // Closing tag for unordered list
    </section> // Closing tag for services section
    <section id="contact"> // Opening tag for contact section
      <h2>Contact</h2> // Heading 2 element for contact section title
      <p>You can reach me at my email or phone number.</p> // Paragraph element for contact information
    </section> // Closing tag for contact section
  </main> // Closing tag for main section
  <footer> // Opening tag for footer section
    <p>Copyright 2020 My Website.</p> // Paragraph element for copyright
  </footer> // Closing tag for footer section
</body> // Closing tag for body section
</html> // Closing tag for HTML document
