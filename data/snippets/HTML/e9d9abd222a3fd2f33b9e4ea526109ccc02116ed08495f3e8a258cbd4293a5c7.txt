<!-- This is a comment in HTML -->
<!DOCTYPE html>  <!-- Defines that this is an HTML5 document -->
<html>  <!-- Defines the root of an HTML document -->

<head>  <!-- Defines the head of the HTML document -->
  <meta charset="UTF-8"/>  <!-- Defines the character encoding for the document to be UTF-8 -->
  <title>This is a Title</title>  <!-- Sets the title of the document -->
  <style> <!-- Defines the style section of the document -->
    h1 { <!-- Styles the h1 element -->
      color: red; <!-- Sets the text color to red -->
      font-size: 24px; <!-- Sets the font size to 24 pixels -->
      text-align: center; <!-- Aligns the text to the center -->
    }
  </style>
</head>

<body>  <!-- Defines the body section of the document -->
  <div>  <!-- Defines a div element -->
    <h1>This is a Heading</h1>  <!-- Inserts a h1 element with the text "This is a Heading" -->
    <p>This is a paragraph.</p>  <!-- Inserts a p element with the text "This is a paragraph." -->
  </div>
</body>

</html>
