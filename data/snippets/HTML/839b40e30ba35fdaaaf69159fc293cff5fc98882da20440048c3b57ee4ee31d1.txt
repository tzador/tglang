<!DOCTYPE html> <!-- this is the doctype declaration for HTML5 -->

<html> <!-- opening tag for the root element -->

<head> <!-- opening tag for the head element -->

    <meta charset="UTF-8"> <!-- sets the character encoding for the document -->

    <title>My Website</title> <!-- sets the title of the website -->

    <style> <!-- opening tag for the style element -->

        /* CSS style declarations for the body element */

        body { /* opening curly bracket for the body style rules */

            font-family: Arial, sans-serif; /* sets the font family for all the text in the body */

            background-color: #ffffff; /* sets the background color of the body to white */

        } /* closing curly bracket for the body style rules */

        /* CSS style declarations for the heading element */

        h1 { /* opening curly bracket for the h1 style rules */

            color: #000000; /* sets the text color to black */

            font-size: 2em; /* sets the font size to 2em */

            text-align: center; /* aligns the text in the center */

            margin: 2em 0; /* sets a margin of 2em on the top and bottom */

        } /* closing curly bracket for the h1 style rules */

    </style> <!-- closing tag for the style element -->

</head> <!-- closing tag for the head element -->

<body> <!-- opening tag for the body element -->

    <h1>Welcome to my website!</h1> <!-- h1 heading element with text -->

    <p>This is my first HTML website. I hope you like it!</p> <!-- paragraph element with text -->

    <img src="images/my_image.png" alt="My Image"> <!-- image element with source attribute and alt text -->

    <a href="https://www.example.com">Click here to visit Example.com</a> <!-- anchor element with link and text -->

</body> <!-- closing tag for the body element -->

</html> <!-- closing tag for the root element -->
