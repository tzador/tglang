<html> <!-- Opening tag for HTML document -->
<head> <!-- Opening tag for the head section -->
  <meta charset="UTF-8"> <!-- Declaring the character encoding for the document -->
  <title>My Website</title> <!-- Declaring the title of the document -->
  <style> <!-- Opening tag for CSS styles -->
    body { /* CSS selector for the body element */
      background-color: #f2f2f2; /* CSS property for background color */
      font-family: Arial, sans-serif; /* CSS property for font family */
    }
    h1 { /* CSS selector for h1 element */
      text-align: center; /* CSS property for text alignment */
      color: #333; /* CSS property for text color */
      font-size: 36px; /* CSS property for font size */
      margin-top: 50px; /* CSS property for margin */
    }
  </style> <!-- Closing tag for CSS styles -->

  <h1>Welcome to My Website</h1> <!-- h1 element for main heading -->
  <p>This is a paragraph in my website.</p> <!-- p element for text -->
  <a href="https://www.example.com">My Link</a> <!-- a element for linking to another webpage -->
