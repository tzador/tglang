<html> <!--opening tag for html document-->
  <head> <!--opening tag for document head-->
    <title>My Website</title> <!--title tag for webpage title-->
    <meta charset="UTF-8"> <!--meta tag for declaring character encoding-->
    <style> /*opening tag for styling, CSS code can be written here*/
      body { /*selector for body element*/
        font-family: sans-serif; /*property to set font-family to sans-serif*/
      }
      h1 { /*selector for h1 heading*/
        color: blue; /*property to set color to blue*/
      }
    </style> <!--closing tag for styling-->
  </head> <!--closing tag for document head-->
  <body> <!--opening tag for document body-->
    <h1>Welcome to My Website</h1> <!--h1 heading for welcome message-->
    <p>This is a paragraph on my website.</p> <!--paragraph tag for text-->
    <ul> <!--opening tag for unordered list-->
      <li>Item 1</li> <!--list item 1-->
      <li>Item 2</li> <!--list item 2-->
      <li>Item 3</li> <!--list item 3-->
    </ul> <!--closing tag for unordered list-->
  </body> <!--closing tag for document body-->
</html> <!--closing tag for html document-->
