<!-- This is a comment in HTML -->
<!DOCTYPE html> <!-- This tag indicates the document type is HTML -->
<html>
  <head> <!-- This tag contains the header information for the webpage -->
    <title>My Website</title> <!-- This tag sets the title of the webpage -->
    <style> /* This tag is used for adding CSS styles to the HTML document */
      h1 { /* This style rule is for the <h1> tag */
        color: blue; /* This sets the color of the <h1> text to blue */
        font-size: 36px; /* This sets the font size of the <h1> text to 36px */
      }
      p { /* This style rule is for the <p> tag */
        font-size: 18px; /* This sets the font size of the <p> text to 18px */
        text-align: justify; /* This aligns the <p> text to justify */
      }
    </style>
  </head>
  <body> <!-- This tag contains the body content of the webpage -->
    <h1>Welcome to my website!</h1> <!-- This tag displays the heading text in <h1> tag -->
    <p>This is a paragraph that contains some text about my website.
      It is a place for me to showcase my skills and share my knowledge.</p> <!-- This tag adds text in <p> tag -->
  </body>
</html> <!-- This marks the end of the HTML document -->
