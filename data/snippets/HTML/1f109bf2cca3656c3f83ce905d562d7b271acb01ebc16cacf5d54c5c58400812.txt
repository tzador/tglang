<!DOCTYPE html> <!-- Specifies the file type -->
<html> <!-- Defines the start of an HTML document -->
<head> <!-- Contains metadata about the document -->
    <meta charset="UTF-8"> <!-- Specifies the character encoding -->
    <meta name="description" content="Example HTML code snippet"> <!-- Provides a description of the document -->
    <meta name="keywords" content="HTML, code, snippet"> <!-- Provides keywords for search engines -->
    <meta name="author" content="John Doe"> <!-- Identifies the author of the document -->
    <title>HTML Code Snippet</title> <!-- Specifies the title of the document -->
</head>
<body> <!-- Contains the visible content of the document -->
    <header> <!-- Defines the header section of the document -->
        <h1>This is a HTML code snippet</h1> <!-- Specifies the heading of the document -->
        <nav> <!-- Contains navigation links -->
            <ul> <!-- Defines an unordered list of links -->
                <li><a href="#">Home</a></li> <!-- Defines a link to the Home page -->
                <li><a href="#">About</a></li> <!-- Defines a link to the About page -->
                <li><a href="#">Contact</a></li> <!-- Defines a link to the Contact page -->
            </ul>
        </nav>
    </header>
    <main> <!-- Defines the main content of the document -->
        <h2>Hello World!</h2> <!-- Another heading of the document -->
        <p>This is an example of a code snippet written in HTML.</p> <!-- Paragraph of text -->
        <img src="image.jpg" alt="Example image"> <!-- Inserts an image into the document -->
        <table> <!-- Defines a table -->
            <tr> <!-- Defines a table row -->
                <th>Column 1</th> <!-- Defines a table heading cell -->
                <th>Column 2</th>
                <th>Column 3</th>
            </tr>
            <tr>
                <td>Row 1, Column 1</td> <!-- Defines a table data cell -->
                <td>Row 1, Column 2</td>
                <td>Row 1, Column 3</td>
            </tr>
            <tr>
                <td>Row 2, Column 1</td>
                <td>Row 2, Column 2</td>
                <td>Row 2, Column 3</td>
            </tr>
        </table>
    </main>
    <footer> <!-- Defines the footer section of the document -->
        <p>&copy; 2021 John Doe</p> <!-- Inserts the copyright symbol and the current year -->
    </footer>
</body>
</html>
