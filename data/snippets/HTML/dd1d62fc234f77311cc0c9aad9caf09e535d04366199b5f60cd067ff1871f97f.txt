<!-- This is a comment in HTML code -->
<!DOCTYPE html> <!-- HTML5 doctype declaration -->
<html> <!-- opening tag for the html element -->
<head> <!-- opening tag for the head element -->
  <title>Website Title</title> <!-- title tag to specify the title of the webpage -->
  <meta charset="UTF-8"> <!-- declaration of character set for the webpage -->
  <meta name="description" content="This is a description of the website"> <!-- meta tag for website description -->
  <meta name="keywords" content="HTML, code, snippet"> <!-- meta tag for website keywords -->
  <link rel="stylesheet" type="text/css" href="style.css"> <!-- link tag to link external CSS file -->
</head> <!-- closing tag for the head element -->
<body> <!-- opening tag for the body element -->
  <h1>Welcome to my website!</h1> <!-- heading tag to create a main heading on the webpage -->
  <p>This is a paragraph element</p> <!-- paragraph tag to create a paragraph element -->
  <img src="image.jpg" alt="Image Description"> <!-- img tag to insert an image on the webpage -->
  <ul> <!-- opening tag for an unordered list -->
    <li>Item 1</li> <!-- list item 1 -->
    <li>Item 2</li> <!-- list item 2 -->
    <li>Item 3</li> <!-- list item 3 -->
  </ul> <!-- closing tag for the unordered list -->
  <ol> <!-- opening tag for an ordered list -->
    <li>Item 1</li> <!-- list item 1 -->
    <li>Item 2</li> <!-- list item 2 -->
    <li>Item 3</li> <!-- list item 3 -->
  </ol> <!-- closing tag for the ordered list -->
  <a href="https://www.example.com">Click here</a> <!-- anchor tag to create a link to another webpage -->
</body> <!-- closing tag for the body element -->
</html> <!-- closing tag for the html element -->
