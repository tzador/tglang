<!-- This is a comment in HTML -->
<!DOCTYPE html> <!-- This declares the document type as HTML -->
<html> <!-- This is the opening tag for the HTML document -->
<head> <!-- This is the opening tag for the document's head section -->
  <title>My Website</title> <!-- This is the title of the document -->
  <style> <!-- This is the opening tag for the style section -->
    /* This is a CSS comment within the style section */
    h1 { /* This is a CSS selector */
      font-size: 24px; /* This is a CSS property and value */
      color: #000000; /* This is another CSS property and value */
    }
  </style> <!-- This is the closing tag for the style section -->
</head> <!-- This is the closing tag for the document's head section -->
<body> <!-- This is the opening tag for the document's body section -->
  <h1>Hello, World!</h1> <!-- This is a heading element with the text "Hello, World!" -->
  <p>This is a paragraph element with some <strong>bold</strong> and <em>italicized</em> text.</p> <!-- This is a paragraph element with nested strong and em elements for formatting -->
  <ul> <!-- This is the opening tag for an unordered list -->
    <li>Item 1</li> <!-- This is a list item element with the text "Item 1" -->
    <li>Item 2</li> <!-- This is a list item element with the text "Item 2" -->
    <li>Item 3</li> <!-- This is a list item element with the text "Item 3" -->
  </ul> <!-- This is the closing tag for the unordered list -->
  <a href="https://www.example.com">This is a link to another website</a> <!-- This is a link element with the URI attribute pointing to "https://www.example.com" and the text "This is a link to another website" -->
</body> <!-- This is the closing tag for the document's body section -->
</html> <!-- This is the closing tag for the HTML document -->
