<!-- This is an example of a complex HTML code snippet -->
<!DOCTYPE html> <!-- Indicates the type of document -->
<html> <!-- Indicates the start of the document -->
    <head> <!-- Contains information about the document -->
        <title>This is the document title</title> <!-- Sets the title of the document -->
        <meta charset="UTF-8"> <!-- Specifies the character set used in the document -->
        <link rel="stylesheet" type="text/css" href="style.css"> <!-- Links the document to an external CSS file for styling -->
    </head>
    <body> <!-- Contains the visible content of the document -->
        <h1>This is a heading</h1> <!-- Defines a heading -->
        <p>This is a paragraph</p> <!-- Defines a paragraph -->
        <ul> <!-- Defines an unordered list -->
            <li>List item 1</li> <!-- Defines a list item -->
            <li>List item 2</li>
            <li>List item 3</li>
        </ul>
        <form> <!-- Defines a form element -->
            <label for="name">Name: </label> <!-- Defines a label for an input field -->
            <input type="text" name="name" id="name" /> <!-- Defines a text input field -->
            <label for="email">Email: </label>
            <input type="email" name="email" id="email" /> <!-- Defines an email input field -->
            <label for="message">Message: </label>
            <textarea name="message" id="message"></textarea> <!-- Defines a textarea input field -->
            <input type="submit" value="Submit" /> <!-- Defines a submit button -->
        </form>
        <script src="script.js"></script> <!-- Links the document to an external JavaScript file for scripting -->
    </body>
</html>
