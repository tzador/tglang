<!-- This is a multi-line comment in HTML -->

<!DOCTYPE html> 
<!-- This declares the document type as HTML -->

<html>
<!-- The opening tag for the HTML document -->

<head>
<!-- This contains the head element of the document -->

<title>Code Snippet</title>
<!-- This sets the title of the document -->

<style>
/* This is a multi-line CSS comment */
/* This is used to add style to the document */

body {
  background-color: #fff;
  /* Sets the background color of the body to white */
  font-family: sans-serif;
  /* Sets the font family to a generic sans-serif font */
}

h1 { 
  /* Styles the h1 tag */
  color: #000;
  /* Sets the text color of h1 to black */
  text-align: center;
  /* Aligns the text in the center */
  font-size: 36px;
  /* Sets the font size to 36px */
}
</style>
</head>

<body>
<!-- This is the body element of the document -->

<h1>Code Snippet</h1>
<!-- This creates an h1 heading with the text "Code Snippet" -->

<p>

<!-- This is a paragraph with the text "This is a complex code snippet in HTML." where the word HTML is italicized with the <em> tag -->

<!-- This adds an empty line break -->

<table>
<!-- This creates a table element -->

<tr>
<!-- This creates a table row element -->

<!-- This creates a table header cell with the text "Name" -->

<!-- This creates a table header cell with the text "Language" -->
</tr>

<tr>
<td>Code Snippet</td>
<!-- This creates a table data cell with the text "Code Snippet" -->

<!-- This creates a table data cell with the text "HTML" -->
