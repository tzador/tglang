<!-- This is a comment in HTML code -->
<!DOCTYPE html> <!-- This is a document type declaration for HTML5 -->
<html> <!-- This is the opening tag for the HTML document -->
<head> <!-- This is the opening tag for the HTML document head section -->
    <meta charset="UTF-8"> <!-- This sets the character encoding to UTF-8 -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- This sets the viewport to adjust according to device width and initial scale -->
    <title>HTML Code Snippet</title> <!-- This sets the title for the HTML document -->
    <style> <!-- This is the opening tag for the HTML style section -->
        body { background-color: #fff; } <!-- This sets the background color of the document body to white -->
        p { color: #000; font-size: 14px; } <!-- This sets the font color and size for paragraph elements -->
    </style> <!-- This is the closing tag for the HTML style section -->
</head> <!-- This is the closing tag for the HTML document head section -->
<body> <!-- This is the opening tag for the HTML document body section -->
    <p>This is a sample paragraph for the HTML code snippet.</p> <!-- This is a sample paragraph within the body section -->
    <button onclick="alert('Hello World!')">Click Me</button> <!-- This creates a button with a click event that triggers an alert message -->
</body> <!-- This is the closing tag for the HTML document body section -->
</html> <!-- This is the closing tag for the HTML document -->
