<!DOCTYPE html> <!--Defines the document type as HTML-->

<html> <!--Start of HTML document-->

<head> <!--Start of head section-->
  <title>My Website</title> <!--Adds a title to the webpage-->
  <meta charset="UTF8"> <!--Specifies the character encoding of the document-->
  <meta name="description" content="My first website"> <!--Provides a description of the webpage for search engines-->
  <meta name="keywords" content="HTML, CSS, JavaScript"> <!--Specifies keywords related to the webpage for search engines-->
  
  <style> <!--Start of style section-->
    body { <!--Defines styling for the entire webpage-->
      background-color: #e6e6e6; <!--Sets the background color to light grey-->
      font-family: Arial, sans-serif; <!--Specifies the font family for the webpage-->
    }
    
    h1 { <!--Defines styling for heading level 1-->
      text-align: center; <!--Centers the heading-->
      color: #333333; <!--Sets the text color to dark grey-->
      font-size: 36px; <!--Sets the font size to 36 pixels-->
    }
    
    p { <!--Defines styling for paragraphs-->
      line-height: 1.5; <!--Sets the line height to 1.5 times the font size-->
      color: #666666; <!--Sets the text color to light grey-->
      font-size: 18px; <!--Sets the font size to 18 pixels-->
    }
  </style> <!--End of style section-->
  
</head> <!--End of head section-->

<body> <!--Start of body section-->
  <h1>Welcome to my website!</h1> <!--Heading level 1 with text-->
  
  <p>This is my first website. I am learning how to use HTML, CSS, and JavaScript.</p> <!--Paragraph with text-->
  
  <p>Here are some things I like to do:</p> <!--Paragraph with text-->
  <ul> <!--Start of unordered list-->
    <li>Read</li> <!--List item with text-->
    <li>Watch movies</li> <!--List item with text-->
    <li>Play video games</li> <!--List item with text-->
  </ul> <!--End of unordered list-->
  
  <p>Check out my <a href="https://www.example.com">other website</a> for more information.</p> <!--Paragraph with link-->
  
  <script> <!--Start of JavaScript section-->
    alert("Thanks for visiting my website!"); <!--Displays an alert message when the page is loaded-->
  </script> <!--End of JavaScript section-->
  
</body> <!--End of body section-->

</html> <!--End of HTML document-->
