<!-- This is a comment in HTML -->
<!DOCTYPE html> <!-- Defines the document type as HTML -->
<html> <!-- Starts the HTML document -->

<head> <!-- Starts the head section of the document -->
  <title>My Website</title> <!-- Sets the title of the document -->
  <style> <!-- Starts the style section for CSS -->
    body { <!-- Styles the body element -->
      font-family: sans-serif;
      background-color: #fff;
    }

    h1 { <!-- Styles the h1 element -->
      font-size: 2em;
      text-align: center;
    }

    p { <!-- Styles the p element -->
      font-size: 1.2em;
      color: #333;
    }
  </style> <!-- Closes the style section -->

</head> <!-- Closes the head section -->

<body> <!-- Starts the body section of the document -->
  <h1>Welcome to my website</h1> <!-- Creates a heading -->
  <p>This is a paragraph on my website</p> <!-- Creates a paragraph -->
  <a href="#about">Learn more about me</a> <!-- Creates a link to a section on the page -->
  
  <section id="about"> <!-- Creates a section with the ID 'about' -->
    <h2>About Me</h2>
    <p>I am a web developer with a passion for coding and creating beautiful websites.</p>
  </section>
  
  <script> <!-- Starts the script section -->
    alert("Thanks for visiting my website!"); <!-- Displays an alert box when the page is loaded -->
  </script> <!-- Closes the script section -->

</body> <!-- Closes the body section -->

</html> <!-- Closes the HTML document -->
