<!DOCTYPE html>                           <!--declaration of HTML5 doctype-->
<html lang="en">                          <!--opening tag for html element with language attribute-->
<head>                                    <!--opening tag for head element-->
  <meta charset="UTF-8">                  <!--meta tag specifying character encoding for document-->
  <meta name="viewport" content="width=device-width, initial-scale=1.0">  <!--meta tag specifying viewport behaviour-->
  <title>Document</title>                 <!--title element for page-->
</head>                                   <!--closing tag for head element-->
<body>                                    <!--opening tag for body element-->
  <h1>Hello, World!</h1>                   <!--opening and closing h1 element for heading-->
  <p>This is a paragraph.</p>              <!--opening and closing p element for paragraph-->
  <a href="#section1">Jump to Section 1</a>  <!--opening and closing a element for link to section 1-->
  <img src="image.png" alt="Image">        <!--opening and self-closing img element for image with alt attribute-->
  <ul>                                     <!--opening tag for ul element-->
    <li>List item 1</li>                   <!--opening and closing li element for list item-->
    <li>List item 2</li>                   <!--opening and closing li element for list item-->
    <li>List item 3</li>                   <!--opening and closing li element for list item-->
    <li>List item 4</li>                   <!--opening and closing li element for list item-->
  </ul>                                    <!--closing tag for ul element-->
  <form>                                   <!--opening tag for form element-->
    <label for="textInput">Enter Text:</label>  <!--opening and closing label element with for attribute for text input-->
    <input type="text" id="textInput" name="textInput">  <!--opening and self-closing input element with type, id, and name attributes-->
    <input type="submit" value="Submit">    <!--opening and self-closing input element with type and value attributes-->
  </form>                                  <!--closing tag for form element-->
</body>                                   <!--closing tag for body element-->
</html>                                   <!--closing tag for html element-->
