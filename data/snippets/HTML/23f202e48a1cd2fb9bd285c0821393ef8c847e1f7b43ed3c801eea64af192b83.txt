<!DOCTYPE html> <!-- Defines the docment type as HTML -->
<html> <!-- Opening tag for the root element of HTML document -->
<head> <!-- Opening tag for the header section of the document -->
    <title>My Website</title> <!-- Sets the title of the webpage -->
    <meta charset="UTF-8"> <!-- Specifies the character encoding for the document -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- Specifies the viewport for mobile devices -->
    <link rel="stylesheet" type="text/css" href="style.css"> <!-- Links to an external stylesheet -->
    <script src="script.js" defer></script> <!-- Links to an external JavaScript file and defers its execution -->
    <style> <!-- Opening tag for inline CSS styles -->
        body { <!-- Styles for the body element -->
            background-color: #ffffff;  <!-- Sets the background color to white -->
            font-family: Arial, sans-serif; <!-- Sets the font family to Arial, sans-serif -->
        }
        h1 { <!-- Styles for the h1 element -->
            color: #2980b9; <!-- Sets the text color to a shade of blue -->
            font-size: 36px; <!-- Sets the font size to 36px -->
        }
        p { <!-- Styles for the p element -->
            color: #2c3e50; <!-- Sets the text color to a shade of dark blue -->
            font-size: 18px; <!-- Sets the font size to 18px -->
        }
    </style> <!-- Closing tag for inline CSS styles -->
</head> <!-- Closing tag for the header section of the document -->
<body> <!-- Opening tag for the body section of the document -->
    <h1>Welcome to my website!</h1> <!-- Heading 1 element with text -->
    <p>This is a long and complex code snippet in HTML.</p> <!-- Paragraph element with text -->
    <button onclick="alert('Hello world!')">Click Me</button> <!-- Button element with onclick attribute to create an alert when clicked -->
    <img src="image.jpg" alt="A beautiful image"> <!-- Image element with a source and alternative text -->
    <ul> <!-- Opening tag for an unordered list -->
        <li>Item 1</li> <!-- List item 1 -->
        <li>Item 2</li> <!-- List item 2 -->
        <li>Item 3</li> <!-- List item 3 -->
    </ul> <!-- Closing tag for an unordered list -->
</body> <!-- Closing tag for the body section of the document -->
</html> <!-- Closing tag for the root element of the HTML document -->
