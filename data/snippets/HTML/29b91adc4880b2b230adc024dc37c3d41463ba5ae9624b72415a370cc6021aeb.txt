<html> <!-- opening tag for the HTML document -->
  <head> <!-- opening tag for the head section -->
    <title>My Website</title> <!-- title of the website -->
    <style> <!-- opening tag for the style element -->
      body { <!-- rules for the body element -->
        background-color: #fff; <!-- sets background color to white -->
        font-family: Arial, sans-serif; <!-- sets font family to Arial or a sans-serif font -->
      }
      h1 { <!-- rules for the h1 element -->
        color: #000; <!-- sets text color to black -->
        text-align: center; <!-- aligns text in the center -->
      }
      p { <!-- rules for the p element -->
        font-size: 1.2em; <!-- sets font size to 1.2em -->
        line-height: 1.5; <!-- sets line height to 1.5 times the font size -->
      }
    </style> <!-- closing tag for the style element -->
  </head> <!-- closing tag for the head section -->
  <body> <!-- opening tag for the body section -->
    <h1>Welcome to my website!</h1> <!-- heading for the website -->
    <p>This is a paragraph on my website.</p> <!-- paragraph on the website -->
  </body> <!-- closing tag for the body section -->
</html> <!-- closing tag for the HTML document -->
