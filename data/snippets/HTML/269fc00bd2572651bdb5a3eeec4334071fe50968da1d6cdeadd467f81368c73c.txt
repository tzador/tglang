<!-- This is a sample HTML code snippet -->

<!DOCTYPE html> <!-- Defines the document type as HTML -->
<html> <!-- Indicates the start of the HTML document -->

<head> <!-- Contains metadata about the HTML document -->
    <title>Sample HTML Code Snippet</title> <!-- Sets the title of the document -->
    <style> /* Starts the style block for CSS code */
        h1 { /* Creates a style for heading 1 */
            color: red; /* Sets the text color to red */
            font-size: 24px; /* Sets the font size to 24 pixels */
        }
        p { /* Creates a style for paragraphs */
            font-size: 18px; /* Sets the font size to 18 pixels */
            font-family: Arial, sans-serif; /* Sets the font family to Arial or any sans-serif font */
            line-height: 1.5; /* Sets the line height to 1.5 times the font size */
        }
        .important { /* Creates a style for elements with class "important" */
            font-weight: bold; /* Sets the font weight to bold */
        }
    </style> <!-- Ends the style block -->
</head>

<body> <!-- Contains the visible content of the HTML document -->
    <h1>This is a heading</h1> <!-- Displays a heading with text "This is a heading" -->
    <p>This is a paragraph.</p> <!-- Displays a paragraph with text "This is a paragraph" -->
    <p class="important">This is an important paragraph.</p> <!-- Displays an important paragraph with text "This is an important paragraph." -->
</body>

</html> <!-- Indicates the end of the HTML document -->
