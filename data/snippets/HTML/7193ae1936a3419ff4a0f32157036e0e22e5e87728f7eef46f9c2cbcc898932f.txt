<!-- This is a complex and long code snippet in HTML -->
<!DOCTYPE html>
<!-- The <!DOCTYPE> tag declares the document type of the HTML document -->
<html>
<!-- The <html> tag represents the root of the document -->
<head>
  <!-- The <head> tag contains metadata about the HTML document -->
  <title>My Website</title>
  <!-- The <title> tag defines the title of the document -->
  <meta charset="UTF-8">
  <!-- The <meta> tag contains metadata about the HTML document -->
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <!-- The <meta> tag specifies the viewport for responsive web design -->
  <link rel="stylesheet" href="style.css">
  <!-- The <link> tag links external style sheets to the HTML document -->
  <script src="script.js"></script>
  <!-- The <script> tag is used to define client-side JavaScript -->
  <style>
    /* This is an internal style sheet with CSS code */
    body {
      background-color: lightblue;
      /* Sets the background color of the body element */
      font-family: Arial, sans-serif;
      /* Sets the font family for all text on the page */
    }

    #title {
      font-size: 32px;
      /* Sets the font size for the element with id="title" */
      font-weight: bold;
      /* Sets the font weight for the element with id="title" */
    }
  </style>
</head>
<body>
  <!-- The <body> tag represents the content of the HTML document -->
  <h1 id="title">Welcome to my website!</h1>
  <!-- The <h1> tag defines a heading -->
  <p>This is a paragraph with <strong>strong</strong> and <em>emphasized</em> text.</p>
  <!-- The <p> tag defines a paragraph -->
  <ul>
    <li>Item 1</li>
    <li>Item 2</li>
    <li>Item 3</li>
    <!-- The <ul> tag creates an unordered list -->
  </ul>
  <img src="image.jpg" alt="An image">
  <!-- The <img> tag inserts an image with the specified source and alternative text -->
  <a href="https://www.example.com">Visit Our Website</a>
  <!-- The <a> tag defines a hyperlink -->
</body>
</html>
