<!DOCTYPE html>
<!-- The above line declares the document type as HTML -->
<html>
  <head>
    <!-- The <head> tag contains metadata about the HTML document -->
    <title>My Webpage</title>
    <!-- The <title> tag sets the title of the webpage that appears in the browser tab -->
    <meta charset="UTF-8">
    <!-- The <meta> tag sets the character encoding for the HTML document -->
    <meta name="description" content="This is a sample webpage">
    <!-- The <meta> tag sets the description of the webpage for SEO purposes -->
    <meta name="keywords" content="sample, webpage, HTML">
    <!-- The <meta> tag sets the keywords for the webpage for SEO purposes -->
    <meta name="author" content="John Doe">
    <!-- The <meta> tag sets the author of the webpage -->
    <link rel="stylesheet" href="styles.css">
    <!-- The <link> tag links the webpage to an external style sheet -->
    <script src="script.js"></script>
    <!-- The <script> tag links the webpage to an external JavaScript file -->
  </head>
  <body>
    <!-- The <body> tag contains the visible content of the webpage -->
    <h1>Welcome to My Webpage!</h1>
    <!-- The <h1> tag sets the main heading of the webpage -->
    <p>This is a sample paragraph.</p>
    <!-- The <p> tag sets a paragraph of text -->
    <img src="image.jpg" alt="Sample Image">
    <!-- The <img> tag displays an image with alternate text -->
  </body>
</html>
