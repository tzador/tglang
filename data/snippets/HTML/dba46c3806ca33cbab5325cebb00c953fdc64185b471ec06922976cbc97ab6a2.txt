<!-- This is a comment in HTML code -->

<!DOCTYPE html> <!-- Defines the document type as HTML -->

<html lang="en-US"> <!-- Defines the language of the document -->

<head> <!-- Begins the header section of the document -->
  <meta charset="UTF-8"> <!-- Defines the character encoding for the document -->
  <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- Defines the viewport for responsive design -->
  <title>My Website</title> <!-- Sets the title of the document -->
</head> <!-- Ends the header section of the document -->

<body> <!-- Begins the body section of the document -->
  <h1>Welcome to my website!</h1> <!-- Creates a heading element -->
  <p>This is a sample paragraph on my website.</p> <!-- Creates a paragraph element -->
  <a href="https://www.example.com">Click here to visit Example.com</a> <!-- Creates a link to an external website -->

  <div id="container"> <!-- Starts a container element with an id attribute set to "container" -->
    <img src="image.jpg" alt="Example Image"> <!-- Inserts an image with a source attribute and an alternative text -->
    <ul> <!-- Starts an unordered list element -->
      <li>Item 1</li> <!-- Creates a list item -->
      <li>Item 2</li> <!-- Creates a list item -->
      <li>Item 3</li> <!-- Creates a list item -->
    </ul> <!-- Ends the unordered list element -->
  </div> <!-- Ends the container element -->
</body> <!-- Ends the body section of the document -->

</html> <!-- Ends the HTML document -->
