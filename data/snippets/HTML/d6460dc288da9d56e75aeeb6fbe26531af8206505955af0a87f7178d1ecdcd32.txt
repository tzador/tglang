<html> <!-- Denotes the start of an HTML document -->
<head> <!-- Contains metadata for the document -->
<title>My Website</title> <!-- Title tag that appears in the browser tab -->
<style> <!-- Defines CSS styles for the document -->
h1 { <!-- Selects heading level 1 elements -->
  font-family: sans-serif; <!-- Specifies font for heading -->
  font-size: 32px; <!-- Specifies font size for heading -->
  color: #123456; <!-- Specifies color for heading -->
}
p { <!-- Selects paragraph elements -->
  font-family: serif; <!-- Specifies font for paragraph -->
  font-size: 16px; <!-- Specifies font size for paragraph -->
  color: #789012; <!-- Specifies color for paragraph -->
}
a { <!-- Selects anchor elements -->
  font-family: monospace; <!-- Specifies font for anchor -->
  font-size: 12px; <!-- Specifies font size for anchor -->
  color: #ababab; <!-- Specifies color for anchor -->
  text-decoration: none; <!-- Removes underline from anchor -->
}
img { <!-- Selects image elements -->
  display: block; <!-- Displays image as a block element -->
  max-width: 100%; <!-- Specifies maximum width for image -->
  height: auto; <!-- Automatically adjusts height of image -->
}
</style>
</head>
<body> <!-- Contains the visible content of the document -->
<h1>Welcome to my website!</h1> <!-- Heading level 1 element with text -->
<p>This is my first website, created with HTML and CSS.</p> <!-- Paragraph element with text -->
<a href="#about">Learn more about me</a> <!-- Anchor element with text and link to another section -->
<img src="mywebsite.jpg" alt="Image of my website"> <!-- Image element with source and alternate text attribute -->
<h2 id="about">About Me</h2> <!-- Heading level 2 element with id for linking -->
<p>I am a web developer passionate about creating beautiful and functional websites.</p> <!-- Paragraph element with text -->
</body>
</html>
