<!-- This is a comment in HTML -->

<!DOCTYPE html> <!-- This is a doctype declaration -->

<html lang="en"> 
<head> <!-- The head element contains metadata about the document -->
    <meta charset="UTF-8">  <!-- This meta tag specifies the character encoding for the document -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- This meta tag sets the viewport for responsive design -->
    <title>My Website</title> <!-- The <title> element sets the title of the document shown in the browser tab -->
</head> 

<body> <!-- The body element contains all the visible content of the document -->
    <header> <!-- The header element represents the introductory content of a page or section -->
        <nav> <!-- The nav element represents a section of a document that links to other documents or pages -->
            <ul> <!-- The ul element represents an unordered list of items, typically used in a navigation menu -->
                <li><a href="#home">Home</a></li> <!-- The li element represents a list item, and the a element creates a hyperlink to the specified URL -->
                <li><a href="#about">About</a></li>
                <li><a href="#contact">Contact</a></li>
            </ul>
        </nav>
    </header>

    <main> <!-- The main element contains the main content of the document -->
        <h1>Welcome to My Website</h1> <!-- The h1 element represents a heading of the highest importance -->
        <p>This is an example paragraph.</p> <!-- The p element represents a paragraph of text -->
        <img src="example.jpg" alt="Example Image"> <!-- The img element inserts an image into the document -->
    </main>

    <footer> <!-- The footer element represents a footer for a document or section -->
        <p>Follow us on social media!</p>
        <div class="social-media">
            <a href="https://www.facebook.com"><i class="fab fa-facebook"></i></a>
            <a href="https://www.twitter.com"><i class="fab fa-twitter"></i></a>
            <a href="https://www.instagram.com"><i class="fab fa-instagram"></i></a>
        </div>
    </footer>
</body>

</html> <!-- HTML documents must be enclosed in the <html> element with closing tags -->
