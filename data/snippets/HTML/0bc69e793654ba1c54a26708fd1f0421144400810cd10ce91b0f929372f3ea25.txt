<!DOCTYPE html>
<html> <!-- The main document container -->

<head> <!-- The head section of the document -->
  <meta charset="UTF-8">
  <title>Hello World</title>
  <style> /* Inline styles to be applied to the HTML elements */
    body {
      font-family: sans-serif; /* Setting the font for the document */
    } 
    h1 {
      color: blue; /* Changing the color of the heading */
    }
  </style>
  <script> /* Embedding JavaScript code to be executed */
    alert("Hello World"); // Displaying an alert pop-up with the message "Hello World"
  </script>
</head>

<body> <!-- The body section of the document -->
  <h1>This is my first HTML document</h1> <!-- Heading element -->
  <p>This is a paragraph element with some <strong>bold</strong> and <em>italic</em> text.</p> <!-- Paragraph element with some inline styling -->
  <ul> <!-- Unordered list element -->
    <li>List item 1</li> <!-- List item 1 -->
    <li>List item 2</li> <!-- List item 2 -->
    <li>List item 3</li> <!-- List item 3 -->
  </ul>
  
  <img src="https://example.com/image.jpg" alt="Example Image"> <!-- Image element with a source and alternate text -->
  
  <a href="https://www.example.com">Link to Example Website</a> <!-- Anchor element with a link to a website -->
</body>

</html>
