<!-- This is a comment in HTML -->
<!DOCTYPE html> <!-- Defines the document type as HTML -->
<html> <!-- The root element of an HTML document -->
<head> <!-- Contains metadata about the document -->
  <meta charset="UTF-8"> <!-- Specifies the character encoding for the document -->
  <title>Example Page</title> <!-- Defines the title of the document -->
  <style> <!-- Defines the style of the document -->
    h1 { <!-- Selects all h1 elements -->
      color: blue; <!-- Sets the color of the selected elements -->
      font-size: 20px; <!-- Sets the font size of the selected elements -->
    }
  </style>
</head>
<body> <!-- Contains the visible content of the document -->
  <h1>Title</h1> <!-- Heading element with the text "Title" -->
  <h2>Subtitle</h2> <!-- Heading element with the text "Subtitle" -->
  <p>This is an example paragraph.</p> <!-- Paragraph element with the text "This is an example paragraph." -->
  <img src="example.jpg" alt="Example image"> <!-- Image element with the source "example.jpg" and alternate text "Example image" -->
</body>
</html>
