<!-- This is a comment in HTML -->

<!DOCTYPE html> <!-- HTML document type declaration -->
<html> <!-- The root element of an HTML document -->
<head> <!-- The head element contains metadata about the document -->
<meta charset="UTF-8"> <!-- Defines the character encoding for the HTML document -->
<title>My Website</title> <!-- Defines the title of the document; displayed in the browser's title bar or page tab -->
<style> <!-- Defines styles for the HTML document -->
body { <!-- Selects and styles the body element -->
	font-family: sans-serif; <!-- Sets the font family for the body to sans-serif -->
	background-color: #ffffff; <!-- Sets the background color of the body to white -->
	color: #000000; <!-- Sets the text color of the body to black -->
}
h1 { <!-- Selects and styles all h1 elements -->
	font-size: 24px; <!-- Sets the font size of the h1 elements to 24px -->
	font-weight: bold; <!-- Sets the font weight of the h1 elements to bold -->
	text-align: center; <!-- Aligns the h1 elements to the center -->
}
p { <!-- Selects and styles all p elements -->
	font-size: 16px; <!-- Sets the font size of the p elements to 16px -->
	line-height: 1.5; <!-- Sets the line height of the p elements to 1.5 times the font size -->
	margin-bottom: 20px; <!-- Sets the bottom margin of the p elements to 20px -->
}
a { <!-- Selects and styles all a elements -->
	color: #0000ff; <!-- Sets the text color of the a elements to blue -->
	text-decoration: none; <!-- Removes the underline decoration from the a elements -->
}
</style>
</head>
<body> <!-- The body element contains the visible content of the document -->
<h1>Welcome to My Website</h1>
<p>This is a paragraph on my website. <a href="#">Click here</a> to learn more.</p>
</body>
</html>
