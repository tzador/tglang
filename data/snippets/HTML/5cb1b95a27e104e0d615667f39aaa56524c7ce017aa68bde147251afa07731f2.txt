<!DOCTYPE html> <!-- Declares the document type as HTML -->
<html> <!-- Opens the HTML document -->
<head> <!-- Head section of the document -->
<title>My Website</title> <!-- Sets the title of the webpage -->
<link rel="stylesheet" type="text/css" href="mystyle.css"/> <!-- Links external CSS stylesheet -->
<script src="myscript.js"></script> <!-- Links external JavaScript file -->
<style> <!-- Internal CSS styling -->
    h1 {
        color: blue; <!-- Sets the color of the heading to blue -->
    }
</style>
</head>
<body> <!-- Body section of the document -->
<h1>Hello World!</h1> <!-- Heading with text "Hello World!" -->
<p>This is my website.</p> <!-- Paragraph of text -->
</body>
</html> <!-- Closes the HTML document -->
