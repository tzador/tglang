<!DOCTYPE html> <!--Defining document type as HTML-->
<html> <!--Opening HTML tag-->

<head> <!--Opening head tag-->
  <title>This is my HTML page</title> <!--Adding title to the page-->
  <meta charset="UTF-8"> <!--Specifying character encoding of the document-->
</head> <!--Closing head tag-->

<body> <!--Opening body tag-->
  <h1>Welcome to my first HTML page!</h1> <!--Adding a heading to the page-->
  <p>This is a paragraph of text.</p> <!--Adding a paragraph of text-->
  
  <ul> <!--Opening unordered list tag-->
    <li>Item 1</li> <!--Adding list item 1-->
    <li>Item 2</li> <!--Adding list item 2-->
    <li>Item 3</li> <!--Adding list item 3-->
  </ul> <!--Closing unordered list tag-->
  
  <table> <!--Opening table tag-->
    <tr> <!--Opening table row tag-->
      <th>Column 1</th> <!--Adding table header for column 1-->
      <th>Column 2</th> <!--Adding table header for column 2-->
      <th>Column 3</th> <!--Adding table header for column 3-->
    </tr> <!--Closing table row tag-->
    <tr> <!--Opening table row tag-->
      <td>Row 1, Column 1</td> <!--Adding table data for row 1, column 1-->
      <td>Row 1, Column 2</td> <!--Adding table data for row 1, column 2-->
      <td>Row 1, Column 3</td> <!--Adding table data for row 1, column 3-->
    </tr> <!--Closing table row tag-->
    <tr> <!--Opening table row tag-->
      <td>Row 2, Column 1</td> <!--Adding table data for row 2, column 1-->
      <td>Row 2, Column 2</td> <!--Adding table data for row 2, column 2-->
      <td>Row 2, Column 3</td> <!--Adding table data for row 2, column 3-->
    </tr> <!--Closing table row tag-->
  </table> <!--Closing table tag-->
</body> <!--Closing body tag-->

</html> <!--Closing HTML tag-->
