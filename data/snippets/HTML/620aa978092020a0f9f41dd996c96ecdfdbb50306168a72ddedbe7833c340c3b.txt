<!DOCTYPE html> <!--Defines the document type as HTML-->
<html> <!--Beginning of the HTML document-->
  <head> <!--Defines the head section of the document (for metadata)-->
    <title>Code Snippet</title> <!--Sets the document title-->
    <link rel="stylesheet" href="styles.css"> <!--Links an external stylesheet-->
    <style> <!--Defines internal CSS styles for the document-->
      body { <!--Defines styles for the body element-->
        background-color: #ffffff; /*Sets background color to white*/
        font-family: Arial, sans-serif; /*Sets font family to Arial*/
        font-size: 16px; /*Sets font size to 16px*/
      }
      h1 { /*Defines styles for heading level 1*/
        color: #00a5e3; /*Sets heading color to blue*/
        font-size: 24px; /*Sets heading font size to 24px*/
      }
      p { /*Defines styles for paragraph elements*/
        color: #000000; /*Sets paragraph color to black*/
        font-size: 14px; /*Sets paragraph font size to 14px*/
        line-height: 1.5; /*Sets line height to 1.5 times the font size*/
      }
    </style>
  </head>
  <body> <!--Beginning of the body section of the document-->
    <h1>Code Snippet</h1> <!--Heading level 1-->
    <p>This is a code snippet written in HTML. It demonstrates how to use various HTML elements and attributes to structure a web page and add styles to its content.</p> <!--Paragraph element-->
    <ul> <!--Unordered list element-->
      <li>Headings</li> <!--List item-->
      <li>Paragraphs</li> <!--List item-->
      <li>Links</li> <!--List item-->
      <li>Images</li> <!--List item-->
    </ul>
    <a href="https://www.example.com">Visit our website</a> <!--Link to an external website-->
    <img src="snippet.png" alt="HTML code snippet"> <!--Image element with an alternative text-->
    <script> <!--Defines a script to be executed on the document-->
      alert("Thank you for checking out our code snippet!"); //Displays an alert message
    </script>
  </body>
</html>
