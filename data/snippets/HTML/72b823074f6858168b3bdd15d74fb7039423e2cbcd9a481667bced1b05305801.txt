<!DOCTYPE html> <!-- Declares this as an HTML5 document -->
<html> <!-- Start tag for the HTML document -->
<head> <!-- Start tag for the document's head -->
  <meta charset="UTF-8"> <!-- Declares the character encoding for the document -->
  <title>My Website</title> <!-- Sets the title for the document -->
  <style> <!-- Defines the document's styles, in this case setting the background color to red -->
      body {
          background-color: red;
      }
  </style>
</head> <!-- End tag for the document's head -->
<body> <!-- Start tag for the document's body -->
  <h1>Welcome to My Website!</h1> <!-- Heading 1 tag for the main heading of the page -->
  <p>This is a paragraph on my website.</p> <!-- Paragraph tag for text content -->
  <a href="https://www.example.com">Click here</a> <!-- Anchor tag for creating a hyperlink -->
  <img src="example.jpg" alt="Example Image"> <!-- Image tag for displaying an image with alternative text -->
  <ul> <!-- Start tag for unordered list -->
      <li>Item 1</li> <!-- List item 1 -->
      <li>Item 2</li> <!-- List item 2 -->
      <li>Item 3</li> <!-- List item 3 -->
  </ul> <!-- End tag for unordered list -->
  <script> <!-- Defines scripts or programming code for the document -->
      alert("Hello World!"); <!-- Displays a message box with the text "Hello World!" -->
  </script>
</body> <!-- End tag for the document's body -->
</html> <!-- End tag for the HTML document -->
