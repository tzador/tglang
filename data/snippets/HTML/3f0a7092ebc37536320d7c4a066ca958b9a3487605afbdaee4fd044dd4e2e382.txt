<!-- This is a comment in HTML -->
<!DOCTYPE html> <!-- A document type declaration at the beginning of an HTML file -->
<html> <!-- The opening tag for an HTML document -->
<head> <!-- The opening tag for the head section of an HTML document -->
    <title>This is a Title</title> <!-- The title of the HTML document -->
    <meta charset="UTF-8"> <!-- A meta tag for specifying the character encoding of the HTML document -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- A meta tag for setting the viewport of the HTML document -->
</head> <!-- The closing tag for the head section of an HTML document -->
<body> <!-- The opening tag for the body section of an HTML document -->
    <h1>This is a Heading</h1> <!-- A heading tag for the main heading of an HTML document -->
    <p>This is a paragraph. It can contain <b>bold text</b>, <i>italic text</i> or <u>underlined text</u>.</p> <!-- A paragraph tag for adding text content to an HTML document -->
    <ul> <!-- The opening tag for an unordered list -->
        <li>Item 1</li> <!-- A list item tag for an item in an unordered list -->
        <li>Item 2</li> <!-- A list item tag for an item in an unordered list -->
        <li>Item 3</li> <!-- A list item tag for an item in an unordered list -->
    </ul> <!-- The closing tag for an unordered list -->
    <ol> <!-- The opening tag for an ordered list -->
        <li>Item 1</li> <!-- A list item tag for an item in an ordered list -->
        <li>Item 2</li> <!-- A list item tag for an item in an ordered list -->
        <li>Item 3</li> <!-- A list item tag for an item in an ordered list -->
    </ol> <!-- The closing tag for an ordered list -->
    <img src="image.jpg" alt="Image"> <!-- An image tag for inserting an image into an HTML document -->
</body> <!-- The closing tag for the body section of an HTML document -->
</html> <!-- The closing tag for an HTML document -->
