<!-- This is a comment in HTML code -->
<!DOCTYPE html> <!-- Doc type declaration for HTML5 -->
<html> <!-- Opening root tag for HTML document -->
<head> <!-- Opening head tag for document metadata -->
  <meta charset="UTF-8"> <!-- Declaring character encoding for document -->
  <title>My HTML Document</title> <!-- Setting title for document -->
</head> <!-- Closing head tag for document metadata -->

  <h1>Heading 1</h1> <!-- Header tag for main heading -->
  <p>This is a paragraph.</p> <!-- Paragraph tag for text -->
  <ul> <!-- Opening unordered list tag -->
    <li>List item 1</li> <!-- List item 1 -->
    <li>List item 2</li> <!-- List item 2 -->
    <li>List item 3</li> <!-- List item 3 -->
  </ul> <!-- Closing unordered list tag -->
  <table> <!-- Opening table tag -->
    <thead> <!-- Opening table head tag -->
      <tr> <!-- Table row tag -->
        <th>Column 1</th> <!-- Table header cell 1 -->
        <th>Column 2</th> <!-- Table header cell 2 -->
      </tr>
    </thead> <!-- Closing table head tag -->
    <tbody> <!-- Opening table body tag -->
      <tr> <!-- Table row tag -->
        <td>Row 1, Column 1</td> <!-- Table data cell 1 -->
        <td>Row 1, Column 2</td> <!-- Table data cell 2 -->
      </tr>
      <tr> <!-- Table row tag -->
        <td>Row 2, Column 1</td> <!-- Table data cell 1 -->
        <td>Row 2, Column 2</td> <!-- Table data cell 2 -->
      </tr>
    </tbody> <!-- Closing table body tag -->
  </table> <!-- Closing table tag -->
