<!DOCTYPE html> 
<html> <!--This is the opening tag for the HTML document--> 
<head> <!--This is the opening tag for the head section--> 
  <title>Example Code</title> <!--This is the title tag that will be displayed in the browser tab--> 
  <meta charset="UTF-8"> <!--This is a meta tag that defines the character encoding for the document-->
  <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!--This is a meta tag that sets the viewport to the width of the device and scales the page to fit the screen-->
  <style> <!--This is the opening tag for the style section-->
      p { <!--This is a CSS selector that targets all paragraph elements-->
        color: blue; <!--This is a CSS property that sets the font color-->
        font-size: 18px; <!--This is a CSS property that sets the font size-->
        text-align: center; <!--This is a CSS property that centers the text-->
      } <!--This is the closing tag for the paragraph CSS rule-->
    </style> <!--This is the closing tag for the style section-->
</head> <!--This is the closing tag for the head section--> 
<body> <!--This is the opening tag for the body section--> 
  <h1>This is a heading</h1> <!--This is the opening tag for a level 1 heading--> 
  <p>This is a paragraph</p> <!--This is the opening tag for a paragraph--> 
  <!--This is a comment in HTML--> 
  <img src="image.jpg" alt="Example Image"> <!--This is an image tag that specifies the image source and alternative text--> 
  <a href="https://www.example.com">Link to example website</a> <!--This is an anchor tag that creates a link to another webpage--> 
</body> <!--This is the closing tag for the body section--> 
</html> <!--This is the closing tag for the HTML document-->
