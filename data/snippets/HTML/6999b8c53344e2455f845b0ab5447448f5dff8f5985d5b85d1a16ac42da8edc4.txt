<!DOCTYPE html> //Specifies the document type as HTML
<html> //Opening tag for the HTML document
<head> //Opening tag for the head section
  <title>Annotated HTML Code</title> //Specifies the title of the webpage
</head> //Closing tag for the head section
<body> //Opening tag for the body section
  <h1>Hello World!</h1> //Heading 1 tag to display text
  <p>This is an example of annotated HTML code.</p> //Paragraph tag to display text
  <ul> //Opening tag for unordered list
    <li>Item 1</li> //List item 1
    <li>Item 2</li> //List item 2
    <li>Item 3</li> //List item 3
  </ul> //Closing tag for unordered list
  <br> //Line break tag
  <a href="https://www.example.com">Link to Example Website</a> //Link tag to display a hyperlink
  <img src="example_image.png" alt="Example Image"> //Image tag to display an image with alternative text
</body> //Closing tag for the body section
</html> //Closing tag for the HTML document
