<!DOCTYPE html>   <!-- This is the first line of an HTML document, which defines the document type -->
<html>   <!-- This <html> tag indicates the start of the HTML document -->
<head>   <!-- This is the <head> tag, which is used to contain all the head elements (title, style, etc.) -->
    <meta charset="UTF-8">   <!-- This <meta> tag specifies the character encoding for the HTML document -->
    <title>Code Snippet</title>   <!-- This is the <title> tag, which defines the title of the HTML document -->
</head>   <!-- This closes the <head> tag -->
<body>   <!-- This is the <body> tag, which includes all the visible content of the HTML document -->
    <h1>Hello World!</h1>   <!-- This <h1> tag defines a heading with the text "Hello World!" -->
    <p>This is a code snippet in HTML.</p>   <!-- This <p> tag defines a paragraph with the text "This is a code snippet in HTML." -->
    <ul>   <!-- This <ul> tag defines an unordered list -->
        <li>Item 1</li>   <!-- This <li> tag defines a list item with the text "Item 1" -->
        <li>Item 2</li>   <!-- This <li> tag defines a list item with the text "Item 2" -->
        <li>Item 3</li>   <!-- This <li> tag defines a list item with the text "Item 3" -->
    </ul>   <!-- This closes the <ul> tag -->
    <script>   <!-- This is the <script> tag, which is used to define client-side JavaScript -->
        alert("This is a code snippet!");   <!-- This is a JavaScript function that will display an alert with the text "This is a code snippet!" -->
    </script>   <!-- This closes the <script> tag -->
</body>   <!-- This closes the <body> tag -->
</html>   <!-- This closes the <html> tag -->
