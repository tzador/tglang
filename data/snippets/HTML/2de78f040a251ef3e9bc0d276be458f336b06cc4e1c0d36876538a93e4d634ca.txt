<!-- This is a comment in HTML code -->
<!DOCTYPE html> <!--This is the declaration of the document type as HTML-->
<html> <!--This is the opening tag for the HTML document-->

<head> <!--This is the opening tag for the head section-->
  <title>Hello World</title> <!--This is the title of the document-->
  <meta charset="UTF-8"> <!--This is the character set for the document-->
  <meta name="description" content="Example HTML document"> <!--This is the description of the document-->
  <meta name="author" content="John Doe"> <!--This is the author of the document-->
</head> <!--This is the closing tag for the head section-->

<body> <!--This is the opening tag for the body section-->
  <h1>Hello World</h1> <!--This is the heading tag for level 1 heading-->
  <p>This is an example of an HTML document.</p> <!--This is a paragraph tag-->
  <ul> <!--This is the opening tag for an unordered list-->
    <li>Item 1</li> <!--This is a list item-->
    <li>Item 2</li>
    <li>Item 3</li>
  </ul> <!--This is the closing tag for the unordered list-->
</body> <!--This is the closing tag for the body section-->

</html> <!--This is the closing tag for the HTML document-->
