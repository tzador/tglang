<!-- This is a comment in HTML -->

<!DOCTYPE html> <!-- This declares the document type as HTML -->

<html> <!-- This is the opening tag for the root element of the HTML document -->

  <head> <!-- This is the opening tag for the head element -->
    <meta charset="UTF-8"> <!-- This specifies the character encoding for the document -->
    <title>This is the title of the webpage</title> <!-- This sets the title for the webpage -->
    <link rel="stylesheet" type="text/css" href="style.css"> <!-- This links to an external CSS file for styling -->
  </head> <!-- This is the closing tag for the head element -->

  <body> <!-- This is the opening tag for the body element -->
    <h1>This is a heading</h1> <!-- This creates a heading element -->
    <p>This is a paragraph</p> <!-- This creates a paragraph element -->
    <img src="image.jpg" alt="An image"> <!-- This creates an image element with a source and alternate text -->
    <a href="https://www.example.com">Click here</a> <!-- This creates a link to a web address -->
    <ul> <!-- This is the opening tag for an unordered list element -->
      <li>Item 1</li> <!-- This creates a list item within the unordered list -->
      <li>Item 2</li>
      <li>Item 3</li>
    </ul> <!-- This is the closing tag for the unordered list element -->
  </body> <!-- This is the closing tag for the body element -->

</html> <!-- This is the closing tag for the root element -->
