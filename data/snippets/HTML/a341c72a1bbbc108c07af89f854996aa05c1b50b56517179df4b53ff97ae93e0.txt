<head> <!-- This is the head section of the HTML document -->
    <meta charset="utf-8"> <!-- This specifies the character encoding for the document -->
    <title>My Website</title> <!-- This is the title of the document -->
    <style> <!-- This is an inline CSS style that will be applied to the entire document -->
        body {
            font-family: sans-serif; /* Sets the font to sans-serif for all text in the body */
            background-color: #ffffff; /* Sets the background color to white */
        }
        h1 {
            font-size: 2em; /* Sets the font size of h1 headings to 2em */
            text-align: center; /* Centers the text in h1 headings */
            color: #000000; /* Sets the color of h1 headings to black */
        }
        p {
            font-size: 1em; /* Sets the font size of paragraphs to 1em */
            line-height: 1.5; /* Sets the line height of paragraphs to 1.5 */
            text-align: justify; /* Justifies the text in paragraphs */
        }
    </style>
</head>
<body> <!-- This is the body section of the HTML document -->
    <h1>Welcome to My Website</h1> <!-- This is a heading in the document -->
    <p>This is a sample paragraph on my website that will demonstrate how CSS styles are applied to text.</p> <!-- This is a paragraph in the document -->
    <p>This is another paragraph on my website. It will show how text can be formatted and aligned using CSS.</p>
</body>
