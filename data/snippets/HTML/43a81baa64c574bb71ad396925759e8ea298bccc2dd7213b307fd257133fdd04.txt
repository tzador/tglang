<!DOCTYPE html> <!-- defining document type as HTML -->
<html> <!-- root element of HTML document -->

<head> <!-- contains metadata about the document -->
  <title>This is a complex HTML code snippet</title> <!-- sets the title of the document -->
  <link rel="stylesheet" type="text/css" href="styles.css"> <!-- links to external CSS style sheet -->
  <script type="text/javascript" src="main.js"></script> <!-- links to external JavaScript file -->
</head>

<body> <!-- contains the visible content of the document -->
  <header> <!-- defines a header section -->
    <h1>This is a complex HTML code snippet</h1> <!-- heading element with text -->
    <nav> <!-- defines navigation links within the header -->
      <a href="#">Home</a> <!-- link to home page -->
      <a href="#">About</a> <!-- link to about page -->
      <a href="#">Contact</a> <!-- link to contact page -->
    </nav>
  </header>

  <main> <!-- defines the main content of the document -->
    <section> <!-- defines a section within the main content -->
      <h2>Section 1</h2> <!-- heading element with text -->
      <p>This is the first section of the document.</p> <!-- paragraph element with text -->
      <img src="image.jpg" alt="An image"> <!-- image element with source and alternative text -->
    </section>
    <section> <!-- defines another section within the main content -->
      <h2>Section 2</h2> <!-- heading element with text -->
      <p>This is the second section of the document.</p> <!-- paragraph element with text -->
      <ul> <!-- unordered list element -->
        <li>Item 1</li> <!-- list item element with text -->
        <li>Item 2</li> <!-- list item element with text -->
        <li>Item 3</li> <!-- list item element with text -->
      </ul>
    </section>
  </main>

  <footer> <!-- contains information about the document -->
    <p>Created by John Smith</p> <!-- paragraph element with text -->
    <small>&copy; 2020</small> <!-- small element with text -->
  </footer>
</html>
