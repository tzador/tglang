<!DOCTYPE html> <!-- Defines the document type as HTML -->
<html lang="en"> <!-- Specifies the language of the document as English -->
<head> <!-- Begins the head section -->
  <meta charset="UTF-8"> <!-- Specifies the character encoding for the document -->
  <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- Specifies the viewport properties -->
  <meta http-equiv="X-UA-Compatible" content="ie=edge"> <!-- Specifies the compatibility with Internet Explorer -->
  <title>Complex Code Snippet</title> <!-- Sets the title of the document -->
  <style> <!-- Begins the style section -->
    body { <!-- Sets the style for the body element -->
      font-size: 16px; <!-- Sets the font size to 16px -->
      color: #333; <!-- Sets the font color to dark gray -->
      font-family: Arial, sans-serif; <!-- Specifies the fallback fonts to use -->
    }

    h1 { <!-- Sets the style for the h1 element -->
      font-size: 24px; <!-- Sets the font size to 24px -->
      font-weight: bold; <!-- Sets the font weight to bold -->
      color: #000; <!-- Sets the font color to black -->
      text-decoration: underline; <!-- Adds an underline to the text -->
    }

    p { <!-- Sets the style for the p element -->
      margin-bottom: 20px; <!-- Sets the bottom margin to 20px -->
    }

    a { <!-- Sets the style for the a element -->
      color: #007bff; <!-- Sets the font color to blue -->
      text-decoration: none; <!-- Removes any text decoration -->
    }

    img { <!-- Sets the style for the img element -->
      max-width: 100%; <!-- Sets the maximum width to 100% to ensure responsiveness -->
    }
  </style> <!-- Ends the style section -->
</head> <!-- Ends the head section -->

<body> <!-- Begins the body section -->
  <h1>Welcome to My Website</h1> <!-- Adds a header element with the text "Welcome to My Website" -->
  <p>This is a complex code snippet in HTML. Here, you can see how different elements are styled using CSS. <a href="#">Click here</a> to learn more about CSS.</p> <!-- Adds a paragraph element with the text and a link -->
  <img src="image.jpg" alt="An image"> <!-- Adds an image with the source and alternate text -->
  <script> <!-- Begins the script section -->
    console.log("Hello world!"); <!-- Prints "Hello world!" to the console -->
    document.querySelector("h1").style.color = "red"; <!-- Selects the h1 element and changes its color to red -->
  </script> <!-- Ends the script section -->
</body> <!-- Ends the body section -->

</html> <!-- Ends the HTML document -->
