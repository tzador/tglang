<!-- This is an example of a complex HTML code snippet -->
<!DOCTYPE html>
<html lang="en">
<head>
    <!-- Specifies the title of the document -->
    <title>Code Snippet Example</title>

    <!-- This is how you include a CSS file -->
    <link rel="stylesheet" href="styles.css">

    <!-- This is how you insert JavaScript files -->
    <script src="scripts.js"></script>

    <!-- This is how you add metadata to your document -->
    <meta charset="UTF-8">
    <meta name="description" content="Example of a complex HTML code snippet">
    <meta name="author" content="John Smith">

    <!-- This is how you add a favicon -->
    <link rel="icon" href="favicon.ico">
</head>
<body>
    <!-- This is how you add a header to your document -->
    <header>
        <h1>Welcome to my Code Snippet Example</h1>
    </header>

    <!-- This is how you create a navigation bar -->
    <nav>
        <ul>
            <li><a href="#introduction">Introduction</a></li>
            <li><a href="#syntax">Syntax</a></li>
            <li><a href="#examples">Examples</a></li>
        </ul>
    </nav>

    <!-- This is how you create a section with an id -->
    <section id="introduction">
        <h2>Introduction</h2>
        <p>This is an example of a complex HTML code snippet.</p>
        <p>HTML stands for HyperText Markup Language and is used to create web pages.</p>
    </section>

    <!-- This is how you create another section with an id -->
    <section id="syntax">
        <h2>Syntax</h2>
        <p>HTML uses tags to define the structure and content of a web page.</p>
        <p>Tags are written inside angle brackets, like this: &lt;tag&gt;</p>
        <p>Each HTML document starts with the &lt;!DOCTYPE&gt; declaration and is then followed by the &lt;html&gt; element, which contains the &lt;head&gt; and &lt;body&gt; elements.</p>
    </section>

    <!-- This is how you create a third section with an id -->
    <section id="examples">
        <h2>Examples</h2>
        <p>Here are some examples of common HTML tags:</p>
        <ul>
            <li>&lt;h1&gt; - &lt;h6&gt;: headings</li>
            <li>&lt;p&gt;: paragraphs</li>
            <li>&lt;img&gt;: images</li>
            <li>&lt;a&gt;: links</li>
        </ul>
    </section>

    <!-- This is how you add a footer to your document -->
    <footer>
        <p>Created by John Smith</p>
    </footer>
</body>
</html>
