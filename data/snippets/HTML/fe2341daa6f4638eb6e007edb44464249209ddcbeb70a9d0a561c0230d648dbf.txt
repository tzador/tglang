<!-- This is a comment in HTML -->
<!DOCTYPE html>
<html> <!-- The opening tag for the HTML document -->
  <head> <!-- The opening tag for the document's head section -->
    <meta charset="UTF-8"> <!-- This element sets the character encoding for the document to UTF-8 -->
    <title>This is the title of the document</title> <!-- This element specifies the title of the document -->
    <style> <!-- This element contains CSS code for styling the document -->
      body { <!-- This rule sets the properties for the body element -->
        font-family: "Arial", sans-serif; <!-- This property sets the font family for the text -->
        background-color: #FFF; <!-- This property sets the background color for the document -->
      }
      h1 { <!-- This rule sets the properties for the h1 element -->
        color: #4183D7; <!-- This property sets the color for the heading text -->
      }
    </style>
  </head>
  <body> <!-- The opening tag for the document's body section -->
    <h1>This is a heading</h1> <!-- This element creates a level 1 heading -->
    <p>This is a paragraph</p> <!-- This element creates a paragraph -->
    <a href="https://www.example.com">This is a link</a> <!-- This element creates a link with the specified URL -->
    <img src="image.jpg" alt="An image"> <!-- This element inserts an image with the specified source and alt text -->
  </body>
</html>
