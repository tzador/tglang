<!DOCTYPE html> 

<html> <!-- This is the opening tag for the main HTML element -->
<head> <!-- This is the opening tag for the head element -->
    <title>Hello World</title> <!-- This is a title tag, which sets the title of the document -->
    <style> <!-- This is the opening tag for the style element -->
        h1 { /* This is a CSS selector, which selects all h1 elements */
            color: red; /* This is a CSS property, which sets the color of the text to red */
            font-size: 24px; /* This is a CSS property, which sets the font size to 24 pixels */
        }
    </style> <!-- This is the closing tag for the style element -->
</head> <!-- This is the closing tag for the head element -->
<body> <!-- This is the opening tag for the body element -->
    <h1>Hello World</h1> <!-- This is a heading tag, which displays the text "Hello World" -->
    <p>This is a paragraph tag, which displays some text as a paragraph</p> <!-- This is a paragraph tag, which displays the text within it as a paragraph -->
    <ul> <!-- This is the opening tag for the unordered list element -->
        <li>Item 1</li> <!-- This is a list item tag, which displays "Item 1" as a list item -->
        <li>Item 2</li> <!-- This is a list item tag, which displays "Item 2" as a list item -->
        <li>Item 3</li> <!-- This is a list item tag, which displays "Item 3" as a list item -->
    </ul> <!-- This is the closing tag for the unordered list element -->
</body> <!-- This is the closing tag for the body element -->
</html> <!-- This is the closing tag for the main HTML element -->
