<!DOCTYPE html> //Defines the document type as HTML
<html lang="en"> //Defines the language of the document as English
<head> //Head section of the document
  <meta charset="UTF-8"> //Specifies the character encoding for the document
  <meta http-equiv="X-UA-Compatible" content="IE=edge"> //Specifies the version of IE to render the document
  <meta name="viewport" content="width=device-width, initial-scale=1.0"> //Specifies the viewport for mobile devices
  <title>My Website</title> //Specifies the title of the document
  <style> //Style section for HTML elements
    body { //Styles the body element
      font-family: "Arial", sans-serif; //Sets the font family for the body text
    }
    p { //Styles the paragraph element
      font-size: 18px; //Sets the font size for the paragraph text
    }
    .container { //Styles the container class
      width: 80%; //Sets the width of the container
      margin: 20px auto; //Sets the margins for the container
      border: 1px solid black; //Adds a border around the container
    }
  </style>
</head>
<body> //Body section of the document
  <div class="container"> //A div element with a class of "container"
    <h1>Hello World!</h1> //A heading element with text "Hello World!"
    <p>Lorem ipsum dolor sit amet, consectetur adipiscing elit.</p> //A paragraph element with Lorem Ipsum text
  </div>
</body>
</html>
