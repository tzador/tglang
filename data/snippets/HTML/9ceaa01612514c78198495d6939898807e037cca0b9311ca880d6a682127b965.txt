<!DOCTYPE html> <!-- This declares the document type as HTML -->
<html lang="en"> <!-- This starts the HTML document and specifies language as English -->
<head> <!-- This marks the start of the head section -->
	<title>My Website</title> <!-- This adds a title for the website -->
	<meta charset="UTF-8"> <!-- This specifies the character encoding for the document -->
	<!-- This is a comment, it is not rendered in the browser -->
	<style> <!-- This starts the style section -->
		p {color: blue;} <!-- This styles all paragraphs to have blue text -->
	</style>
</head> <!-- This marks the end of the head section -->
<body> <!-- This marks the start of the body section -->
	<h1>Welcome to My Website</h1> <!-- This adds a heading to the webpage -->
	<p>This is my first webpage!</p> <!-- This adds a paragraph to the webpage -->
	<ul> <!-- This starts an unordered list -->
		<li>Item 1</li> <!-- This adds a list item -->
		<li>Item 2</li> <!-- This adds another list item -->
	</ul> <!-- This ends the unordered list -->
	<!-- This is an example of a self-closing tag, it doesn't need a closing tag -->
	<br> <!-- This adds a line break -->
	<a href="https://www.example.com">Link</a> <!-- This creates a hyperlink to another webpage -->
</body> <!-- This marks the end of the body section -->
</html> <!-- This marks the end of the HTML document -->
