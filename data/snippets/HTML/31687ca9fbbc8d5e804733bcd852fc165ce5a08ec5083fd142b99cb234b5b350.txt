<!-- This is a comment in HTML -->
<!DOCTYPE html> <!-- Declares the document type as HTML -->
<html> <!-- Opening tag for the HTML document -->
  <head> <!-- Opening tag for the head section -->
    <title>My Website</title> <!-- Sets the title of the document -->
    <meta charset="UTF-8"> <!-- Sets the character encoding for the document -->
    <link rel="stylesheet" href="styles.css"> <!-- Links the external stylesheet to the document -->
  </head> <!-- Closing tag for the head section -->
  <body> <!-- Opening tag for the body section -->
    <h1>Welcome to my website!</h1> <!-- Heading tag for the main heading -->
    <p>This is my first website and it's written in HTML.</p> <!-- Paragraph tag for text content -->
    <img src="mylogo.jpeg" alt="My Logo"> <!-- Image tag to insert an image with alternative text -->
    <ul> <!-- Opening tag for an unordered list -->
      <li>HTML</li> <!-- List item -->
      <li>CSS</li> <!-- List item -->
      <li>Javascript</li> <!-- List item -->
    </ul> <!-- Closing tag for the unordered list -->
    <a href="about.html">Learn more about me</a> <!-- Anchor tag to create a link to another webpage -->
  </body> <!-- Closing tag for the body section -->
</html> <!-- Closing tag for the HTML document -->
