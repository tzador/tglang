<!-- This is a multi-line comment in HTML -->

<!DOCTYPE html> <!-- This is the document type declaration -->

<html> <!-- This is the opening tag for the HTML document -->

<head> <!-- This is the head element where we define metadata for the document -->

    <title>This is a complex HTML code snippet</title> <!-- This is the title element, which is displayed in the browser tab -->

    <meta charset="UTF-8"> <!-- This is the metadata for character encoding of the document -->

    <meta name="description" content="This is a complex code snippet in HTML"> <!-- This is the metadata for the document's description -->

    <style>  /* This is the element used to define CSS styles for the document */

        body { /* This is the CSS selector for the body element */
            font-size: 16px; /* This is the CSS property that sets the font size */
            font-family: Arial, sans-serif; /* This is the CSS property that sets the font family */
        }

        .container { /* This is the CSS selector for the container class */
            width: 80%; /* This is the CSS property that sets the width */
            margin: 0 auto; /* This is the CSS property that sets the margin */
            background-color: lightgray; /* This is the CSS property that sets the background color */
            padding: 2rem; /* This is the CSS property that sets the padding */
        }

        h1 { /* This is the CSS selector for the h1 element */
            font-size: 3rem; /* This is the CSS property that sets the font size */
            text-align: center; /* This is the CSS property that sets the text alignment */
        }

        p { /* This is the CSS selector for the p element */
            line-height: 1.5; /* This is the CSS property that sets the line height */
        }

        a { /* This is the CSS selector for the a element */
            color: blue; /* This is the CSS property that sets the color */
            text-decoration: none; /* This is the CSS property that removes the default underline on links */
        }

        a:hover { /* This is the CSS selector for the a element when hovering over it */
            text-decoration: underline; /* This is the CSS property that adds an underline on hover */
        }

    </style>

</head>

<body> <!-- This is the body element, where we define the visible content of the document -->

    <div class="container"> <!-- This is a div element with the class "container", which is used for styling purposes -->

        <h1>This is a Complex HTML Code Snippet</h1> <!-- This is the heading of the document -->

        <p>This is a paragraph with <em>emphasis</em> and <strong>strong text</strong>.</p> <!-- This is a paragraph with emphasized and strong text -->

        <ul> <!-- This is an unordered list element -->
            <li>Item 1</li> <!-- Each list item is defined with an li element -->
            <li>Item 2</li>
            <li>Item 3</li>
        </ul>

        <ol> <!-- This is an ordered list element -->
            <li>Step 1</li>
            <li>Step 2</li>
            <li>Step 3</li>
        </ol>

        <a href="https://www.example.com">This is a link to an external website</a> <!-- This is a link element with the URL specified in the href attribute -->

    </div> <!-- This is the end of the div element with the class "container" -->

</body>

</html> <!-- This is the closing tag for the HTML document -->
