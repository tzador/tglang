<!-- This is a comment in HTML code -->

<!DOCTYPE html>
<html>  <!-- HTML root element -->
<head>  <!-- HTML header element -->
  <title>My Website</title>
  <style>  <!-- CSS style element for internal styling -->
    h1 {
      color: blue;
      font-size: 24px;
      font-weight: bold;
    }
  </style>
</head>
<body>  <!-- HTML body element -->
  <h1>Welcome to my website!</h1>  <!-- main heading using h1 tag -->
  <p>This is a paragraph with a <a href="https://www.example.com" target="_blank">link</a> to another website.</p>  <!-- p tag for paragraph and a tag for linking -->
  <ul>  <!-- unordered list using ul tag -->
    <li>First item</li>
    <li>Second item</li>
    <li>Third item</li>
  </ul>
  <script>  <!-- JavaScript script element for embedded code -->
    alert("Hello, world!");
  </script>
  <div id="container" class="main-content">  <!-- div element with id and class attributes -->
    <h2>This is another heading inside a div element</h2> <!-- secondary heading using h2 tag -->
    <p>This is another paragraph inside the div. It also has a <strong>strong</strong> element for bold text.</p> <!-- p tag with strong tag for bold text -->
  </div>
</body>
</html>
