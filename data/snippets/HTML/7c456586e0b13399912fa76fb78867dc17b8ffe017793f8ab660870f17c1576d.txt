<!-- This is a comment in HTML -->

<!DOCTYPE html> <!-- This is the doctype declaration -->

<html> <!-- The opening <html> tag indicates the start of the HTML document -->

<head> <!-- This section contains information about the document itself -->
    <title>My Website</title> <!-- This is used to specify the title of the webpage -->

    <!-- This is a meta tag used to specify the character set of the document -->
    <meta charset="UTF-8"> 

	<!-- This is a meta tag used to specify the viewport settings -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> 

    <!-- This is a link tag used to link an external stylesheet -->
    <link rel="stylesheet" href="style.css">

    <!-- This is a script tag used to add JavaScript to the document -->
    <script src="app.js"></script> 
</head>

<body> <!-- The opening <body> tag indicates the start of the document's body -->

    <!-- This is a header element, used to represent introductory content -->
    <header>
        <h1>Welcome to My Website</h1> <!-- This is a heading element -->
        <nav> <!-- This is a navigation element containing links to other pages -->
            <a href="about.html">About</a>
            <a href="contact.html">Contact</a>
        </nav>
    </header>

    <!-- This is a section element, used to group related content -->
    <section>
        <h2>Meet the Team</h2>
        <p>Here you can find information about our team members.</p>

        <!-- This is an unordered list containing several list items -->
        <ul>
            <li>John Smith</li>
            <li>Jessica Brown</li>
            <li>Michael Johnson</li>
        </ul>

        <!-- This is an image element, used to display images on the webpage -->
        <img src="team.jpg" alt="Team Photo">
    </section>

    <!-- This is a footer element, used to contain information about the document -->
    <footer>
        &copy; 2020 My Website
    </footer>

</body> 

</html> <!-- The closing </html> tag indicates the end of the HTML document -->
