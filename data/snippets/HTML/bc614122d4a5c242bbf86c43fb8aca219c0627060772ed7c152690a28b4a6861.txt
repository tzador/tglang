<!DOCTYPE html> <!-- Declares document type as HTML 5 -->
<html> <!-- Opening tag for HTML document -->
  <head> <!-- Head element to contain metadata for the document -->
    <meta charset="UTF-8"> <!-- Defines character encoding for the document -->
    <title>Sample Website</title> <!-- Sets title for the document -->
    <style> <!-- Defines CSS styles for the document -->
      body { <!-- Styles the body element -->
        font-family: Arial, sans-serif; <!-- Sets font family for text -->
        background-color: #eee; <!-- Sets background color -->
      }
      
      h1 { <!-- Styles h1 element -->
        color: blue; <!-- Sets text color -->
      }
      
      p { <!-- Styles p element -->
        font-size: 16px; <!-- Sets font size -->
      }
      
      #main-section { <!-- Styles element with id of 'main-section' -->
        border: 1px solid black; <!-- Sets border style -->
        padding: 10px; <!-- Sets padding -->
        margin: 10px; <!-- Sets margin -->
      }
    </style>
  </head>
  <body> <!-- Body element holds all visible content for the page -->
    <h1>Welcome to my sample website!</h1> <!-- H1 heading -->
    <p>This is a paragraph element with some sample text.</p> <!-- P paragraph -->
    <div id="main-section"> <!-- Div element with id of 'main-section' -->
      <h2>Section 1</h2> <!-- H2 heading -->
      <p>This is some sample text inside the first section.</p>
      <h2>Section 2</h2> <!-- H2 heading -->
      <p>Here is some more text in the second section.</p>
    </div>
  </body>
</html> <!-- Closing tag for HTML document -->
