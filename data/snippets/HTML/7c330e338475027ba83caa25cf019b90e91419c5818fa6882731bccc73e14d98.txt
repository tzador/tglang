<!DOCTYPE html> <!-- defining document type for HTML5 -->
<html> <!-- start of HTML document -->

<head> <!-- start of head section -->
  <title>My Website</title> <!-- title of the webpage -->
  <link rel="stylesheet" href="style.css"> <!-- linking external style sheet -->
  <script src="script.js"></script> <!-- linking external JavaScript file -->
</head> <!-- end of head section -->

<body> <!-- start of body section -->
  <h1>Welcome to My Website</h1> <!-- heading element -->
  <p>This is a paragraph element with <em>emphasis</em> and <strong>strong</strong> text.</p> <!-- paragraph element with emphasized and strong text -->
  <ul> <!-- start of unordered list -->
    <li>Item 1</li> <!-- list item 1 -->
    <li>Item 2</li> <!-- list item 2 -->
    <li>Item 3</li> <!-- list item 3 -->
  </ul> <!-- end of unordered list -->
  <img src="image.jpg" alt="Image"> <!-- image element with source and alternate text -->
</body> <!-- end of body section -->

</html> <!-- end of HTML document -->
