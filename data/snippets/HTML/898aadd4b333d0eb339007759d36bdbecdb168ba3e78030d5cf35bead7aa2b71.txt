<!DOCTYPE html> <!-- doctype declaration -->
<html> <!-- opening html tag -->
<head> <!-- opening head tag -->
<meta charset="UTF-8"> <!-- sets document character encoding to UTF-8 -->
<title>My Website</title> <!-- sets document title -->
<link rel="stylesheet" type="text/css" href="styles.css"> <!-- links external stylesheet -->
<style> <!-- opening style tag -->
body { <!-- opening body selector -->
font-family: Arial, sans-serif; <!-- sets font for body -->
background-color: #eaeaea; <!-- sets background color for body -->
}
h1 { <!-- opening h1 selector -->
color: #333; <!-- sets text color for h1 -->
text-align: center; <!-- centers h1 text -->
}
</style> <!-- closing style tag -->
</head> <!-- closing head tag -->
<body> <!-- opening body tag -->
<h1>Welcome to my website!</h1> <!-- h1 heading with text -->
<p>My name is John and I am a web developer.</p> <!-- paragraph with text -->
<hr> <!-- creates horizontal line -->
<a href="about.html">About Me</a> <!-- link to about page -->
<p>This is my <strong>personal</strong> website.</p> <!-- paragraph with bolded text -->
<img src="profile.jpg" alt="Profile Picture"> <!-- inserts image with alt attribute -->
</body> <!-- closing body tag -->
