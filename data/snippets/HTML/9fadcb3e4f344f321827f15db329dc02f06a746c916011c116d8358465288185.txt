<!-- This is a basic HTML page -->
<!DOCTYPE html> <!-- Document type declaration -->
<html> <!-- Start of HTML document -->
<head> <!-- Head section of the document -->
	<meta charset="UTF-8"> <!-- Character encoding for the document -->
	<title>Hello World!</title> <!-- Title of the page -->
	<style> /* Internal CSS for styling */
		body {
			font-family: Arial, sans-serif; /* Setting font to Arial or any sans-serif font */
			color: blue;
		}
		h1 {
			color: green; /* Setting color of heading to green */
		}
	</style> <!-- End of internal CSS -->
</head> <!-- End of head section -->
<body> <!-- Body of the page -->
	<h1>Hello World!</h1> <!-- Main heading of the page -->
	<p>This is a basic HTML page.</p> <!-- Paragraph explaining page content -->
	<ul> <!-- Unordered list -->
		<li>Item 1</li> <!-- List item 1 -->
		<li>Item 2</li> <!-- List item 2 -->
		<li>Item 3</li> <!-- List item 3 -->
	</ul> <!-- End of unordered list -->

	<script> /* Script section for JavaScript */
		alert("Hello World!"); /* Displaying "Hello World!" in an alert box */
	</script> <!-- End of script section -->
