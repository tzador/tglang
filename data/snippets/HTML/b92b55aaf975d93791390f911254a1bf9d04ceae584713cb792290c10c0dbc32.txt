<!DOCTYPE html> <!-- doctype declaration for HTML5 -->
<html> <!-- opening tag for the html element -->
  <head> <!-- opening tag for the head element -->
    <meta charset="UTF-8"> <!-- character set declaration -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- responsive design meta tag -->
    <title>My Website</title> <!-- title of the webpage -->
    <style> <!-- opening tag for the style element -->
      body { <!-- opening tag for css body rule -->
        font-family: 'Roboto', sans-serif; <!-- font type for body text -->
        background-color: #f5f5f5; <!-- background color setting -->
      } <!-- closing tag for css body rule -->
      h1 { <!-- opening tag for css h1 rule -->
        color: #009688; <!-- text color setting -->
      } <!-- closing tag for css h1 rule -->
    </style> <!-- closing tag for the style element -->
  </head> <!-- closing tag for the head element -->
  <body> <!-- opening tag for the body element -->
    <div> <!-- opening tag for the div element -->
      <h1>Welcome to My Website</h1> <!-- heading for the webpage -->
      <p>This is a sample paragraph to show how to use HTML.</p> <!-- sample paragraph for the webpage -->
      <button type="button">Click Me</button> <!-- button element for user interaction -->
    </div> <!-- closing tag for the div element -->
    <script> <!-- opening tag for the script element -->
      alert("Hello, World!"); <!-- javascript code to display a popup message -->
    </script> <!-- closing tag for the script element -->
  </body> <!-- closing tag for the body element -->
</html> <!-- closing tag for the html element -->
