<!DOCTYPE html> <!-- Declares the document type as HTML -->
<html> <!-- Root element of an HTML document -->
<head> <!-- Contains metadata about the document -->
  <title>Example</title> <!-- Specifies the document's title -->
  <meta charset="UTF-8"> <!-- Specifies the character encoding for the document -->
  <meta name="description" content="Code Snippet Example"> <!-- Specifies a description of the document -->
  <meta name="author" content="John Doe"> <!-- Specifies the author of the document -->
  <style> /* Defines the style for the document */
    body {
      background-color: #ffffff; /* Sets the background color of the body */
    }
    h1 {
      color: #000000; /* Sets the color of the heading */
      font-size: 32px; /* Sets the font size of the heading */
      font-family: Arial, sans-serif; /* Sets the font family of the heading */
    }
    p {
      color: #666666; /* Sets the color of the paragraph */
      font-size: 14px; /* Sets the font size of the paragraph */
      font-family: Times New Roman, serif; /* Sets the font family of the paragraph */
    }
  </style>
</head>
<body> <!-- Contains the main content of the document -->
  <h1>Hello, World!</h1> <!-- Heading element with the text "Hello, World!" -->
  <p>This is an example of a code snippet in HTML.</p> <!-- Paragraph element with the text "This is an example of a code snippet in HTML." -->
</body>
</html>
