<!DOCTYPE html> <!--defines the document type as HTML-->
<html> <!--denotes the start of the HTML document-->
  <head> <!--denotes the head section of the document-->
    <title>Code Snippet</title> <!--sets the title of the document-->
    <meta charset="UTF-8"> <!--specifies the character encoding-->
    <meta name="description" content="Code Snippet Example"> <!--provides a description of the document-->
    <meta name="keywords" content="HTML, code snippet, example"> <!--provides keywords for search engines-->
    <meta name="author" content="John Doe"> <!--indicates the author of the document-->
    <link rel="stylesheet" href="styles.css"> <!--links an external style sheet-->
  </head>
  <body> <!--denotes the body section of the document-->
    <h1>Welcome to my Code Snippet!</h1> <!--creates a heading element-->
    <p>This is an example of a long and complex code snippet in HTML.</p> <!--creates a paragraph element-->
    <table> <!--creates a table element-->
      <tr> <!--creates a table row-->
        <th>Language</th> <!--creates a table header cell-->
        <th>Code Snippet</th>
      </tr>
      <tr>
        <td>HTML</td> <!--creates a table data cell-->
        <td><code>&lt;html&gt; ... &lt;/html&gt;</code></td> <!--creates a code element with HTML tags-->
      </tr>
      <tr>
        <td>CSS</td>
        <td><code>body { color: red; }</code></td> <!--creates a code element with CSS syntax-->
      </tr>
      <tr>
        <td>JavaScript</td>
        <td><code>alert("Hello World!");</code></td> <!--creates a code element with JavaScript code-->
      </tr>
    </table>
    <script src="scripts.js"></script> <!--links an external JavaScript file-->
  </body>
</html>
