<!-- This is a comment in HTML -->
<!DOCTYPE html> <!-- Specifies the document type as HTML -->
<html> <!-- Opens the HTML document -->

<head> <!-- Defines the document's header section -->
    <title>My Website</title> <!-- Adds a title for the document -->
    <meta charset="UTF-8"> <!-- Specifies the character encoding for the document -->
    <meta name="description" content="This is my personal website"> <!-- Adds a description for the document -->
    <meta name="author" content="John Doe"> <!-- Adds the author's name for the document -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- Specifies the viewport settings for responsive design -->
</head> <!-- Closes the header section -->

<body> <!-- Defines the document's body section -->
    <header> <!-- Defines a header for the document -->
        <h1>Welcome to my website</h1> <!-- Adds a heading tag with text for the header -->
        <nav> <!-- Defines a navigation bar for the header -->
            <ul> <!-- Creates an unordered list for the navigation links -->
                <li><a href="#about">About</a></li> <!-- Adds a link to the About section in the document -->
                <li><a href="#services">Services</a></li> <!-- Adds a link to the Services section in the document -->
                <li><a href="#contact">Contact</a></li> <!-- Adds a link to the Contact section in the document -->
            </ul> <!-- Closes the unordered list -->
        </nav> <!-- Closes the navigation bar -->
    </header> <!-- Closes the header -->

    <main> <!-- Defines the main content area of the document -->
        <section id="about"> <!-- Defines a section with an id for the About section -->
            <h2>About Me</h2> <!-- Adds a heading tag with text for the About section -->
            <p>I am a web developer with a passion for creating beautiful and functional websites.</p> <!-- Adds a paragraph with information about the author -->
            <img src="profilepicture.png" alt="Profile Picture"> <!-- Inserts an image for the author's profile picture -->
        </section> <!-- Closes the About section -->

        <section id="services"> <!-- Defines a section with an id for the Services section -->
            <h2>Services</h2> <!-- Adds a heading tag with text for the Services section -->
            <ul> <!-- Creates an unordered list for the services offered -->
                <li>Web Design</li> <!-- Adds a list item for the Web Design service -->
                <li>Front-end Development</li> <!-- Adds a list item for the Front-end Development service -->
                <li>Back-end Development</li> <!-- Adds a list item for the Back-end Development service -->
            </ul> <!-- Closes the unordered list -->
        </section> <!-- Closes the Services section -->

        <section id="contact"> <!-- Defines a section with an id for the Contact section -->
            <h2>Contact Me</h2> <!-- Adds a heading tag with text for the Contact section -->
            <p>You can reach me at john.doe@example.com or follow me on social media:</p> <!-- Adds a paragraph with contact information -->
            <ul> <!-- Creates an unordered list for the social media icons -->
                <li><a href="https://www.twitter.com"><img src="twitter.png" alt="Twitter Icon"></a></li> <!-- Adds a link to the author's Twitter account with an image of the Twitter icon -->
                <li><a href="https://www.instagram.com"><img src="instagram.png" alt="Instagram Icon"></a></li> <!-- Adds a link to the author's Instagram account with an image of the Instagram icon -->
            </ul> <!-- Closes the unordered list -->
        </section> <!-- Closes the Contact section -->
    </main> <!-- Closes the main content area -->

    <footer> <!-- Defines the document's footer section -->
        <p>&copy; 2020 John Doe</p> <!-- Adds a paragraph with the current year and author's name for the footer -->
    </footer> <!-- Closes the footer -->
</body> <!-- Closes the body section -->

</html> <!-- Closes the HTML document -->
