<!-- This is an HTML comment -->
<!DOCTYPE html> <!-- This defines the document type as HTML5 -->
<html><!-- This is the root element -->

<head><!-- This is the head section of the webpage -->
  <title>My Website</title> <!-- This defines the title of the webpage -->
  <link rel="stylesheet" href="styles.css"> <!-- This links the external stylesheet -->
  <meta charset="UTF-8"> <!-- This defines the character encoding of the webpage -->
</head>

<body><!-- This is the body section of the webpage -->
  <h1>Welcome to My Website!</h1> <!-- This is a heading -->
  <p>This is a paragraph with <strong>strong</strong> and <em>emphasized</em> text.</p> <!-- This is a paragraph with formatted text -->
  <ul> <!-- This is an unordered list -->
    <li>Item 1</li> <!-- This is the first list item -->
    <li>Item 2</li> <!-- This is the second list item -->
    <li>Item 3</li> <!-- This is the third list item -->
  </ul>
  <img src="image.jpg" alt="Image"> <!-- This is an image with an alternate text -->
  <a href="https://www.example.com">Link to Example Website</a> <!-- This is a link to another webpage -->
</body>

</html><!-- This is the closing tag for the root element -->
