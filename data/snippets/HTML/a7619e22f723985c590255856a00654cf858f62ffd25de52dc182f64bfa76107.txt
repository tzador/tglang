<html> <!-- opening tag for HTML document -->
  <head> <!-- opening tag for the head section -->
    <title>My Webpage</title> <!-- defining the title of the webpage -->
    <style type="text/css"> <!-- defining the style for the webpage -->
      body {
        color: blue;
        background-color: lightgrey;
      }
    </style>
  </head> <!-- closing tag for the head section -->
  
  <body> <!-- opening tag for the body section -->
    <h1>Welcome to my webpage!</h1> <!-- defining a heading in the body -->
    <p>This is a paragraph with some <strong>bold</strong> text and <em>italic</em> text.</p> <!-- defining a paragraph with some styling -->
    
    <ul> <!-- opening tag for an unordered list -->
      <li>First item</li> <!-- defining the first item in the list -->
      <li>Second item</li> <!-- defining the second item in the list -->
      <li>Third item</li> <!-- defining the third item in the list -->
    </ul> <!-- closing tag for the unordered list -->
    
    <ol> <!-- opening tag for an ordered list -->
      <li>Item one</li> <!-- defining the first item in the list -->
      <li>Item two</li> <!-- defining the second item in the list -->
      <li>Item three</li> <!-- defining the third item in the list -->
    </ol> <!-- closing tag for the ordered list -->
    
    <img src="myimage.jpg" alt="My Image"> <!-- inserting an image with a source and alternate text -->
    
    <a href="https://www.example.com">Link to Example Website</a> <!-- adding a link to an external website -->
  </body> <!-- closing tag for the body section -->
</html> <!-- closing tag for the HTML document -->
