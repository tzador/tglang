<!DOCTYPE html> <!-- This denotes the document type as HTML -->
<html> <!-- This is the opening tag for the HTML document -->
  <head> <!-- This is the head element of the document -->
    <title>My Website</title> <!-- This is the title of the webpage -->
    <meta charset="UTF-8"> <!-- This specifies the character encoding of the document -->
    <meta name="description" content="Welcome to my website!"> <!-- This provides a brief description of the webpage -->
    <meta name="keywords" content="HTML, website, coding"> <!-- This provides keywords for search engine optimization -->
    <meta name="author" content="John Smith"> <!-- This specifies the author of the webpage -->
    <link rel="stylesheet" type="text/css" href="styles.css"> <!-- This links the document to an external stylesheet for styling purposes -->
  </head>
  <body> <!-- This is the body element of the document -->
    <header> <!-- This is the header section of the webpage -->
      <h1>My Website</h1> <!-- This is the main heading of the webpage -->

        <ul> <!-- This is an unordered list of links for navigation -->
          <li><a href="#about">About</a></li> <!-- This is a list item with a link to the about section -->
          <li><a href="#services">Services</a></li> <!-- This is a list item with a link to the services section -->

      </nav>

    <main> <!-- This is the main section of the webpage -->
      <section id="about"> <!-- This is the about section of the webpage with a specified id for linking purposes -->
