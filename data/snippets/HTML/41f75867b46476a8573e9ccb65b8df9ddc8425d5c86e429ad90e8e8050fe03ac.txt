<!--This is a complex code snippet in HTML-->
<!DOCTYPE html> <!--Declares the document type as HTML-->
<html> <!--Starts the HTML document-->
<head> <!--The head element contains information about the document-->
    <title>My Website</title> <!--The title element is used to define the title of the document-->
    <link rel="stylesheet" href="styles.css"> <!--The link element is used to link an external CSS file-->
    <script src="script.js"></script> <!--The script element is used to link an external JavaScript file-->
</head>
<body> <!--The body element contains the visible content of the document-->
    <header> <!--The header element defines a header for a document or section-->
        <nav> <!--The nav element defines a navigation bar-->
            <a href="#home">Home</a> <!--The a element is used to create a hyperlink-->
            <a href="#about">About</a>
            <a href="#contact">Contact</a>
        </nav>
    </header>
    <main> <!--The main element defines the main content of a document-->
        <section> <!--The section element defines a section in a document-->
            <h1>Welcome to my website!</h1> <!--The h1 element defines a heading-->
            <p>This is a complex code snippet in HTML. It uses different elements to create a webpage.</p> <!--The p element defines a paragraph-->
        </section>
        <section>
            <h2>About Me</h2>
            <p>I am a web developer with a passion for coding and creating beautiful websites.</p>
        </section>
    </main>
    <footer> <!--The footer element defines a footer for a document or section-->
        <p>Created by John Doe &copy; 2021</p> <!--The &copy; character entity is used to create a copyright symbol-->
    </footer>
</html> <!--Ends the HTML document-->
