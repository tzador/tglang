<!-- This is a comment in HTML code -->
<!DOCTYPE html>
<!-- This is a declaration of HTML document type -->

<html lang="en">
<!-- This is the opening HTML tag -->

<head>
    <!-- This is the head tag, where we include metadata and external resources -->
    <meta charset="UTF-8">
    <!-- This specifies the character set for the document -->
    <title>My Website</title>
    <!-- This is the title of our HTML document -->

    <!-- This is a link to an external CSS stylesheet -->
    <link rel="stylesheet" href="style.css">
</head>

<body>
    <!-- This is the body tag, where we put all the visible content -->
    <h1>Hello World!</h1>
    <!-- This is a heading element with the text "Hello World!" -->

    <!-- This is a paragraph element with some text -->
    <p>Welcome to my website. This is a long and complex code snippet, but it showcases the power of HTML as a markup language. It allows us to structure and organize our content in a meaningful way.</p>

    <!-- This is an unordered list element with three list items -->
    <ul>
        <li>Item 1</li>
        <li>Item 2</li>
        <li>Item 3</li>
    </ul>

    <!-- This is an image element with a source and alt attribute -->
    <img src="image.jpg" alt="A beautiful landscape">
</body>

</html>
<!-- This is the closing HTML tag -->
