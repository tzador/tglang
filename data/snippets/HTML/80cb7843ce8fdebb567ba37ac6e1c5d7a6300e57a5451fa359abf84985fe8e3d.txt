<!-- This is a comment in HTML code -->
<!DOCTYPE html> <!-- this line specifies the document type as HTML -->
<html> <!-- this is the opening tag for the HTML document -->
<head> <!-- this is the head section of the document, contains metadata and other information -->
    <title>Page Title</title> <!-- this is where the title of the page goes -->
    <meta charset="UTF-8"> <!-- this specifies the character encoding for the document -->
    <meta name="description" content="This is an example website"> <!-- this provides a description for the webpage -->
    <meta name="keywords" content="HTML, CSS, JavaScript"> <!-- this provides keywords for search engines -->
    <meta name="author" content="John Doe"> <!-- this identifies the author of the webpage -->
</head> <!-- end of head section -->
<body> <!-- this is the body section of the document, contains the visible content -->
    <h1>Main Heading</h1> <!-- this is a top-level heading -->
    <p>This is a paragraph.</p> <!-- this is a paragraph element -->
    <a href="https://www.example.com">Example Link</a> <!-- this is a hyperlink -->
    <img src="image.jpg" alt="Image Description"> <!-- this is an image element -->
    <ul> <!-- this is an unordered list -->
        <li>List item 1</li> <!-- this is a list item -->
        <li>List item 2</li>
        <li>List item 3</li>
    </ul>
    <ol> <!-- this is an ordered list -->
        <li>List item 1</li>
        <li>List item 2</li>
        <li>List item 3</li>
    </ol>
    <table> <!-- this is a table -->
        <tr> <!-- this is a table row -->
            <th>Column 1</th> <!-- this is a table header -->
            <th>Column 2</th>
        </tr>
        <tr>
            <td>Data 1</td> <!-- this is a table data cell -->
            <td>Data 2</td>
        </tr>
    </table>
</body> <!-- end of body section -->
</html> <!-- end of HTML document -->
