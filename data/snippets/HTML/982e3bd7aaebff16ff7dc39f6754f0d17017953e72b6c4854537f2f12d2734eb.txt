<html> <!--Opening tag for the root element of the html document-->
<body> <!--Opening tag for the body of the document-->
<h1 style="color:red;">Hello World!</h1> <!--Opening and closing tags for a level one heading with inline styling to change font color to red-->
<ul> <!--Opening tag for an unordered list-->
<li>List Item 1</li> <!--List item within the unordered list-->
<li>List Item 2</li> <!--List item within the unordered list-->
<li>List Item 3</li> <!--List item within the unordered list-->
</ul> <!--Closing tag for the unordered list-->
<p>This is a paragraph of text within the body of the document. It may contain <strong>bold</strong> and <em>italicized</em> words.</p> <!--Opening and closing tags for a paragraph with nested tags for adding emphasis and strong emphasis-->
<a href="https://www.example.com">Link to Example Website</a> <!--Opening and closing tags for a hyperlink with the URL specified in the "href" attribute-->
<div class="container"> <!--Opening tag for a div element with a class of "container"-->
<p>This is another paragraph of text within the div element.</p> <!--Paragraph within the div element-->
</div> <!--Closing tag for the div element-->
<img src="image.jpg" alt="Image of a mountain"> <!--Self-closing tag for an image element with the source file specified in the "src" attribute and alternate text in the "alt" attribute-->
</body> <!--Closing tag for the body of the document-->
</html> <!--Closing tag for the root element of the html document-->
