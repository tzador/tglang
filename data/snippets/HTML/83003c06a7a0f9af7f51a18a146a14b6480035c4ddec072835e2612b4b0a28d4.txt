<!-- This is a comment -->
<!-- HTML document declaration -->
<!DOCTYPE html>
<html>
  <head>
    <!-- The <head> element contains meta information and page resources -->
    <!-- The <meta> element sets the character set for HTML document -->
    <meta charset="UTF-8">
    <!-- The <title> element sets the title of the page -->
    <title>My HTML Page</title>
    <!-- CSS stylesheets can be linked to the HTML document using the <link> element -->
    <link rel="stylesheet" href="style.css">
    <!-- JavaScript files can be linked to the HTML document using the <script> element -->
    <script src="script.js"></script>
  </head>
  <body>
    <!-- The <body> element contains the visible content of the HTML document -->
    <h1>Welcome to My HTML Page</h1>
    <!-- The <p> element defines a paragraph -->
    <p>This is a long and complex code snippet in HTML.</p>
    <!-- The <a> element creates a link with the specified attributes -->
    <a href="https://www.example.com">Example Link</a>
    <!-- The <img> element embeds an image into the HTML document -->
    <img src="example.jpg" alt="Example Image">
    <!-- The <ul> element creates an unordered list -->
    <ul>
      <!-- The <li> element creates a list item -->
      <li>Item 1</li>
      <li>Item 2</li>
      <li>Item 3</li>
    </ul>
    <!-- The <table> element creates a table -->
    <table>
      <!-- The <tr> element creates a row -->
      <tr>
        <!-- The <td> element creates a cell in a row -->
        <td>Cell 1</td>
        <td>Cell 2</td>
      </tr>
      <tr>
        <td>Cell 3</td>
        <td>Cell 4</td>
      </tr>
    </table>
  </body>
</html>
