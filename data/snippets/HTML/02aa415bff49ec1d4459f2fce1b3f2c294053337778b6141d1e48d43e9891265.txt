<!DOCTYPE html> <!-- Defines the document type as HTML -->
<html> <!-- Opening tag for the HTML document -->
<head> <!-- The 'head' element contains information
                about the document, such as the title -->
    <title>My Complex HTML Document</title> <!-- Sets the title of the document -->
    <meta charset="UTF-8"/> <!-- Defines the character encoding for the document -->
    <meta name="description" content="A complex HTML document"/> <!-- Sets the description of the document for search engines -->
    <meta name="author" content="John Smith"/> <!-- Sets the author of the document -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"/> <!-- Defines the viewport for responsive design -->
</head> <!-- Closing tag for the 'head' element -->
<body> <!-- The 'body' element contains all visible content of the document -->
    <header> <!-- The 'header' element usually contains a logo or heading -->
        <h1>Welcome to My Complex HTML Document</h1> <!-- Sets the main heading of the document -->
    </header> <!-- Closing tag for the 'header' element -->
    <nav> <!-- The 'nav' element contains navigation links for the document -->
        <a href="#section1">Section 1</a> <!-- Links to a specific section of the document -->
        <a href="#section2">Section 2</a>
        <a href="#section3">Section 3</a>
    </nav> <!-- Closing tag for the 'nav' element -->
    <main> <!-- The 'main' element contains the main content of the document -->
        <section id="section1"> <!-- The 'section' element organizes content into different sections -->
            <h2>Section 1</h2> <!-- Sets the heading for this section -->
            <p>This is some text in section 1.</p> <!-- Sets a paragraph of text for this section -->
            <ul> <!-- The 'ul' element creates an unordered list -->
                <li>List item 1</li> <!-- Adds a list item to the unordered list -->
                <li>List item 2</li>
            </ul> <!-- Closing tag for the 'ul' element -->
        </section> <!-- Closing tag for the 'section1' element -->
        <section id="section2">
            <h2>Section 2</h2>
            <p>This is some text in section 2.</p>
            <ol> <!-- The 'ol' element creates an ordered list -->
                <li>List item 1</li>
                <li>List item 2</li>
            </ol> <!-- Closing tag for the 'ol' element -->
        </section> <!-- Closing tag for the 'section2' element -->
        <section id="section3">
            <h2>Section 3</h2>
            <p>This is some text in section 3.</p>
            <dl> <!-- The 'dl' element creates a definition list -->
                <dt>Term 1</dt> <!-- Defines a term in the list -->
                <dd>This is the definition for term 1.</dd> <!-- Defines the definition for term 1 -->
                <dt>Term 2</dt>
                <dd>This is the definition for term 2.</dd>
            </dl> <!-- Closing tag for the 'dl' element -->
        </section> <!-- Closing tag for the 'section3' element -->
    </main> <!-- Closing tag for the 'main' element -->
    <footer> <!-- The 'footer' element contains information about the document's author or copyright -->
        <p>Created by John Smith</p> <!-- Displays the author's name -->
        <small>&copy; 2021 My Complex HTML Document</small> <!-- Displays the document's copyright information -->
    </footer> <!-- Closing tag for the 'footer' element -->
</body> <!-- Closing tag for the 'body' element -->
</html> <!-- Closing tag for the HTML document -->
