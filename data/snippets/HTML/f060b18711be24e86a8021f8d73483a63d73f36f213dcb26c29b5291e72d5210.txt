<!DOCTYPE html> <!-- Defines the document type as HTML5 -->
<html> <!-- Opening tag for the <html> element -->
<head> <!-- Opening tag for the <head> element -->
    <title>Welcome to My Website</title> <!-- Sets the title of the document to "Welcome to My Website" -->
    <meta charset="UTF-8"> <!-- Sets the character encoding to UTF-8 -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- Sets the viewport to the initial scale of 1 -->
    <link rel="stylesheet" type="text/css" href="style.css"> <!-- Links the CSS stylesheet "style.css" to the document -->
</head> <!-- Closing tag for the <head> element -->
<body> <!-- Opening tag for the <body> element -->
    <h1>Hello World!</h1> <!-- Displays the heading "Hello World!" -->
    <p>This is a paragraph.</p> <!-- Displays the text "This is a paragraph." -->
    <img src="image.jpg" alt="Image"> <!-- Displays the image "image.jpg" with the alt text "Image" -->
    <ul> <!-- Opening tag for the <ul> element, creates an unordered list -->
        <li>Item 1</li> <!-- Displays list item 1 -->
        <li>Item 2</li> <!-- Displays list item 2 -->
        <li>Item 3</li> <!-- Displays list item 3 -->
    </ul> <!-- Closing tag for the <ul> element -->
    <form> <!-- Opening tag for the <form> element -->
        <label for="name">Name:</label> <!-- Creates a label for the input field with the id "name" -->
        <input type="text" id="name" name="name"> <!-- Creates a text input field with the id "name" and name "name" -->
        <input type="submit" value="Submit"> <!-- Creates a submit button with the text "Submit" -->
    </form> <!-- Closing tag for the <form> element -->
</body> <!-- Closing tag for the <body> element -->
</html> <!-- Closing tag for the <html> element -->
