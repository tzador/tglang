<!DOCTYPE html>
<html>
<!-- This is the opening tag of the HTML document
and includes the DOCTYPE declaration and the start of the HTML element. -->

<head>
  <title>My Website</title>
  <!-- This is the opening tag of the head element which contains metadata
  about the HTML document. The title element specifies the title of the webpage
  that will appear in the browser's tab. -->
</head>

<body>
  <!-- This is the opening tag of the body element which contains the visible
  parts of the webpage. -->

  <header>
    <!-- This is the opening tag of the header element which is used to
    structure and organize the content of the header section of the webpage. -->
    <nav>
      <!-- This is the opening tag of the nav element which is used to define
      a set of navigation links for the webpage. -->
      <ul>
        <!-- This is an unordered list element used to create a bulleted list. -->
        <li><a href="#">Home</a></li>
        <!-- This is a list item element which contains a link to the homepage.
        The '#' symbol is a placeholder for the link destination. -->
        <li><a href="#">About</a></li>
        <!-- This is another list item element which contains a link to the
        about page. -->
        <li><a href="#">Contact</a></li>
        <!-- This is a third list item element which contains a link to the
        contact page. -->
      </ul>
    </nav>
  </header>

  <main>
    <!-- This is the opening tag of the main element which is used to contain
    the main content of the webpage. -->
    <h1>Welcome to My Website!</h1>
    <!-- This is the opening tag of the h1 element which is used to define
    a heading on the webpage. The text within the tags will appear as the
    heading. -->
    <p>This is a paragraph on my website.</p>
    <!-- This is the opening tag of the p element which is used to define
    a paragraph on the webpage. The text within the tags will appear as the
    paragraph. -->
    <img src="myimage.jpg" alt="My Image">
    <!-- This is an image element which is used to include an image on the
    webpage. The 'src' attribute specifies the location of the image and the
    'alt' attribute provides alternative text for the image. -->
  </main>

  <footer>
    <!-- This is the opening tag of the footer element which is used to
    structure and organize the content of the footer section of the webpage. -->
    <p>&copy; 2021 My Website</p>
    <!-- This is another paragraph element which includes the current year and
    the name of the website, with the copyright symbol. -->
  </footer>

</body>

</html>
<!-- This is the closing tag of the HTML document. -->
