<!DOCTYPE html> <!-- The doctype declaration is used to instruct the browser which version of HTML to use -->

<html> <!-- Opening tag for the HTML document -->

<head> <!-- Contains meta information about the HTML document -->

    <title>Code Snippet</title> <!-- Specifies the title of the HTML document -->

    <meta charset="UTF-8"> <!-- Specifies the character encoding for the HTML document -->

</head> <!-- Closing tag for the head section -->

<body> <!-- Contains the visible content of the HTML document -->

    <h1>This is a heading</h1> <!-- Defines a heading for the HTML document -->

    <p>This is a paragraph</p> <!-- Defines a paragraph of text for the HTML document -->

    <ul> <!-- Used to create a list of items -->
        <li>First item</li> <!-- Defines the first item in the list -->
        <li>Second item</li> <!-- Defines the second item in the list -->
        <li>Third item</li> <!-- Defines the third item in the list -->
    </ul> <!-- Closing tag for the unordered list -->

    <a href="https://www.example.com">Link to Example Website</a> <!-- Creates a hyperlink to the specified webpage -->

    <img src="image.jpg" alt="Image description"> <!-- Inserts an image into the HTML document -->

    <table> <!-- Used to create a table -->
        <thead> <!-- Contains the table header information -->
            <tr> <!-- Defines a row in the table -->
                <th>First column</th> <!-- Defines the first column in the table header -->
                <th>Second column</th> <!-- Defines the second column in the table header -->
            </tr> <!-- Closing tag for the table row -->
        </thead> <!-- Closing tag for the table header -->
        <tbody> <!-- Contains the table body information -->
            <tr> <!-- Defines a row in the table -->
                <td>First column, first row</td> <!-- Defines the first cell in the first row -->
                <td>Second column, first row</td> <!-- Defines the second cell in the first row -->
            </tr> <!-- Closing tag for the table row -->
            <tr> <!-- Defines a row in the table -->
                <td>First column, second row</td> <!-- Defines the first cell in the second row -->
                <td>Second column, second row</td> <!-- Defines the second cell in the second row -->
            </tr> <!-- Closing tag for the table row -->
        </tbody> <!-- Closing tag for the table body -->
    </table> <!-- Closing tag for the table -->
