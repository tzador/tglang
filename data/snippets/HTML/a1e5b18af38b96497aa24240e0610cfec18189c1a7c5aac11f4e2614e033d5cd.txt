<!DOCTYPE html> <!-- This line indicates that the document is an HTML file -->
<html> <!-- Defines the root element of an HTML page -->

<head> <!-- Contains the metadata of the document such as title, styles, scripts-->
  <meta charset="UTF-8"> <!-- Specifies the character encoding for the document -->
  <title>My HTML Page</title> <!-- Defines the title of the document -->
  <link rel="stylesheet" href="style.css"> <!-- Link to an external style sheet -->
  <script src="script.js"></script> <!-- Link to an external JavaScript file -->
</head>

<body> <!-- Defines the main content of the document -->
  <h1>Welcome</h1> <!-- Defines a heading level 1 -->
  <p>Hello, World!</p> <!-- Defines a paragraph -->
  <ul> <!-- Defines an unordered list -->
    <li>Item 1</li> <!-- Defines a list item -->
    <li>Item 2</li>
    <li>Item 3</li>
  </ul>
  <a href="https://www.example.com">Link to Example</a> <!-- Defines a hyperlink to a URL -->
  <img src="image.jpg" alt="Example Image"> <!-- Defines an image element -->
</body>

</html>
