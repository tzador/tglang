<DOCTYPE html> <!-- Declares the document type as HTML -->
<html> <!-- Begins the HTML document -->
<head> <!-- Starts the head section -->
  <title>My Website</title> <!-- Sets the title of the webpage -->
  <style> <!-- Begins the style section -->
    /* Sets the font family for the entire document */
    body {
      font-family: Arial, sans-serif; 
    }

    /* Styles the headings */
    h1 {
      color: blue;
      font-size: 36px;
      text-align: center;
    }

    /* Styles the paragraphs */
    p {
      color: black;
      font-size: 18px;
      margin: 10px 0; /* Sets the top and bottom margins to 10px and the left and right margins to 0 */
    }

    /* Styles the links */
    a {
      color: purple;
      text-decoration: none; /* Removes the underline from the links */
    }

    /* Styles the images */
    img {
      width: 100%; /* Makes the image take up the full width of its parent element */
      height: auto; /* Keeps the aspect ratio of the image */
      display: block; /* Displays the image as a block element to remove any extra spacing around it */
      margin: 0 auto; /* Centers the image horizontally */
    }
  </style> <!-- Ends the style section -->
</head> <!-- Ends the head section -->
<body> <!-- Starts the body section -->
  <h1>Welcome to my website!</h1> <!-- Adds a heading to the page -->
  <p>This is my personal website where I share my thoughts and experiences.</p> <!-- Adds a paragraph to the page -->
  <p>You can also check out <a href="https://www.example.com">my blog</a> for more of my writing.</p> <!-- Adds a link to the page -->
  <img src="https://www.example.com/images/my-photo.jpg" alt="My Photo"> <!-- Adds an image to the page with a alternate text -->
</body> <!-- Ends the body section -->
</html> <!-- Ends the HTML document -->
