<html> <!-- HTML document -->
  <head> <!-- Contains metadata for the document -->
    <title>My website</title> <!-- Sets the title for the document -->
    <style> <!-- Contains CSS for the document -->
      /* This is comment in CSS */
      body {
        background-color: #FFFFFF; /* Sets the background color to white */
        font-family: "Arial", sans-serif; /* Sets the font family to Arial */
      }
      h1 {
        color: #000000; /* Sets the color of heading 1 to black */
        font-size: 48px; /* Sets the font size to 48 pixels */
      }
      p {
        color: #333333; /* Sets the color of paragraphs to dark grey */
        font-size: 16px; /* Sets the font size to 16 pixels */
        line-height: 1.5; /* Sets the line height to 1.5 */
      }
    </style>
  </head>
  <body> <!-- Contains the visible content of the document -->
    <h1>Welcome to My Website</h1> <!-- Displays the heading 1 -->
    <p>This is a paragraph on my website. <!-- Displays a paragraph -->
      It contains some text and will have a font size of 16 pixels.
    </p>
    <p>This is another paragraph with a line break below. <br> <!-- Displays a paragraph with a line break -->
      The line break is achieved using the br tag.
    </p>
  </body>
</html>
