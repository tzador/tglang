<!-- This is a comment in HTML. Comments are used to explain the code and are not visible on the webpage. -->

<!-- HTML uses tags to define the structure of a webpage. Here, we are creating a web page using the <html> tag. -->
<html>

<!-- The <head> tag contains meta information about the webpage such as the title, character encoding, stylesheets, etc. -->
<head>
  <title>My Webpage</title>
  <!-- This meta tag specifies the character encoding of the webpage. -->
  <meta charset="UTF-8">
  <!-- This link tag imports a stylesheet named style.css to add styles to the webpage. -->
  <link rel="stylesheet" href="style.css">
</head>

<!-- The <body> tag contains all the visible content of the webpage. -->
<body>
  <!-- The <h1> tag is used to define headings, with <h1> being the largest and <h6> being the smallest. -->
  <h1>Welcome to my webpage</h1>

  <!-- The <p> tag is used to define paragraphs. -->
  <p>This is a paragraph. Lorem ipsum dolor sit amet, consectetur adipiscing elit. In dictum ullamcorper mi, ac porttitor odio consequat nec. Phasellus rutrum dictum elit, at eleifend lectus sodales quis. Sed eu justo tellus. Proin sed tempor risus, et egestas orci. In tempor, sem vel laoreet blandit.</p>

  <!-- The <ul> tag defines an unordered list. Inside, <li> tags are used to define list items. -->
  <ul>
    <li>Item 1</li>
    <li>Item 2</li>
    <li>Item 3</li>
  </ul>

  <!-- The <img> tag is used to display images on the webpage. -->
  <img src="image.jpg" alt="Image description">

  <!-- The <a> tag defines links, with the href attribute specifying the URL of the webpage being linked. -->
  <a href="https://www.example.com">Visit my website</a>

  <!-- The <table> tag is used to create a table. Inside, <tr> defines rows and <td> defines cells in the table. -->
  <table>
    <tr>
      <td>Cell 1</td>
      <td>Cell 2</td>
    </tr>

    <tr>
      <td>Cell 3</td>
      <td>Cell 4</td>
    </tr>
  </table>
</body>

</html>
