<!-- This is a comment in HTML -->
<!DOCTYPE html> <!-- Defines the document type as HTML -->
<html> <!-- Opening tag for the HTML document -->
<head> <!-- Head section of the HTML document -->
  <title>This is the document title</title> <!-- Title of the document -->
  <script type="text/javascript" src="script.js"></script> <!-- Link to external JavaScript file -->
  <style> /* Inline style for the document */
    body {
      background-color: #fff; /* Sets the background color of the body */
    }
  </style>
</head> <!-- End of head section -->
<body> <!-- Body section of the HTML document -->
  <h1>This is a heading</h1> <!-- Heading element -->
  <p>This is a paragraph</p> <!-- Paragraph element -->
  <a href="https://www.example.com">This is a link</a> <!-- Link element -->
  <img src="image.png" alt="This is an image"> <!-- Image element with alt attribute for accessibility -->
</body> <!-- End of body section -->
</html> <!-- Closing tag for the HTML document -->
