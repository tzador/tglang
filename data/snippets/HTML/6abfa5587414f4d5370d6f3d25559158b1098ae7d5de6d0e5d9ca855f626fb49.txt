<!DOCTYPE html>
<!-- This comment line denotes the type of the document. -->

<html>
<!-- This tag encloses the entire HTML document. -->

<head>
  <!-- This tag contains meta information about the document. -->
  <meta charset="UTF-8">
  <!-- This specifies the character encoding for the document. -->
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <!-- This tag defines the viewport for responsive design. -->
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <!-- This tag specifies compatibility with Internet Explorer. -->

  <title>Code Snippet</title>
  <!-- This tag sets the title of the document. -->
  
  <style>
    /* This tag is used to write inline CSS for the document. */
    body {
      /* This CSS rule specifies the font family for the entire document. */
      font-family: Arial, sans-serif;
    }
    
    h1 {
      /* This CSS rule specifies the font size and color for all h1 headers. */
      font-size: 36px;
      color: #1e90ff;
    }
  </style>
</head>

<body>
  <!-- This tag contains the body of the HTML document. -->

  <h1>Code Snippet</h1>
  <!-- This tag creates a header with the text "Code Snippet". -->

  <p>This is an example of a code snippet in HTML.</p>
  <!-- This tag creates a paragraph with the text "This is an example of a code snippet in HTML." -->

  <ul>
    <!-- This tag creates an unordered list. -->
    <li>HTML: Hypertext Markup Language</li>
    <!-- This tag creates a list item with the text "HTML: Hypertext Markup Language". -->
    <li>CSS: Cascading Style Sheets</li>
    <!-- This tag creates a list item with the text "CSS: Cascading Style Sheets". -->
    <li>JS: JavaScript</li>
    <!-- This tag creates a list item with the text "JS: JavaScript". -->
  </ul>

  <a href="https://www.example.com">Click here</a>
  <!-- This tag creates a hyperlink with the text "Click here" and the URL "https://www.example.com". -->

  <img src="image.jpg" alt="Image">
  <!-- This tag creates an image with the source "image.jpg" and the alternative text "Image". -->

</body>

</html>
<!-- This tag marks the end of the HTML document. -->
