<!DOCTYPE html> <!-- declares the document type as HTML -->
<html> <!-- main root element of an HTML page -->
  <head> <!-- contains metadata and title of the page -->
    <meta charset="UTF-8"> <!-- defines the character set for the document -->
    <title>My First HTML Page</title> <!-- sets the title of the page -->
    <style> /* optional element for applying CSS styles to the page */
      body {
        font-family: Arial, sans-serif; /* sets the font family for the body text */
      }
      h1 {
        color: blue; /* sets the color of the heading text to blue */
      }
    </style>
  </head>
  <body> <!-- contains the main content of the page -->
    <h1>Welcome to My Page</h1> <!-- heading element with the text "Welcome to My Page" -->
    <p>This is my first HTML page.</p> <!-- paragraph element with the text "This is my first HTML page." -->
    <ul> <!-- unordered list element -->
      <li>Item 1</li> <!-- list item with the text "Item 1" -->
      <li>Item 2</li> <!-- list item with the text "Item 2" -->
      <li>Item 3</li> <!-- list item with the text "Item 3" -->
    </ul>
    <img src="image.jpg" alt="Image"> <!-- image element with a source and alternate text -->
    <a href="https://www.example.com">Visit Example Website</a> <!-- anchor element with a link and anchor text -->
    <input type="text" placeholder="Enter your name"> <!-- input element for user input -->
  </body>
</html>
