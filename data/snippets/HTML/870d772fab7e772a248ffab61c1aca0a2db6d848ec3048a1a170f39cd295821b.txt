<!DOCTYPE html>    <!-- declares document type as HTML -->
<html>    <!-- opening tag for HTML document -->
<head>    <!-- opening tag for head section -->
<title>This is a complex HTML code snippet</title>    <!-- sets the title of the document -->
</head>     <!-- closing tag for head section -->
<body>    <!-- opening tag for body section -->
<header>    <!-- opening tag for header section -->
<nav>    <!-- opening tag for navigation section -->
<ul>    <!-- opening tag for unordered list -->
<li><a href="#home">Home</a></li>    <!-- creates a list item with a link to "home" section -->
<li><a href="#about">About</a></li>    <!-- creates a list item with a link to "about" section -->
<li><a href="#contact">Contact</a></li>    <!-- creates a list item with a link to "contact" section -->
</ul>    <!-- closing tag for unordered list -->
</nav>    <!-- closing tag for navigation section -->
</header>    <!-- closing tag for header section -->
<main>    <!-- opening tag for main section -->
<h1>Hello, World!</h1>    <!-- creates a heading in main section -->
<p>This is a complex <em>HTML</em> code snippet that showcases various tags and their usage.</p>    <!-- creates a paragraph in main section with emphasized "HTML" text -->
<section id="home">    <!-- defines a section with an id of "home" -->
<h2>Welcome!</h2>    <!-- creates a heading in "home" section -->
<p>This is the home section of the document.</p>    <!-- creates a paragraph in "home" section -->
</section>    <!-- closing tag for "home" section -->
<section id="about">    <!-- defines a section with an id of "about" -->
<h2>About Us</h2>    <!-- creates a heading in "about" section -->
<p>We are a team of developers who love coding and creating amazing things with HTML.</p>    <!-- creates a paragraph in "about" section -->
</section>    <!-- closing tag for "about" section -->
<section id="contact">    <!-- defines a section with an id of "contact" -->
<h2>Contact Us</h2>    <!-- creates a heading in "contact" section -->
<p>If you have any questions or want to collaborate, feel free to reach out to us!</p>    <!-- creates a paragraph in "contact" section -->
<address>    <!-- opening tag for address element -->
<p><strong>Address:</strong> 123 Main St, Anytown, USA</p>    <!-- creates a paragraph with a bolded "address" text -->
<p><strong>Email:</strong> contact@htmlcode.com</p>    <!-- creates a paragraph with a bolded "email" text -->
<p><strong>Phone:</strong> 555-555-1234</p>    <!-- creates a paragraph with a bolded "phone" text -->
</address>    <!-- closing tag for address element -->
</section>    <!-- closing tag for "contact" section -->
</main>    <!-- closing tag for main section -->
<footer>    <!-- opening tag for footer section -->
<p>Made with ❤️ and HTML</p>    <!-- creates a paragraph in footer section -->
</footer>    <!-- closing tag for footer section -->
</body>    <!-- closing tag for body section -->
</html>    <!-- closing tag for HTML document -->
