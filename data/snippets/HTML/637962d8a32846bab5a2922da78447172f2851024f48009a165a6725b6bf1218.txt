<!-- This is a comment in HTML -->
<!DOCTYPE html> 
<html> <!-- This is the opening tag for HTML document -->
<head> <!-- This is the start of the head section -->
    <title>My Website</title> <!-- This is the title of the HTML document -->
    <meta charset="UTF-8"> <!-- This declares the character encoding for the document -->
    <meta name="description" content="This is a sample website."> <!-- This is a meta tag for description -->
    <meta name="author" content="John Doe"> <!-- This is a meta tag for author -->
    <link rel="stylesheet" type="text/css" href="mystyle.css"> <!-- This is a link tag to link an external CSS file -->
</head> <!-- End of the head section -->
<body> <!-- This is the start of the body section -->
    <h1>Welcome to my website!</h1> <!-- This is a heading element -->
    <p>This is a sample paragraph.</p> <!-- This is a paragraph element -->
    <img src="image.jpg" alt="Sample Image"> <!-- This is an image tag with an alt attribute -->
    <ul> <!-- This is an unordered list -->
        <li>Item 1</li> <!-- This is a list item -->
        <li>Item 2</li> <!-- This is a list item -->
        <li>Item 3</li> <!-- This is a list item -->
    </ul> <!-- End of unordered list -->
    <script src="myscript.js"></script> <!-- This is a script tag to link an external JavaScript file -->
</body> <!-- End of the body section -->
</html> <!-- End of the HTML document -->
