<!-- This is a comment in HTML code -->
<!DOCTYPE html> <!-- declaring the document type as HTML -->
<html> <!-- opening the HTML document -->
    <head> <!-- opening the head of the document -->
        <title>My Webpage</title> <!-- setting the title of the webpage, to be displayed in the browser tab -->
        <style> <!-- adding a style element in the head to add CSS styles -->
            body { <!-- CSS code for the body of the document -->
                font-family: Arial, sans-serif; <!-- setting the font of the webpage to Arial or sans-serif as a fallback -->
                background-color: #282c34; <!-- setting the background color to a dark shade -->
                color: #fff; <!-- setting the text color to white -->
            }
        </style>
    </head>
    <body> <!-- opening the body of the document -->
        <h1>Hello, world!</h1> <!-- adding a heading element with the text "Hello, world!" -->
        <p>This is my webpage. Welcome!</p> <!-- adding a paragraph element with some text -->
        <img src="https://example.com/image.jpg" alt="An example image"> <!-- adding an image element with a source and an alt attribute -->
        <a href="https://example.com">Click here</a> <!-- adding an anchor element with a link to the URL specified in the href attribute -->
    </body>
</html> <!-- closing the HTML document -->
