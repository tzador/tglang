<!DOCTYPE html> <!--declaration of HTML document-->
<html> <!--opening tag for the html element-->
<head> <!--opening tag for the head element-->
    <meta charset="UTF-8"> <!--sets the character encoding for the document to UTF-8-->
    <title>My Cool Website</title> <!--sets the title of the webpage-->
    <style> <!--opening tag for the style element-->
        h1 {text-align: center; font-family: sans-serif;} /*sets the style for the heading element*/
        p {color: blue; font-size: 18px; font-family: Arial;} /*sets the style for the paragraph element*/
        #main {width: 80%; margin: 0 auto;} /*sets the style for the element with id "main"*/
    </style> <!--closing tag for the style element-->
</head> <!--closing tag for the head element-->
<body> <!--opening tag for the body element-->
    <div id="main"> <!--opening tag for the div element with id "main"-->
        <h1>Welcome to my cool website!</h1> <!--heading element with text-->
        <p>This is a <strong>long</strong> and <em>complex</em> code snippet.</p> <!--paragraph element with text and nested elements-->
    </div> <!--closing tag for the div element-->
</body> <!--closing tag for the body element-->
</html> <!--closing tag for the html element-->
