<!-- This is a comment in HTML -->
<!DOCTYPE html>
<html> <!-- This is the opening tag for the root element -->
<head> <!-- This is the opening tag for the head element -->
  <meta charset="UTF-8"> <!-- This is the meta tag for specifying the character encoding -->
  <title>My Website</title> <!-- This is the title tag for the webpage -->
</head> <!-- This is the closing tag for the head element -->
<body> <!-- This is the opening tag for the body element -->
  <h1>Welcome to My Website</h1> <!-- This is a heading tag for the main title -->
  <p>This is a paragraph element with a <a href="https://www.example.com">link</a> to another website.</p> <!-- This is a paragraph element with an anchor tag for a link -->
  <img src="image.jpg" alt="Image"> <!-- This is an image tag with the source and alternate text attributes -->
  <ul> <!-- This is an unordered list element -->
    <li>Item 1</li> <!-- This is a list item element -->
    <li>Item 2</li> <!-- This is a list item element -->
    <li>Item 3</li> <!-- This is a list item element -->
  </ul> <!-- This is the closing tag for the unordered list element -->
</body> <!-- This is the closing tag for the body element -->
</html> <!-- This is the closing tag for the root element -->
