<!-- This is a multi-line comment in HTML -->

<!-- This is the head element, which contains important information about the page -->
<head>
  <!-- The title element sets the title of the page that appears in the browser tab -->
  <title>My Website</title>

  <!-- This meta tag sets the character set used for the page -->
  <meta charset="UTF-8">

  <!-- This link tag references an external stylesheet for styling the page -->
  <link rel="stylesheet" href="styles.css">

</head>

<!-- This is the body element, which contains all of the visible content on the page -->
<body>
  <!-- This is a heading element, used for headings and page titles -->
  <h1>Welcome to my website!</h1>

  <!-- This is a paragraph element, used for text content -->
  <p>This is my first website and I'm excited to share it with you.</p>

  <!-- This is an unordered list element, used for displaying a list of items -->
  <ul>
    <!-- Each list item is represented by an li element -->
    <li>Item 1</li>
    <li>Item 2</li>
    <li>Item 3</li>
  </ul>

  <!-- This is a link element, used for linking to other pages or resources -->
  <a href="about.html">Learn more about me!</a>

</body>

<!-- This is the closing tag for the HTML element -->
</html>
