// Declare three variables to hold the three numbers
int num1 = 10;
int num2 = 15;
int num3 = 20;

// Calculate the sum of the three numbers
int sum = num1 + num2 + num3;

// Calculate the average by dividing the sum with 3
float average = sum / 3.0;

// Print the average to the console
stdout.printf("The average of num1, num2 and num3 is: %f\n", average);

// Output:
// The average of num1, num2 and num3 is: 15.000000
