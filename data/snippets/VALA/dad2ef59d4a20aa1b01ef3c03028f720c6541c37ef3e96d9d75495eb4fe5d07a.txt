// This code snippet generates a random number between 0 and 100
int num = Random.int_range(0, 100); // creates a variable 'num' and assigns a random integer between 0 and 100
if (num >= 50) { // checks if 'num' is greater than or equal to 50
    print ("The generated number is greater than or equal to 50"); // prints a message if 'num' is greater than or equal to 50
} else { // executes if 'num' is less than 50
    print ("The generated number is less than 50"); // prints a message if 'num' is less than 50
}
