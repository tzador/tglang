int main (string[] args) {
    // Define variables
    int x = 5;
    int y = 10;
    
    // Perform arithmetic operations
    int sum = x + y;
    int difference = y - x;
    int product = x * y;
    int quotient = y / x;

    // Print results
    print("The sum of x and y is: ${sum}");
    print("The difference of y and x is: ${difference}");
    print("The product of x and y is: ${product}");
    print("The quotient of y and x is: ${quotient}");
    return 0;
}
