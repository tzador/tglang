int i; 
string s; 

for (i = 0; i < 10; i++) { // for loop to iterate through numbers 0-9
    s = "Number: " + i.to_string(); // concatenate i to the string "Number: "
    print(s); // print the concatenated string
} 

/* Output:
Number: 0
Number: 1
Number: 2
Number: 3
Number: 4
Number: 5
Number: 6
Number: 7
Number: 8
Number: 9
*/
