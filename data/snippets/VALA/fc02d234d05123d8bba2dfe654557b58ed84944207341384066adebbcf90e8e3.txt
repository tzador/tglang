// Simple program to find the average of three numbers

// Define three variables of type float
double num1 = 5.0;
double num2 = 4.5;
double num3 = 7.2;

// Calculate the sum of the three numbers
double sum = num1 + num2 + num3;

// Calculate the average
double average = sum / 3.0;

// Print the result
stdout.printf("The average of %f, %f, and %f is %f", num1, num2, num3, average);
