public class Example : Object { // define a public class named Example that inherits from the base class Object
    public static void main() { // define a public static method named main with no parameters
        int num = 5; // declare and initialize a variable of type int with value 5
        for (int i = 0; i < num; i++) { // create a for loop that iterates i from 0 to num-1
            print(i); // call the print function and pass in the value of i
        }
    }
}
