| myArray |                               "Declares the variable 'myArray' to hold an array"
myArray := #(2 4 6 8 10).                  "Assigns an array with elements 2, 4, 6, 8, 10 to 'myArray'"
myArray do: [ :each |                      "Iterates through the elements of 'myArray'"
    each * 2.                             "Multiplies each element by 2"
    Transcript show: each printString.    "Prints each element to the Transcript output"
    Transcript cr.                        "Creates a new line in the output"
].                                         "End of iteration"
^myArray.                                 "Returns the modified array as the output"
