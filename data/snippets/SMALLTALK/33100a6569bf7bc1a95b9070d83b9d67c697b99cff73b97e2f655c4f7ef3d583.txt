| values | "create a variable called values"
values := OrderedCollection new: 5. "instantiate the variable with an ordered collection of size 5"
values add: 'Hello'. "add the string 'Hello' as the first element in the collection"
values add: 12345. "add the number 12345 as the second element in the collection"
values add: 3.14. "add the decimal number 3.14 as the third element in the collection"
values add: true. "add the boolean value true as the fourth element in the collection"
values add: #symbol. "add the symbol #symbol as the fifth element in the collection"
values do: [ :each | Transcript show: each printString; cr ]. "iterate through the collection and print each element to the Transcript in their string representation"
