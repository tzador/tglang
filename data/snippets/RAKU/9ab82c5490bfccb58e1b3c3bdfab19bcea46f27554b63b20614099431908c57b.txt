my $num1 = 10; # assigns integer value 10 to variable $num1
my $num2 = 20; # assigns integer value 20 to variable $num2
say "Addition of $num1 and $num2 is: " ~ ($num1 + $num2); # prints the addition of both variables
say "Subtraction of $num2 from $num1 is: " ~ ($num1 - $num2); # prints the subtraction of $num2 from $num1
say "Multiplication of $num1 and $num2 is: " ~ ($num1 * $num2); # prints the multiplication of both variables
say "Division of $num1 by $num2 is: " ~ ($num1 / $num2); # prints the division of $num1 by $num2
my $remainder = $num1 % $num2; # assigns remainder of $num1 divided by $num2 to variable $remainder
say "The remainder of $num1 divided by $num2 is: $remainder"; # prints the remainder
say "Increment of $num1 is: " ~ ++$num1; # increments $num1 by 1 and prints the new value
say "Decrement of $num2 is: " ~ --$num2; # decrements $num2 by 1 and prints the new value
