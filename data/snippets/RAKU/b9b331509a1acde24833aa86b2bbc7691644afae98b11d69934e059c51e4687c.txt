# This is a code snippet in Raku which calculates
# the area of a circle with a specified radius

my $radius = 5; # Initialize the radius variable
my $pi = 3.14;  # Define the value of pi

# Calculate the area of the circle using the formula
my $area = $pi * $radius ** 2;

# Print the result
say "The area of the circle with radius $radius is $area";
