my $num1 = 10; # Assigning the value '10' to the variable '$num1'
my $num2 = 5; # Assigning the value '5' to the variable '$num2'
my $sum = $num1 + $num2; # Adding the values of '$num1' and '$num2' and storing the result in the variable '$sum'
say "The sum of $num1 and $num2 is $sum"; # Printing the result using the 'say' function
