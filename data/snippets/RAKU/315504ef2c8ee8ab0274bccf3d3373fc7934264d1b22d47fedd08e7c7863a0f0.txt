# This code snippet creates a subroutine that prints the sum of two numbers
sub add_numbers($num1, $num2) {
  my $sum = $num1 + $num2;
  say "The sum of $num1 and $num2 is $sum";
}

# This code snippet uses the subroutine to calculate the sum of 5 and 7
add_numbers(5,7);

# Output: The sum of 5 and 7 is 12
