# This code snippet uses RAKU to calculate

my $num1 = 10;                  # Declaring a variable and assigning value 10
my $num2 = 5;                   # Declaring a variable and assigning value 5
my $sum = $num1 + $num2;        # Adding num1 and num2 and storing the value in sum variable
my $product = $num1 * $num2;    # Multiplying num1 and num2 and storing the value in product variable

say "The sum of $num1 and $num2 is $sum.";        # Output the sum of variables
say "The product of $num1 and $num2 is $product."; # Output the product of variables
