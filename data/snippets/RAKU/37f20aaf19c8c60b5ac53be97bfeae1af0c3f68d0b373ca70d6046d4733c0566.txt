#!/usr/bin/raku

=begin

This is a simple calculator program written in RAKU.
It performs addition, subtraction, multiplication, and division operations on two numbers.

=end

my $num1 = 5; # first number
my $num2 = 10; # second number

# addition operation
say "$num1 + $num2 = {$num1 + $num2}";

# subtraction operation
say "$num1 - $num2 = {$num1 - $num2}";

# multiplication operation
say "$num1 * $num2 = {$num1 * $num2}";

# division operation
say "$num1 / $num2 = {$num1 / $num2}";
