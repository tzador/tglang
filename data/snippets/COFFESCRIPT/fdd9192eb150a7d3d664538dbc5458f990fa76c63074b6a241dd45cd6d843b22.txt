for i in [1..10]
    if i % 2 == 0
        console.log "#{i} is even!"
    else
        console.log "#{i} is odd!"
         
// This code snippet creates a for loop that iterates through numbers 1 to 10
// The loop checks if the current number is even or odd using the modulus operator
// and prints out a string statement depending on the result
// The string includes the current number using string interpolation
