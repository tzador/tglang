###
Function to calculate the sum of two numbers
Params:
- num1: the first number to be added
- num2: the second number to be added
Returns:
- sum: the sum of the two numbers
###
sum = (num1, num2) ->  # Defines a function named "sum" with two parameters "num1" and "num2"
  num1 + num2          # Adds num1 and num2 and returns the sum

# Example usage:
console.log(sum(5, 10))  # Prints 15 to the console

# Output:
###
15
###
