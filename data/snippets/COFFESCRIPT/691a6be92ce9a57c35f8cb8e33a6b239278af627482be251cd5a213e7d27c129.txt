# A function that adds two numbers stored in variables
# Input: two numbers stored in variables
# Output: the sum of the two numbers

sum = (num1, num2) ->
  return num1 + num2

# Variables to store the two numbers
num1 = 10
num2 = 20

# Print the sum of the two numbers using the function
console.log(sum(num1, num2)) # Output: 30
