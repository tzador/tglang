# This code snippet shows an example of a loop using CoffeeScript syntax.
# In this case, the loop goes through a list of numbers and prints out each number multiplied by 2.
numbers = [1,2,3,4,5]
for num in numbers # loop through the numbers list
  x = num * 2
  console.log("The number " + num + " multiplied by 2 is: " + x) # print the result
