# This is a sample code snippet in TCL.
set name "John"

# By using `expr` we can perform mathematical operations.
set age [expr 2020 - 1992]

# We can use `incr` to increment a variable by a certain amount.
incr age 4

# We can use the `puts` command to print out the values of our variables.
puts "Hello, my name is $name and I am $age years old."
