% This program calculates the average of five numbers and displays the result

% Initializing variables
a = 1;
b = 2;
c = 3;
d = 4;
e = 5;

% Calculating the sum of the numbers
sum = a + b + c + d + e;

% Calculating the average
avg = sum / 5;

% Displaying the result
disp('The average of the five numbers is:');
disp(avg);
