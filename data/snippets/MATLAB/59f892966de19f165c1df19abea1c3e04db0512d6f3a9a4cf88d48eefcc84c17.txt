% Define variables
a = 10; % Initialize variable a with value 10
b = 5; % Initialize variable b with value 5

% Perform operations
c = a + b; % Add a and b, store result in variable c
d = a * b; % Multiply a by b, store result in variable d
e = a / b; % Divide a by b, store result in variable e
f = a ^ b; % Raise a to the power of b, store result in variable f
g = sqrt(a); % Find the square root of a, store result in variable g

% Print output
disp("The value of c is: " + c); % Print the value of c
disp("The value of d is: " + d); % Print the value of d
disp("The value of e is: " + e); % Print the value of e
disp("The value of f is: " + f); % Print the value of f
disp("The value of g is: " + g); % Print the value of g

% Output: 
% The value of c is: 15
% The value of d is: 50
% The value of e is: 2
% The value of f is: 100000
% The value of g is: 3.1623
