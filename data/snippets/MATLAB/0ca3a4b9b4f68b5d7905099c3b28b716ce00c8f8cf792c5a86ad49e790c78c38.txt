% Define variables
A = [1, 2, 3; 4, 5, 6];
B = [10, 20, 30; 40, 50, 60];
C = [100, 200, 300; 400, 500, 600];
D = [1000, 2000, 3000; 4000, 5000, 6000];
E = 3; % Scalar

% Perform mathematical operations
x = A * B; % Matrix multiplication
y = (A + B) / C; % Element-wise addition and division
z = (D * E)^2; % Scalar multiplication and power operation

% Display results
disp(x); % Output result of x
disp(y); % Output result of y
disp(z); % Output result of z
