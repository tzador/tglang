; This code snippet uses the cond and let statements to implement a binary search algorithm
(defun binary-search (list target)
  "Implements a binary search algorithm on a sorted list to find a target value"
  (let ((start 0)
        (end (- (length list) 1)))
    (cond ((< start end)
           (let ((mid (/ (+ start end) 2)))
             (cond ((< (nth mid list) target)
                    (binary-search (subseq list (1+ mid)) target))
                   ((> (nth mid list) target)
                    (binary-search (subseq list 0 mid) target))
                   (t mid))))
          ((= start end)
           (if (= (nth start list) target) start nil))
          (t nil))))
