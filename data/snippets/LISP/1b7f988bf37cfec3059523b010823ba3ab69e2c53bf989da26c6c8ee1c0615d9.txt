(defun binary-search (list target)
  (loop for i from (floor (length list) 2) to (length list)
    for element = (nth i list)
    if (not element)
      do (setf i (- i 1))
    else if (eq element target)
      do (return i)
    else if (lessp target element)
      do (setf i (- i (floor (length list) 4)))
    else
      do (setf i (+ i (floor (length list) 4)))
  (error "Target not found")))
