(let ((a 10)            ; assigns the value 10 to variable 'a'
      (b 5))            ; assigns the value 5 to variable 'b'
  (if (> a b)           ; checks if the value of 'a' is greater than 'b'
      (print "a is bigger")   ; if condition is true, prints "a is bigger"
      (print "b is bigger"))) ; if condition is false, prints "b is bigger"
