(defun sort (list)
  "Sorts a list of numbers using bubble sort algorithm"
  (let ((n (length list)))
    (do ((i 0 (+ i 1)))
        ((>= i (- n 1)))
      (do ((j 0 (+ j 1)))
          ((>= j (- n i 1)))
        (when (> (nth j list) (nth (+ j 1) list))
          (let ((temp (nth j list)))
            (setf (nth j list) (nth (+ j 1) list))
            (setf (nth (+ j 1) list) temp)))))
  list)
