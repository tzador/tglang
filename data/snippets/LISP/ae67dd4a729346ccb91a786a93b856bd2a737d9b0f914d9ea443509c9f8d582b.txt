(defun pig-latin-word (word)
  "Takes in a word and converts it to pig-latin."
  (let ((vowels '(#\a #\e #\i #\o #\u)))
    (let* ((char-list (coerce word 'list))
           (first-char (car char-list))
           (rest-chars (cdr char-list))
           (output (concatenate 'string "ay" (string first-char))))
      (if (member first-char vowels)
          (concatenate 'string word "-way")
          (concatenate 'string (coerce rest-chars 'string) output "-way")))))
