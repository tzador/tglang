(defun print-fib (n)
  "Function to print the first n numbers in the Fibonacci sequence"
  (let ((fib-1 0) (fib-2 1) (fib-list '(1)))
    (when (> n 1)
      (dotimes (i (- n 1))
        (let ((fib-next (+ fib-1 fib-2)))
          (push fib-next fib-list)
          (setf fib-1 fib-2)
          (setf fib-2 fib-next))))
    (print fib-list)))
