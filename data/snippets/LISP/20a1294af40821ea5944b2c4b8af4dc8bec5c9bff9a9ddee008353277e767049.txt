(let ((x 5) (y 10)) ; defining a let form with variables x and y 
  (if (< x y)       ; using a if condition to check if x is less than y
    ;; if condition body
    (+ x y)         ; if condition is true, add x and y
    ;; else condition body
    (* x y)))       ; if condition is false, multiply x and y to get the result
