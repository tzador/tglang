(defun binary-search (target list)
  "A function to perform binary search on a sorted list"
  (let ((start 0)
        (end (length list)-1)
        (mid (/ (length list) 2)))
    (while (<= start end)
      (if (equal target (nth mid list))
          (return-from binary-search mid)
          (if (> target (nth mid list))
              (setq start (1+ mid))
              (setq end (1- mid)))
          (setq mid (/ (+ start end) 2)))))
