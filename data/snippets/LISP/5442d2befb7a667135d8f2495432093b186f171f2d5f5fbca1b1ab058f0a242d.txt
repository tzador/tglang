(defun quicksort (lst)
    (if (null lst)
        '()
        (let ((pivot (first lst)))
            (append (quicksort (remove-if-not #'(lambda (x)
                                                    (< x pivot))
                                                (rest lst)))
                    (list pivot)
                    (quicksort (remove-if-not #'(lambda (x)
                                                    (> x pivot))
                                                (rest lst)))))))
