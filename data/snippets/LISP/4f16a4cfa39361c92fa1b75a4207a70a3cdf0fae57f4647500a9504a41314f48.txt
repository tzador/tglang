; A function to generate a random number between 1 and 10
(defun random-number ()
  (random 10)) ; Calls the random function with a max value of 10

; A function to check if the random number is even or odd
(defun even-or-odd (num)
  (if (equal (mod num 2) 0) ; Checks if the remainder of num divided by 2 is 0
      (format nil "The number ~a is even." num) ; If remainder is 0, formats a string with the number
    (format nil "The number ~a is odd." num))) ; If remainder is not 0, formats a string with the number

; A loop to generate 5 random numbers and check if they are even or odd
(loop for i from 1 to 5 do
  (let ((num (random-number))) ; Assigns a random number generated by the random-number function to num
    (print (even-or-odd num)))) ; Calls the even-or-odd function with num as an argument and prints the output
