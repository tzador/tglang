def a = 10 // defining variable a with value 10
def b = ++a // incrementing a and assigning its value to b
def c = 10 // defining variable c with value 10
def d = --c // decrementing c and assigning its value to d
println "a = $a" // printing the value of variable a
println "b = $b" // printing the value of variable b
println "c = $c" // printing the value of variable c
println "d = $d" // printing the value of variable d
