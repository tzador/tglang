// This is a Groovy Script that calculates the area of a circle

// Define the radius of the circle
def radius = 5.5

// Define the value of pi
def pi = Math.PI

// Calculate the area of the circle using the formula A = pi * r^2
def area = pi * (radius ** 2)

// Print out the result
println "The area of the circle with a radius of ${radius} is ${area}"
