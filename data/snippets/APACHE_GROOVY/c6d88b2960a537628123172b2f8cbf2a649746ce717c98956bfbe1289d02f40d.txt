/**
 * A class representing a person.
 * Contains basic information such as name, age, and address.
 */
class Person {
    // attributes
    String name
    int age
    String address

    // constructor with parameters
    Person(String name, int age, String address) {
        this.name = name
        this.age = age
        this.address = address
    }

    // getters and setters
    String getName() {
        return this.name
    }

    void setName(String name) {
        this.name = name
    }

    int getAge() {
        return this.age
    }

    void setAge(int age) {
        this.age = age
    }

    String getAddress() {
        return this.address
    }

    void setAddress(String address) {
        this.address = address
    }

    // method to print person's information
    void printInfo() {
        println("Name: " + this.getName())
    	println("Age: " + this.getAge())
    	println("Address: " + this.getAddress())
    }
}

// create an instance of Person
Person person1 = new Person("John Smith", 25, "123 Main St")

// call printInfo() method to print person's information
person1.printInfo()
