// Define a function that takes in two integers and returns their sum
def sum(int1, int2) {
    int sum = int1 + int2 // Calculate sum of the two integers
    println "The sum of $int1 and $int2 is $sum." // Print the result
    return sum // Return the sum
}

// Call the function with two integers as arguments
int num1 = 5
int num2 = 10
sum(num1, num2)

// Output:
// The sum of 5 and 10 is 15.
