/**
 * This program calculates the average of three numbers.
 */

def num1 = 5 // first number
def num2 = 10 // second number
def num3 = 15 // third number

def sum = num1 + num2 + num3 // calculate the sum of the three numbers

def avg = sum / 3 // calculate the average by dividing the sum by 3

println "The average of the three numbers is: " + avg // display the average
