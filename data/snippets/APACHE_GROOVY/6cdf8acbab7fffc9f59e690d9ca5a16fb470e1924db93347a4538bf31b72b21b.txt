// This code snippet is a Groovy script that takes in two numbers and performs basic arithmetic operations on them.
// The output is then printed to the console.

int num1 = 5 // Define first number
int num2 = 10 // Define second number

// Addition operation
int sum = num1 + num2
println("Sum: ${sum}") // Print sum to console

// Subtraction operation
int difference = num1 - num2
println("Difference: ${difference}") // Print difference to console

// Multiplication operation
int product = num1 * num2
println("Product: ${product}") // Print product to console

// Division operation
double quotient = (double) num1 / num2 // Convert to double for accurate division
println("Quotient: ${quotient}") // Print quotient to console
