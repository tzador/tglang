/*This program calculates the average of three numbers.*/

//Defining three variables with initial values
def num1 = 5.8
def num2 = 3.2
def num3 = 7.9

//Calculating the sum of the three numbers
def sum = num1 + num2 + num3

//Calculating the average
def average = sum / 3

/*Printing the result with a message and the average value using string interpolation*/
println "The average of $num1, $num2 and $num3 is: $average"
