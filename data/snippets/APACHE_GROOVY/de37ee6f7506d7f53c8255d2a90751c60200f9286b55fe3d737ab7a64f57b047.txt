/* Example groovy code */

// Defining a variable
def greeting = "Hello "

// Defining a function that takes in a name parameter
def sayHello(name) {
    // Concatenating the greeting with the name parameter
    def message = greeting + name

    // Printing out the message
    println(message)
}

// Calling the sayHello function with a name parameter
sayHello("John")

// Output: Hello John
