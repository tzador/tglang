/**
* This is a complex function that takes in three parameters
* and returns the sum of the parameters.
*
* @param a The first parameter.
* @param b The second parameter.
* @param c The third parameter.
* @return The sum of the three parameters.
**/
def complexFunction(a, b, c) {
    def sum = a + b + c
    return sum
}

// Example usage:
def result = complexFunction(2, 3, 4)
println(result)
