defmodule Calculator do                 # Defines a module called Calculator
   
  def add(x,y) do                       # Defines a function called add with parameters x and y
    x + y                               # Returns the sum of x and y
  end                                   # End of add function definition
  
  def subtract(x,y) do                  # Defines a function called subtract with parameters x and y
    x - y                               # Returns the difference of x and y
  end                                   # End of subtract function definition
  
  def multiply(x,y) do                  # Defines a function called multiply with parameters x and y
    x * y                               # Returns the product of x and y
  end                                   # End of multiply function definition
  
  def divide(x,y) do                    # Defines a function called divide with parameters x and y
    x / y                               # Returns the quotient of x and y
  end                                   # End of divide function definition
  
end                                     # End of Calculator module definition
