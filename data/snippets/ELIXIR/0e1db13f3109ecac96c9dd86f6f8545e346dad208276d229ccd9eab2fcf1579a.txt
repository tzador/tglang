defmodule Calculator do
  # This module creates a calculator that can perform basic mathematical operations
  def add(number1, number2) do
    # This function adds the two given numbers and returns the sum
    number1 + number2
  end

  def subtract(number1, number2) do
    # This function subtracts the second number from the first number and returns the difference
    number1 - number2
  end

  def multiply(number1, number2) do
    # This function multiplies the two given numbers and returns the product
    number1 * number2
  end

  def divide(number1, number2) do
    # This function divides the first number by the second number and returns the quotient
    number1 / number2
  end
end
