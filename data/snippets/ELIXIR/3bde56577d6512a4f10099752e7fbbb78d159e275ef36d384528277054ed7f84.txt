# Example module
defmodule Example do
  @doc """
  A module for calculating various mathematical operations.
  """
  @moduledoc """
  A module for calculating various mathematical operations.
  """
  
  # Function for calculating the sum of two numbers
  def sum(a, b) do
    a + b
  end
  
  # Function for calculating the difference of two numbers
  def diff(a, b) do
    a - b
  end
  
  # Function for calculating the product of two numbers
  def product(a, b) do
    a * b
  end
  
  # Function for calculating the division of two numbers
  def quotient(a, b) do
    a / b
  end
end
