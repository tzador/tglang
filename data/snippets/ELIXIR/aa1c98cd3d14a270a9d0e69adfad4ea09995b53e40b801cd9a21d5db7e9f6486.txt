defmodule Example do
  @moduledoc """
  This module contains functions that help with basic arithmetic operations.
  """

  def add(x, y) do
    """
    This function adds two numbers together and returns the result.
    It accepts two integers as arguments.
    """
    x + y
  end

  def subtract(x, y) do
    """
    This function subtracts the second number from the first number and returns the result.
    It accepts two integers as arguments.
    """
    x - y
  end

  def multiply(x, y) do
    """
    This function multiplies two numbers together and returns the result.
    It accepts two integers as arguments.
    """
    x * y
  end

  def divide(x, y) do
    """
    This function divides the first number by the second number and returns the result.
    It accepts two integers as arguments.
    """
    x / y
  end
end
