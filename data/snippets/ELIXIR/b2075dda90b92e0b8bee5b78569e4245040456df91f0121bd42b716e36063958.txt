defmodule MyModule do
  @moduledoc """
  This is a module that defines mathematical operations.
  """
  def add(x, y) do
    """
    Function that adds two numbers.
    """
    x + y
  end

  def subtract(x, y) do
    """
    Function that subtracts two numbers.
    """
    x - y
  end

  def multiply(x, y) do
    """
    Function that multiplies two numbers.
    """
    x * y
  end

  def divide(x, y) do
    """
    Function that divides two numbers.
    """
    x / y
  end
end
