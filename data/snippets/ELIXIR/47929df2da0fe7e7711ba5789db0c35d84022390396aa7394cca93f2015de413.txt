defmodule Operations do
  ## This is a module named 'Operations' which defines a collection of functions.

  # Function: add
  # Description: This function takes two integers as input and returns their sum.
  # Parameters:
  #   a: the first number
  #   b: the second number
  # Returns:
  #   the sum of a and b
  def add(a, b) do
    a + b
  end

  # Function: subtract
  # Description: This function takes two integers as input and returns their difference.
  # Parameters:
  #   a: the first number
  #   b: the second number
  # Returns:
  #   the difference of a and b
  def subtract(a, b) do
    a - b
  end

  # Function: multiply
  # Description: This function takes two integers as input and returns their product.
  # Parameters:
  #   a: the first number
  #   b: the second number
  # Returns:
  #   the product of a and b
  def multiply(a, b) do
    a * b
  end

  # Function: divide
  # Description: This function takes two integers as input and returns their quotient.
  # Parameters:
  #   a: the first number
  #   b: the second number
  # Returns:
  #   the quotient of a and b
  def divide(a, b) do
    a / b
  end
end
