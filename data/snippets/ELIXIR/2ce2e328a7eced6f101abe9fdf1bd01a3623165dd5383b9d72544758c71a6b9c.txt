defmodule Math do
  # Documentation for the `add` function
  #
  # Adds two numbers together
  #
  # Examples:
  # iex> Math.add(1, 2)
  # 3
  # iex> Math.add(5, 10)
  # 15
  #
  # ### Parameters
  #
  # * `x` - The first number to be added
  # * `y` - The second number to be added
  #
  # ### Examples
  #
  # iex> Math.add(3, 5)
  # 8
  def add(x, y) do
    x + y
  end
end
