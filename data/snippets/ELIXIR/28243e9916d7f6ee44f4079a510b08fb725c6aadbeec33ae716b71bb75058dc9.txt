defmodule Math do
  @moduledoc """
  Module that contains basic math functions.
  """

  @doc """
  Returns the sum of two numbers.
  """
  def add(x, y) do
    x + y
  end

  @doc """
  Returns the product of two numbers.
  """
  def multiply(x, y) do
    x * y
  end

  @doc """
  Returns the quotient of two numbers.
  """
  def divide(x, y) do
    x / y
  end

  @doc """
  Returns the difference of two numbers.
  """
  def subtract(x, y) do
    x - y
  end
end
