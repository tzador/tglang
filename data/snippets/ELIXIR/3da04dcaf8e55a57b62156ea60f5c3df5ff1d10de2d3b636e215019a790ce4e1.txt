defmodule Math do
  # Defines a module named Math

  def add(x, y) do
    # Defines a function named add with two parameters x and y
    x + y
    # The function adds the two parameters and returns the sum
  end

  def subtract(x, y) do
    # Defines a function named subtract with two parameters x and y
    x - y
    # The function subtracts the two parameters and returns the difference
  end

  def multiply(x, y) do
    # Defines a function named multiply with two parameters x and y
    x * y
    # The function multiplies the two parameters and returns the product
  end

  def divide(x, y) do
    # Defines a function named divide with two parameters x and y
    x / y
    # The function divides the two parameters and returns the quotient
  end
end
