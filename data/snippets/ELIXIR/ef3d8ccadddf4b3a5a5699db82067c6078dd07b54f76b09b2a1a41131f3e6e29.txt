# Define a module called "MathFunctions"
defmodule MathFunctions do
  # Define a function named "add"
  def add(x, y) do
    x + y
  end

  # Define a function named "subtract"
  def subtract(x, y) do
    x - y
  end

  # Define a function named "multiply"
  def multiply(x, y) do
    x * y
  end

  # Define a function named "divide"
  def divide(x, y) do
    x / y
  end
end
