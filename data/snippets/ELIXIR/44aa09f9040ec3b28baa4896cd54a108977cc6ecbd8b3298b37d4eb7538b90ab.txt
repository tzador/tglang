# Define a module named "Math" that contains mathematical operations
defmodule Math do
  
  # Define a function named "add" that takes two parameters and adds them
  def add(num1, num2) do
    num1 + num2
  end
  
  # Define a function named "subtract" that takes two parameters and subtracts them
  def subtract(num1, num2) do
    num1 - num2
  end
  
  # Define a function named "multiply" that takes two parameters and multiplies them
  def multiply(num1, num2) do
    num1 * num2
  end
  
  # Define a function named "divide" that takes two parameters and divides them
  def divide(num1, num2) do
    num1 / num2
  end
  
end
