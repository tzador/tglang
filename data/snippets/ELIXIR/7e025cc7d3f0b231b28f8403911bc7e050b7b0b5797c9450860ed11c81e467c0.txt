# defines a module Calculator
defmodule Calculator do 
  
  # defines a function add with two parameters, a and b
  def add(a, b) do 
    # adds the parameters a and b
    a + b 
  end 
  
  # defines a function subtract with two parameters, a and b
  def subtract(a, b) do 
    # subtracts the parameter b from a
    a - b 
  end 
  
  # defines a function multiply with two parameters, a and b
  def multiply(a, b) do 
    # multiplies the parameters a and b
    a * b 
  end 
  
  # defines a function divide with two parameters, a and b
  def divide(a, b) do 
    # divides the parameter a by b
    a / b 
  end 
end
