defmodule MathFunctions do
  @moduledoc """
  A collection of math functions for basic arithmetic operations.
  """

  @doc """
  Adds two numbers together.
  """
  def add(a, b) do
    a + b
  end

  @doc """
  Subtracts two numbers from each other.
  """
  def subtract(a, b) do
    a - b
  end

  @doc """
  Multiplies two numbers together.
  """
  def multiply(a, b) do
    a * b
  end

  @doc """
  Divides a number by another number.
  """
  def divide(a, b) do
    a / b
  end
end
