defmodule Math do
  @moduledoc """
  A collection of functions for mathematical operations.
  """

  @doc """
  Returns the sum of two numbers.
  """
  def sum(a, b) do
    a + b
  end

  @doc """
  Returns the difference between two numbers.
  """
  def difference(a, b) do
    a - b
  end

  @doc """
  Returns the product of two numbers.
  """
  def product(a, b) do
    a * b
  end

  @doc """
  Returns the quotient of two numbers.
  """
  def quotient(a, b) do
    a / b
  end
end
