# Define a module named "Calculator"
defmodule Calculator do
  # Define an addition function that takes in two parameters and returns their sum
  def add(x, y) do
    x + y
  end
  # Define a subtraction function that takes in two parameters and returns their difference
  def subtract(x, y) do
    x - y
  end
  # Define a multiplication function that takes in two parameters and returns their product
  def multiply(x, y) do
    x * y
  end
  # Define a division function that takes in two parameters and returns their quotient
  def divide(x, y) do
    x / y
  end
end
