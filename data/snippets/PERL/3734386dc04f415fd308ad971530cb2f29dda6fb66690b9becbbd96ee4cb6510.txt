# This is a simple Perl code
# It takes in two numbers and calculates their sum and product

#!/usr/bin/perl
use strict;
use warnings;

# Assign input numbers to variables
my $num1 = 5;
my $num2 = 10;

# Calculate sum and product
my $sum = $num1 + $num2;
my $product = $num1 * $num2;

# Print result
print "Sum: $sum\n";
print "Product: $product\n";
