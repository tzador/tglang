#set the starting values
my $first = 0; 
my $second = 1; 
my $fib = 0;
my $num = 10; #the number of terms we want to print 
  
#loop to print the sequence 
for(my $i=0; $i<$num; $i++){ 
    #check for the first and second terms 
    if($i == 0){ 
        print $first." ".$second." "; 
    } else{ 
        #calculate the next term 
        $fib = $first + $second; 
        $first = $second; 
        $second = $fib; 
        print $fib." "; 
    } 
}
