### This code snippet is a loop that calculates the sum of numbers from 1 to 10

my $sum = 0; # initialize sum variable to 0
for (my $i = 1; $i <= 10; $i++) { # loop from 1 to 10
    $sum = $sum + $i; # add current number to sum
}
print "The sum of numbers from 1 to 10 is $sum"; # print final sum with message
