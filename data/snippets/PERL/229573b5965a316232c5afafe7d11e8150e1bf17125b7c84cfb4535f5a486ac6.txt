my $message = "Hello World!";
my $num = 1234;
my @letters = ("a", "b", "c");
print $message; # This prints out the string "Hello World!"
print $num; # This prints out the number 1234
foreach my $letter (@letters) { # Loop through each element in the @letters array
    print $letter; # Prints out each individual letter ("a", "b", "c")
}
