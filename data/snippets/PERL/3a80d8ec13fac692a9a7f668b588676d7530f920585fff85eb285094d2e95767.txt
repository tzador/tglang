use strict;                              # This line enables strict compilation mode

my $num1 = 10;                           # Declare and initialize a variable named 'num1' with value of 10
my $num2 = 5;                            # Declare and initialize a variable named 'num2' with value of 5

my $sum = $num1 + $num2;                 # Create a third variable named 'sum' and assign it the sum of 'num1' and 'num2'
my $product = $num1 * $num2;             # Create a fourth variable named 'product' and assign it the product of 'num1' and 'num2'

print "The sum of $num1 and $num2 is $sum";       # Print the sum of 'num1' and 'num2'
print "The product of $num1 and $num2 is $product"; # Print the product of 'num1' and 'num2'
