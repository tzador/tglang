# This program takes two numbers as input and outputs the result of the first number divided by the second number
# Declare two variables and set their values to the input
my $num1 = <>;
my $num2 = <>;
# Calculate the result by dividing the first number by the second number
my $result = $num1 / $num2;

# Print the result to the console, with a message
print "The result of $num1 divided by $num2 is: $result";
