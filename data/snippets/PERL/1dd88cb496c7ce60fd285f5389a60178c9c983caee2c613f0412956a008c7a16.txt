# This is a comment in PERL
use strict; # Enable strict mode
my $x = 42; # Assign the value 42 to variable x
if ($x == 42) { # If statement to check if x is equal to 42
    print "This is the answer to life, the universe, and everything!\n"; # Print statement if the if statement is true
} else { # If x is not equal to 42
    print "This is not the answer we're looking for.\n"; # Print statement if if statement is false
} # End of if statement
while ($x < 50) { # While loop to increment x until it reaches 50
    $x++; # Increment x by 1
} # End of while loop
print "X is now $x.\n"; # Print statement to confirm that x is now 50
