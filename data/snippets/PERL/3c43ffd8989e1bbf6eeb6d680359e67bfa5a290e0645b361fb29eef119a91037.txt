#!/usr/bin/perl
use strict;
use warnings;

# declare and initialize variables
my $string1 = "Hello,";
my $string2 = "world!";
my $number1 = 5;
my $number2 = 10;

# perform operations on variables
my $concatenated_string = $string1 . " " . $string2;
my $sum = $number1 + $number2;
my $product = $number1 * $number2;

# print out results
print "The concatenated string is: $concatenated_string\n";
print "The sum is: $sum\n";
print "The product is: $product\n";
