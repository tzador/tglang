#!/usr/bin/perl
use strict;
use warnings;

# This code snippet uses the while loop to print a series of numbers
# from 1 to 10.

my $num = 1;

# while loop to iterate through numbers
# print statement to display the current number
# increment the number by 1 in each iteration

while ($num <= 10) {
    print $num ."\n";
    $num++;
}
