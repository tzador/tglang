%% This code calculates the sum of two numbers
\documentclass{article}
\usepackage{amsmath}

\begin{document}

\section{Sum Calculation}

\begin{equation*} % Begins a math environment
a = 10 % Assigns the value 10 to variable a
\end{equation*}

\begin{equation*}
b = 7 % Assigns the value 7 to variable b
\end{equation*}

\begin{equation*}
\text{Sum} = a + b % Adds a and b and assigns the result to the variable Sum
\end{equation*}

The sum of $a$ and $b$ is: 
\begin{equation*}
\text{Sum} = 10 + 7 = 17 % Prints the value of Sum using the assigned values of a and b
\end{equation*}
