\def\sum{k = 1}^{n}                    %This line defines the summation notation from k=1 to n
\frac{k(k+1)}{2}                       %This line is the actual formula for sum of first n natural numbers
    = \frac{n(n+1)}{2}                 %This is the simplified result of the summation
    = \sum(k+1)                        %This is the alternative representation of the summation
    = \frac{(n+1)(n+2)}{2}             %This is the simplified result of the alternative representation
