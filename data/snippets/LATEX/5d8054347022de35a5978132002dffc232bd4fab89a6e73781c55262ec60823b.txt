\documentclass{article} %creates a new document with "article" class
\usepackage{amsmath} %imports the AMS math package
\begin{document} %starts the document
Hello! %prints "Hello!" in the document
\begin{align*} %opens a math environment for aligning equations
a &= b + c \\ %first equation, "a equals b plus c"
&= d - e \\ %second equation, "equals d minus e"
&= f \times g \\ %third equation, "equals f times g"
&= \frac{h}{i} %fourth equation, "equals h divided by i"
\end{align*} %closes the math environment
\end{document} %ends the document
