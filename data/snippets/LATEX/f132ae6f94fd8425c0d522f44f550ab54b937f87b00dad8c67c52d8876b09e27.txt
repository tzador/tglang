\documentclass{article} % Defining the document class
\usepackage[utf8]{inputenc} % Setting the document encoding to UTF-8
\usepackage{amsmath} % Importing the AMS math package

\begin{document}
\title{My First Document} % Setting the document title
\author{John Doe} % Setting the document author
\date{\today} % Setting the document date to today

\maketitle % Creating the document title page

\section{Introduction} % Creating a section

\subsection{Mathematics} % Creating a subsection
Here is an equation: % Adding text
\[ % Starting a math environment
E = mc^2 % Actual equation
\] % Ending the math environment

\section{Conclusion} % Creating another section
That's it! % Adding text

\end{document} % Ending the document
