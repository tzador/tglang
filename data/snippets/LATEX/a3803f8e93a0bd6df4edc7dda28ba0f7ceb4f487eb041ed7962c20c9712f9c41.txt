% This is a code snippet in LATEX
% It creates a basic document with a title, author, and text content

\documentclass{article} % defines the document type as an article
\title{My First Document} % sets the title of the document
\author{John Doe} % sets the author of the document
\begin{document} % begins the document
\maketitle % generates the title that was defined earlier
This is my first document in \LaTeX. % adds text content to the document
\end{document} % ends the document
