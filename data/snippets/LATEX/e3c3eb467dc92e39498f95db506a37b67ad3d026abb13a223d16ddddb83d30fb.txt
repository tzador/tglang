\documentclass{article} %defining document class as an article

\usepackage{amsmath} % importing amsmath package

\begin{document} % starting the document

\section{Introduction} % creating a section titled Introduction

This is a sample document written in \LaTeX{} % inserting text and using formatting for \LaTeX

\begin{equation} % starting an equation
x + y = z % writing the equation
\end{equation} % closing the equation 

\subsection{Subsection 1} % creating a subsection titled "Subsection 1"

Here is some text with a \textbf{bold} and \textit{italic} word. % inserting text with formatting 

\subsubsection*{Subsubsection 1.1} % creating a subsubsection title "Subsubsection 1.1" without numbering

\begin{itemize} % starting a list
\item First item % adding first item to list
\item Second item % adding second item to list
\end{itemize} % closing the list

\subsection{Subsection 2} % creating another subsection titled "Subsection 2"

Here is some more text with a \texttt{monospaced} word. % inserting text with formatting 

\end{document} % closing the document
