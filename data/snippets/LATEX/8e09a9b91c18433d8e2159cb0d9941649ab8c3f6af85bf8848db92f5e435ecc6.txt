\begin{document} %defines the beginning of the document
\usepackage{graphicx} %imports the graphicx package for image handling
\begin{figure}[h] %starts a floating figure
\centering %centers the figure
\includegraphics[width=0.5\textwidth]{example.jpg} %includes an image from the specified file and sets it to half the text width
\caption{Example Image} %adds a caption to the image
\label{fig:example} %adds a label to the image
\end{figure} %ends the floating figure
\begin{equation} %starts an equation
y = mx + b %adds the equation with variables m, x, and b
\end{equation} %ends the equation
\section{Introduction} %creates a new section titled 'Introduction'
This is a section of text. %adds text to the section
\textbf{This text is bold.} %changes the font to bold for this section
\textit{This text is italicized.} %changes the font to italic for this section
\end{document} %ends the document
