\documentclass{article} 
%% This is a basic LaTeX document class
\usepackage[utf8]{inputenc} 
%% This package allows the use of unicode characters 
\title{My First LaTeX Document} % Sets the title of the document
\author{John Doe} % Sets the author of the document
\date{\today} % Sets the date of the document
\begin{document} % Starts the document
\maketitle % Adds the title, author, and date to the document 
\section{Introduction} % Begins a new section
This is my first \LaTeX{} document. % Outputs text, with proper indentation
\subsection{Subsection 1} % Begins a subsection
This is a subsection within the introduction section. % Outputs text, with proper indentation
\subsection{Subsection 2} % Begins another subsection
Lorem ipsum dolor sit amet, consectetur adipiscing elit. % Outputs filler text
\end{document} % Ends the document
