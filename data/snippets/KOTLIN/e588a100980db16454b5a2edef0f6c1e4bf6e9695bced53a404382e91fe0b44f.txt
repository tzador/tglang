/***************** 
* Function to calculate the sum of two numbers 
* @param num1 first number 
* @param num2 second number 
* @return sum of num1 and num2
*****************/
fun sum(num1: Int, num2: Int): Int {
    return num1 + num2
}

// Example usage
val a = 2
val b = 5
val result = sum(a, b)
println("The sum of $a and $b is $result")
