// This is a function that calculates the sum of three numbers.
// The function takes in three parameters, all integers.
fun sum(firstNumber: Int, secondNumber: Int, thirdNumber: Int) : Int {
    return firstNumber + secondNumber + thirdNumber
}
// Here we call the function and store the result in a variable.
val result = sum(5, 10, 15)
// Lastly, we print the result to the console.
println("The sum is $result")
