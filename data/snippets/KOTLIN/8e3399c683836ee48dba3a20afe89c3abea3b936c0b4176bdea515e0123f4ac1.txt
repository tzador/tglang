// This program calculates the sum of two numbers and prints the result

fun main() {
    // initialize variables
    val num1 = 10
    val num2 = 5

    // calculate sum
    val sum = num1 + num2

    // print result
    println("The sum of $num1 and $num2 is $sum")
}
// Output: The sum of 10 and 5 is 15
