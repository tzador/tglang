/**
 * This is a program that calculates the area of a circle given its radius.
 */
fun main() {
    val pi = 3.14159265358979 // constant value of pi
    val radius = 5 // input radius

    // calculate area of circle
    val area = pi * radius * radius

    // print result
    println("The area of the circle with radius $radius is $area")
}
