fun main() {
    val firstNumber = 5
    val secondNumber = 10
    var result = 0

    if (firstNumber > secondNumber) {
        result = firstNumber - secondNumber
    } else {
        result = firstNumber + secondNumber
    }
    println("The result is: $result")
}

// This code snippet compares two numbers and calculates the difference between them
// if the first number is larger than the second, or the sum of the two numbers if the
// second number is larger. The result is stored in a variable and then printed to the console.
