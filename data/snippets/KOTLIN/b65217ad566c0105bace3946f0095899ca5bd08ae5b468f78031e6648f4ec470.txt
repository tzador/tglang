/** This is a complex function that takes in two integers and returns the sum of their squares */
fun complexFunction(num1: Int, num2: Int): Int {
    val square1 = num1 * num1
    val square2 = num2 * num2
    return square1 + square2
}

// calling the function and printing the result
println(complexFunction(4, 5))

// output:
// 41
