fun main(args: Array<String>) {
    // This is a comment in Kotlin
    val num1 = 10
    val num2 = 5
    val sum = num1 + num2 // Adding two numbers in Kotlin
    val diff = num1 - num2 // Subtracting two numbers in Kotlin
    val product = num1 * num2 // Multiplying two numbers in Kotlin
    val quotient = num1 / num2 // Dividing two numbers in Kotlin
    println("The sum of $num1 and $num2 is: $sum") // Printing the sum
    println("The difference between $num1 and $num2 is: $diff") // Printing the difference
    println("The product of $num1 and $num2 is: $product") // Printing the product
    println("The quotient of $num1 and $num2 is: $quotient") // Printing the quotient
}
