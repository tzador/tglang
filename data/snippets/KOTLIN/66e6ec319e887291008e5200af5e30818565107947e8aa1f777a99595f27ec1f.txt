// This is a simple Kotlin program
// to print the sum of two numbers

fun main(args: Array<String>) {
    val num1 = 10
    val num2 = 20
    val sum = num1 + num2 // formula for sum

    // print the result
    println("The sum of $num1 and $num2 is $sum")
}

/* Output:
The sum of 10 and 20 is 30
*/
