// This is a function that calculates the sum of two numbers
// Inputs:
// a - first number
// b - second number
// Output:
// result - sum of the two numbers

fun sum(a: Int, b: Int): Int {
    // declare a variable to hold the sum
    var result: Int
    
    // initialize the variable to the sum of the two numbers
    result = a + b
    
    // return the result
    return result
}
