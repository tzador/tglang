fun main(args: Array<String>) {
    //Initialize variables
    val num1: Int = 10
    var num2: Int = 2
    var result: Int
    
    //Add two numbers
    result = num1 + num2
    println("The sum of $num1 and $num2 is $result")

    //Subtract two numbers
    result = num1 - num2
    println("The difference of $num1 and $num2 is $result")
    
    //Multiply two numbers
    result = num1 * num2
    println("The product of $num1 and $num2 is $result")
    
    //Divide two numbers
    result = num1 / num2
    println("The quotient of $num1 and $num2 is $result")
}
