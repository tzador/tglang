/*
 * This is a program that calculates the average of three numbers
 * Author: John Smith
 */

fun main(args: Array<String>) {

    // initialize three numbers
    val num1 = 2.5
    val num2 = 4.9
    val num3 = 6.3

    // calculate average
    val average = (num1 + num2 + num3) / 3

    // printing the result
    println("The average of $num1, $num2, and $num3 is $average")
}
