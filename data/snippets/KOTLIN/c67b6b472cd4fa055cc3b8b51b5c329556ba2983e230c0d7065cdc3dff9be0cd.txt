/* Define a function named "difference" that takes in two integer parameters, x and y */
fun difference(x: Int, y: Int) {
    /* Calculate the difference of x and y */
    val result = x - y
    /* Print the result */
    println(result)
}

/* Call the function and pass in the values 10 and 5 to get the difference of 5 */
difference(10, 5)
