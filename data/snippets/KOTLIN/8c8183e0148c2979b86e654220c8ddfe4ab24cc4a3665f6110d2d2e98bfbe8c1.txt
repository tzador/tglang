// This code snippet is a function that calculates the sum of two numbers
fun sum(x: Int, y: Int): Int {
    return x + y
}

// This code snippet calls the sum function with the numbers 5 and 7 and assigns the result to the variable 'result'
val result = sum(5, 7)

// The result variable is then printed to the console
println(result)

// Output: 12
