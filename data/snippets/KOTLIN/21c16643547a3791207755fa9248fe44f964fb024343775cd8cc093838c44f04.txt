fun main() {
    // This is the main function in Kotlin
    println("Hello World!") // Prints "Hello World!" to the console
    var num1 = 10 // Variable declaration and initialization
    var num2 = 5
    num1 += num2 // Using the shorthand operator '+=' to add num2 to num1
    println("num1 is now $num1") // Prints "num1 is now 15" to the console
}
