# This code snippet uses the PowerShell If Statement to check if a number is greater than 10

# Declare a variable 'num' and assign a value
$num = 15

# Use the If Statement to check if 'num' is greater than 10
if($num -gt 10)
{
    Write-Host "The number is greater than 10"
}
# If the condition is not met, use the Else statement to display a different message
else
{
    Write-Host "The number is not greater than 10"
}
