# Define variables
$number1 = 10
$number2 = 3

# Calculate Sum
$sum = $number1 + $number2

# Calculate Difference
$diff = $number1 - $number2

# Calculate Product
$prod = $number1 * $number2

# Calculate Quotient
$quot = $number1 / $number2

# Calculate Modulus
$mod = $number1 % $number2

# Print Output
Write-Host "The sum of $number1 and $number2 is $sum"
Write-Host "The difference between $number1 and $number2 is $diff"
Write-Host "The product of $number1 and $number2 is $prod"
Write-Host "The quotient of $number1 divided by $number2 is $quot"
Write-Host "The modulus of $number1 and $number2 is $mod"
