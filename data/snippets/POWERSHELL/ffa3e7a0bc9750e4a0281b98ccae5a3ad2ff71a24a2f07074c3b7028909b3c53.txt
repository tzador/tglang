# This cmdlet adds a new Active Directory user and sets their attributes
New-ADUser -Name JohnSmith -GivenName John -Surname Smith -DisplayName JohnSmith -SamAccountName JSmith -UserPrincipalName jsmith@example.com -EmailAddress jsmith@example.com -Enabled $true -AccountPassword (ConvertTo-SecureString -AsPlainText "Passw0rd!" -Force) -Path "OU=Sales,OU=UserAccounts,DC=example,DC=com"
