//This is a code snippet in QML to create a complex calculator app with various functions

//Importing necessary QML modules
import QtQuick 2.12
import QtQuick.Controls 2.12

//Defining the main window
Window {
    visible: true
    width: 500
    height: 700
    title: "Complex Calculator"

    //Defining the main rectangle to hold all components of the calculator
    Rectangle {
        id: mainRect
        anchors.fill: parent
        color: "#f4f4f4"

        //Defining the input and output displays for the calculator
        Column {
            id: displayColumn
            spacing: 5
            anchors.centerIn: parent
            TextField {
                id: inputField
                placeholderText: "Enter a complex number"
                width: parent.width * 0.9
                height: parent.height * 0.15
                font.pointSize: 20
                font.bold: true
                font.capitalization: Font.AllUppercase
                border.width: 2
                Material.theme: Material.Dark
                style: TextFieldStyle {
                    inputTextStyle.fontSize: 20
                    inputTextStyle.textColor: Qt.lighter("yellow")
                }
            }

            Text {
                id: symbolText
                text: "+"
                font.pointSize: 30
                font.bold: true
                color: "#2ec4b6"
            }

            Text {
                id: resultText
                text: "= 0"
                font.pointSize: 30
                font.bold: true
                color: "#2ec4b6"
            }
        }

        //Defining the keypad for the calculator
        Grid {
            id: keypadGrid
            anchors {
                top: displayColumn.bottom
                left: parent.left
                right: parent.right
                bottom: parent.bottom
                margins: 10
            }
            rows: 5
            columns: 4
            spacing: 10
            Repeater {
                delegate: Button {
                    text: model.display
                    font.pointSize: 20
                    font.bold: true
                    anchors.centerIn: parent
                    background: Rectangle {
                        color: "#264653"
                        radius: 10
                        border.color: "#264653"
                        border.width: 1
                    }
                    onClicked: {
                        //Defining the functionality of each button when clicked
                        switch(displayColumn.children.length) {
                        case 1:
                            //Handling input for real part of complex number
                            inputField.text += text
                            break
                        case 2:
                            //Handling input for imaginary part of complex number
                            inputField.text += text + "i"
                            break
                        case 3:
                            //Resetting the input and output displays when "=" button is clicked
                            inputField.text = ""
                            symbolText.text = "+"
                            resultText.text = "= 0"
                            break
                        default:
                            break
                        }
                    }
                }
                //Defining the list of buttons to be displayed in the keypad
                model: [
                    {display: "7"},
                    {display: "8"},
                    {display: "9"},
                    {display: "รท"},
                    {display: "4"},
                    {display: "5"},
                    {display: "6"},
                    {display: "ร"},
                    {display: "1"},
                    {display: "2"},
                    {display: "3"},
                    {display: "-"},
                    {display: "0"},
                    {display: "."},
                    {display: "+/-"},
                    {display: "+"},
                    {display: "C"},
                    {display: "="}
                ]
            }
        }
    }
}
