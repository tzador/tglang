* Set the library location and data set names;
LIBNAME mylib "/path/to/mydata";
DATA mylib.mydataset;
* Read in the data from a CSV file;
INFILE "/path/to/mydata/myfile.csv" DLM=',' MISSOVER DSD LRECL=32767 firstobs=2;
* Specify the variables from the file to be included in the dataset;
INPUT var1 var2 var3 var4 var5 var6 var7 var8 var9 var10;
* Create a new variable that sums two existing variables;
my_new_var = var1 + var2;
* Apply a format to the variable to ensure proper display;
FORMAT my_new_var dollar10.2;
* Filter the data to only include observations where var3 is greater than or equal to 50;
WHERE var3 >= 50;
* Sort the data by var2 in descending order;
PROC SORT DATA=mylib.mydataset;
BY descending var2;
* Generate a new dataset containing only the observations that meet the filter criteria;
DATA mylib.filtered_dataset;
SET mylib.mydataset;
* Subset the data to only include certain variables;
KEEP var1 var3 var5 var7 var9 my_new_var;
RUN;
