* This code snippet uses an array to store and process multiple values within a SAS data set ;
data patient_data;
set patient_info;
* Create an array with size of 10 to store weight values ;
array weight[10] weight1-weight10;
* Loop through each weight variable and calculate the BMI ;
do i=1 to 10;
    BMI = weight[i] / (height * height);
end;
* Calculate the average BMI for all patients ;
sum_bmi = mean(BMI);
* Sort the data set by BMI in descending order ;
proc sort data=patient_data;
by descending BMI;
run;
* Print the data set with BMI and weight variables ;
proc print data=patient_data;
var BMI weight:;
run;
