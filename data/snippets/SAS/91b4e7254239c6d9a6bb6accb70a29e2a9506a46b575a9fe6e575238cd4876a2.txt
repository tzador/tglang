* This is a sample SAS program;
options nodate nonumber nonotes;
* Set the library for the data source;
libname mydata '/users/data/'
* Create a dataset with two variables;
data mydata.newdata;
  input var1;
  input var2;
  cards;
  * Set the format and informat for the variables;
  format var1 best12.;
  informat var2 date9.;
run;
quit;
* Run a proc sort to sort the data in descending order of var1;
proc sort data=mydata.newdata out=mydata.sorted;
  by descending var1;
run;
* Create a new dataset with only values for var2 that are greater than 200;
data mydata.newdataset;
  set mydata.sorted;
  where var2 > 200;
run;
