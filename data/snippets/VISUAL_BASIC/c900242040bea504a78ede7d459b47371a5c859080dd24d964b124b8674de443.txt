Option Explicit

Sub main()
    'This is a simple subroutine in Visual Basic.
    Dim num1 As Integer 'declare a variable of type integer
    Dim num2 As Integer 'declare another integer variable
    Dim result As Integer 'declare variable to store result
    num1 = 10 'assign value of 10 to num1
    num2 = 5 'assign value of 5 to num2
    result = num1 + num2 'additive operation, store result in variable
    Debug.Print "The sum is: " & result 'print result in debug console with annotation
End Sub
