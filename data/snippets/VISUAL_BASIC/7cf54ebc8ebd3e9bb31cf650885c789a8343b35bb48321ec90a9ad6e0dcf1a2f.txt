' This program calculates the sum of two numbers

Dim num1 As Integer ' declare variable for first number
Dim num2 As Integer ' declare variable for second number
Dim sum As Integer ' declare variable for sum

num1 = 10 ' assign value of 10 to num1
num2 = 5 ' assign value of 5 to num2
sum = num1 + num2 ' calculate sum and assign to sum variable

Console.WriteLine("The sum of " & num1 & " and " & num2 & " is " & sum) ' print sum to console

' Output: The sum of 10 and 5 is 15
