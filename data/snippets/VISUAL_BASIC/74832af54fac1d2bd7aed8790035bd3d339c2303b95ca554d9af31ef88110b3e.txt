Dim firstNum As Integer 'declare new variable for first number
Dim secondNum As Integer 'declare new variable for second number
Dim result As Integer 'declare new variable for result

firstNum = 5 'assign value of 5 to firstNum variable
secondNum = 10 'assign value of 10 to secondNum variable

'perform addition operation
result = firstNum + secondNum 

'print result to console
Console.WriteLine("The result of the addition is: " & result)

'perform multiplication operation
result = firstNum * secondNum 

'print result to console
Console.WriteLine("The result of the multiplication is: " & result)
