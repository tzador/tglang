Dim x As Integer 'Declares a variable named 'x' with data type Integer
Dim y As Integer 'Declares a variable named 'y' with data type Integer
Dim z As Integer 'Declares a variable named 'z' with data type Integer
Dim result As Integer 'Declares a variable named 'result' with data type Integer
x = 5 'Assigns the value 5 to 'x'
y = 10 'Assigns the value 10 to 'y'
z = 7 'Assigns the value 7 to 'z'
result = x + y * z 'Calculates the result of 'x + y * z' and stores it in 'result'
Console.WriteLine("The result is: " & result) 'Prints the value of 'result' to the console
