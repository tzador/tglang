'This code snippet uses Visual Basic to calculate the average of three numbers and display the result

'Define variables for the three numbers
Dim num1 As Integer
Dim num2 As Integer
Dim num3 As Integer

'Assign values to the variables
num1 = 5
num2 = 8
num3 = 12

'Calculate the average
Dim average As Double
average = (num1 + num2 + num3) / 3

'Display the result
Console.WriteLine("The average of " & num1 & ", " & num2 & ", and " & num3 & " is: " & average)

'The output of this code snippet is: "The average of 5, 8, and 12 is: 8.33333333333333"
