' This program calculates the sum of two numbers and outputs the result
Const num1 As Integer = 5 ' first number
Const num2 As Integer = 10 ' second number
Dim sum As Integer ' variable to store the sum
sum = num1 + num2 ' calculate sum
Console.WriteLine("The sum of {0} and {1} is {2}.", num1, num2, sum) ' output result
' Output: The sum of 5 and 10 is 15.
