REM Set variables
set num1=10
set num2=20
set num3=30

REM Arithmetic operations
set /a sum=num1+num2
set /a difference=num3-num2
set /a product=num1*num2
set /a quotient=num3/num1

REM Display results
echo "The sum of %num1% and %num2% is %sum%."
echo "The difference between %num3% and %num2% is %difference%."
echo "The product of %num1% and %num2% is %product%."
echo "The quotient of %num3% and %num1% is %quotient%."
