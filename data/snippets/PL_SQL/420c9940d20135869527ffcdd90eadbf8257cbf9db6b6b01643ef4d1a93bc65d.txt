CREATE OR REPLACE PROCEDURE CALCULATE_SALARY 
(
   EMP_ID IN NUMBER, 
   SALARY OUT NUMBER
) 
AS
BEGIN
   SELECT EMPLOYEE_SALARY INTO SALARY
   FROM EMPLOYEE_TABLE
   WHERE EMP_ID = EMPLOYEE_ID;
   
   SALARY := SALARY * 1.1;
   IF (SALARY > 5000) THEN
      SALARY := SALARY + 500;
   ELSEIF (SALARY > 3000) THEN
      SALARY := SALARY + 300;
   ELSEIF (SALARY > 1000) THEN
      SALARY := SALARY + 100;
   END IF;
   
   DBMS_OUTPUT.PUT_LINE('The salary for employee ID ' || EMP_ID || ' is $' || SALARY);
EXCEPTION
   WHEN NO_DATA_FOUND THEN
      DBMS_OUTPUT.PUT_LINE('No employee with ID ' || EMP_ID || ' found.');
   WHEN OTHERS THEN
      DBMS_OUTPUT.PUT_LINE('Error calculating salary for employee ID ' || EMP_ID || '.');
END;
