DECLARE
  EMP_NAME VARCHAR2(255);
  EMP_ID NUMBER;
  EMP_SALARY NUMBER;
BEGIN
  -- Get employee name
  SELECT FIRST_NAME || ' ' || LAST_NAME INTO EMP_NAME FROM EMPLOYEES WHERE EMP_ID = 123;
  
  -- Get employee ID
  SELECT EMPLOYEE_ID INTO EMP_ID FROM EMPLOYEES WHERE FIRST_NAME = 'John' AND LAST_NAME = 'Doe';
  
  -- Get employee salary
  SELECT SALARY INTO EMP_SALARY FROM EMPLOYEES WHERE EMP_ID > 100 AND DEPARTMENT_ID = 50;
  
  -- Print employee details
  DBMS_OUTPUT.PUT_LINE('Employee Name: ' || EMP_NAME);
  DBMS_OUTPUT.PUT_LINE('Employee ID: ' || EMP_ID);
  DBMS_OUTPUT.PUT_LINE('Employee Salary: ' || EMP_SALARY);
  
  EXCEPTION
    WHEN NO_DATA_FOUND THEN
      DBMS_OUTPUT.PUT_LINE('Employee not found');
END;
