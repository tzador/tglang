CREATE OR REPLACE PROCEDURE GET_EMP_INFO (P_EMP_ID IN NUMBER, P_DEPT_ID IN NUMBER, P_EMP_CURSOR OUT SYS_REFCURSOR)
  IS
  BEGIN
    OPEN P_EMP_CURSOR FOR
      SELECT E.EMP_ID, E.EMP_NAME, D.DEPARTMENT_NAME
      FROM EMPLOYEES E
      JOIN DEPARTMENTS D ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
      WHERE E.EMP_ID = P_EMP_ID
      AND D.DEPARTMENT_ID = P_DEPT_ID;
END;
