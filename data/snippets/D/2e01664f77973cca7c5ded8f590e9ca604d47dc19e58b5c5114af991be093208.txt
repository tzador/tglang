import std.stdio; // Importing the standard input/output library

void main()
{
    int a = 10; // Initializing an integer variable with value 10
    int b = 20; // Initializing another integer variable with value 20
    writeln("a + b = ", a + b); // Printing the sum of a and b
    writeln("a * b = ", a * b); // Printing the product of a and b
    writeln("a / b = ", a / b); // Printing the result of a divided by b
    writeln("b - a = ", b - a); // Printing the result of b subtracted by a
}
