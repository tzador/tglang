import std.stdio;

void main()
{
    int a = 10;
    int b = 20;
    
    // calculate the sum of a and b
    int sum = a + b;
    
    // print the sum to the console
    writeln("The sum of a and b is: ", sum);
    
    // calculate the difference of a and b
    int diff = a - b;
    
    // print the difference to the console
    writeln("The difference of a and b is: ", diff);
    
    // calculate the product of a and b
    int product = a * b;
    
    // print the product to the console
    writeln("The product of a and b is: ", product);
    
    // calculate the quotient of a and b
    double quotient = a / b;
    
    // print the quotient to the console
    writeln("The quotient of a and b is: ", quotient);
}
