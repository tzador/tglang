// This program finds the sum of two numbers

import std.stdio;            // importing the standard input/output module

void main() {                // main function
    int num_1 = 5;           // first number
    int num_2 = 8;           // second number
    
    int sum = num_1 + num_2; // calculating the sum
    
    writeln("The sum is: ", sum);    // printing the result
}
