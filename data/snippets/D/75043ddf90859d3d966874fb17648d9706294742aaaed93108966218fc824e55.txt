import std.stdio; // Importing the standard input/output module

void main() // Main function
{
    writeln("Hello, world!"); // Printing "Hello, world!"
    int a = 5; // Declaring an integer variable 'a' and assigning it the value 5
    int b = 10; // Declaring an integer variable 'b' and assigning it the value 10
    int c = a + b; // Declaring an integer variable 'c' and assigning it the value of 'a' plus 'b'
    writeln("The sum of a and b is: ", c); // Printing the sum of 'a' and 'b'
}
