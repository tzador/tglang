// Delphi code snippet to calculate the sum of two numbers

// Declare variables to store user inputs
var
  num1, num2: Integer;
  sum: Integer;

// Prompt user for input
WriteLn('Enter first number: ');
ReadLn(num1);
WriteLn('Enter second number: ');
ReadLn(num2);

// Calculate sum and store result in 'sum' variable
sum := num1 + num2;

// Display result to user
WriteLn('Sum of ', num1, ' + ', num2, ' = ', sum);
