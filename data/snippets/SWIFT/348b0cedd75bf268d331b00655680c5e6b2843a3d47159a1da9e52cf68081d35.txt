// This is a program that calculates the average of three numbers
var num1 = 5.5 // First number
var num2 = 3.2 // Second number
var num3 = 7.8 // Third number

// Calculate sum of the three numbers
var sum = num1 + num2 + num3

// Calculate average
var avg = sum / 3.0

// Print the result
print("The average is: \(avg)")
