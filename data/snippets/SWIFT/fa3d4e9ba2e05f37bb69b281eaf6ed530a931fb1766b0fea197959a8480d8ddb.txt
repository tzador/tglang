let greeting = "Hello, world!" // declaring a variable 'greeting' with the value "Hello, world!"
var greetingCount = 0 // declaring a variable 'greetingCount' with the value 0
while greetingCount < 5 { // creating a loop that runs until the value of 'greetingCount' is less than 5
    print(greeting) // printing the value of 'greeting' 
    greetingCount += 1 // incrementing the value of 'greetingCount' by 1
}
// Output:
Hello, world!
Hello, world!
Hello, world!
Hello, world!
Hello, world!
