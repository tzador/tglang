var numArray: [Int] = [5,7,2,9,3] // Declaring and initializing an array of integers
var sum = 0 // Initializing a variable for sum
for num in numArray { // Looping through each element in the array
    sum += num // Adding each element to the sum variable
}
print("The sum of the numbers is \(sum)") // Printing the sum of the numbers
// Output: The sum of the numbers is 26
