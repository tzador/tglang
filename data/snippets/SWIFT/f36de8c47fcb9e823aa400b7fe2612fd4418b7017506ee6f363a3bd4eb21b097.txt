// This is a program that calculates the average of three numbers

// Creating three variables to store the numbers
var num1 = 4
var num2 = 8
var num3 = 12

// Calculating the sum of the numbers
var sum = num1 + num2 + num3

// Calculating the average by dividing the sum by 3
var average = sum / 3

// Printing the result
print("The average of \(num1), \(num2), and \(num3) is \(average)")


// Output: The average of 4, 8, and 12 is 8
