let numbers = [1, 2, 3, 4, 5]

// loop through numbers using for-in loop
for number in numbers {
    // check if number is divisible by 2
    if number % 2 == 0 {
        // print "even" if number is even
        print("even")
    } else {
        // print "odd" if number is odd
        print("odd")
    }
}
