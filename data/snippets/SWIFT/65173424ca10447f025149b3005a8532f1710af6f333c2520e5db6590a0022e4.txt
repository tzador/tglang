*// String Interpolation*
let name = "Jane"
let age = 25
let introMessage = "Hi, my name is \(name) and I am \(age) years old."

/* The above code uses string interpolation to insert the values of the 'name' and 'age' variables into the 'introMessage' string. */

print(introMessage)
// Output: Hi, my name is Jane and I am 25 years old.
