var numbers = [1, 2, 3, 4, 5] //This is an array of integers
var total = 0 //This is a variable to store the total sum
for num in numbers { //This is a for loop to iterate through the numbers array
    total += num //Add each number to the total sum variable
}
print("The sum of the numbers is \(total)") //Print the final result
