// Declare an array of strings
var fruitList: [String] = ["apple", "orange", "banana", "strawberry"]

// Use a for loop to iterate through the array
for fruit in fruitList {
    // Add the fruit to the console output
    print(fruit)
    
    // Check if the fruit is equal to "banana"
    if fruit == "banana" {
        // If it is, print a special message
        print("I love bananas!")
    }
}
