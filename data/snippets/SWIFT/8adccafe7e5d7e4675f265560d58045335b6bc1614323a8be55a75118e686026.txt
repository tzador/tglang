let array = [1, 2, 3, 4, 5] // create an array with five integers
var sum = 0 // initialize a variable to hold the sum of the elements in the array

for num in array { // loop through each element in the array
    sum += num // add the element to the sum variable
}

print("The sum of the array is: \(sum)") // output the sum of the array

/*
Output:
The sum of the array is: 15
*/
