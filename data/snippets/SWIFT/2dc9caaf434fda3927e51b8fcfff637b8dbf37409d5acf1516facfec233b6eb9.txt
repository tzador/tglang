let rainbowColors = ["Red", "Orange", "Yellow", "Green", "Blue", "Indigo", "Violet"]
// Creates an array of rainbow colors

func printRainbowColors() {
    for color in rainbowColors {
        print(color)
    }
}
// Defines a function that prints each color in the rainbowColors array

printRainbowColors()
// Calls the function to print the rainbow colors
