// This variable holds the value of a first name
var firstName: String = "John"

// This variable holds the value of a last name
var lastName: String = "Doe"

// This function concatenates the first and last name and prints a greeting
func generateGreeting() {
    print("Hello, \(firstName) \(lastName)!")
}

// This array holds a list of ages
var ages: [Int] = [20, 30, 40, 50]

// This loop iterates through the ages and prints each age
for age in ages {
    print(age)
}

// This variable holds the value of a boolean
var isRaining: Bool = true

// This condition checks if it is raining and prints an appropriate message
if isRaining {
    print("Don't forget your umbrella!")
} else {
    print("Enjoy the sunshine!")
}
