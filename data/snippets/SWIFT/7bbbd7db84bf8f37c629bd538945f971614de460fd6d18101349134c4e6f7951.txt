let aircraft = ["Boeing 747", "Airbus A380", "Boeing 777", "Airbus A350", "Boeing 767"]

func printAircraft(_ planes: [String]) {
    for plane in planes {
        print("The \(plane) is a popular choice for long-haul flights.")
    }
}

printAircraft(aircraft)

// Output:
// The Boeing 747 is a popular choice for long-haul flights.
// The Airbus A380 is a popular choice for long-haul flights.
// The Boeing 777 is a popular choice for long-haul flights.
// The Airbus A350 is a popular choice for long-haul flights.
// The Boeing 767 is a popular choice for long-haul flights.
