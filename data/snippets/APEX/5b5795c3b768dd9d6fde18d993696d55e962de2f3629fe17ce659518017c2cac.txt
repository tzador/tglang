// Declaration of a custom class
public class CustomClass {
    
    // Declaration of a private property
    private String name;
    
    // Declaration of a constructor
    public CustomClass(String newName) {
        this.name = newName;
    }
    
    // Declaration of a public method
    public String getName() {
        return this.name;
    }
    
    // Declaration of a setter method
    public void setName(String newName) {
        this.name = newName;
    }
}
