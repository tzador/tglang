/* This program calculates the sum of two given numbers */
public class AddNumbers {
    /* Main method that takes in two numbers and prints the sum */
    public static void main(Integer num1, Integer num2) {
        /* Variables to store the numbers and the sum */
        Integer number1 = num1;
        Integer number2 = num2;
        Integer sum = number1 + number2;

        /* Print the sum to the console */
        System.out.println("The sum of " + number1 + " and " + number2 + " is: " + sum);
    }
}
