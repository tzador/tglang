/**
 * This class calculates the sum of two given integers.
 *
 * @param int1 The first integer to be added.
 * @param int2 The second integer to be added.
 * @return The sum of the two integers.
 */
public class IntegerSumCalculator {
    public Integer calculateSum(Integer int1, Integer int2) {
        return int1 + int2;
    }
}
