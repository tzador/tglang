//This code snippet generates a random string of characters
//using the built-in Apex Crypto class

//Declare a String variable to store the generated random string
String randomString;

//Generate a random number within the character range
Integer randomLen = (Math.random() * 20).intValue();

//Specify the character range used in the random string
String charSet = 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';

//Create a new instance of the Crypto.RandomByteGenerator class
//to generate the random bytes
Crypto.RandomByteGenerator generator = Crypto.getRandomByteGenerator();

//Create a new array to store the random bytes
Blob randomBytes = new Blob(new Integer[]{});

//Loop through the randomLen and add randomly generated bytes to the array
for (Integer i=0; i<randomLen; i++){
    randomBytes.add(generator.nextBytes(1)[0]);
}

//Convert the bytes into a string using the specified charSet
randomString = randomBytes.toString().toUpperCase().getChars(charSet, 0, randomLen).toString();

//Print the generated random string
System.debug(randomString);
