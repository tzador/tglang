Public class ControllerClass
{
    // Define Variables 
    Integer num1 = 5;
    Integer num2 = 10;
    
    // Create a method to calculate the sum of num1 and num2
    Public Integer calculateSum()
    {
        // Add num1 and num2 
        Integer sum = num1 + num2;
        return sum;
    }
    
    // Create a method to calculate the product of num1 and num2
    Public Integer calculateProduct()
    {
        // Multiply num1 and num2 
        Integer product = num1 * num2;
        return product;
    }
    
    // Create a method to calculate the difference of num1 and num2
    Public Integer calculateDifference()
    {
        // Subtract num2 from num1 
        Integer difference = num1 - num2;
        return difference;
    }
}
