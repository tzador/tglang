// This program calculates the sum of first 100 natural numbers

// Declaring an integer variable
Integer sum = 0;

// Using a for loop to iterate through first 100 natural numbers
for(Integer i = 1; i <= 100; i++){
    // Adding current number to sum variable
    sum += i;
}

// Printing the final result
System.debug('The sum of first 100 natural numbers is: ' + sum);

// Output: The sum of first 100 natural numbers is: 5050
