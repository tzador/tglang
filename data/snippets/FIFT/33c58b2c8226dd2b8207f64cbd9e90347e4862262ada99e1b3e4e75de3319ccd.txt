<?fift

    // Define a function that calculates the sum of two numbers
    fun sum(a, b) {
        return a + b
    }

    // Assign variables for input numbers
    val num1 = 5
    val num2 = 10

    // Call the sum function and save the output
    val result = sum(num1, num2)

    // Print the result to the console
    print("The sum of " + num1 + " and " + num2 + " is: " + result)

?>
