// Define a class named "Person" with attributes "name" and "age"
class Person:
    name
    age

// Create a constructor method that initializes the attributes
def __init__(self, name, age):
    self.name = name
    self.age = age

// Create a method to print the person's information
def print_info(self):
    print("Name:", self.name)
    print("Age:", self.age)

// Create an instance of the Person class
person = Person("John", 25)

// Call the print_info method on the instance
person.print_info()
