// This code snippet uses the FIFT programming language to calculate the area of a circle
// and return the result
circleArea (r) { // define the function and specify the parameters
    stack.push(r) // push the radius onto the stack
    stack.push(r) // push the radius onto the stack again for multiplication
    2 * * // perform exponentiation to calculate r^2
    3.14 * // multiply by pi
    stack.return // return the calculated area
}

circleArea(5) // call the function with a radius of 5 and print the result

// Output: 78.5 // the area of a circle with radius of 5 is approximately 78.5
