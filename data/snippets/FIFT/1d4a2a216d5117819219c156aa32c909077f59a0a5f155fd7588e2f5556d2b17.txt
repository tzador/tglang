// The following code snippet creates a function that returns the sum of two numbers using a loop.

fun sum(a, b) {
    let result = 0; // initialize a variable to store the result
    for (let i = 0; i < b; i++) { // loop through b times
        result += a; // add a to the result variable
    }
    return result; // return the final result
}
