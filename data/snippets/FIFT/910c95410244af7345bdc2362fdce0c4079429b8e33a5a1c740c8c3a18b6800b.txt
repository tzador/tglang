// This program calculates and prints the sum of two numbers
// Declare and initialize the first number
num1 = 10

// Declare and initialize the second number
num2 = 20

/* Declare and initialize a variable to store the sum
   This variable will be used to display the result */
res = 0

// Add num1 and num2 and store the result in the res variable
res = num1 + num2

// Print the result
PRINT "The sum of " + num1 + " and " + num2 + " is " + res
