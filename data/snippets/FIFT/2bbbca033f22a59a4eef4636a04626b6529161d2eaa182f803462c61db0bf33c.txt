{
  // Define an array of numbers
  nums = [1, 2, 3, 4, 5];
  // Loop through the array and increment each number by 1
  for i in nums {
    nums[i] = nums[i] + 1;
    // Print out the updated number
    "# Updated number: " + nums[i];
  }
  // Print out the full array
  nums;
}
// Output: [2, 3, 4, 5, 6]
