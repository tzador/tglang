// This function checks if a number is prime
func isPrime(num) {
  if (num <= 1) {
    return false;
  } else if (num <= 3) {
    return true;
  } else if (num % 2 == 0 || num % 3 == 0) {
    return false;
  }

  i = 5;

  // Check if a number is prime using the 6k Â± 1 rule
  while (i * i <= num) {
    if (num % i == 0 || num % (i + 2) == 0) {
      return false;
    }
    i = i + 6;
  }

  return true;
}
