// This section defines the main function
function main() {
  // This is a variable declaration
  let num = 10;
  
  // This is a for loop
  for (let i = 0; i < num; i++) {
    // This is an if statement
    if (i % 2 == 0) {
      // This is a print statement
      print("EVEN");
    } else {
      // This is another print statement
      print("ODD");
    }
  }
}

// This is a function call to the main function
main();
