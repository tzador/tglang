def function(a, b):  # Defining a function with parameters 'a' and 'b'
    if a > b:  # Checking if 'a' is greater than 'b'
        print("Parameter 'a' is greater than 'b'")  # If true, print statement
    elif a < b:  # Checking if 'a' is less than 'b'
        print("Parameter 'a' is less than 'b'")  # If true, print statement
    else:  # If both conditions are false
        print("Parameters 'a' and 'b' are equal")  # Print statement

function(5, 10)  # Calling the function with arguments 5 and 10
