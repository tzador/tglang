<{exe}>:                 ( Define a new executable function )
   <{stack}>: $!          ( Set stack pointer )
   <len> 0<              ( Push 0 as length of array )
   <array> generate-array ( Generate a new empty array )
   <{stack}>: #!          ( Reset stack pointer to default )
;
<{add}>:                  ( Define a new function to add two numbers )
   <{stack}>: $!          ( Set stack pointer )
   #<x>!                  ( Push first argument onto stack )
   #<y>!                  ( Push second argument onto stack )
   <{stack}>: >r          ( Move stack pointer to r-register )
   <x> <y> i+ r>          ( Add the two numbers and save result in r-register )
   r<                     ( Move result from r-register to stack )
   <{stack}>: #!          ( Reset stack pointer to default )
   s>                     ( Return result )
;
<{divide}>:               ( Define a new function to divide two numbers )
   <{stack}>: $!          ( Set stack pointer )
   #<x>!                  ( Push first argument onto stack )
   #<y>!                  ( Push second argument onto stack )
   <{stack}>: >r          ( Move stack pointer to r-register )
   <x> <y> i/ r>          ( Divide the two numbers and save quotient result in r-register )
   r<                     ( Move result from r-register to stack )
   <{stack}>: #!          ( Reset stack pointer to default )
   s>                     ( Return result )
;
<{main}>:                 ( Define the main function )
   <{stack}>: $!          ( Set stack pointer )
   <{add}>                 ( Call add function )
   <{divide}>              ( Call divide function )
;
