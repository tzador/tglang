; This code snippet creates a custom class in FIFT
; with a few methods and properties
Class MyClass:
    Strings:
        name
        address
        email

    Methods:
        SetName:
            ; sets the value of the name property
            :name: ; store the passed argument in the name property
        SetAddress:
            ; sets the value of the address property
            :address: ; store the passed argument in the address property
        SetEmail:
            ; sets the value of the email property
            :email: ; store the passed argument in the email property
        GetName:
            ; returns the value of the name property
            name ; return the value of the name property
        GetAddress:
            ; returns the value of the address property
            address ; return the value of the address property
        GetEmail:
            ; returns the value of the email property
            email ; return the value of the email property
