#FIFT PROGRAM TO CALCULATE THE PERIMETER AND AREA OF A RECTANGLE

# Define input variables for length and width of rectangle
length := 5
width := 10

# Calculate perimeter using FIFT formula
perimeter := (2 * length) + (2 * width)

# Calculate area using FIFT formula
area := length * width

# Print out results with labels using FIFT print function
print "The perimeter of the rectangle is ", perimeter
print "The area of the rectangle is ", area
