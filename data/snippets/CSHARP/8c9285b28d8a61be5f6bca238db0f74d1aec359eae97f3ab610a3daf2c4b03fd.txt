using System;                        // Importing the System namespace  
class Program                         // Defining a class named Program  
{                                     // Opening curly bracket for class definition  
    static void Main(string[] args)   // Defining the Main method with string array as input  
    {                                 // Opening curly bracket for Main method definition  
        Console.WriteLine("Hello World!");    // Printing "Hello World!" to the console  
        int num1 = 5;                  // Initializing an integer variable with value 5  
        int num2 = 7;                  // Initializing another integer variable with value 7  
        int result = num1 + num2;      // Adding the two variables and storing the result in a third variable  
        Console.WriteLine("The sum of " + num1 + " and " + num2 + " is " + result + ".");  // Printing the result to the console  
    }                                 // Closing curly bracket for Main method definition  
}                                     // Closing curly bracket for class definition
