// This code snippet is a basic example of a C# program that prints "Hello, World!" to the console

using System; // Imports the System namespace, which contains the Console class

class Program // Creates a new class called "Program"
{
    static void Main() // Creates a static method called "Main" that does not return anything
    {
        Console.WriteLine("Hello, World!"); // Calls the WriteLine method from the Console class to print "Hello, World!" to the console
    }
}

// Output: Hello, World!
