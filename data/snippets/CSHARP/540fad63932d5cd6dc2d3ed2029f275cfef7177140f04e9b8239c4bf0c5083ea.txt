/* This code snippet is an example of a complex if-else statement in CSHARP */

// First, we declare two variables
int num1 = 5;
int num2 = 10;

// Then we use an if-else statement to check if num1 is greater than num2
if (num1 > num2)
{
    // If num1 is greater, we print a message stating so
    Console.WriteLine("num1 is greater than num2");
}
else if (num2 > num1)
{
    // If num2 is greater, we print a message stating so
    Console.WriteLine("num2 is greater than num1");
}
else
{
    // If neither condition is true, we print a message stating they are equal
    Console.WriteLine("num1 and num2 are equal");
}
// The output would be "num2 is greater than num1"
