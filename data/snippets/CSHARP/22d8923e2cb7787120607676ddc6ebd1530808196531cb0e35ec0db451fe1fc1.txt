using System;  // Importing the necessary libraries

namespace HelloWorld // Creating a namespace to avoid naming conflicts
{
    public class Program // Creating a public class named Program
    {
        static void Main(string[] args) // Creating a main method, which serves as the entry point of the program
        {
            Console.WriteLine("Hello, world!"); // Printing "Hello, world!" to the console
            Console.ReadLine(); // Allowing the user to input a value before the console closes
        }
    }
}
