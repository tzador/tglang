/// <summary>
        /// This is a method that takes in two strings and concatenates them together.
        /// </summary>
        /// <param name="str1">The first string to be concatenated.</param>
        /// <param name="str2">The second string to be concatenated.</param>
        /// <returns>The concatenated string of the two input strings.</returns>
        public string ConcatenateStrings(string str1, string str2)
        {
            string result = str1 + str2;
            return result;
        }
