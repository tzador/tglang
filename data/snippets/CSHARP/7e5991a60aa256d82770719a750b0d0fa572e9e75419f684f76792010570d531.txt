// This code snippet creates a method that takes in two integer parameters
// and returns their sum as an integer value.

// Define the Add method with int return type and two int parameters, a and b
public int Add(int a, int b)
{
    // Initialize a variable to store the sum
    int sum = 0;
    
    // Add the two parameters and store the result in the sum variable
    sum = a + b;
    
    // Return the sum as the method output
    return sum;
}
