int[] numbers = {1, 2, 3, 4, 5}; // Declare an integer array with 5 elements
for(int i = 0; i < numbers.Length; i++) // Use a for loop to iterate through the array
{
   Console.WriteLine($"Number at index {i}: {numbers[i]}"); // Print the index and value of each element
}

// Output:
// Number at index 0: 1
// Number at index 1: 2
// Number at index 2: 3
// Number at index 3: 4
// Number at index 4: 5
