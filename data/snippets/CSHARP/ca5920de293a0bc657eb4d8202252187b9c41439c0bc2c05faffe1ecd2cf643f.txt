/// <summary>
/// This method takes in two integers and returns their average.
/// </summary>
/// <param name="num1">The first integer to be averaged</param>
/// <param name="num2">The second integer to be averaged</param>
/// <returns>The average of the two integers as a double</returns>
public double CalculateAverage(int num1, int num2)
{
    double average = (num1 + num2) / 2.0;
    return average;
}
