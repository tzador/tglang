class Program
{
    static void Main(string[] args) // declaring main method
    {
        int x = 5; // declaring an integer variable with initial value of 5
        int y = 10; // declaring another integer variable with initial value of 10
        int z = x + y; // adding the values of x and y and storing it in z
        Console.WriteLine("The sum of {0} and {1} is {2}", x, y, z); 
        // printing the values of x, y and z in the specified format
        Console.ReadLine(); // waiting for user input before closing the program
    }
}
