//This program calculates and displays the sum of two numbers

using System;

public class Program
{
    public static void Main()
    {
        //declare and initialize variables 
        int num1 = 5;
        int num2 = 10;
        
        //calculate the sum 
        int sum = num1 + num2;
        
        //display output 
        Console.WriteLine($"The sum of {num1} and {num2} is {sum}.");
    }
}
