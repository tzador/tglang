//This program prints out a list of numbers from 1 to 10

using System; //imports the necessary libraries

class Program //class declaration
{
    static void Main() //main function declaration
    {
        for (int i = 0; i <= 10; i++) //for loop, loop will run 10 times
        {
            Console.WriteLine(i); //prints out the current value of i
        }
    }
}
