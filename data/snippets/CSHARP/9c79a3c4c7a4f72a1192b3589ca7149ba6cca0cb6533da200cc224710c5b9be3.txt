//Creates a class for a book with properties for title, author, number of pages and publisher
public class Book
{
    public string Title { get; set; } //Property for the title of the book
    public string Author { get; set; } //Property for the author of the book
    public int Pages { get; set; } //Property for the number of pages in the book
    public string Publisher { get; set; } //Property for the publisher of the book
}
