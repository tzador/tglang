using System;

public class Program
{
    public static void Main()
    {
        // This variable will store the sum of two numbers
        int sum = 0;

        // This variable will hold the first number
        int num1 = 5;

        // This variable will hold the second number
        int num2 = 10;

        // Add the two numbers and store the result in the sum variable
        sum = num1 + num2;

        // Print out the sum
        Console.WriteLine("The sum of {0} and {1} is {2}", num1, num2, sum);
    }
}
