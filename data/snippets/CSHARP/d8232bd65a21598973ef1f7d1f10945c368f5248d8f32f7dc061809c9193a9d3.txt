//Set up variables
var num1 = 2;
var num2 = 5;

//Check if num1 is greater than num2
if(num1 > num2)
{
    // If true, display a message
    Console.WriteLine("Number 1 is greater than Number 2");
}
// If false, move onto next part of code
else
{
    //Add num1 and num2 together
    var sum = num1 + num2;
    // Display result
    Console.WriteLine($"The sum of Number 1 and Number 2 is: {sum}");
}
