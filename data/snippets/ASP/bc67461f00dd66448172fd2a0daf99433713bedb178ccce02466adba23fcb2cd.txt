<% Option Explicit %>
<!-- Initializing variables -->
<% Dim a, b, c %>
<!-- Declaring variables -->
<% Dim num1, num2, num3 %>
<!-- Assigning values to variables -->
<% num1 = 5 %>
<% num2 = 10 %>
<% num3 = 15 %>
<!-- Performing mathematical operations -->
<% a = num1 + num2 %>
<% b = num2 * num3 %>
<% c = num3 - num1 %>
<!-- Outputting values -->
<%= "The sum of num1 and num2 is " & a %>
<%= "The product of num2 and num3 is " & b %>
<%= "The difference between num3 and num1 is " & c %>
