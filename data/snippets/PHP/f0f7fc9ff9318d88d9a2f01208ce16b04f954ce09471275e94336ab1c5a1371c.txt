<?php
// This is a simple PHP program that performs basic arithmetic operations and output the results
$number1 = 12;
$number2 = 4;
// Addition
$result = $number1 + $number2;
echo "Sum of {$number1} and {$number2} is: {$result}" . PHP_EOL;
// Subtraction
$result = $number1 - $number2;
echo "Difference between {$number1} and {$number2} is: {$result}" . PHP_EOL;
// Multiplication
$result = $number1 * $number2;
echo "Product of {$number1} and {$number2} is: {$result}" . PHP_EOL;
// Division
$result = $number1 / $number2;
echo "Quotient of {$number1} and {$number2} is: {$result}" . PHP_EOL;
// Modulus
$result = $number1 % $number2;
echo "Remainder when {$number1} is divided by {$number2} is: {$result}" . PHP_EOL;
// Exponentiation
$result = $number1 ** $number2;
echo "{$number1} raised to the power of {$number2} is: {$result}" . PHP_EOL;
