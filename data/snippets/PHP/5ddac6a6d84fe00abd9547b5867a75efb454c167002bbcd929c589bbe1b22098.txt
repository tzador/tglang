<?php
// This is a PHP code snippet
$numbers = array(1,2,3,4,5); // An array of numbers
$sum = 0; // Variable to store the sum of numbers
foreach($numbers as $num){ // Loop through each number in the array
	$sum += $num; // Add the number to the sum
}
echo "The sum of the numbers is: " . $sum; // Print the result
?>
