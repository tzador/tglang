<?php

// An array containing numbers
$numbers = [1, 2, 3, 4, 5]; 

// A variable to store the sum
$sum = 0; 

// Loops through the numbers array and adds each element to the sum variable
foreach($numbers as $num) { 
    $sum += $num; 
} 

// Prints the sum variable
echo "The sum of the numbers is: $sum"; 

?>
