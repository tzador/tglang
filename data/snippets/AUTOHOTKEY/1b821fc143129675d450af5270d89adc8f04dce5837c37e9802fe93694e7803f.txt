SetBatchLines, -1 ; Disable waiting for user input 

#IfWinActive, ahk_exe notepad.exe ; Only run this code if Notepad is the active window 

F1:: ; Assign F1 key to trigger script 

Loop
{
	Send, This code snippet is generated by me! ; Sends the specified text to the active window 
	Sleep, 1000 ; Pauses for 1 second 
	Send, %A_ScreenWidth% x %A_ScreenHeight% ; Sends the screen width and height to the active window 
	Sleep, 1000 ; Pauses for 1 second 
	Send, %A_NumGetKeyState("F1", "P") ; Sends the current state of the F1 key to the active window 
	Sleep, 1000 ; Pauses for 1 second 
	Send, {Enter} ; Sends an enter keystroke to the active window 
	Sleep, 1000 ; Pauses for 1 second 
} 

F2:: ; Assign F2 key to kill script 
ExitApp ; Exits the current script
