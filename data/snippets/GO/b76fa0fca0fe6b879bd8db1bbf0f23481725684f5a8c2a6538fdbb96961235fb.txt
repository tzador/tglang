// Define a function named 'calculate' that takes in two integer parameters and returns an integer
func calculate(x int, y int) int {
	// Declare a variable 'result' that will store the result of the calculation
	var result int 
	// Assign the value x + y to the variable 'result'
	result = x + y 
	// Return the value of 'result' 
	return result
}

// Define a variable 'a' and assign the value 5 to it. 
var a = 5

// Define a variable 'b' and assign the value 10 to it.
var b = 10

// Call the 'calculate' function and pass in the values of 'a' and 'b'.
calculate(a, b)

// Output: The output of the function 'calculate' will be 15.
