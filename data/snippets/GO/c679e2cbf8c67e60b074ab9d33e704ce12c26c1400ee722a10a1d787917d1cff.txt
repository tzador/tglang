package main
import "fmt"

func main() {
  var x int = 10
  fmt.Println("The value of x is:", x) //printing initial value of x 
  var y int = 5
  fmt.Println("The value of y is:", y) //printing initial value of y
  var result int = x + y
  fmt.Println("The result of x + y is:", result) //printing the result of x + y
}
