// This is a complex code snippet in GO language with annotations

package main // package keyword for creating a main package

import "fmt" // import fmt package for formatting and printing

func main() { // main function declaration
	// initializing two variables with integer values
	num1 := 10
	num2 := 5

	// performing basic arithmetic operations
	sum := num1 + num2
	sub := num1 - num2
	mul := num1 * num2
	div := num1 / num2
	mod := num1 % num2

	// printing the results with annotations
	fmt.Println("Sum:", sum) // printing the sum
	fmt.Println("Subtraction:", sub) // printing the subtraction
	fmt.Println("Multiplication:", mul) // printing the multiplication
	fmt.Println("Division:", div) // printing the division
	fmt.Println("Modulo:", mod) // printing the modulo
}
