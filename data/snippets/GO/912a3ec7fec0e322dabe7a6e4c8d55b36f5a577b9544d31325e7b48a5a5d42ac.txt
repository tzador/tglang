// Declare a function named "calculateProduct" that takes in two floats, "a" and "b", as parameters
func calculateProduct(a float64, b float64) float64 {
    // Declare a variable "product" that will hold the product of variables "a" and "b"
    var product float64

    // Multiply the two variables and store the result in "product" variable
    product = a * b

    // Return the value of "product" variable as output
    return product
}
