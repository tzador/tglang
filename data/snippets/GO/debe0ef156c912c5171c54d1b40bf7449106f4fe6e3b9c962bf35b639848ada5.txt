package main

import "fmt"

func main() {
   // assigning two variables
   var a int = 10
   var b int = 20

   // using if statement
   if a < b {
      // if condition is true, print the following
      fmt.Printf("a is less than b")
   } else {
      // if condition is false, print the following
      fmt.Printf("a is greater than b")
   }
}
