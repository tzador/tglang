package main  

// declaring the variables
var number1 = 10  
var number2 = 5  

// performing addition
var addition = number1 + number2  

// taking input from user  
fmt.Printf("Enter a number: ")  
fmt.Scanf("%d", &number1)  
fmt.Printf("Enter another number: ")  
fmt.Scanf("%d", &number2)  

// calculating the difference  
var difference = number1 - number2  

// printing the result  
fmt.Printf("Result of addition: %d\n", addition)  
fmt.Printf("Result of subtraction: %d\n", difference)
