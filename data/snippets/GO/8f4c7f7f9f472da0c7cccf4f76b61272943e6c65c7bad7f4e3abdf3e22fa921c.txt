import (
    "fmt"
    "time"
)

// creates a function that returns the square root of a number
func findSquareRoot(num int) int {
    counter := 0
    for i := 0; i * i <= num; i++ {
        counter++
    }
    return counter
}

func main() {
    startTime := time.Now()
    squareRoot := findSquareRoot(25)
    fmt.Println(squareRoot)
    fmt.Println("Function took", time.Since(startTime))
}
