// Define a function that takes in two ints and returns the sum of those ints
func sum(x, y int) int {
    // Declare a variable to store the sum
    var result int
    // Add the two ints and store the result in the variable
    result = x + y
    // Return the result
    return result
}

// Call the function and print the result
fmt.Println(sum(5, 10))

// Output: 15
