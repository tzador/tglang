fn main() { // Defines the main function
    println!("Hello, world!"); // Prints "Hello, world!" to the console
    let x = 5; // Declares and initializes a variable with value 5
    let y: i32 = 10; // Declares and initializes a variable with value 10, explicitly specifying it as a 32-bit integer
    let z = x + y; // Declares and initializes a variable as the sum of x and y
    println!("The value of z is {}", z); // Prints the value of z to the console using string interpolation
} // Closes the main function
