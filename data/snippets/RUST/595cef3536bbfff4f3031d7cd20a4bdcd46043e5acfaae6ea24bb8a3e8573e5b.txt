fn main() {
    // Initialize variables
    let mut num1 = 10;
    let mut num2 = 20;
    
    // Calculate product of num1 and num2
    let product = num1 * num2;
    
    // Calculate quotient of num1 and num2
    let quotient = num2 / num1;
    
    // Print results
    println!("The product of {} and {} is {}", num1, num2, product);
    println!("The quotient of {} and {} is {}", num2, num1, quotient);
}
