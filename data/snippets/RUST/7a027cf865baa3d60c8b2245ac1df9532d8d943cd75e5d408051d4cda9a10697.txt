fn main() {  //declare a main function
    let a = 5;  //initialize a variable 'a' with value of integer '5'
    let b = 10;  //initialize a variable 'b' with value of integer '10'
    let c = a + b;  //declare variable 'c' and assign it the sum of 'a' and 'b'
    println!("The sum of {} and {} is {}", a, b, c);  //print the sum of 'a' and 'b' with their respective values
}
//output: The sum of 5 and 10 is 15
