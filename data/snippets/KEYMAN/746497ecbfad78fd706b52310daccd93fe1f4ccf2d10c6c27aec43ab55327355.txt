[KEYMAN-VIEW
    NAME='exampleView' // creates a named view
    KEYBOARD='exampleKeyboard' // assigns a keyboard to the view
    OVERLAY='exampleOverlay' // optionally, adds an overlay to the view
    LAYOUT='exampleLayout' // defines the layout of the view, can be a keyboard or a language-specific layout
    DISPLAY='exampleDisplay' // optional, specifies a displayable object for the view
    OVERRIDE='exampleOverride' // optional, specifies an override map for the view
    MENU='exampleMenu' // optionally, adds a context menu to the view
    ONKEY='exampleOnKey' // optional, adds a function to handle individual keypresses in the view
    ONUI='exampleOnUI' // optional, adds a function to handle UI events in the view
    ONLOAD='exampleOnLoad' // optional, adds a function to execute when the view is loaded
    ONSHOW='exampleOnShow' // optional, adds a function to execute when the view is shown
    ONSAVE='exampleOnSave' // optional, adds a function to execute when the view is saved
    ONCLOSE='exampleOnClose' // optional, adds a function to execute when the view is closed
]
