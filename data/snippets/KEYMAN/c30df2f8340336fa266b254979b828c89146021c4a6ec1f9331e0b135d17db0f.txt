begin_program
  variable i as integer //Declare an integer variable
  for i = 1 to 10 //Start a for loop
    if i % 2 = 0 then //Check if i is even
      output("Even") //Output "Even" if i is even
    else //If i is odd
      output("Odd") //Output "Odd" if i is odd
    end if //End if statement
  end for //End for loop
end_program //End program
