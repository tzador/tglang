store(&filename) 'Text file name'
store(&file) "<% filename %>' Text file content"
begin 'Start of the program'
    set(&index, 0) 'Variable for indexing'
    for(&line, &file) 'Loop through each line in the text file'
        store(&words) 'Array for storing words'
        for(&word, &line) 'Loop through each word in the line'
            if(&index < 10) 'Check if index is less than 10'
                store(&words, &word) 'Add word to the array'
                set(&index, &index + 1) 'Update index'
            endif 'End of if statement'
        endfor 'End of word loop'
        call(encrypt, &words) 'Call encrypt function with word array as parameter'
    endfor 'End of line loop'
    save(&filename, "<% decrypt() %>") 'Save decrypted words to text file'
end 'End of program'

function encrypt (&words) 'Function for encrypting words'
    store(&encrypted) 'Array for storing encrypted words'
    for(&word, &words) 'Loop through each word in the array'
        set(&encryptedWord, "") 'Variable for storing encrypted word'
        for(&char, &word) 'Loop through each character in the word'
            set(&charCode, asc(&char)) 'Get ASCII code for character'
            set(&encryptedChar, &charCode + 1) 'Add 1 to ASCII code for encryption'
            set(&encryptedWord, &encryptedWord + chr(&encryptedChar)) 'Convert ASCII code back to character and add to encrypted word'
        endfor 'End of character loop'
        store(&encrypted, &encryptedWord) 'Add encrypted word to array'
    endfor 'End of word loop'
    return (&encrypted) 'Return array of encrypted words'
endfunction 'End of encrypt function'

function decrypt () 'Function for decrypting words'
    for(&word, &encrypted) 'Loop through each word in the encrypted array'
        set(&decryptedWord, "") 'Variable for storing decrypted word'
        for(&char, &word) 'Loop through each character in the word'
            set(&charCode, asc(&char)) 'Get ASCII code for character'
            set(&decryptedChar, &charCode - 1) 'Subtract 1 from ASCII code for decryption'
            set(&decryptedWord, &decryptedWord + chr(&decryptedChar)) 'Convert ASCII code back to character and add to decrypted word'
        endfor 'End of character loop'
        return (&decryptedWord) 'Return decrypted word'
    endfor 'End of word loop'
endfunction 'End of decrypt function'
