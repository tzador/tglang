/* Declaring a variable of type integer */
Variable(1) i;
/* Assigning the value of 10 to the variable */
i = 10;
/* Looping through the values from 1 to 10 */
For(1) j {
    /* Conditional statement using the modulus operator */
    if(i % j == 0) {
        /* Output the value of i and j */
        "i is divisible by j"
    }
}
