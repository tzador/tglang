+ [MIK Character Map] > store(&KMWD) 'KEYMAN'

    store(&KMWD) 'KEYMAN'
    # This command stores the string 'KEYMAN' in the system variable KMWD.
    # KMWD is used to define the keyboard for the keyboard manager in Keyman Developer. 

    alerter("Hello, World!")
    # This command displays a pop-up message with the text "Hello, World!".
    # The message will appear when SHIFT is pressed in the keyboard defined by KMWD.
