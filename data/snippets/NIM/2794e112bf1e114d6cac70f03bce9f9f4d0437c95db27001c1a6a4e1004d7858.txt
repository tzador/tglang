# This is a code snippet for a program in NIM
var
    num1, num2, sum : int # declaring variables 'num1' and 'num2' as integers, and 'sum' as an integer to store the result
echo "Enter first number: " # asking user to input first number
read num1 # storing the first number in the variable 'num1'

read num2 # storing the second number in the variable 'num2'
sum = num1 + num2 # calculating sum and storing the result in the variable 'sum'
echo "The sum of" , num1, "and" , num2, "is", sum # displaying the result to the user
