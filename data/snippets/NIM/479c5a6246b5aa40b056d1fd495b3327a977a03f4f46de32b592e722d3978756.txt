# NIM implementation of a binary search algorithm
proc binarySearch(arr: seq[int], key: int): int {.exportc.} =
  var low, high, mid: int
  low = 0
  high = len(arr) - 1
  while low <= high:
    mid = (low + high) div 2
    if arr[mid] == key:
      return mid
    elif arr[mid] < key:
      low = mid + 1
    else:
      high = mid - 1
  return -1
