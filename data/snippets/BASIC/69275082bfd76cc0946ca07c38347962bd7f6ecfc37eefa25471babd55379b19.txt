' This program will calculate the area of a trapezoid

' prompt user for the bases and height of the trapezoid
INPUT "Enter length of base 1: ", B1
INPUT "Enter length of base 2: ", B2
INPUT "Enter height: ", H

' calculate the area using the formula
A = (B1 + B2) / 2 * H

' display the result
PRINT "The area of the trapezoid is "; A
END
