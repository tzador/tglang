10 DIM A(10), B(10), C(10)
20 FOR I = 1 TO 10
30 A(I) = 2 * I - 1 : B(I) = (I - 3) ^ 2
40 FOR J = 1 TO 10
50 C(I, J) = A(I) + B(J)
60 NEXT J
70 NEXT I
80 PRINT "A:" : FOR I = 1 TO 10 : PRINT A(I) : NEXT I
90 PRINT "B:" : FOR J = 1 TO 10 : PRINT B(J) : NEXT J
100 PRINT "C:"
110 FOR I = 1 TO 10
120 FOR J = 1 TO 10
130 PRINT C(I, J),
140 NEXT J
150 PRINT
160 NEXT I
