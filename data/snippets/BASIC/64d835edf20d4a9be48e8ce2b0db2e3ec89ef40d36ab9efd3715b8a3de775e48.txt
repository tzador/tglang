' This program will calculate the area of a trapezoid
' Input the values for the base, top, and height
INPUT "Enter the value for the base: "; b
INPUT "Enter the value for the top: "; t
INPUT "Enter the value for the height: "; h
' Calculate the area using the formula (base + top) * height / 2
LET area = (b + t) * h / 2
' Output the calculated area
PRINT "The area of the trapezoid is: "; area
' {This program uses user input and a specific formula to calculate the area of a trapezoid}
