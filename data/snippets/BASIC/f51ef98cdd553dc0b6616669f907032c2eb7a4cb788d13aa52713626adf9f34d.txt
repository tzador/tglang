' This program calculates the area of a rectangle  
DIM length, width, area AS INTEGER 'set variables for length, width, and area  
INPUT "Enter length: "; length 'get input for length from user  
INPUT "Enter width: "; width 'get input for width from user  
area = length * width 'calculate area by multiplying length and width  
PRINT "Area of rectangle = "; area 'output the result to the user  
END 'end the program
