REM This is a sample code snippet in BASIC

' This is a comment
'------------------------------------------------
' This program converts feet to inches
'
' VARIABLES
DIM length AS INTEGER 'stores length in feet
DIM inches AS INTEGER 'stores converted length in inches
'
' ASK FOR INPUT
PRINT "Enter length in feet: "
INPUT length
'
' CONVERT TO INCHES
inches = length * 12
'
' OUTPUT RESULTS
PRINT "That is equal to "; inches; " inches."
'
' END PROGRAM
END
