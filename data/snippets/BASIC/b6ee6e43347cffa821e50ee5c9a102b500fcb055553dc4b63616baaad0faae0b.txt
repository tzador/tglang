REM This is a BASIC program that calculates the sum of two numbers.
INPUT "Enter first number"; num1       'User inputs first number
INPUT "Enter second number"; num2      'User inputs second number
REM num1 and num2 are stored as floating point variables
sum = num1 + num2                      'Calculates sum and stores it in sum variable
PRINT "The sum of ", num1, " and ", num2, " is "; sum    'Prints out the sum of the two numbers
END                                   'End of program
