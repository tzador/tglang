REM INITIALIZE ARRAY DIMENSIONS
DIM X(10), Y(10)

REM FILL ARRAY WITH RANDOM VALUES
RANDOMIZE TIMER
FOR I = 1 TO 10
  X(I) = INT(RND * 100)
  Y(I) = INT(RND * 100)
NEXT I

REM OUTPUT ARRAYS
PRINT "X ARRAY:"
FOR I = 1 TO 10
  PRINT X(I)
NEXT I

PRINT "Y ARRAY:"
FOR I = 1 TO 10
  PRINT Y(I)
NEXT I

REM CALCULATE AVERAGE OF X ARRAY
DIM AVGX
FOR I = 1 TO 10
  AVGX = AVGX + X(I)
NEXT I
AVGX = AVGX / 10

REM CALCULATE STANDARD DEVIATION OF X ARRAY
DIM STDX, SUMSQ
FOR I = 1 TO 10
  SUMSQ = SUMSQ + (X(I) - AVGX)^2
NEXT I
STDX = SQR(SUMSQ / 10)

REM CALCULATE AVERAGE OF Y ARRAY
DIM AVGY
FOR I = 1 TO 10
  AVGY = AVGY + Y(I)
NEXT I
AVGY = AVGY / 10

REM CALCULATE STANDARD DEVIATION OF Y ARRAY
DIM STDY, SUMSQ
FOR I = 1 TO 10
  SUMSQ = SUMSQ + (Y(I) - AVGY)^2
NEXT I
STDY = SQR(SUMSQ / 10)

REM PRINT RESULTS
PRINT "Average of X array: " & AVGX
PRINT "Standard deviation of X array: " & STDX
PRINT "Average of Y array: " & AVGY
PRINT "Standard deviation of Y array: " & STDY
