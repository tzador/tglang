REM This program calculates the product of two numbers and displays the result.

REM Initialize variables
DIM A AS INTEGER
DIM B AS INTEGER
DIM PRODUCT AS INTEGER

REM Get input from user
PRINT "Enter first number:"
INPUT A
PRINT "Enter second number:"
INPUT B

REM Calculate product
PRODUCT = A * B

REM Display result to user
PRINT "The product is:"
PRINT PRODUCT

END
