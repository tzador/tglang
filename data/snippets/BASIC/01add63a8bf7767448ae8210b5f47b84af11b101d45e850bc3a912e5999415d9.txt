' This code snippet calculates the area of a circle
' and prints the result to the screen.

' Declare variables
DIM radius, area, pi AS SINGLE

' Assign values to variables
radius = 5
pi = 3.14159

' Calculate area using the formula: A = πr²
area = pi * radius * radius

' Display the result to the screen
PRINT "The area of the circle with a radius of "; radius; " is "; area
