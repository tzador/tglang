For x = 1 to 10
    If x Mod 2 = 0 Then
        Print "Even Number: " x 'This line prints the even number
    Else
        Print "Odd Number: " x 'This line prints the odd number
    End If
    If x = 5 Then
        Print "Midpoint reached" 'This line prints a message when the loop reaches 5
    End If
Next x
