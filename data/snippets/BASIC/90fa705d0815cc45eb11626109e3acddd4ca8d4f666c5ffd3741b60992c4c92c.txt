# This is a BASIC program to calculate the average of three numbers
# Declare variables to store three numbers and their sum
DIM num1, num2, num3, sum
# Input the three numbers from the user
PRINT "Enter the first number:"
INPUT num1
PRINT "Enter the second number:"
INPUT num2
PRINT "Enter the third number:"
INPUT num3
# Calculate the sum of the three numbers
sum = num1 + num2 + num3
# Calculate the average by dividing the sum by three
avg = sum / 3
# Print the result to the user
PRINT "The average is: " + avg
# End of program
END
