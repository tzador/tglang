' This program calculates the area of a rectangle

' Declaring variables
DECLARE WIDTH AS INTEGER
DECLARE LENGTH AS INTEGER
DECLARE AREA AS INTEGER

' Prompt user for input
INPUT "Enter the width of the rectangle: ", WIDTH
INPUT "Enter the length of the rectangle: ", LENGTH

' Calculate the area using the formula area = length * width
AREA = LENGTH * WIDTH

' Print the result
PRINT "The area of the rectangle is: "; AREA
END
