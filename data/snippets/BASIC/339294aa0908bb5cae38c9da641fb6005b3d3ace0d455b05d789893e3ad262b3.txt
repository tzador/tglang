REM This program will calculate the average of three numbers
LET num1 = 40       'defining variable num1 with value 40
LET num2 = 25       'defining variable num2 with value 25
LET num3 = 10       'defining variable num3 with value 10

LET total = num1 + num2 + num3     'calculating the total of all three numbers
LET average = total / 3            ' calculating the average by dividing the total by 3

PRINT "The average of " + num1 + ", " + num2 + ", and " + num3 + " is " + average + "."   'displaying the result
END                              'ending the program
