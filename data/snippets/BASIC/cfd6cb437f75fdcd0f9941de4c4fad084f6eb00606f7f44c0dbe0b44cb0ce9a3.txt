REM This is a program that calculates the sum of two numbers.
REM First, we define two variables, x and y.
LET x = 5
LET y = 7
REM Next, we prompt the user to input two numbers.
INPUT "Enter first number: ", x
INPUT "Enter second number: ", y
REM Then, we perform the addition operation and store the result in the variable z.
LET z = x + y
REM Finally, we display the result to the user.
PRINT "The sum of " + x + " and " + y + " is " + z + "."
END
