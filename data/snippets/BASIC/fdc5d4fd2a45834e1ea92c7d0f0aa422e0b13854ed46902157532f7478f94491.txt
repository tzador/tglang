10 REM ***BASIC PROGRAM***
20 DIM FIRST(10), SECOND(10), RESULT(10)
30 FOR I = 1 TO 10
40 INPUT "ENTER A NUMBER: ", FIRST(I)
50 NEXT I
60 FOR J = 1 TO 10
70 INPUT "ENTER ANOTHER NUMBER: ", SECOND(J)
80 NEXT J
90 FOR K = 1 TO 10
100 RESULT(K) = FIRST(K) * SECOND(K)
110 NEXT K
120 PRINT "MULTIPLICATION RESULTS:"
130 FOR L = 1 TO 10
140 PRINT "FIRST NUMBER: "; FIRST(L)
150 PRINT "SECOND NUMBER: "; SECOND(L)
160 PRINT "RESULT: "; RESULT(L)
170 PRINT
180 NEXT L
190 PRINT "END OF PROGRAM"
200 END
