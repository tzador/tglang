' This is a program written in BASIC
' It calculates the average of four numbers

' Initialize variables
number1 = 10
number2 = 25
number3 = 36
number4 = 49
average = 0

' Calculate average
average = (number1 + number2 + number3 + number4) / 4

' Print the result
PRINT "The average of the four numbers is " ; average
END
