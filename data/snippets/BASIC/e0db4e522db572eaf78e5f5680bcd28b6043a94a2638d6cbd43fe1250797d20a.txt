REM GENERATE RANDOM NUMBER
PROCEDURE RAND(MIN, MAX, COUNT)
DIM RESULT(COUNT) AS INTEGER
FOR I = 0 TO COUNT - 1
RESULT(I) = RND * (MAX - MIN) + MIN
NEXT I
END PROCEDURE

REM SORT NUMBERS IN ASCENDING ORDER
PROCEDURE QUICKSORT(ARRAY(), LEFT, RIGHT)
IF LEFT > RIGHT THEN GOTO EXIT
SWAP ARRAY(LEFT), ARRAY(FIX((LEFT + RIGHT) / 2))
SYC = LEFT
FOR I = LEFT + 1 TO RIGHT
IF ARRAY(I) < ARRAY(LEFT) THEN
SYC = SYC + 1
SWAP ARRAY(SYC), ARRAY(I)
ENDIF
NEXT I
SWAP ARRAY(SYC), ARRAY(LEFT)
CALL QUICKSORT(ARRAY, LEFT, SYC - 1)
CALL QUICKSORT(ARRAY, SYC + 1, RIGHT)
EXIT:
END PROCEDURE

REM MAIN FUNCTION
PROCEDURE MAIN()
DIM NUMBERS(10) AS INTEGER
CALL RAND(0, 100, 10)
CALL QUICKSORT(NUMBERS, 0, 9)
FOR I = 0 TO 9
PRINT NUMBERS(I)
NEXT I
END PROCEDURE

REM CALLING MAIN FUNCTION
CALL MAIN
