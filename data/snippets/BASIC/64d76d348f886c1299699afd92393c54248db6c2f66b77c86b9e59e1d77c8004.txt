REM This program calculates the factorial of a given number.

INPUT "Enter a number: "; num 
REM PROMPTS USER TO ENTER A NUMBER AND STORES IT IN VARIABLE 'num'

factorial = 1
REM INITIALIZE FACTORIAL VARIABLE TO 1

FOR i = 1 TO num 
REM LOOP FROM 1 TO THE GIVEN NUMBER

factorial = factorial * i
REM MULTIPLY FACTORIAL VARIABLE WITH LOOP COUNTER 'i' AND STORE IN FACTORIAL VARIABLE

NEXT i 
REM END OF LOOP

PRINT "The factorial of "; num; " is "; factorial
REM PRINT FINAL RESULT TO THE USER

END 
REM END OF PROGRAM
