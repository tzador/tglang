REM this code snippet calculates the average of three numbers
DIM num1, num2, num3, sum, average AS INTEGER
INPUT "Enter the first number: ", num1
INPUT "Enter the second number: ", num2
INPUT "Enter the third number: ", num3
sum = num1 + num2 + num3
average = sum / 3
PRINT "The average of", num1, num2, "and", num3, "is", average
END
