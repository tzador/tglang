' Program to calculate the area of a trapezoid
' INPUT: side a, side b, height h
' OUTPUT: area

INPUT "Enter value for side a: "; a
INPUT "Enter value for side b: "; b
INPUT "Enter value for height h: "; h

REM Calculate the area
area = ((a + b) / 2) * h

PRINT "The area of the trapezoid is: "; area
