REM FUNCTION TO CALCULATE THE SUM OF TWO NUMBERS
REM **This function uses the SUM instruction to add two numbers**
10 INPUT "ENTER FIRST NUMBER: ", A
20 INPUT "ENTER SECOND NUMBER: ", B
30 SUM = A + B
35 PRINT TAB(15); "- CALCULATING SUM -"
40 FOR C = 1 TO 100
50 PRINT "LOADING... " + STR$(C); "%"
60 NEXT C
70 IF SUM < 0 THEN GOTO 100
80 PRINT "THE SUM OF " + STR$(A); " AND " + STR$(B); " IS: ";
90 PRINT SUM
100 END
