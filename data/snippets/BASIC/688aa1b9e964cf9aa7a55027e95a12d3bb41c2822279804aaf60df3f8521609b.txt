REM DEFINE CONSTANTS FOR INPUT VALUES
DIM PI AS SINGLE
CONST ANGLE1 = 45
CONST ANGLE2 = 90
CONST ANGLE3 = 135

REM DECLARE VARIABLES
DIM SIDE1, SIDE2, SIDE3, TRIANGLE_TYPE AS STRING
DIM ANGLE1_RAD, ANGLE2_RAD, ANGLE3_RAD AS SINGLE

REM INPUT AND CONVERT ANGLES TO RADIANS
INPUT "ENTER LENGTH OF SIDE 1: ", SIDE1
INPUT "ENTER LENGTH OF SIDE 2: ", SIDE2
INPUT "ENTER LENGTH OF SIDE 3: ", SIDE3
ANGLE1_RAD = ANGLE1 * PI / 180
ANGLE2_RAD = ANGLE2 * PI / 180
ANGLE3_RAD = ANGLE3 * PI / 180

REM CALCULATE TRIANGLE TYPE
IF (SIDE1 = SIDE2 AND SIDE1 = SIDE3) THEN
    TRIANGLE_TYPE = "EQUILATERAL TRIANGLE"
ELSE IF (ANGLE1_RAD = ANGLE2_RAD OR ANGLE1_RAD = ANGLE3_RAD OR ANGLE2_RAD = ANGLE3_RAD) THEN
    TRIANGLE_TYPE = "ISOSCELES TRIANGLE"
ELSE IF (SIDE1^2 + SIDE2^2 = SIDE3^2 OR SIDE1^2 + SIDE3^2 = SIDE2^2 OR SIDE2^2 + SIDE3^2 = SIDE1^2) THEN
    TRIANGLE_TYPE = "RIGHT ANGLED TRIANGLE"
ELSE
    TRIANGLE_TYPE = "SCALABLE TRIANGLE"
END IF

REM OUTPUT TRIANGLE TYPE
PRINT "THE TRIANGLE YOU ENTERED IS A " + TRIANGLE_TYPE
