' *** Program to find the average of three numbers without using arrays ***

' Define variables for input values and average
a, b, c, avg AS Integer

' Get input from user
PRINT "Enter three numbers:"
INPUT a, b, c

' Calculate average
avg = (a + b + c) / 3

' Print result
PRINT "The average is: "; avg

' Example output:
' Enter three numbers:
' 3
' 5
' 7
' The average is: 5
