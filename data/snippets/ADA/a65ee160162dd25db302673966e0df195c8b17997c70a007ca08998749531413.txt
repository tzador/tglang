with Ada.Text_IO;                     -- import Text_IO package

procedure Main is                      -- main procedure
begin
   Ada.Text_IO.Put_Line("Hello World!");      -- print "Hello World!"
   Ada.Text_IO.Put_Line("This is a code snippet in ADA.");       -- print "This is a code snippet in ADA."
   Ada.Text_IO.New_Line;                  -- print new line
end Main;                               -- end of main procedure
