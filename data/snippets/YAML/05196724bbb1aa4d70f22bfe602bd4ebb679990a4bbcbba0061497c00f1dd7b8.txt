# This is a YAML code snippet that illustrates the usage of lists and dictionaries
# Lists are denoted by a hyphen (-) followed by an item, and dictionaries use key: value pairs

# Example of a list with strings as items
fruits:
  - apple
  - banana
  - orange

# Example of a dictionary with strings and integers as values
people:
  John: 32
  Jane: 27
  Bob: 45

# Example of a list with dictionaries as items
movies:
  - title: The Shawshank Redemption
    year: 1994
  - title: The Godfather
    year: 1972
  - title: The Dark Knight
    year: 2008

# Example of a dictionary with lists as values
car_brands:
  Toyota: [Camry, Corolla, Prius]
  Honda: [Accord, Civic, CR-V]
  Ford: [F-150, Mustang, Explorer]

# Example of a more complex dictionary with nested structures
students:
  Alice:
    age: 21
    major: Computer Science
    courses: [Programming, Data Structures, Algorithms]
  Bob:
    age: 23
    major: Economics
    courses: [Microeconomics, Macroeconomics]
    clubs: [Economics Club, Debate Club]

# The indentation in YAML is significant and should be consistent within each level
# In this snippet, lists and dictionaries are used to store and organize data in a readable and structured manner.
