### This is a YAML file example
---
# Comments start with a hash symbol
# YAML stands for "YAML Ain't Markup Language"
# It's a human-readable data serialization language

name: John Smith
age: 28
address:
  city: New York City
  state: New York
  country: USA
hobbies:
  - hiking # YAML supports lists
  - reading
  - coding
  - playing video games
education:
  degree: Bachelor of Science # YAML allows for nested values
  major: Computer Science
  university: New York University
  graduation_year: 2015
  courses: # YAML also supports nested lists
    - Introduction to Programming
    - Data Structures
    - Algorithms
    - Web Development
  awards: 
    - Dean's List
    - Outstanding Computer Science Student Award
    - Hackathon Winner
# YAML is whitespace sensitive, indentation determines hierarchy
