# Defines a list of employees in a company
employees:
  - name: John Smith # Employee's name
    age: 35 # Employee's age
    position: Senior Developer # Employee's job position
    projects: # Employee's list of projects
      - name: Project A # Project's name
        description: A web app for managing customer orders # Project's description
        status: In Progress # Project's current status
      - name: Project B
        description: A mobile app for tracking inventory
        status: Completed
  - name: Jane Doe
    age: 28
    position: Marketing Manager
    projects:
      - name: Project C
        description: A marketing campaign for new product launch
        status: On Hold
      - name: Project D
        description: A social media strategy for increasing brand awareness
        status: In Progress
      - name: Project E
        description: An email marketing campaign for promoting holiday sales
        status: Completed
