Key: value # This is a basic key-value pair
list_key:
  - item1 # This is the first item in the list
  - item2 # This is the second item in the list
nested_key:
  key1: value1 # This is a nested key-value pair
  key2: value2
  list_key2:
    - item3 # This is the first item in the second list
    - item4 # This is the second item in the second list
  key3:
    nested_key2:
      key4: value4 # This is a deeply nested key-value pair
      list_key3:
        - item5 # First item in the third list
        - item6 # Second item in the third list
