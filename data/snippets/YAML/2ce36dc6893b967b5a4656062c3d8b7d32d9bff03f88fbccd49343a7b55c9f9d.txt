# Define a data structure for a recipe
recipe:
  name: Spaghetti Bolognese
  cuisine: Italian
  difficulty: medium
  ingredients:
    - spaghetti
    - ground beef
    - diced tomatoes
    - onions
    - garlic
    - basil
    - oregano
    - grated parmesan cheese
  instructions:
    - Cook spaghetti according to package instructions
    - Brown ground beef in a pan
    - Add diced tomatoes, onions, and garlic to the pan and cook for 5 minutes
    - Add basil and oregano to the pan and continue cooking for an additional 5 minutes
    - Serve sauce over cooked spaghetti and top with grated parmesan cheese
