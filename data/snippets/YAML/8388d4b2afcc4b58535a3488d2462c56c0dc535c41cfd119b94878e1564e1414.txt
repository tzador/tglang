# This is a sample YAML file
# It contains basic information about a person

person:
  name: Sarah Jones  # name of the person
  age: 29            # age of the person
  address:           # home address of the person
    - street: 123 Main St
      city: New York
      state: NY
      zip: 10001
  occupation: Doctor # occupation of the person
  hobbies:           # list of hobbies
    - reading
    - hiking
    - cooking
  spouse: John Jones  # name of person's spouse
  
# Comments are denoted by the # symbol
# Lists are denoted by a hyphen (-) 
# Nested objects are created through indentation 
# Key-value pairs are denoted by a colon (:)
# Strings can be written in either single or double quotes
# Numeric values do not need quotes
# Root level objects are separated by line breaks
