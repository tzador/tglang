# This is a sample code snippet in YAML
# It shows how to define variables and maps in YAML

# Variable declaration
variables:
  - name: "Language"
    value: "YAML"

  - name: "Version"
    value: "1.2"

# Map declaration
map:
  key1: value1
  key2: value2
  key3: value3

# Nested maps
nested_map:
  key1:
    key1a: value1a
    key1b: value1b
  key2:
    key2a: value2a
    key2b: value2b
  key3:
    key3a: value3a
    key3b: value3b

# List declaration
list:
  - item1
  - item2
  - item3

# Nested lists
nested_list:
  - item1:
    - sublist1
    - sublist1
  - item2:
    - sublist2
    - sublist2
  - item3:
    - sublist3
    - sublist3
