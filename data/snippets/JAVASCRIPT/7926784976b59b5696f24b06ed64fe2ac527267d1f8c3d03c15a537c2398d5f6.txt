const name = "John"; // declaring a variable and assigning a string value
const age = 30; // declaring a variable and assigning a numeric value

// creating a function that takes in two parameters
function greeting(name, age) {
  console.log(`Hello, my name is ${name} and I am ${age} years old.`); // prints out a greeting message with the name and age
}

greeting(name, age); // calling the function and passing in the variables
