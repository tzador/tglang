// This program calculates the average of three numbers

let num1 = 5; // first number
let num2 = 10; // second number
let num3 = 15; // third number
let sum = num1 + num2 + num3; // sum of all three numbers

let average = sum / 3; // calculates the average by dividing the sum by 3

console.log(average); // prints the average to the console
