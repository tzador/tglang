const num = 5; // declaring a constant variable and assigning it a value of 5
let count = 0; // declaring a variable using let keyword and assigning it a value of 0
for (let i = 1; i <= num; i++) { // creating a for loop that iterates from 1 to the value of num
  count += i; // adding the current value of i to count
}

console.log(count); // printing the final value of count after the for loop has finished

// Output: 15 (since 1+2+3+4+5 = 15)

// This code snippet demonstrates the use of constants, variables, and a for loop in JavaScript. It also shows how to update a variable within a loop and how to print the final result in the console.
