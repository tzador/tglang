let count = 0; // initializing a variable

/**
 * This function increases the value of the count
 * by one every time it is called.
 */
function increaseCount() {
  count++; // incrementing the count by one
}

// calling the function four times
increaseCount();
increaseCount();
increaseCount();
increaseCount();

console.log(count); // prints the value of count which is 4
