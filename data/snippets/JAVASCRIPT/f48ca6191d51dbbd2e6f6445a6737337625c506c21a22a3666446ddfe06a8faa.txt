// This code creates a function called "multiply" that takes in two numbers as parameters
// and returns the product of the two numbers.
function multiply(num1, num2) {
  return num1 * num2; // This line multiplies the two numbers and returns the product
}

// This code creates a variable called "result" and assigns it the value of calling the "multiply" function
// with the arguments 5 and 10.
let result = multiply(5, 10);

// This code logs the value of the "result" variable to the console.
console.log(`The result of multiplying 5 and 10 is: ${result}`);
