const addNumbers = (x, y) => { // defining a function with parameters x and y
  return x + y; // returning the sum of x and y
}; // end of function
let num1 = 5; // declaring a variable with value 5
let num2 = 10; // declaring a variable with value 10
let result = addNumbers(num1, num2); // calling the addNumbers function with arguments num1 and num2 and storing the result in variable result
console.log('The result of adding ' + num1 + ' and ' + num2 + ' is ' + result); // printing the result to the console with concatenated strings
