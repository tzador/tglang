// Declare a variable called "num1" and assign it a value of 5
let num1 = 5;

// Declare a variable called "num2" and assign it a value of 7
let num2 = 7;

// Declare a function named "multiply" that takes in two parameters, "num1" and "num2"
function multiply(num1, num2) {
  // Multiply "num1" and "num2" and store the result in a variable called "result"
  let result = num1 * num2;
  // Return the "result" variable
  return result;
}

// Call the "multiply" function with the values of "num1" and "num2" as arguments and store the returned value in a variable called "output"
let output = multiply(num1, num2);

// Print the value of "output" to the console
console.log(output);

// Output: 35
