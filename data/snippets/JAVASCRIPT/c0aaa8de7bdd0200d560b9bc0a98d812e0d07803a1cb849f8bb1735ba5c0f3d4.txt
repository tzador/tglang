function calculateSum(number1, number2) {
  // This function calculates the sum of two numbers
  return number1 + number2;
}

const num1 = 5; // Assigning value 5 to variable num1
const num2 = 7; // Assigning value 7 to variable num2

console.log(calculateSum(num1, num2)); // Calling the calculateSum function and passing in variables num1 and num2 as arguments

// Output: 12
