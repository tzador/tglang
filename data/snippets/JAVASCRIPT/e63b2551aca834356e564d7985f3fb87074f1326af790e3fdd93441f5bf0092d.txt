// This code snippet is a function that calculates the area of a circle
// Function parameters: radius (number)
// Output: area of the circle (number)

function calculateArea(radius) {
  let area = Math.PI * radius * radius; // calculate the area using the formula πr²
  return area; // return the calculated area
}

console.log(calculateArea(5)); // calling the function with a radius of 5 and logging the output to the console
// Output: 78.53981633974483
