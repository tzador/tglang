/*
This function adds two numbers
@param {number} num1 - the first number
@param {number} num2 - the second number
@param {number} - the result of adding num1 and num2
*/
function addNumbers(num1, num2) {
  // declare a variable "result" and assign it the sum of num1 and num2
  const result = num1 + num2;
  // return the result
  return result;
}
// call the addNumbers function and pass in the numbers 5 and 10 as arguments
addNumbers(5, 10);
// expected output: 15
