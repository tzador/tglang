/**
 * Prompts user for their name, age, and favorite color
 */
let name = prompt("What is your name?"); // Gets user input for name
let age = prompt("How old are you?"); // Gets user input for age
let color = prompt("What is your favorite color?"); // Gets user input for favorite color

/**
 * Prints out greeting using template literals
 */
console.log(`Hello ${name}, nice to meet you!`); // Uses template literal to print out personalized greeting
console.log(`You are ${age} years old and your favorite color is ${color}.`); // Uses template literal to print out user's information
