let arr = [1, 2, 3, 4, 5]; // initializes an array with 5 elements
let sum = 0; // initializes a variable to store the sum
for (let i = 0; i < arr.length; i++) { //loops through the array
  sum += arr[i]; //adds the current element to the sum variable
}
console.log(sum); //prints the sum to the console
