//This program calculates the average of three numbers and displays it on the console
//declaring and initializing variables
let num1 = 9;
let num2 = 7;
let num3 = 12;

//calculating the total sum
let total = num1 + num2 + num3;

//calculating the average
let average = total / 3;

//displaying the average on the console
console.log(`The average of ${num1}, ${num2}, and ${num3} is ${average}.`);

//output: The average of 9, 7, and 12 is 9.333333333333334.
