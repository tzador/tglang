// Define a GraphQL schema
const typeDefs = `
  type Query {
    greeting: String!
  }
`;

// Define resolvers
const resolvers = {
  Query: {
    greeting: () => {
      return "Hello World!";
    }
  }
};

// Create an Apollo server
const server = new ApolloServer({
  typeDefs,
  resolvers
});

// Start the server
server.listen().then(({ url }) => {
  console.log(`ğŸš€ Server ready at ${url}`);
});
