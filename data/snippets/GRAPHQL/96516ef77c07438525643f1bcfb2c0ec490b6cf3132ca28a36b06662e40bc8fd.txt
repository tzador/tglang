type User {
  id: ID!
  name: String!
  email: String!
  age: Int!
  posts: [Post!]!
}

type Post {
  id: ID!
  title: String!
  content: String!
  author: User!
}

type Mutation {
  createUser(name: String!, email: String!, age: Int!): User!
}

type Query {
  users: [User!]!
  user(id: ID!): User
}
